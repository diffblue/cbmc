32 exprt field_sensitivityt::apply(
33  const namespacet &ns,
34  goto_symex_statet &state,
35  exprt expr,
36  bool write) const
37 {
38  if(expr.id() != ID_address_of)
39  {
40  Forall_operands(it, expr)
41  *it = apply(ns, state, std::move(*it), write);
42  }
43 
44  if(!write && is_ssa_expr(expr))
45  {
46  return get_fields(ns, state, to_ssa_expr(expr));
47  }
48  else if(
49  !write && expr.id() == ID_member &&
50  to_member_expr(expr).struct_op().id() == ID_struct)
51  {
52  return simplify_opt(std::move(expr), ns);
53  }
54 #ifdef ENABLE_ARRAY_FIELD_SENSITIVITY
55  else if(
56  !write && expr.id() == ID_index &&
57  to_index_expr(expr).array().id() == ID_array)
58  {
59  return simplify_opt(std::move(expr), ns);
60  }
61 #endif // ENABLE_ARRAY_FIELD_SENSITIVITY
62  else if(expr.id() == ID_member)
63  {
64  // turn a member-of-an-SSA-expression into a single SSA expression, thus
65  // encoding the member as an individual symbol rather than only the full
66  // struct
67  member_exprt &member = to_member_expr(expr);
68 
69  if(
70  is_ssa_expr(member.struct_op()) &&
71  (member.struct_op().type().id() == ID_struct ||
72  member.struct_op().type().id() == ID_struct_tag))
73  {
74  // place the entire member expression, not just the struct operand, in an
75  // SSA expression
76  ssa_exprt tmp = to_ssa_expr(member.struct_op());
77  bool was_l2 = !tmp.get_level_2().empty();
78 
79  tmp.remove_level_2();
80  member.struct_op() = tmp.get_original_expr();
81  tmp.set_expression(member);
82  if(was_l2)
83  return state.rename(std::move(tmp), ns).get();
84  else
85  return std::move(tmp);
86  }
87  }
88 #ifdef ENABLE_ARRAY_FIELD_SENSITIVITY
89  else if(expr.id() == ID_index)
90  {
91  // turn a index-of-an-SSA-expression into a single SSA expression, thus
92  // encoding the index access into an array as an individual symbol rather
93  // than only the full array
94  index_exprt &index = to_index_expr(expr);
95  if(should_simplify)
96  simplify(index.index(), ns);
97 
98  if(
99  is_ssa_expr(index.array()) && index.array().type().id() == ID_array &&
100  index.index().id() == ID_constant)
101  {
102  // place the entire index expression, not just the array operand, in an
103  // SSA expression
104  ssa_exprt tmp = to_ssa_expr(index.array());
105  auto l2_index = state.rename(index.index(), ns);
106  if(should_simplify)
107  l2_index.simplify(ns);
108  bool was_l2 = !tmp.get_level_2().empty();
109  exprt l2_size =
110  state.rename(to_array_type(index.array().type()).size(), ns).get();
111  if(l2_size.is_nil() && index.array().id() == ID_symbol)
112  {
113  // In case the array type was incomplete, attempt to retrieve it from
114  // the symbol table.
115  const symbolt *array_from_symbol_table = ns.get_symbol_table().lookup(
116  to_symbol_expr(index.array()).get_identifier());
117  if(array_from_symbol_table != nullptr)
118  l2_size = to_array_type(array_from_symbol_table->type).size();
119  }
120 
121  if(
122  l2_size.id() == ID_constant &&
123  numeric_cast_v<mp_integer>(to_constant_expr(l2_size)) <=
124  max_field_sensitivity_array_size)
125  {
126  if(l2_index.get().id() == ID_constant)
127  {
128  // place the entire index expression, not just the array operand,
129  // in an SSA expression
130  ssa_exprt ssa_array = to_ssa_expr(index.array());
131  ssa_array.remove_level_2();
132  index.array() = ssa_array.get_original_expr();
133  index.index() = l2_index.get();
134  tmp.set_expression(index);
135  if(was_l2)
136  return state.rename(std::move(tmp), ns).get();
137  else
138  return std::move(tmp);
139  }
140  else if(!write)
141  {
142  // Expand the array and return `{array[0]; array[1]; ...}[index]`
143  exprt expanded_array =
144  get_fields(ns, state, to_ssa_expr(index.array()));
145  return index_exprt{std::move(expanded_array), index.index()};
146  }
147  }
148  }
149  }
150 #endif // ENABLE_ARRAY_FIELD_SENSITIVITY
151  return expr;
152 }

154 exprt field_sensitivityt::get_fields(
155  const namespacet &ns,
156  goto_symex_statet &state,
157  const ssa_exprt &ssa_expr) const
158 {
159  if(ssa_expr.type().id() == ID_struct || ssa_expr.type().id() == ID_struct_tag)
160  {
161  const struct_typet &type = to_struct_type(ns.follow(ssa_expr.type()));
162  const struct_union_typet::componentst &components = type.components();
163 
164  struct_exprt::operandst fields;
165  fields.reserve(components.size());
166 
167  const exprt &struct_op = ssa_expr.get_original_expr();
168 
169  for(const auto &comp : components)
170  {
171  ssa_exprt tmp = ssa_expr;
172  bool was_l2 = !tmp.get_level_2().empty();
173  tmp.remove_level_2();
174  tmp.set_expression(member_exprt{struct_op, comp.get_name(), comp.type()});
175  if(was_l2)
176  {
177  fields.push_back(state.rename(get_fields(ns, state, tmp), ns).get());
178  }
179  else
180  fields.push_back(get_fields(ns, state, tmp));
181  }
182 
183  return struct_exprt(std::move(fields), ssa_expr.type());
184  }
185 #ifdef ENABLE_ARRAY_FIELD_SENSITIVITY
186  else if(
187  ssa_expr.type().id() == ID_array &&
188  to_array_type(ssa_expr.type()).size().id() == ID_constant)
189  {
190  const mp_integer mp_array_size = numeric_cast_v<mp_integer>(
191  to_constant_expr(to_array_type(ssa_expr.type()).size()));
192  if(mp_array_size < 0 || mp_array_size > max_field_sensitivity_array_size)
193  return ssa_expr;
194 
195  const array_typet &type = to_array_type(ssa_expr.type());
196  const std::size_t array_size = numeric_cast_v<std::size_t>(mp_array_size);
197 
198  array_exprt::operandst elements;
199  elements.reserve(array_size);
200 
201  const exprt &array = ssa_expr.get_original_expr();
202 
203  for(std::size_t i = 0; i < array_size; ++i)
204  {
205  const index_exprt index(array, from_integer(i, type.index_type()));
206  ssa_exprt tmp = ssa_expr;
207  bool was_l2 = !tmp.get_level_2().empty();
208  tmp.remove_level_2();
209  tmp.set_expression(index);
210  if(was_l2)
211  {
212  elements.push_back(state.rename(get_fields(ns, state, tmp), ns).get());
213  }
214  else
215  elements.push_back(get_fields(ns, state, tmp));
216  }
217 
218  return array_exprt(std::move(elements), type);
219  }
220 #endif // ENABLE_ARRAY_FIELD_SENSITIVITY
221  else
222  return ssa_expr;
223 }

252 void field_sensitivityt::field_assignments_rec(
253  const namespacet &ns,
254  goto_symex_statet &state,
255  const exprt &lhs_fs,
256  const exprt &ssa_rhs,
257  symex_targett &target,
258  bool allow_pointer_unsoundness) const
259 {

262  const ssa_exprt ssa_lhs = state

264  to_ssa_expr(lhs_fs),

267  true,
268  true,
269  allow_pointer_unsoundness)
270  .get();
271 
272  // do the assignment
273  target.assignment(
274  state.guard.as_expr(),
275  ssa_lhs,
276  ssa_lhs,
277  ssa_lhs.get_original_expr(),
278  ssa_rhs,
279  state.source,
280  symex_targett::assignment_typet::STATE);
281  }
282  else if(
283  ssa_rhs.type().id() == ID_struct || ssa_rhs.type().id() == ID_struct_tag)
284  {
285  const struct_typet &type = to_struct_type(ns.follow(ssa_rhs.type()));
286  const struct_union_typet::componentst &components = type.components();
287 
288  PRECONDITION(
289  components.empty() ||
290  (lhs_fs.has_operands() && lhs_fs.operands().size() == components.size()));
291 
292  exprt::operandst::const_iterator fs_it = lhs_fs.operands().begin();
293  for(const auto &comp : components)
294  {
295  const exprt member_rhs =
296  simplify_opt(member_exprt{ssa_rhs, comp.get_name(), comp.type()}, ns);
297  const exprt &member_lhs = *fs_it;
298 
299  field_assignments_rec(
300  ns, state, member_lhs, member_rhs, target, allow_pointer_unsoundness);
301  ++fs_it;
302  }
303  }
304 #ifdef ENABLE_ARRAY_FIELD_SENSITIVITY
305  else if(const auto &type = type_try_dynamic_cast<array_typet>(ssa_rhs.type()))
306  {
307  const std::size_t array_size =
308  numeric_cast_v<std::size_t>(to_constant_expr(type->size()));
309  PRECONDITION(lhs_fs.operands().size() == array_size);
310 
311  if(array_size > max_field_sensitivity_array_size)
312  return;
313 
314  exprt::operandst::const_iterator fs_it = lhs_fs.operands().begin();
315  for(std::size_t i = 0; i < array_size; ++i)
316  {
317  const exprt index_rhs = simplify_opt(
318  index_exprt{ssa_rhs, from_integer(i, type->index_type())}, ns);
319  const exprt &index_lhs = *fs_it;
320 
321  field_assignments_rec(
322  ns, state, index_lhs, index_rhs, target, allow_pointer_unsoundness);
323  ++fs_it;
324  }
325  }
326 #endif // ENABLE_ARRAY_FIELD_SENSITIVITY
327  else if(lhs_fs.has_operands())
328  {
329  PRECONDITION(
330  ssa_rhs.has_operands() &&
331  lhs_fs.operands().size() == ssa_rhs.operands().size());
332 
333  exprt::operandst::const_iterator fs_it = lhs_fs.operands().begin();
334  for(const exprt &op : ssa_rhs.operands())
335  {
336  field_assignments_rec(
337  ns, state, *fs_it, op, target, allow_pointer_unsoundness);
338  ++fs_it;
339  }
340  }
341  else
342  {
343  UNREACHABLE;
344  }
