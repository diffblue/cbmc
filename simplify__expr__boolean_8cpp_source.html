228 simplify_exprt::resultt<> simplify_exprt::simplify_not(const not_exprt &expr)
229 {
230  const exprt &op = expr.op();
231 
232  if(!expr.is_boolean() || !op.is_boolean())
233  {
234  return unchanged(expr);
235  }
236 
237  if(op.id()==ID_not) // (not not a) == a
238  {
239  return to_not_expr(op).op();
240  }
241  else if(op.is_false())
242  {
243  return true_exprt();
244  }
245  else if(op.is_true())
246  {
247  return false_exprt();
248  }
249  else if(op.id()==ID_and ||
250  op.id()==ID_or)
251  {
252  exprt tmp = op;
253 
254  Forall_operands(it, tmp)
255  {
256  *it = simplify_not(not_exprt(*it));
257  }
258 
259  tmp.id(tmp.id() == ID_and ? ID_or : ID_and);
260 
261  return std::move(tmp);
262  }
263  else if(op.id()==ID_notequal) // !(a!=b) <-> a==b
264  {
265  exprt tmp = op;
266  tmp.id(ID_equal);
267  return std::move(tmp);
268  }
269  else if(op.id()==ID_exists) // !(exists: a) <-> forall: not a
270  {
271  auto const &op_as_exists = to_exists_expr(op);
272  return forall_exprt{op_as_exists.variables(),
273  simplify_not(not_exprt(op_as_exists.where()))};
274  }
275  else if(op.id() == ID_forall) // !(forall: a) <-> exists: not a
276  {
277  auto const &op_as_forall = to_forall_expr(op);
278  return exists_exprt{op_as_forall.variables(),
279  simplify_not(not_exprt(op_as_forall.where()))};
280  }
