53 exprt simplify_evaluate_update(
54  evaluate_exprt evaluate_expr,
55  const std::unordered_set<symbol_exprt, irep_hash> &address_taken,
56  const namespacet &ns)
57 {
58  PRECONDITION(evaluate_expr.state().id() == ID_update_state);
59 
60  const auto &update_state_expr = to_update_state_expr(evaluate_expr.state());
61 
62 #if 0
63  std::cout << "U: " << format(update_state_expr) << "\n";
64  std::cout << "u: " << format(update_state_expr.address()) << "\n";
65  std::cout << "T: " << format(update_state_expr.address().type()) << "\n";
66  std::cout << "E: " << format(evaluate_expr.address()) << "\n";
67  std::cout << "T: " << format(evaluate_expr.address().type()) << "\n";
68 #endif
69 
70  auto may_alias = ::may_alias(
71  evaluate_expr.address(), update_state_expr.address(), address_taken, ns);
72 
73 #if 0
74  if(may_alias.has_value())
75  std::cout << "M: " << format(*may_alias) << "\n";
76  else
77  std::cout << "M: ?\n";
78 #endif
79 
80  if(may_alias.has_value())
81  {
82  // 'simple' case
83  if(may_alias->is_true())
84  {
85  // The object is known to be the same.
86  // (ς[A:=V])(A) --> V
87  return simplify_state_expr_node(
88  typecast_exprt::conditional_cast(
89  update_state_expr.new_value(), evaluate_expr.type()),
90  address_taken,
91  ns);
92  }
93  else if(may_alias->is_false())
94  {
95  // The object is known to be different.
96  // (ς[❝x❞:=V])(❝y❞) --> ς(❝y❞)
97  // It might be possible to further simplify ς(❝y❞).
98  return simplify_state_expr_node(
99  evaluate_expr.with_state(update_state_expr.state()), address_taken, ns);
100  }
101  else
102  {
103  // The object may or may not be the same.
104  // (ς[A:=V])(B) --> if cond then V else ς(B) endif
105  auto simplified_cond = simplify_state_expr(*may_alias, address_taken, ns);
106  auto new_evaluate_expr =
107  evaluate_expr.with_state(update_state_expr.state());
108  auto simplified_new_evaluate_expr = simplify_state_expr_node(
109  new_evaluate_expr, address_taken, ns); // rec. call
110  auto if_expr = if_exprt(
111  std::move(simplified_cond),
112  simplify_state_expr_node(
113  typecast_exprt::conditional_cast(
114  update_state_expr.new_value(), evaluate_expr.type()),
115  address_taken,
116  ns),
117  std::move(simplified_new_evaluate_expr));
118  return simplify_expr(if_expr, ns);
119  }
120  }
121 
122  auto new_evaluate_expr = evaluate_expr.with_state(update_state_expr.state());
123  auto simplified_new_evaluate_expr =
124  simplify_state_expr(new_evaluate_expr, address_taken, ns); // rec. call
125 
126  // Types are compatible?
127  if(types_are_compatible(
128  update_state_expr.new_value().type(), evaluate_expr.type()))
129  {
130  // Disregard case where the two memory regions overlap.
131  //
132  // (ς[w:=v])(r) -->
133  // IF same_object(w, r) ∧ offset(w) = offset(r) THEN
134  // v
135  // ELSE
136  // ς(r)
137  // ENDIF
138  auto same_object =
139  ::same_object(evaluate_expr.address(), update_state_expr.address());
140 
141  auto simplified_same_object =
142  simplify_state_expr(same_object, address_taken, ns);
143 
144  auto offset_w = simplify_state_expr(
145  pointer_offset(evaluate_expr.address()), address_taken, ns);
146 
147  auto offset_r = simplify_state_expr(
148  pointer_offset(update_state_expr.address()), address_taken, ns);
149 
150  auto same_offset = equal_exprt(offset_w, offset_r);
151 
152  auto same = and_exprt(simplified_same_object, same_offset);
153 
154  auto simplified_same =
155  simplify_state_expr(simplify_expr(same, ns), address_taken, ns);
156 
157  auto new_value = typecast_exprt::conditional_cast(
158  update_state_expr.new_value(), evaluate_expr.type());
159 
160  auto if_expr =
161  if_exprt(simplified_same, new_value, simplified_new_evaluate_expr);
162 
163  return simplify_expr(if_expr, ns);
164  }
165  else
166  {
167  // Complex case. Types don't match.
168  return simplified_new_evaluate_expr;
169 
170 #if 0
171  auto object = update_state_expr.new_value();
172 
173  auto offset = simplify_state_expr_node(
174  pointer_offset(evaluate_expr.address()), address_taken, ns);
175 
176  auto byte_extract = make_byte_extract(object, offset, evaluate_expr.type());
177 
178  return if_exprt(
179  std::move(simplified_same_object),
180  std::move(byte_extract),
181  std::move(simplified_new_evaluate_expr));
182 #endif
183  }

316 exprt simplify_live_object_expr(
317  state_live_object_exprt src,
318  const std::unordered_set<symbol_exprt, irep_hash> &address_taken,
319  const namespacet &ns)
320 {
321  const auto &pointer = src.address();
322 
323  auto object = simplify_object_expression(pointer);
324 
325  if(object.id() == ID_object_address)
326  {
327  auto identifier = to_object_address_expr(object).object_identifier();
328 
329  if(has_prefix(id2string(identifier), "allocate-"))
330  {
331  }
332  else if(identifier == "return_value")
333  {
334  return true_exprt(); // never dies
335  }
336  else if(has_prefix(id2string(identifier), "va_args::"))
337  {
338  return true_exprt(); // might be 'dead'
339  }
340  else
341  {
342  const auto &symbol = ns.lookup(identifier);
343  if(symbol.is_static_lifetime)
344  {
345  return true_exprt(); // always live
346  }
347  }
348  }
349 
350  // A store does not affect the result.
351  // live_object(ς[A:=V]), p) --> live_object(ς, p)
352  if(src.state().id() == ID_update_state)
353  {
354  src.state() = to_update_state_expr(src.state()).state();
355 
356  // rec. call
357  return simplify_live_object_expr(std::move(src), address_taken, ns);
358  }
359  else if(src.state().id() == ID_deallocate_state)
360  {
361  const auto &deallocate_state_expr = to_deallocate_state_expr(src.state());
362  // live_object(deallocate_state(ς, p), q) -->
363  // IF same_object(p, q) THEN false ELSE live_object(ς, q) ENDIF
364  auto same_object = ::same_object(object, deallocate_state_expr.address());
365  auto simplified_same_object =
366  simplify_state_expr(same_object, address_taken, ns);
367  auto new_live_object_expr = simplify_live_object_expr(
368  src.with_state(deallocate_state_expr.state()), address_taken, ns);
369  return if_exprt(
370  simplified_same_object, false_exprt(), new_live_object_expr);
371  }
372  else if(src.state().id() == ID_enter_scope_state)
373  {
374  // This begins the life of a local-scoped variable.
375  const auto &enter_scope_state_expr = to_enter_scope_state_expr(src.state());
376  if(
377  address_taken.find(enter_scope_state_expr.symbol()) !=
378  address_taken.end())
379  {
380  // live_object(enter_scope_state(ς, p), q) -->
381  // IF same_object(p, q) THEN true ELSE live_object(ς, q) ENDIF
382  auto same_object =
383  ::same_object(object, enter_scope_state_expr.address());
384  auto simplified_same_object =
385  simplify_state_expr(same_object, address_taken, ns);
386  auto new_live_object_expr = simplify_live_object_expr( // rec. call
387  src.with_state(enter_scope_state_expr.state()),
388  address_taken,
389  ns);
390  return if_exprt(
391  simplified_same_object, true_exprt(), new_live_object_expr);
392  }
393  else
394  {
395  return simplify_live_object_expr( // rec. call
396  src.with_state(enter_scope_state_expr.state()),
397  address_taken,
398  ns);
399  }
400  }
401  else if(src.state().id() == ID_exit_scope_state)
402  {
403  // This ends the life of a local-scoped variable.
404  const auto &exit_scope_state_expr = to_exit_scope_state_expr(src.state());
405  if(
406  address_taken.find(exit_scope_state_expr.symbol()) != address_taken.end())
407  {
408  // live_object(exit_scope_state(ς, p), q) -->
409  // IF same_object(p, q) THEN false ELSE live_object(ς, q) ENDIF
410  auto same_object = ::same_object(object, exit_scope_state_expr.address());
411  auto simplified_same_object =
412  simplify_state_expr(same_object, address_taken, ns);
413  auto new_live_object_expr = simplify_live_object_expr( // rec. call
414  src.with_state(exit_scope_state_expr.state()),
415  address_taken,
416  ns);
417  return if_exprt(
418  simplified_same_object, false_exprt(), new_live_object_expr);
419  }
420  else
421  {
422  return simplify_live_object_expr( // rec. call
423  src.with_state(exit_scope_state_expr.state()),
424  address_taken,
425  ns);
426  }
427  }
428 
429  return std::move(src);
430 }

901  pointer_offset_exprt src,
902  const std::unordered_set<symbol_exprt, irep_hash> &address_taken,
903  const namespacet &ns)
904 {
905  if(src.pointer().id() == ID_object_address)
906  {
907  // pointer_offset(❝x❞) -> 0
908  return from_integer(0, src.type());
909  }
910  else if(src.pointer().id() == ID_element_address)
911  {
912  const auto &element_address_expr = to_element_address_expr(src.pointer());
913  // pointer_offset(element_address(Z, y)) -->
914  // pointer_offset(Z) + y*sizeof(x)
915  auto size_opt = size_of_expr(element_address_expr.element_type(), ns);
916  if(size_opt.has_value())
917  {
918  auto product = mult_exprt(
919  typecast_exprt::conditional_cast(
920  element_address_expr.index(), src.type()),
921  typecast_exprt::conditional_cast(*size_opt, src.type()));
922  auto pointer_offset = simplify_pointer_offset_expr(
923  pointer_offset_exprt(element_address_expr.base(), src.type()),
924  address_taken,
925  ns);
926  auto sum = plus_exprt(pointer_offset, std::move(product));
927  return std::move(sum);
928  }
929  }
930  else if(src.pointer().id() == ID_address_of)
931  {
932  const auto &address_of_expr = to_address_of_expr(src.pointer());
933  if(address_of_expr.object().id() == ID_string_constant)

941  return simplify_pointer_offset_expr(
942  pointer_offset_exprt(to_typecast_expr(src.pointer()).op(), src.type()),
943  address_taken,
944  ns);
945  }
