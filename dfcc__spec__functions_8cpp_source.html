48 void dfcc_spec_functionst::generate_havoc_function(
49  const irep_idt &function_id,
50  const irep_idt &havoc_function_id,
51  std::size_t &nof_targets)
52 {
53  INVARIANT(
54  !goto_model.symbol_table.has_symbol(havoc_function_id),
55  "DFCC: havoc function id '" + id2string(havoc_function_id) +
56  "' already exists");
57 
58  const auto &function_symbol = utils.get_function_symbol(function_id);
59 
60  // create the write_set symbol used as input by the havoc function
61  const auto &write_set_symbol = utils.create_symbol(
62  library.dfcc_type[dfcc_typet::CAR_SET_PTR],
63  id2string(havoc_function_id),
64  "__write_set_to_havoc",
65  function_symbol.location,
66  function_symbol.mode,
67  function_symbol.module,
68  true);
69 
70  // create the code type that goes on the function symbol
71  code_typet::parametert write_set_param(write_set_symbol.type);
72  write_set_param.set_base_name(write_set_symbol.base_name);
73  write_set_param.set_identifier(write_set_symbol.name);
74  code_typet havoc_code_type({write_set_param}, empty_typet());
75 
76  // create the havoc function symbol
77  symbolt havoc_function_symbol;
78  havoc_function_symbol.base_name = havoc_function_id;
79  havoc_function_symbol.name = havoc_function_id;
80  havoc_function_symbol.pretty_name = havoc_function_id;
81  havoc_function_symbol.type = havoc_code_type;
82  havoc_function_symbol.mode = function_symbol.mode;
83  havoc_function_symbol.module = function_symbol.module;
84  havoc_function_symbol.location = function_symbol.location;
85  havoc_function_symbol.set_compiled();
86  bool add_function_failed = goto_model.symbol_table.add(havoc_function_symbol);
87  INVARIANT(
88  !add_function_failed,
89  "DFCC: could not insert havoc function '" + id2string(havoc_function_id) +
90  "' in the symbol table");
91 
92  // create new goto_function
93  goto_functiont dummy_havoc_function;
94  dummy_havoc_function.set_parameter_identifiers(havoc_code_type);
95  goto_model.goto_functions.function_map[havoc_function_id].copy_from(
96  dummy_havoc_function);
97 
98  // body will be filled with instructions
99  auto &body =
100  goto_model.goto_functions.function_map.at(havoc_function_id).body;
101 
102  // index of the CAR to havoc in the write set
103  std::size_t next_idx = 0;
104 
105  // iterate on the body of the original function and emit one havoc instruction
106  // per target
107  Forall_goto_program_instructions(
108  ins_it, goto_model.goto_functions.function_map.at(function_id).body)
109  {
110  if(ins_it->is_function_call())
111  {
112  if(ins_it->call_function().id() != ID_symbol)
113  {
114  throw invalid_source_file_exceptiont(
115  "Function pointer call '" +
116  from_expr(ns, function_id, ins_it->call_function()) +
117  "' in function '" + id2string(function_id) + "' is not supported",
118  ins_it->source_location());
119  }
120 
121  const irep_idt &callee_id =
122  to_symbol_expr(ins_it->call_function()).get_identifier();
123 
124  // only process built-in functions that return assignable_t,
125  // error out on any other function call
126  // find the corresponding instrumentation hook
127  auto hook_opt = library.get_havoc_hook(callee_id);
128  INVARIANT(
129  hook_opt.has_value(),
130  "dfcc_spec_functionst::generate_havoc_function: function calls must "
131  "be inlined before calling this function");
132 
133  // Use same source location as original call
134  source_locationt location(ins_it->source_location());
135  auto hook = hook_opt.value();
136  code_function_callt call(
137  library.dfcc_fun_symbol.at(hook).symbol_expr(),
138  {write_set_symbol.symbol_expr(), from_integer(next_idx, size_type())});
139 
140  if(hook == dfcc_funt::WRITE_SET_HAVOC_GET_ASSIGNABLE_TARGET)
141  {
142  // ```
143  // DECL __havoc_target;
144  // CALL __havoc_target = havoc_hook(set, next_idx);
145  // IF !__havoc_target GOTO label;
146  // ASSIGN *__havoc_target = nondet(target_type);
147  // label: DEAD __havoc_target;
148  // ```
149  // declare a local var to store targets havoced via nondet assignment
150  auto &target_type = get_target_type(ins_it->call_arguments().at(0));
151 
152  const auto &target_symbol = utils.create_symbol(
153  pointer_type(target_type),
154  id2string(havoc_function_id),
155  "__havoc_target",
156  location,
157  function_symbol.mode,
158  function_symbol.module,
159  false);
160 
161  auto target_expr = target_symbol.symbol_expr();
162  body.add(goto_programt::make_decl(target_expr));
163 
164  call.lhs() = target_expr;
165  body.add(goto_programt::make_function_call(call, location));
166 
167  auto goto_instruction = body.add(goto_programt::make_incomplete_goto(
168  utils.make_null_check_expr(target_expr), location));
169 
170  // create nondet assignment to the target
171  side_effect_expr_nondett nondet(target_type, location);
172  body.add(goto_programt::make_assignment(
173  dereference_exprt{typecast_exprt::conditional_cast(
174  target_expr, pointer_type(target_type))},
175  nondet,
176  location));
177  auto label = body.add(goto_programt::make_dead(target_expr));
178  goto_instruction->complete_goto(label);
179  }
180  else if(
181  hook == dfcc_funt::WRITE_SET_HAVOC_OBJECT_WHOLE ||
182  hook == dfcc_funt::WRITE_SET_HAVOC_SLICE)
183  {
184  // ```
185  // CALL havoc_hook(set, next_idx);
186  // ```
187  body.add(goto_programt::make_function_call(call, location));
188  }
189  else
190  {
191  UNREACHABLE;
192  }
193  ++next_idx;
194  }
195  nof_targets = next_idx;
196  }
197 
198  body.add(goto_programt::make_end_function());
199 
200  goto_model.goto_functions.update();
201 
202  std::set<irep_idt> no_body;
203  std::set<irep_idt> missing_function;
204  std::set<irep_idt> recursive_call;
205  std::set<irep_idt> not_enough_arguments;
206  utils.inline_function(
207  havoc_function_id,
208  no_body,
209  recursive_call,
210  missing_function,
211  not_enough_arguments);
212  INVARIANT(
213  no_body.empty(),
214  "no body warnings when inlining " + id2string(havoc_function_id));
215  INVARIANT(
216  missing_function.empty(),
217  "missing function warnings when inlining " + id2string(havoc_function_id));
218  INVARIANT(
219  recursive_call.empty(),
220  "recursive calls when inlining " + id2string(havoc_function_id));
221  INVARIANT(
222  not_enough_arguments.empty(),
223  "not enough arguments when inlining " + id2string(havoc_function_id));
224 
225  utils.set_hide(havoc_function_id, true);
226 
227  goto_model.goto_functions.update();
228 }

230 void dfcc_spec_functionst::to_spec_assigns_function(
231  const irep_idt &function_id,
232  std::size_t &nof_targets)
233 {
234  // counts the number of calls to built-ins to get an over approximation
235  // of the size of the set
236  std::size_t next_idx = 0;
237 
238  auto &goto_function = goto_model.goto_functions.function_map.at(function_id);
239 
240  // add write_set parameter
241  auto &set_symbol = utils.add_parameter(
242  function_id,
243  "__write_set_to_fill",
244  library.dfcc_type[dfcc_typet::WRITE_SET_PTR]);
245 
246  // rewrite calls
247  Forall_goto_program_instructions(ins_it, goto_function.body)
248  {
249  if(ins_it->is_function_call())
250  {
251  if(ins_it->call_function().id() != ID_symbol)
252  {
253  throw invalid_source_file_exceptiont(
254  "Function pointer call '" +
255  from_expr(ns, function_id, ins_it->call_function()) +
256  "' in function '" + id2string(function_id) + "' is not supported",
257  ins_it->source_location());
258  }
259 
260  const irep_idt &callee_id =
261  to_symbol_expr(ins_it->call_function()).get_identifier();
262 
263  // only process built-in functions that return assignable_t,
264  // error out on any other function call
265  // find the corresponding instrumentation hook
266  INVARIANT(
267  library.is_front_end_builtin(callee_id),
268  "dfcc_spec_functionst::to_spec_assigns_function: function calls must "
269  "be inlined before calling this function");
270 
271  auto hook = library.get_hook(callee_id);
272  // redirect the call to the hook
273  ins_it->call_function() = library.dfcc_fun_symbol.at(hook).symbol_expr();
274  // insert insertion index argument
275  ins_it->call_arguments().insert(
276  ins_it->call_arguments().begin(), from_integer(next_idx, size_type()));
277  // insert write set argument
278  ins_it->call_arguments().insert(
279  ins_it->call_arguments().begin(), set_symbol.symbol_expr());
280 
281  // remove the is_pointer_to_pointer argument which is not used in the
282  // hook for insert assignable
283  if(hook == dfcc_funt::WRITE_SET_INSERT_ASSIGNABLE)
284  ins_it->call_arguments().pop_back();
285 
286  ++next_idx;
287  }
288  }
289 
290  nof_targets = next_idx;
291  goto_model.goto_functions.update();
292 
293  // instrument for side-effects checking
294  std::set<irep_idt> function_pointer_contracts;
295  instrument.instrument_function(function_id, function_pointer_contracts);
296  INVARIANT(
297  function_pointer_contracts.size() == 0,
298  "discovered function pointer contracts unexpectedly");
299  utils.set_hide(function_id, true);
300 }
