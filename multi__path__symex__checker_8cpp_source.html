32 incremental_goto_checkert::resultt multi_path_symex_checkert::
33 operator()(propertiest &properties)
34 {
35  resultt result(resultt::progresst::DONE);
36 
37  // When the equation has been generated, we know all the properties.
38  // Have we got anything to check? Otherwise we return DONE.
39  if(equation_generated && !has_properties_to_check(properties))
40  return result;
41 
42  std::chrono::duration<double> solver_runtime(0);
43 
44  // we haven't got an equation yet
45  if(!equation_generated)
46  {
47  generate_equation();
48 
49  output_coverage_report(
50  options.get_option("symex-coverage-report"),
51  goto_model,
52  symex,
53  ui_message_handler);
54 
55  update_properties(properties, result.updated_properties);
56 
57  // Have we got anything to check? Otherwise we return DONE.
58  if(!has_properties_to_check(properties))
59  return result;
60 
61  solver_runtime += prepare_property_decider(properties);
62 
63  equation_generated = true;
64  }
65 
66  run_property_decider(result, properties, solver_runtime);
67 
68  return result;
69 }
