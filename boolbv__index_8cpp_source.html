282 bvt boolbvt::convert_index(
283  const exprt &array,
284  const mp_integer &index)
285 {
286  const array_typet &array_type = to_array_type(array.type());
287 
288  std::size_t width = boolbv_width(array_type.element_type());
289 
290  // TODO: If the underlying array can use one of the
291  // improvements given above then it may be better to use
292  // the array theory for short chains of updates and then
293  // the improved array handling rather than full flattening.
294  // Note that the calculation is non-trivial as the cost of
295  // full flattening is amortised against all uses of
296  // the array (constant and variable indexes) and updated
297  // versions of it.
298 
299  const bvt &tmp=convert_bv(array); // recursive call
300 
301  mp_integer offset=index*width;
302 
303  if(offset>=0 &&
304  offset+width<=mp_integer(tmp.size()))
305  {
306  // in bounds
307 
308  // The assertion below is disabled as we want to be able
309  // to run CBMC without simplifier.
310  // Expression simplification should remove these cases
311  // assert(array.id()!=ID_array_of &&
312  // array.id()!=ID_array);
313  // If not there are large improvements possible as above
314 
315  std::size_t offset_int = numeric_cast_v<std::size_t>(offset);
316  return bvt(tmp.begin() + offset_int, tmp.begin() + offset_int + width);
317  }
318  else if(array.id() == ID_member || array.id() == ID_index)
319  {
320  // out of bounds for the component, but not necessarily outside the bounds
321  // of the underlying object
322  object_descriptor_exprt o;
323  o.build(array, ns);
324  CHECK_RETURN(o.offset().id() != ID_unknown);
325 
326  const auto subtype_bytes_opt =
327  pointer_offset_size(array_type.element_type(), ns);
328  CHECK_RETURN(subtype_bytes_opt.has_value());
329 
330  exprt new_offset = simplify_expr(
331  plus_exprt(
332  o.offset(), from_integer(index * (*subtype_bytes_opt), o.offset().type())),
333  ns);
334 
335  byte_extract_exprt be =
336  make_byte_extract(o.root_object(), new_offset, array_type.element_type());
337 
338  return convert_bv(be);
339  }
340  else if(
341  array.id() == ID_byte_extract_big_endian ||
342  array.id() == ID_byte_extract_little_endian)
343  {
344  const byte_extract_exprt &byte_extract_expr = to_byte_extract_expr(array);
345 
346  const auto subtype_bytes_opt =
347  pointer_offset_size(array_type.element_type(), ns);
348  CHECK_RETURN(subtype_bytes_opt.has_value());
349 
350  // add offset to index
351  exprt new_offset = simplify_expr(
352  plus_exprt{
353  byte_extract_expr.offset(),
354  from_integer(
355  index * (*subtype_bytes_opt), byte_extract_expr.offset().type())},
356  ns);
357 
358  byte_extract_exprt be = byte_extract_expr;
359  be.offset() = new_offset;
360  be.type() = array_type.element_type();
361 
362  return convert_bv(be);
363  }
364  else
365  {
366  // out of bounds
367  return prop.new_variables(width);
368  }
369 }
