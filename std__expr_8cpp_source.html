172 exprt binding_exprt::instantiate(const operandst &values) const
173 {
174  // number of values must match the number of bound variables
175  auto &variables = this->variables();
176  PRECONDITION(variables.size() == values.size());
177 
178  std::map<symbol_exprt, exprt> value_map;
179 
180  for(std::size_t i = 0; i < variables.size(); i++)
181  {
182  // types must match
183  PRECONDITION(variables[i].type() == values[i].type());
184  value_map[variables[i]] = values[i];
185  }
186 
187  // build a substitution map
188  std::map<irep_idt, exprt> substitutions;
189 
190  for(std::size_t i = 0; i < variables.size(); i++)
191  substitutions[variables[i].get_identifier()] = values[i];
192 
193  // now recurse downwards and substitute in 'where'
194  auto substitute_result = substitute_symbols(substitutions, where());
195 
196  if(substitute_result.has_value())
197  return substitute_result.value();
198  else
199  return where(); // trivial case, variables not used
200 }
