56 static const exprt &
57 get_checked_pointer_from_null_pointer_check(const exprt &violation)
58 {
59  // A NULL-pointer check is the negation of an equation between the checked
60  // pointer and a NULL pointer.
61  // ! (POINTER_OBJECT(NULL) == POINTER_OBJECT(ptr))
62  const equal_exprt &equal_expr = to_equal_expr(to_not_expr(violation).op());
63 
64  const pointer_object_exprt &lhs_pointer_object =
65  to_pointer_object_expr(equal_expr.lhs());
66  const pointer_object_exprt &rhs_pointer_object =
67  to_pointer_object_expr(equal_expr.rhs());
68 
69  const exprt &lhs_pointer = lhs_pointer_object.operands()[0];
70  const exprt &rhs_pointer = rhs_pointer_object.operands()[0];
71 
72  // NULL == ptr
73  if(
74  can_cast_expr<constant_exprt>(lhs_pointer) &&
75  is_null_pointer(*expr_try_dynamic_cast<constant_exprt>(lhs_pointer)))
76  {
77  return rhs_pointer;
78  }
79 
80  // Not a equation with NULL on one side.
81  UNREACHABLE;
82 }

125 cext::violation_typet
126 cegis_verifiert::extract_violation_type(const std::string &description)
127 {
128  // The violation is a pointer OOB check.
129  if((description.find(
130  "dereference failure: pointer outside object bounds in") !=
131  std::string::npos))
132  {
133  return cext::violation_typet::cex_out_of_boundary;
134  }
135 
136  // The violation is a null pointer check.
137  if(description.find("pointer NULL") != std::string::npos)
138  {
139  return cext::violation_typet::cex_null_pointer;
140  }
141 
142  // The violation is a loop-invariant-preservation check.
143  if(description.find("preserved") != std::string::npos)
144  {
145  return cext::violation_typet::cex_not_preserved;
146  }
147 
148  // The violation is a loop-invariant-preservation check.
149  if(description.find("invariant before entry") != std::string::npos)
150  {
151  return cext::violation_typet::cex_not_hold_upon_entry;
152  }
153 
154  // The violation is an assignable check.
155  if(description.find("assignable") != std::string::npos)
156  {
157  return cext::violation_typet::cex_assignable;
158  }
159 
160  return cext::violation_typet::cex_other;

163 std::list<loop_idt>
164 cegis_verifiert::get_cause_loop_id_for_assigns(const goto_tracet &goto_trace)
165 {
166  std::list<loop_idt> result;
167 
168  // We say a loop is the cause loop of an assignable-violation if the inclusion
169  // check is in the loop.
170 
171  // So we check what loops the last step of the trace is in.
172  // Transformed loop head:
173  // ASSIGN entered_loop = false
174  // Transformed loop end:
175  // ASSIGN entered_loop = true
176  for(const auto &step : goto_trace.steps)
177  {
178  // We are entering a loop.
179  if(is_transformed_loop_head(step.pc))
180  {
181  result.push_front(
182  loop_idt(step.function_id, original_loop_number_map[step.pc]));
183  }
184  // We are leaving a loop.
185  else if(is_transformed_loop_end(step.pc))
186  {
187  const loop_idt loop_id(
188  step.function_id, original_loop_number_map[step.pc]);
189  INVARIANT(
190  result.front() == loop_id, "Leaving a loop we haven't entered.");
191  result.pop_front();
192  }
193  }
194 
195  INVARIANT(!result.empty(), "The assignable violation is not in a loop.");
196  return result;
197 }

199 std::list<loop_idt> cegis_verifiert::get_cause_loop_id(
200  const goto_tracet &goto_trace,
201  const goto_programt::const_targett violation)
202 {
203  std::list<loop_idt> result;
204 
205  // build the dependence graph
206  const namespacet ns(goto_model.symbol_table);
207  dependence_grapht dependence_graph(ns);
208  dependence_graph(goto_model);
209 
210  // Checking if `to` is dependent on `from`.
211  // `from` : loop havocing
212  // `to` : violation
213 
214  // Get `to`---the instruction where the violation happens
215  irep_idt to_fun_name = goto_trace.get_last_step().function_id;
216  const goto_functionst::goto_functiont &to_function =
217  goto_model.get_goto_function(to_fun_name);
218  goto_programt::const_targett to = to_function.body.instructions.end();
219  for(goto_programt::const_targett it = to_function.body.instructions.begin();
220  it != to_function.body.instructions.end();
221  ++it)
222  {
223  if(it->location_number == violation->location_number)
224  {
225  to = it;
226  }
227  }
228 
229  INVARIANT(
230  to != to_function.body.instructions.end(),
231  "There must be a violation in a trace.");
232 
233  // Compute the backward reachable set.
234  const auto reachable_vector =
235  dependence_graph.get_reachable(dependence_graph[to].get_node_id(), false);
236  const std::set<size_t> reachable_set =
237  std::set<size_t>(reachable_vector.begin(), reachable_vector.end());
238 
239  // A loop is the cause loop if the violation is dependent on the write set of
240  // the loop.
241  for(const auto &step : goto_trace.steps)
242  {
243  // Being dependent on a write set is equivalent to being dependent on one
244  // of the loop havocing instruction.
245  if(loop_havoc_set.count(step.pc))
246  {
247  // Get `from`---a loop havoc instruction.
248  irep_idt from_fun_name = step.function_id;
249  const goto_functionst::goto_functiont &from_function =
250  goto_model.get_goto_function(from_fun_name);
251  goto_programt::const_targett from = from_function.body.instructions.end();
252  for(goto_programt::const_targett it =
253  from_function.body.instructions.begin();
254  it != from_function.body.instructions.end();
255  ++it)
256  {
257  if(it->location_number == step.pc->location_number)
258  {
259  from = it;
260  }
261  }
262 
263  INVARIANT(
264  from != from_function.body.instructions.end(),
265  "Failed to find the location number of the loop havoc.");
266 
267  // The violation is caused by the loop havoc
268  // if it is dependent on the loop havoc.
269  if(reachable_set.count(dependence_graph[from].get_node_id()))
270  {
271  result.push_back(
272  loop_idt(step.function_id, original_loop_number_map[step.pc]));
273  return result;
274  }
275  }
276  }
277  return result;
278 }

300 bool cegis_verifiert::is_instruction_in_transformed_loop_condition(
301  const loop_idt &loop_id,
302  const goto_functiont &function,
303  unsigned location_number_of_target)
304 {
305  // The transformed loop condition is a set of instructions from
306  // loop havocing instructions
307  // to
308  // if(!guard) GOTO EXIT
309  unsigned location_number_of_havocing = 0;
310  for(auto it = function.body.instructions.begin();
311  it != function.body.instructions.end();
312  ++it)
313  {
314  // Record the location number of the beginning of a transformed loop.
315  if(
316  loop_havoc_set.count(it) &&
317  original_loop_number_map[it] == loop_id.loop_number)
318  {
319  location_number_of_havocing = it->location_number;
320  }
321 
322  // Reach the end of the evaluation of the transformed loop condition.
323  if(location_number_of_havocing != 0 && it->is_goto())
324  {
325  if((location_number_of_havocing < location_number_of_target &&
326  location_number_of_target < it->location_number))
327  {
328  return true;
329  }
330  location_number_of_havocing = 0;
331  }
332  }
333  return false;
334 }

336 bool cegis_verifiert::is_instruction_in_transformed_loop(
337  const loop_idt &loop_id,
338  const goto_functiont &function,
339  unsigned location_number_of_target)
340 {
341  // The loop body after transformation are instructions from
342  // loop havocing instructions
343  // to
344  // loop end of transformed code.
345  unsigned location_number_of_havocing = 0;
346 
347  for(goto_programt::const_targett it = function.body.instructions.begin();
348  it != function.body.instructions.end();
349  ++it)
350  {
351  // Record the location number of the begin of a transformed loop.
352  if(
353  loop_havoc_set.count(it) &&
354  original_loop_number_map[it] == loop_id.loop_number)
355  {
356  location_number_of_havocing = it->location_number;
357  }
358 
359  // Reach the end of a transformed loop.
360  if(
361  is_transformed_loop_end(it) &&
362  original_loop_number_map[it] == loop_id.loop_number)
363  {
364  INVARIANT(
365  location_number_of_havocing != 0,
366  "We must have entered the transformed loop before reaching the end");
367 
368  // Check if `location_number_of_target` is between the begin and the end
369  // of the transformed loop.
370  if((location_number_of_havocing < location_number_of_target &&
371  location_number_of_target < it->location_number))
372  {
373  return true;
374  }
375  }
376  }
377 
378  return false;
379 }

381 cext cegis_verifiert::build_cex(
382  const goto_tracet &goto_trace,
383  const source_locationt &loop_entry_loc)
384 {
385  const namespacet ns(goto_model.symbol_table);
386 
387  // Valuations of havoced variables right after havoc instructions.
388  std::unordered_map<exprt, mp_integer, irep_hash> object_sizes;
389  std::unordered_map<exprt, mp_integer, irep_hash> havoced_values;
390  std::unordered_map<exprt, mp_integer, irep_hash> havoced_pointer_offsets;
391 
392  // Loop-entry valuations of havoced variables.
393  std::unordered_map<exprt, mp_integer, irep_hash> loop_entry_values;
394  std::unordered_map<exprt, mp_integer, irep_hash> loop_entry_offsets;
395 
396  // Live variables upon the loop head.
397  std::set<exprt> live_variables;
398 
399  bool entered_loop = false;
400 
401  // Scan the trace step by step to store needed valuations.
402  for(const auto &step : goto_trace.steps)
403  {
404  switch(step.type)
405  {
406  case goto_trace_stept::typet::DECL:
407  case goto_trace_stept::typet::ASSIGNMENT:
408  {
409  if(!step.full_lhs_value.is_nil())
410  {
411  // Entered loop?
412  if(is_assignment_to_instrumented_variable(step.pc, ENTERED_LOOP))
413  entered_loop = step.full_lhs_value == true_exprt();
414 
415  // skip hidden steps
416  if(step.hidden)
417  break;
418 
419  // Live variables
420  // 1. must be in the same function as the target loop;
421  // 2. alive before entering the target loop;
422  // 3. a pointer or a primitive-typed variable;
423  // TODO: add support for union pointer
424  if(
425  step.pc->source_location().get_function() ==
426  loop_entry_loc.get_function() &&
427  !entered_loop &&
428  (step.full_lhs.type().id() == ID_unsignedbv ||
429  step.full_lhs.type().id() == ID_signedbv ||
430  step.full_lhs.type().id() == ID_pointer) &&
431  step.full_lhs.id() == ID_symbol)
432  {
433  const auto &symbol =
434  expr_try_dynamic_cast<symbol_exprt>(step.full_lhs);
435 
436  // malloc_size is not-hidden tmp variable.
437  if(id2string(symbol->get_identifier()) != "malloc::malloc_size")
438  {
439  live_variables.emplace(step.full_lhs);
440  }
441  }
442 
443  // Record valuation of primitive-typed variable.
444  if(
445  step.full_lhs.type().id() == ID_unsignedbv ||
446  step.full_lhs.type().id() == ID_signedbv)
447  {
448  bool is_signed = step.full_lhs.type().id() == ID_signedbv;
449  const auto &bv_type =
450  type_try_dynamic_cast<bitvector_typet>(step.full_lhs.type());
451  const auto width = bv_type->get_width();
452  // Store the value into the map for loop_entry value if we haven't
453  // entered the loop.
454  if(!entered_loop)
455  {
456  loop_entry_values[step.full_lhs] = bvrep2integer(
457  step.full_lhs_value.get_string(ID_value), width, is_signed);
458  }
459 
460  // Store the value into the the map for havoced value if this step
461  // is a loop havocing instruction.
462  if(loop_havoc_set.count(step.pc))
463  {
464  havoced_values[step.full_lhs] = bvrep2integer(
465  step.full_lhs_value.get_string(ID_value), width, is_signed);
466  }
467  }
468 
469  // Record object_size, pointer_offset, and loop_entry(pointer_offset).
470  if(
471  can_cast_expr<annotated_pointer_constant_exprt>(
472  step.full_lhs_value) &&
473  contains_symbol_prefix(
474  step.full_lhs_value, SYMEX_DYNAMIC_PREFIX "::dynamic_object"))
475  {
476  const auto &pointer_constant_expr =
477  expr_try_dynamic_cast<annotated_pointer_constant_exprt>(
478  step.full_lhs_value);
479 
480  pointer_arithmetict pointer_arithmetic(
481  pointer_constant_expr->symbolic_pointer());
482  if(pointer_constant_expr->symbolic_pointer().id() == ID_typecast)
483  {
484  pointer_arithmetic = pointer_arithmetict(
485  pointer_constant_expr->symbolic_pointer().operands()[0]);
486  }
487 
488  // Extract object size.
489  exprt &underlying_array = pointer_arithmetic.pointer;
490  // Object sizes are stored in the type of underlying arrays.
491  while(!can_cast_type<array_typet>(underlying_array.type()))
492  {
493  if(
494  underlying_array.id() == ID_address_of ||
495  underlying_array.id() == ID_index)
496  {
497  underlying_array = underlying_array.operands()[0];
498  continue;
499  }
500  UNREACHABLE;
501  }
502  mp_integer object_size =
503  pointer_offset_size(to_array_type(underlying_array.type()), ns)
504  .value();
505  object_sizes[step.full_lhs] = object_size;
506 
507  // Extract offsets.
508  mp_integer offset = 0;
509  if(pointer_arithmetic.offset.is_not_nil())
510  {
511  constant_exprt offset_expr =
512  expr_dynamic_cast<constant_exprt>(pointer_arithmetic.offset);
513  offset = bvrep2integer(
514  offset_expr.get_value(), size_type().get_width(), false);
515  }
516 
517  // Store the offset into the map for loop_entry if we haven't
518  // entered the loop.
519  if(!entered_loop)
520  {
521  loop_entry_offsets[step.full_lhs] = offset;
522  }
523 
524  // Store the offset into the the map for havoced offset if this step
525  // is a loop havocing instruction.
526  if(loop_havoc_set.count(step.pc))
527  {
528  havoced_pointer_offsets[step.full_lhs] = offset;
529  }
530  }
531  }
532  }
533 
534  case goto_trace_stept::typet::ASSERT:
535  case goto_trace_stept::typet::FUNCTION_CALL:
536  case goto_trace_stept::typet::FUNCTION_RETURN:
537  case goto_trace_stept::typet::ASSUME:
538  case goto_trace_stept::typet::LOCATION:
539  case goto_trace_stept::typet::GOTO:
540  case goto_trace_stept::typet::OUTPUT:
541  case goto_trace_stept::typet::INPUT:
542  case goto_trace_stept::typet::SPAWN:
543  case goto_trace_stept::typet::MEMORY_BARRIER:
544  case goto_trace_stept::typet::ATOMIC_BEGIN:
545  case goto_trace_stept::typet::ATOMIC_END:
546  case goto_trace_stept::typet::DEAD:
547  case goto_trace_stept::typet::CONSTRAINT:
548  case goto_trace_stept::typet::SHARED_READ:
549  case goto_trace_stept::typet::SHARED_WRITE:
550  break;
551 
552  case goto_trace_stept::typet::NONE:
553  UNREACHABLE;
554  }
555  }
556 
557  return cext(
558  object_sizes,
559  havoced_values,
560  havoced_pointer_offsets,
561  loop_entry_values,
562  loop_entry_offsets,
563  live_variables);
564 }

576 optionalt<cext> cegis_verifiert::verify()
577 {
578  // This method does the following three things to verify the `goto_model` and
579  // return a formatted counterexample if there is any violated property.
580  //
581  // 1. annotate and apply the loop contracts stored in `invariant_candidates`.
582  //
583  // 2. run the CBMC API to verify the instrumented goto model. As the API is
584  // not merged yet, we preprocess the goto model and run the symex checker
585  // on it to simulate CBMC API.
586  // TODO: ^^^ replace the symex checker once the real API is merged.
587  //
588  // 3. construct the formatted counterexample from the violated property and
589  // its trace.
590 
591  const namespacet ns(goto_model.symbol_table);
592 
593  // Store the original functions. We will restore them after the verification.
594  for(const auto &fun_entry : goto_model.goto_functions.function_map)
595  {
596  original_functions[fun_entry.first].copy_from(fun_entry.second.body);
597  }
598 
599  // Annotate the candidates to the goto_model for checking.
600  annotate_invariants(invariant_candidates, goto_model);
601 
602  // Annotate assigns
603  annotate_assigns(assigns_map, goto_model);
604 
605  // Control verbosity. We allow non-error output message only when verbosity
606  // is set to larger than messaget::M_DEBUG.
607  const unsigned original_verbosity = log.get_message_handler().get_verbosity();
608  if(original_verbosity < messaget::M_DEBUG)
609  log.get_message_handler().set_verbosity(messaget::M_ERROR);
610 
611  // Apply loop contracts we annotated.
612  code_contractst cont(goto_model, log);
613  cont.apply_loop_contracts();
614  original_loop_number_map = cont.get_original_loop_number_map();
615  loop_havoc_set = cont.get_loop_havoc_set();
616 
617  // Get the checker same as CBMC api without any flag.
618  // TODO: replace the checker with CBMC api once it is implemented.
619  ui_message_handlert ui_message_handler(log.get_message_handler());
620  const auto options = get_options();
621  std::unique_ptr<
622  all_properties_verifier_with_trace_storaget<multi_path_symex_checkert>>
623  checker = util_make_unique<
624  all_properties_verifier_with_trace_storaget<multi_path_symex_checkert>>(
625  options, ui_message_handler, goto_model);
626 
627  goto_convert(
628  goto_model.symbol_table,
629  goto_model.goto_functions,
630  log.get_message_handler());
631 
632  // Run the checker to get the result.
633  const resultt result = (*checker)();
634 
635  if(original_verbosity >= messaget::M_DEBUG)
636  checker->report();
637 
638  // Restore the verbosity.
639  log.get_message_handler().set_verbosity(original_verbosity);
640 
641  //
642  // Start to construct the counterexample.
643  //
644 
645  if(result == resultt::PASS)
646  {
647  restore_functions();
648  return optionalt<cext>();
649  }
650 
651  if(result == resultt::ERROR || result == resultt::UNKNOWN)
652  {
653  INVARIANT(false, "Verification failed during loop contract synthesis.");
654  }
655 
656  properties = checker->get_properties();
657  auto target_violation = properties.end();
658 
659  // Find target violation---the violation we want to fix next.
660  // A target violation is an assignable violation or the first violation that
661  // is not assignable violation.
662  for(auto it_property = properties.begin(); it_property != properties.end();
663  it_property++)
664  {
665  if(it_property->second.status != property_statust::FAIL)
666  continue;
667 
668  // assignable violation found
669  if(it_property->second.description.find("assignable") != std::string::npos)
670  {
671  target_violation = it_property;
672  break;
673  }
674 
675  // Store the violation that we want to fix with synthesized
676  // assigns/invariant.
677  // ignore ASSERT FALSE
678  if(
679  target_violation == properties.end() &&
680  simplify_expr(it_property->second.pc->condition(), ns) != false_exprt())
681  {
682  target_violation = it_property;
683  }
684  }
685 
686  // All violations are
687  // ASSERT FALSE
688  if(target_violation == properties.end())
689  {
690  restore_functions();
691  return optionalt<cext>();
692  }
693 
694  target_violation_id = target_violation->first;
695 
696  // Decide the violation type from the description of violation
697  cext::violation_typet violation_type =
698  extract_violation_type(target_violation->second.description);
699 
700  // Compute the cause loop---the loop for which we synthesize loop contracts,
701  // and the counterexample.
702 
703  // If the violation is an assignable check, we synthesize assigns targets.
704  // In the case, cause loops are all loops the violation is in. We keep
705  // adding the new assigns target to the most-inner loop that does not
706  // contain the new target until the assignable violation is resolved.
707 
708  // For other cases, we synthesize loop invariant clauses. We synthesize
709  // invariants for one loop at a time. So we return only the first cause loop
710  // although there can be multiple ones.
711 
712  log.debug() << "Start to compute cause loop ids." << messaget::eom;
713  log.debug() << "Violation description: "
714  << target_violation->second.description << messaget::eom;
715 
716  const auto &trace = checker->get_traces()[target_violation->first];
717  // Doing assigns-synthesis or invariant-synthesis
718  if(violation_type == cext::violation_typet::cex_assignable)
719  {
720  cext result(violation_type);
721  result.cause_loop_ids = get_cause_loop_id_for_assigns(trace);
722  result.checked_pointer = static_cast<const exprt &>(
723  target_violation->second.pc->condition().find(ID_checked_assigns));
724  restore_functions();
725  return result;
726  }
727 
728  // We construct the full counterexample only for violations other than
729  // assignable checks.
730 
731  // Although there can be multiple cause loop ids. We only synthesize
732  // loop invariants for the first cause loop.
733  const std::list<loop_idt> cause_loop_ids =
734  get_cause_loop_id(trace, target_violation->second.pc);
735 
736  if(cause_loop_ids.empty())
737  {
738  log.debug() << "No cause loop found!" << messaget::eom;
739  restore_functions();
740 
741  return cext(violation_type);
742  }
743 
744  log.debug() << "Found cause loop with function id: "
745  << cause_loop_ids.front().function_id
746  << ", and loop number: " << cause_loop_ids.front().loop_number
747  << messaget::eom;
748 
749  auto violation_location = cext::violation_locationt::in_loop;
750  // We always strengthen in_clause if the violation is
751  // invariant-not-preserved.
752  if(violation_type != cext::violation_typet::cex_not_preserved)
753  {
754  // Get the location of the violation
755  violation_location = get_violation_location(
756  cause_loop_ids.front(),
757  goto_model.get_goto_function(cause_loop_ids.front().function_id),
758  target_violation->second.pc->location_number);
759  }
760 
761  restore_functions();
762 
763  auto return_cex = build_cex(
764  trace,
765  get_loop_head(
766  cause_loop_ids.front().loop_number,
767  goto_model.goto_functions
768  .function_map[cause_loop_ids.front().function_id])
769  ->source_location());
770  return_cex.violated_predicate = target_violation->second.pc->condition();
771  return_cex.cause_loop_ids = cause_loop_ids;
772  return_cex.violation_location = violation_location;
773  return_cex.violation_type = violation_type;
774 
775  // The pointer checked in the null-pointer-check violation.
776  if(violation_type == cext::violation_typet::cex_null_pointer)
777  {
778  return_cex.checked_pointer = get_checked_pointer_from_null_pointer_check(
779  target_violation->second.pc->condition());
780  }
781 
782  return return_cex;
