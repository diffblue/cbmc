36 bool is_nondet_initializable_static(
37  const symbol_exprt &symbol_expr,
38  const namespacet &ns)
39 {
40  const irep_idt &id = symbol_expr.get_identifier();
41 
42  // is it a __CPROVER_* variable?
43  if(id.starts_with(CPROVER_PREFIX))
44  return false;
45 
46  // variable not in symbol table such as symex variable?
47  if(!ns.get_symbol_table().has_symbol(id))
48  return false;
49 
50  const symbolt &symbol = ns.lookup(id);
51 
52  // is the symbol explicitly marked as not to be nondet initialized?
53  if(symbol.value.get_bool(ID_C_no_nondet_initialization))
54  {
55  return false;
56  }
57 
58  // is the type explicitly marked as not to be nondet initialized?
59  if(symbol.type.get_bool(ID_C_no_nondet_initialization))
60  return false;
61 
62  // is the type explicitly marked as not to be initialized?
63  if(symbol.type.get_bool(ID_C_no_initialization_required))
64  return false;
65 
66  // static lifetime?
67  if(!symbol.is_static_lifetime)
68  return false;
69 
70  // constant?
71  return !is_constant_or_has_constant_components(symbol_expr.type(), ns) &&
72  !is_constant_or_has_constant_components(symbol.type, ns);
73 }

82 static void nondet_static(
83  const namespacet &ns,
84  goto_functionst &goto_functions,
85  const irep_idt &fct_name)
86 {
87  goto_functionst::function_mapt::iterator fct_entry =
88  goto_functions.function_map.find(fct_name);
89  CHECK_RETURN(fct_entry != goto_functions.function_map.end());
90 
91  goto_programt &init = fct_entry->second.body;
92 
93  for(auto &instruction : init.instructions)
94  {
95  if(instruction.is_assign())
96  {
97  const symbol_exprt sym =
98  to_symbol_expr(as_const(instruction).assign_lhs());
99 
100  if(is_nondet_initializable_static(sym, ns))
101  {
102  const auto source_location = instruction.source_location();
103  instruction = goto_programt::make_assignment(
104  code_assignt(
105  sym, side_effect_expr_nondett(sym.type(), source_location)),
106  source_location);
107  }
108  }
109  else if(instruction.is_function_call())
110  {
111  const symbol_exprt &fsym = to_symbol_expr(instruction.call_function());
112 
113  // see cpp/cpp_typecheck.cpp, which creates initialization functions
114  if(fsym.get_identifier().starts_with("#cpp_dynamic_initialization#"))
115  {
116  nondet_static(ns, goto_functions, fsym.get_identifier());
117  }
118  }
119  }
120 
121  // update counters etc.
122  goto_functions.update();
123 }

151 void nondet_static(
152  goto_modelt &goto_model,
153  const std::set<std::string> &except_values)
154 {
155  const namespacet ns(goto_model.symbol_table);
156  std::set<std::string> to_exclude;
157 
158  for(auto const &except : except_values)
159  {
160  const bool file_prefix_found = except.find(":") != std::string::npos;
161 
162  if(file_prefix_found)
163  {
164  to_exclude.insert(except);
165  if(has_prefix(except, "./"))
166  {
167  to_exclude.insert(except.substr(2, except.length() - 2));
168  }
169  else
170  {
171  to_exclude.insert("./" + except);
172  }
173  }
174  else
175  {
176  irep_idt symbol_name(except);
177  symbolt lookup_results = ns.lookup(symbol_name);
178  to_exclude.insert(
179  id2string(lookup_results.location.get_file()) + ":" + except);
180  }
181  }
182 
183  symbol_tablet &symbol_table = goto_model.symbol_table;
184 
185  for(symbol_tablet::iteratort symbol_it = symbol_table.begin();
186  symbol_it != symbol_table.end();
187  symbol_it++)
188  {
189  symbolt &symbol = symbol_it.get_writeable_symbol();
190  std::string qualified_name = id2string(symbol.location.get_file()) + ":" +
191  id2string(symbol.display_name());
192  if(to_exclude.find(qualified_name) != to_exclude.end())
193  {
194  symbol.value.set(ID_C_no_nondet_initialization, 1);
195  }
196  else if(is_nondet_initializable_static(symbol.symbol_expr(), ns))
197  {
198  symbol.value = side_effect_expr_nondett(symbol.type, symbol.location);
199  }
200  }
201 
202  nondet_static(ns, goto_model.goto_functions, INITIALIZE_FUNCTION);
203 }
