302  bool contains(const source_locationt &source_location)
303  {
304  if(source_location.is_not_nil())
305  {
306  return
307  // use pessimistic highest value to compare to min for undefined
308  is_lte(
309  min_line,
310  min_col,

313  col_to_size_t(

315  // use pessimistic lowest value to compare to max for undefined
