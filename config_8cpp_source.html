760 void configt::set_arch(const irep_idt &arch)
761 {
762  ansi_c.arch=arch;
763 
764  if(arch=="none")
765  {
766  // the architecture for people who can't commit
767  ansi_c.endianness=configt::ansi_ct::endiannesst::NO_ENDIANNESS;
768  ansi_c.lib=configt::ansi_ct::libt::LIB_NONE;
769  ansi_c.NULL_is_zero=false;
770 
771  if(sizeof(long int)==8)
772  ansi_c.set_64();
773  else
774  ansi_c.set_32();
775  }
776  else if(arch=="alpha")
777  ansi_c.set_arch_spec_alpha();
778  else if(arch=="arm64" ||
779  arch=="armel" ||
780  arch=="armhf" ||
781  arch=="arm")
782  ansi_c.set_arch_spec_arm(arch);
783  else if(arch=="mips64el" ||
784  arch=="mipsn32el" ||
785  arch=="mipsel" ||
786  arch=="mips64" ||
787  arch=="mipsn32" ||
