702 void configt::set_arch(const irep_idt &arch)
703 {
704  ansi_c.arch=arch;
705 
706  if(arch=="none")
707  {
708  // the architecture for people who can't commit
709  ansi_c.endianness=configt::ansi_ct::endiannesst::NO_ENDIANNESS;
710  ansi_c.lib=configt::ansi_ct::libt::LIB_NONE;
711  ansi_c.NULL_is_zero=false;
712 
713  if(sizeof(long int)==8)
714  ansi_c.set_64();
715  else
716  ansi_c.set_32();
717  }
718  else if(arch=="alpha")
719  ansi_c.set_arch_spec_alpha();
720  else if(arch=="arm64" ||
721  arch=="armel" ||
722  arch=="armhf" ||
723  arch=="arm")
724  ansi_c.set_arch_spec_arm(arch);
725  else if(arch=="mips64el" ||
726  arch=="mipsn32el" ||
727  arch=="mipsel" ||
728  arch=="mips64" ||
729  arch=="mipsn32" ||
730  arch=="mips")
731  ansi_c.set_arch_spec_mips(arch);
732  else if(arch=="powerpc" ||
733  arch=="ppc64" ||
734  arch=="ppc64le")
735  ansi_c.set_arch_spec_power(arch);
736  else if(arch == "riscv64")
737  ansi_c.set_arch_spec_riscv64();
738  else if(arch=="sparc" ||
739  arch=="sparc64")
740  ansi_c.set_arch_spec_sparc(arch);
741  else if(arch=="ia64")
742  ansi_c.set_arch_spec_ia64();
743  else if(arch=="s390x")
744  ansi_c.set_arch_spec_s390x();
745  else if(arch=="s390")
746  ansi_c.set_arch_spec_s390();
747  else if(arch=="x32")
748  ansi_c.set_arch_spec_x32();
749  else if(arch=="v850")
750  ansi_c.set_arch_spec_v850();
751  else if(arch=="hppa")
752  ansi_c.set_arch_spec_hppa();
753  else if(arch=="sh4")
754  ansi_c.set_arch_spec_sh4();
755  else if(arch=="x86_64")
756  ansi_c.set_arch_spec_x86_64();
757  else if(arch=="i386")
758  ansi_c.set_arch_spec_i386();
759  else
760  {
761  // We run on something new and unknown.
762  // We verify for i386 instead.
763  ansi_c.set_arch_spec_i386();
764  ansi_c.arch="i386";
765  }
