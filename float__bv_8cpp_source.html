19 exprt float_bvt::convert(const exprt &expr) const
20 {
21  if(expr.id()==ID_abs)
22  return abs(to_abs_expr(expr).op(), get_spec(expr));
23  else if(expr.id()==ID_unary_minus)
24  return negation(to_unary_minus_expr(expr).op(), get_spec(expr));
25  else if(expr.id()==ID_ieee_float_equal)
26  {
27  const auto &equal_expr = to_ieee_float_equal_expr(expr);
28  return is_equal(
29  equal_expr.lhs(), equal_expr.rhs(), get_spec(equal_expr.lhs()));
30  }
31  else if(expr.id()==ID_ieee_float_notequal)
32  {
33  const auto &notequal_expr = to_ieee_float_notequal_expr(expr);
34  return not_exprt(is_equal(
35  notequal_expr.lhs(), notequal_expr.rhs(), get_spec(notequal_expr.lhs())));
36  }
37  else if(expr.id()==ID_floatbv_typecast)
38  {
39  const auto &floatbv_typecast_expr = to_floatbv_typecast_expr(expr);
40  const auto &op = floatbv_typecast_expr.op();
41  const typet &src_type = floatbv_typecast_expr.op().type();
42  const typet &dest_type = floatbv_typecast_expr.type();
43  const auto &rounding_mode = floatbv_typecast_expr.rounding_mode();
44 
45  if(dest_type.id()==ID_signedbv &&
46  src_type.id()==ID_floatbv) // float -> signed
47  return to_signed_integer(
48  op,
49  to_signedbv_type(dest_type).get_width(),
50  rounding_mode,
51  get_spec(op));
52  else if(dest_type.id()==ID_unsignedbv &&
53  src_type.id()==ID_floatbv) // float -> unsigned
54  return to_unsigned_integer(
55  op,
56  to_unsignedbv_type(dest_type).get_width(),
57  rounding_mode,
58  get_spec(op));
59  else if(src_type.id()==ID_signedbv &&
60  dest_type.id()==ID_floatbv) // signed -> float
61  return from_signed_integer(op, rounding_mode, get_spec(expr));
62  else if(src_type.id()==ID_unsignedbv &&
63  dest_type.id()==ID_floatbv) // unsigned -> float
64  {
65  return from_unsigned_integer(op, rounding_mode, get_spec(expr));
66  }
67  else if(dest_type.id()==ID_floatbv &&
68  src_type.id()==ID_floatbv) // float -> float
69  {
70  return conversion(op, rounding_mode, get_spec(op), get_spec(expr));
71  }
72  else
73  return nil_exprt();
74  }
75  else if(
76  expr.id() == ID_typecast && expr.type().id() == ID_bool &&
77  to_typecast_expr(expr).op().type().id() == ID_floatbv) // float -> bool
78  {
79  return not_exprt(is_zero(to_typecast_expr(expr).op()));
80  }
81  else if(
82  expr.id() == ID_typecast && expr.type().id() == ID_bv &&
83  to_typecast_expr(expr).op().type().id() == ID_floatbv) // float -> raw bv
84  {
85  const typecast_exprt &tc = to_typecast_expr(expr);
86  const bitvector_typet &dest_type = to_bitvector_type(expr.type());
87  const floatbv_typet &src_type = to_floatbv_type(tc.op().type());
88  if(
89  dest_type.get_width() != src_type.get_width() ||
90  dest_type.get_width() == 0)
91  {
92  return nil_exprt{};
93  }
94 
95  return extractbits_exprt{
96  to_typecast_expr(expr).op(), dest_type.get_width() - 1, 0, dest_type};
97  }
98  else if(expr.id()==ID_floatbv_plus)
99  {
100  const auto &float_expr = to_ieee_float_op_expr(expr);
101  return add_sub(
102  false,
103  float_expr.lhs(),
104  float_expr.rhs(),
105  float_expr.rounding_mode(),
106  get_spec(expr));
107  }
108  else if(expr.id()==ID_floatbv_minus)
109  {
110  const auto &float_expr = to_ieee_float_op_expr(expr);
111  return add_sub(
112  true,
113  float_expr.lhs(),
114  float_expr.rhs(),
115  float_expr.rounding_mode(),
116  get_spec(expr));
117  }
118  else if(expr.id()==ID_floatbv_mult)
119  {
120  const auto &float_expr = to_ieee_float_op_expr(expr);
121  return mul(
122  float_expr.lhs(),
123  float_expr.rhs(),
124  float_expr.rounding_mode(),
125  get_spec(expr));
126  }
127  else if(expr.id()==ID_floatbv_div)
128  {
129  const auto &float_expr = to_ieee_float_op_expr(expr);
130  return div(
131  float_expr.lhs(),
132  float_expr.rhs(),
133  float_expr.rounding_mode(),
134  get_spec(expr));
135  }
136  else if(expr.id()==ID_isnan)
137  {
138  const auto &op = to_unary_expr(expr).op();
139  return isnan(op, get_spec(op));
140  }
141  else if(expr.id()==ID_isfinite)
142  {
143  const auto &op = to_unary_expr(expr).op();
144  return isfinite(op, get_spec(op));
145  }
146  else if(expr.id()==ID_isinf)
147  {
148  const auto &op = to_unary_expr(expr).op();
149  return isinf(op, get_spec(op));
150  }
151  else if(expr.id()==ID_isnormal)
152  {
153  const auto &op = to_unary_expr(expr).op();
154  return isnormal(op, get_spec(op));
155  }
156  else if(expr.id()==ID_lt)
157  {
158  const auto &rel_expr = to_binary_relation_expr(expr);
159  return relation(
160  rel_expr.lhs(), relt::LT, rel_expr.rhs(), get_spec(rel_expr.lhs()));
161  }
162  else if(expr.id()==ID_gt)
163  {
164  const auto &rel_expr = to_binary_relation_expr(expr);
165  return relation(
166  rel_expr.lhs(), relt::GT, rel_expr.rhs(), get_spec(rel_expr.lhs()));
167  }
168  else if(expr.id()==ID_le)
169  {
170  const auto &rel_expr = to_binary_relation_expr(expr);
171  return relation(
172  rel_expr.lhs(), relt::LE, rel_expr.rhs(), get_spec(rel_expr.lhs()));
173  }
174  else if(expr.id()==ID_ge)
175  {
176  const auto &rel_expr = to_binary_relation_expr(expr);
177  return relation(
178  rel_expr.lhs(), relt::GE, rel_expr.rhs(), get_spec(rel_expr.lhs()));
179  }
180  else if(expr.id()==ID_sign)
181  return sign_bit(to_unary_expr(expr).op());
182 
183  return nil_exprt();
184 }

364 exprt float_bvt::to_integer(
365  const exprt &src,
366  std::size_t dest_width,
367  bool is_signed,
368  const exprt &rm,
369  const ieee_float_spect &spec)
370 {
371  const unbiased_floatt unpacked=unpack(src, spec);
372 
373  rounding_mode_bitst rounding_mode_bits(rm);
374 
375  // Right now hard-wired to round-to-zero, which is
376  // the usual case in ANSI-C.
377 
378  // if the exponent is positive, shift right
379  exprt offset=from_integer(spec.f, signedbv_typet(spec.e));
380  const minus_exprt distance(offset, unpacked.exponent);
381  const lshr_exprt shift_result(unpacked.fraction, distance);
382 
383  // if the exponent is negative, we have zero anyways
384  exprt result=shift_result;
385  const sign_exprt exponent_sign(unpacked.exponent);
386 
387  result=
388  if_exprt(exponent_sign, from_integer(0, result.type()), result);
389 
390  // chop out the right number of bits from the result
391  typet result_type=
392  is_signed?static_cast<typet>(signedbv_typet(dest_width)):
393  static_cast<typet>(unsignedbv_typet(dest_width));
394 
395  result=typecast_exprt(result, result_type);
396 
397  // if signed, apply sign.
398  if(is_signed)
399  {
400  result=if_exprt(
401  unpacked.sign, unary_minus_exprt(result), result);
402  }
403  else
404  {
405  // It's unclear what the behaviour for negative floats
406  // to integer shall be.
407  }
408 
409  return result;

412 exprt float_bvt::conversion(
413  const exprt &src,
414  const exprt &rm,
415  const ieee_float_spect &src_spec,
416  const ieee_float_spect &dest_spec) const
417 {
418  // Catch the special case in which we extend,
419  // e.g. single to double.
420  // In this case, rounding can be avoided,
421  // but a denormal number may be come normal.
422  // Be careful to exclude the difficult case
423  // when denormalised numbers in the old format
424  // can be converted to denormalised numbers in the
425  // new format. Note that this is rare and will only
426  // happen with very non-standard formats.
427 
428  int sourceSmallestNormalExponent = -((1 << (src_spec.e - 1)) - 1);
429  int sourceSmallestDenormalExponent =
430  sourceSmallestNormalExponent - src_spec.f;
431 
432  // Using the fact that f doesn't include the hidden bit
433 
434  int destSmallestNormalExponent = -((1 << (dest_spec.e - 1)) - 1);
435 
436  if(dest_spec.e>=src_spec.e &&
437  dest_spec.f>=src_spec.f &&
438  !(sourceSmallestDenormalExponent < destSmallestNormalExponent))
439  {
440  unbiased_floatt unpacked_src=unpack(src, src_spec);
441  unbiased_floatt result;
442 
443  // the fraction gets zero-padded
444  std::size_t padding=dest_spec.f-src_spec.f;
445  result.fraction=
446  concatenation_exprt(
447  unpacked_src.fraction,
448  from_integer(0, unsignedbv_typet(padding)),
449  unsignedbv_typet(dest_spec.f+1));
450 
451  // the exponent gets sign-extended
452  INVARIANT(
453  unpacked_src.exponent.type().id() == ID_signedbv,
454  "the exponent needs to have a signed type");
455  result.exponent=
456  typecast_exprt(unpacked_src.exponent, signedbv_typet(dest_spec.e));
457 
458  // if the number was denormal and is normal in the new format,
459  // normalise it!
460  if(dest_spec.e > src_spec.e)
461  normalization_shift(result.fraction, result.exponent);
462 
463  // the flags get copied
464  result.sign=unpacked_src.sign;
465  result.NaN=unpacked_src.NaN;
466  result.infinity=unpacked_src.infinity;
467 
468  // no rounding needed!
469  return pack(bias(result, dest_spec), dest_spec);
470  }
471  else
472  {
473  // we actually need to round
474  unbiased_floatt result=unpack(src, src_spec);
475  return rounder(result, rm, dest_spec);
476  }
477 }

508 exprt float_bvt::add_sub(
509  bool subtract,
510  const exprt &op0,
511  const exprt &op1,
512  const exprt &rm,
513  const ieee_float_spect &spec) const
514 {
515  unbiased_floatt unpacked1=unpack(op0, spec);
516  unbiased_floatt unpacked2=unpack(op1, spec);
517 
518  // subtract?
519  if(subtract)
520  unpacked2.sign=not_exprt(unpacked2.sign);
521 
522  // figure out which operand has the bigger exponent
523  const exprt exponent_difference=subtract_exponents(unpacked1, unpacked2);
524  const sign_exprt src2_bigger(exponent_difference);
525 
526  const exprt bigger_exponent=
527  if_exprt(src2_bigger, unpacked2.exponent, unpacked1.exponent);
528 
529  // swap fractions as needed
530  const exprt new_fraction1=
531  if_exprt(src2_bigger, unpacked2.fraction, unpacked1.fraction);
532 
533  const exprt new_fraction2=
534  if_exprt(src2_bigger, unpacked1.fraction, unpacked2.fraction);
535 
536  // compute distance
537  const exprt distance=
538  typecast_exprt(abs_exprt(exponent_difference), unsignedbv_typet(spec.e));
539 
540  // limit the distance: shifting more than f+3 bits is unnecessary
541  const exprt limited_dist=limit_distance(distance, spec.f+3);
542 
543  // pad fractions with 3 zeros from below
544  exprt three_zeros=from_integer(0, unsignedbv_typet(3));
545  // add 4 to spec.f because unpacked new_fraction has the hidden bit
546  const exprt fraction1_padded=
547  concatenation_exprt(new_fraction1, three_zeros, unsignedbv_typet(spec.f+4));
548  const exprt fraction2_padded=
549  concatenation_exprt(new_fraction2, three_zeros, unsignedbv_typet(spec.f+4));
550 
551  // shift new_fraction2
552  exprt sticky_bit;
553  const exprt fraction1_shifted=fraction1_padded;
554  const exprt fraction2_shifted=sticky_right_shift(
555  fraction2_padded, limited_dist, sticky_bit);
556 
557  // sticky bit: 'or' of the bits lost by the right-shift
558  const bitor_exprt fraction2_stickied(
559  fraction2_shifted,
560  concatenation_exprt(
561  from_integer(0, unsignedbv_typet(spec.f + 3)),
562  sticky_bit,
563  fraction2_shifted.type()));
564 
565  // need to have two extra fraction bits for addition and rounding
566  const exprt fraction1_ext=
567  typecast_exprt(fraction1_shifted, unsignedbv_typet(spec.f+4+2));
568  const exprt fraction2_ext=
569  typecast_exprt(fraction2_stickied, unsignedbv_typet(spec.f+4+2));
570 
571  unbiased_floatt result;
572 
573  // now add/sub them
574  const notequal_exprt subtract_lit(unpacked1.sign, unpacked2.sign);
575 
576  result.fraction=
577  if_exprt(subtract_lit,
578  minus_exprt(fraction1_ext, fraction2_ext),
579  plus_exprt(fraction1_ext, fraction2_ext));
580 
581  // sign of result
582  std::size_t width=to_bitvector_type(result.fraction.type()).get_width();
583  const sign_exprt fraction_sign(
584  typecast_exprt(result.fraction, signedbv_typet(width)));
585  result.fraction=
586  typecast_exprt(
587  abs_exprt(typecast_exprt(result.fraction, signedbv_typet(width))),
588  unsignedbv_typet(width));
589 
590  result.exponent=bigger_exponent;
591 
592  // adjust the exponent for the fact that we added two bits to the fraction
593  result.exponent=
594  plus_exprt(typecast_exprt(result.exponent, signedbv_typet(spec.e+1)),
595  from_integer(2, signedbv_typet(spec.e+1)));
596 
597  // NaN?
598  result.NaN=or_exprt(
599  and_exprt(and_exprt(unpacked1.infinity, unpacked2.infinity),
600  notequal_exprt(unpacked1.sign, unpacked2.sign)),
601  or_exprt(unpacked1.NaN, unpacked2.NaN));
602 
603  // infinity?
604  result.infinity=and_exprt(
605  not_exprt(result.NaN),
606  or_exprt(unpacked1.infinity, unpacked2.infinity));
607 
608  // zero?
609  // Note that:
610  // 1. The zero flag isn't used apart from in divide and
611  // is only set on unpack
612  // 2. Subnormals mean that addition or subtraction can't round to 0,
613  // thus we can perform this test now
614  // 3. The rules for sign are different for zero
615  result.zero=
616  and_exprt(
617  not_exprt(or_exprt(result.infinity, result.NaN)),
618  equal_exprt(
619  result.fraction,
620  from_integer(0, result.fraction.type())));
621 
622  // sign
623  const notequal_exprt add_sub_sign(
624  if_exprt(src2_bigger, unpacked2.sign, unpacked1.sign), fraction_sign);
625 
626  const if_exprt infinity_sign(
627  unpacked1.infinity, unpacked1.sign, unpacked2.sign);
628 
629  #if 1
630  rounding_mode_bitst rounding_mode_bits(rm);
631 
632  const if_exprt zero_sign(

678 exprt float_bvt::mul(
679  const exprt &src1,
680  const exprt &src2,
681  const exprt &rm,
682  const ieee_float_spect &spec) const
683 {
684  // unpack
685  const unbiased_floatt unpacked1=unpack(src1, spec);
686  const unbiased_floatt unpacked2=unpack(src2, spec);
687 
688  // zero-extend the fractions (unpacked fraction has the hidden bit)
689  typet new_fraction_type=unsignedbv_typet((spec.f+1)*2);
690  const exprt fraction1=typecast_exprt(unpacked1.fraction, new_fraction_type);
691  const exprt fraction2=typecast_exprt(unpacked2.fraction, new_fraction_type);
692 
693  // multiply the fractions
694  unbiased_floatt result;
695  result.fraction=mult_exprt(fraction1, fraction2);
696 
697  // extend exponents to account for overflow
698  // add two bits, as we do extra arithmetic on it later
699  typet new_exponent_type=signedbv_typet(spec.e+2);
700  const exprt exponent1=typecast_exprt(unpacked1.exponent, new_exponent_type);
701  const exprt exponent2=typecast_exprt(unpacked2.exponent, new_exponent_type);
702 
703  const plus_exprt added_exponent(exponent1, exponent2);
704 
705  // Adjust exponent; we are thowing in an extra fraction bit,
706  // it has been extended above.
707  result.exponent=
708  plus_exprt(added_exponent, from_integer(1, new_exponent_type));
709 
710  // new sign
711  result.sign=notequal_exprt(unpacked1.sign, unpacked2.sign);
712 
713  // infinity?
714  result.infinity=or_exprt(unpacked1.infinity, unpacked2.infinity);
715 
716  // NaN?
717  result.NaN = disjunction(
718  {isnan(src1, spec),
719  isnan(src2, spec),
720  // infinity * 0 is NaN!
721  and_exprt(unpacked1.zero, unpacked2.infinity),
722  and_exprt(unpacked2.zero, unpacked1.infinity)});
723 
724  return rounder(result, rm, spec);

727 exprt float_bvt::div(
728  const exprt &src1,
729  const exprt &src2,
730  const exprt &rm,
731  const ieee_float_spect &spec) const
732 {
733  // unpack
734  const unbiased_floatt unpacked1=unpack(src1, spec);
735  const unbiased_floatt unpacked2=unpack(src2, spec);
736 
737  std::size_t fraction_width=
738  to_unsignedbv_type(unpacked1.fraction.type()).get_width();
739  std::size_t div_width=fraction_width*2+1;
740 
741  // pad fraction1 with zeros
742  const concatenation_exprt fraction1(
743  unpacked1.fraction,
744  from_integer(0, unsignedbv_typet(div_width - fraction_width)),
745  unsignedbv_typet(div_width));
746 
747  // zero-extend fraction2 to match fraction1
748  const typecast_exprt fraction2(unpacked2.fraction, fraction1.type());
749 
750  // divide fractions
751  unbiased_floatt result;
752  exprt rem;
753 
754  // the below should be merged somehow
755  result.fraction=div_exprt(fraction1, fraction2);
756  rem=mod_exprt(fraction1, fraction2);
757 
758  // is there a remainder?
759  const notequal_exprt have_remainder(rem, from_integer(0, rem.type()));
760 
761  // we throw this into the result, as least-significant bit,
762  // to get the right rounding decision
763  result.fraction=
764  concatenation_exprt(
765  result.fraction, have_remainder, unsignedbv_typet(div_width+1));
766 
767  // We will subtract the exponents;
768  // to account for overflow, we add a bit.
769  const typecast_exprt exponent1(
770  unpacked1.exponent, signedbv_typet(spec.e + 1));
771  const typecast_exprt exponent2(
772  unpacked2.exponent, signedbv_typet(spec.e + 1));
773 
774  // subtract exponents
775  const minus_exprt added_exponent(exponent1, exponent2);
776 
777  // adjust, as we have thown in extra fraction bits
778  result.exponent=plus_exprt(
779  added_exponent,
780  from_integer(spec.f, added_exponent.type()));
781 
782  // new sign
783  result.sign=notequal_exprt(unpacked1.sign, unpacked2.sign);
784 
785  // Infinity? This happens when
786  // 1) dividing a non-nan/non-zero by zero, or
787  // 2) first operand is inf and second is non-nan and non-zero
788  // In particular, inf/0=inf.
789  result.infinity=
790  or_exprt(
791  and_exprt(not_exprt(unpacked1.zero),
792  and_exprt(not_exprt(unpacked1.NaN),
793  unpacked2.zero)),
794  and_exprt(unpacked1.infinity,
795  and_exprt(not_exprt(unpacked2.NaN),
796  not_exprt(unpacked2.zero))));
797 
798  // NaN?
799  result.NaN=or_exprt(unpacked1.NaN,
800  or_exprt(unpacked2.NaN,
801  or_exprt(and_exprt(unpacked1.zero, unpacked2.zero),
802  and_exprt(unpacked1.infinity, unpacked2.infinity))));
803 
804  // Division by infinity produces zero, unless we have NaN
805  const and_exprt force_zero(not_exprt(unpacked1.NaN), unpacked2.infinity);
806 
807  result.fraction=
808  if_exprt(
809  force_zero,
810  from_integer(0, result.fraction.type()),
811  result.fraction);
812 
813  return rounder(result, rm, spec);
814 }

816 exprt float_bvt::relation(
817  const exprt &src1,
818  relt rel,
819  const exprt &src2,
820  const ieee_float_spect &spec)
821 {
822  if(rel==relt::GT)
823  return relation(src2, relt::LT, src1, spec); // swapped
824  else if(rel==relt::GE)
825  return relation(src2, relt::LE, src1, spec); // swapped
826 
827  INVARIANT(
828  rel == relt::EQ || rel == relt::LT || rel == relt::LE,
829  "relation should be equality, less-than, or less-or-equal");
830 
831  // special cases: -0 and 0 are equal
832  const exprt is_zero1 = is_zero(src1);
833  const exprt is_zero2 = is_zero(src2);
834  const and_exprt both_zero(is_zero1, is_zero2);
835 
836  // NaN compares to nothing
837  exprt isnan1=isnan(src1, spec);
838  exprt isnan2=isnan(src2, spec);
839  const or_exprt nan(isnan1, isnan2);
840 
841  if(rel==relt::LT || rel==relt::LE)
842  {
843  const equal_exprt bitwise_equal(src1, src2);
844 
845  // signs different? trivial! Unless Zero.
846 
847  const notequal_exprt signs_different(sign_bit(src1), sign_bit(src2));
848 
849  // as long as the signs match: compare like unsigned numbers
850 
851  // this works due to the BIAS
852  const binary_relation_exprt less_than1(
853  typecast_exprt(
854  typecast_exprt(src1, bv_typet(spec.width())),
855  unsignedbv_typet(spec.width())),
856  ID_lt,
857  typecast_exprt(
858  typecast_exprt(src2, bv_typet(spec.width())),
859  unsignedbv_typet(spec.width())));
860 
861  // if both are negative (and not the same), need to turn around!
862  const notequal_exprt less_than2(
863  less_than1, and_exprt(sign_bit(src1), sign_bit(src2)));
864 
865  const if_exprt less_than3(signs_different, sign_bit(src1), less_than2);
866 
867  if(rel==relt::LT)
868  {
869  and_exprt and_bv{{less_than3,
870  // for the case of two negative numbers
871  not_exprt(bitwise_equal),
872  not_exprt(both_zero),
873  not_exprt(nan)}};
874 
875  return std::move(and_bv);
876  }
877  else if(rel==relt::LE)
878  {
879  or_exprt or_bv{{less_than3, both_zero, bitwise_equal}};
880 
881  return and_exprt(or_bv, not_exprt(nan));
882  }
883  else
884  UNREACHABLE;
885  }
886  else if(rel==relt::EQ)
887  {
888  const equal_exprt bitwise_equal(src1, src2);
889 
890  return and_exprt(
891  or_exprt(bitwise_equal, both_zero),
892  not_exprt(nan));
893  }
894 
895  UNREACHABLE;
896  return false_exprt();
897 }

944 void float_bvt::normalization_shift(
945  exprt &fraction,
946  exprt &exponent)
947 {
948  // n-log-n alignment shifter.
949  // The worst-case shift is the number of fraction
950  // bits minus one, in case the fraction is one exactly.
951  std::size_t fraction_bits=to_unsignedbv_type(fraction.type()).get_width();
952  std::size_t exponent_bits=to_signedbv_type(exponent.type()).get_width();
953  PRECONDITION(fraction_bits != 0);
954 
955  std::size_t depth = address_bits(fraction_bits - 1);
956 
957  if(exponent_bits<depth)
958  exponent=typecast_exprt(exponent, signedbv_typet(depth));
959 
960  exprt exponent_delta=from_integer(0, exponent.type());
961 
962  for(int d=depth-1; d>=0; d--)
963  {
964  unsigned distance=(1<<d);
965  INVARIANT(
966  fraction_bits > distance,
967  "distance must be within the range of fraction bits");
968 
969  // check if first 'distance'-many bits are zeros
970  const extractbits_exprt prefix(
971  fraction,
972  fraction_bits - 1,
973  fraction_bits - distance,
974  unsignedbv_typet(distance));
975  const equal_exprt prefix_is_zero(prefix, from_integer(0, prefix.type()));
976 
977  // If so, shift the zeros out left by 'distance'.
978  // Otherwise, leave as is.
979  const shl_exprt shifted(fraction, distance);
980 
981  fraction=
982  if_exprt(prefix_is_zero, shifted, fraction);
983 
984  // add corresponding weight to exponent
985  INVARIANT(
986  d < (signed int)exponent_bits,
987  "depth must be smaller than exponent bits");
988 
989  exponent_delta=
990  bitor_exprt(exponent_delta,
991  shl_exprt(typecast_exprt(prefix_is_zero, exponent_delta.type()), d));
992  }
993 
994  exponent=minus_exprt(exponent, exponent_delta);
995 }

998 void float_bvt::denormalization_shift(
999  exprt &fraction,
1000  exprt &exponent,
1001  const ieee_float_spect &spec)
1002 {
1003  mp_integer bias=spec.bias();
1004 
1005  // Is the exponent strictly less than -bias+1, i.e., exponent<-bias+1?
1006  // This is transformed to distance=(-bias+1)-exponent
1007  // i.e., distance>0
1008  // Note that 1-bias is the exponent represented by 0...01,
1009  // i.e. the exponent of the smallest normal number and thus the 'base'
1010  // exponent for subnormal numbers.
1011 
1012  std::size_t exponent_bits=to_signedbv_type(exponent.type()).get_width();
1013  PRECONDITION(exponent_bits >= spec.e);
1014 
1015 #if 1
1016  // Need to sign extend to avoid overflow. Note that this is a
1017  // relatively rare problem as the value needs to be close to the top
1018  // of the exponent range and then range must not have been
1019  // previously extended as add, multiply, etc. do. This is primarily
1020  // to handle casting down from larger ranges.
1021  exponent=typecast_exprt(exponent, signedbv_typet(exponent_bits+1));
1022 #endif
1023 
1024  const minus_exprt distance(
1025  from_integer(-bias + 1, exponent.type()), exponent);
1026 
1027  // use sign bit
1028  const and_exprt denormal(
1029  not_exprt(sign_exprt(distance)),
1030  notequal_exprt(distance, from_integer(0, distance.type())));
1031 
1032 #if 1
1033  // Care must be taken to not loose information required for the
1034  // guard and sticky bits. +3 is for the hidden, guard and sticky bits.
1035  std::size_t fraction_bits=to_unsignedbv_type(fraction.type()).get_width();
1036 
1037  if(fraction_bits < spec.f+3)
1038  {
1039  // Add zeros at the LSB end for the guard bit to shift into
1040  fraction=
1041  concatenation_exprt(
1042  fraction, unsignedbv_typet(spec.f + 3 - fraction_bits).zero_expr(),
1043  unsignedbv_typet(spec.f+3));
1044  }
1045 
1046  exprt denormalisedFraction = fraction;
1047 
1048  exprt sticky_bit = false_exprt();
1049  denormalisedFraction =
1050  sticky_right_shift(fraction, distance, sticky_bit);
1051 
1052  denormalisedFraction=
1053  bitor_exprt(denormalisedFraction,
1054  typecast_exprt(sticky_bit, denormalisedFraction.type()));
1055 
1056  fraction=
1057  if_exprt(
1058  denormal,
1059  denormalisedFraction,
1060  fraction);
1061 
1062 #else
1063  fraction=
1064  if_exprt(
1065  denormal,
1066  lshr_exprt(fraction, distance),
1067  fraction);
1068 #endif
1069 
1070  exponent=
1071  if_exprt(denormal,
1072  from_integer(-bias, exponent.type()),
1073  exponent);
1074 }

1077  const unbiased_floatt &src,
1078  const exprt &rm,
1079  const ieee_float_spect &spec) const
1080 {
1081  // incoming: some fraction (with explicit 1),
1082  // some exponent without bias
1083  // outgoing: rounded, with right size, with hidden bit, bias
1084 
1085  exprt aligned_fraction=src.fraction,
1086  aligned_exponent=src.exponent;
1087 
1088  {
1089  std::size_t exponent_bits = std::max(address_bits(spec.f), spec.e) + 1;
1090 
1091  // before normalization, make sure exponent is large enough
1092  if(to_signedbv_type(aligned_exponent.type()).get_width()<exponent_bits)
1093  {
1094  // sign extend
1095  aligned_exponent=
1096  typecast_exprt(aligned_exponent, signedbv_typet(exponent_bits));
1097  }
1098  }
1099 
1100  // align it!
1101  normalization_shift(aligned_fraction, aligned_exponent);
1102  denormalization_shift(aligned_fraction, aligned_exponent, spec);
1103 
1104  unbiased_floatt result;
1105  result.fraction=aligned_fraction;
1106  result.exponent=aligned_exponent;
1107  result.sign=src.sign;
1108  result.NaN=src.NaN;
1109  result.infinity=src.infinity;
1110 
1111  rounding_mode_bitst rounding_mode_bits(rm);
1112  round_fraction(result, rounding_mode_bits, spec);
1113  round_exponent(result, rounding_mode_bits, spec);
1114

1116 }

1119 exprt float_bvt::fraction_rounding_decision(
1120  const std::size_t dest_bits,
1121  const exprt sign,
1122  const exprt &fraction,
1123  const rounding_mode_bitst &rounding_mode_bits)
1124 {
1125  std::size_t fraction_bits=
1126  to_unsignedbv_type(fraction.type()).get_width();
1127 
1128  PRECONDITION(dest_bits < fraction_bits);
1129 
1130  // we have too many fraction bits
1131  std::size_t extra_bits=fraction_bits-dest_bits;
1132 
1133  // more than two extra bits are superflus, and are
1134  // turned into a sticky bit
1135 
1136  exprt sticky_bit=false_exprt();
1137 
1138  if(extra_bits>=2)
1139  {
1140  // We keep most-significant bits, and thus the tail is made
1141  // of least-significant bits.
1142  const extractbits_exprt tail(
1143  fraction, extra_bits - 2, 0, unsignedbv_typet(extra_bits - 2 + 1));
1144  sticky_bit=notequal_exprt(tail, from_integer(0, tail.type()));
1145  }
1146 
1147  // the rounding bit is the last extra bit
1148  INVARIANT(
1149  extra_bits >= 1, "the extra bits contain at least the rounding bit");
1150  const extractbit_exprt rounding_bit(fraction, extra_bits - 1);
1151 
1152  // we get one bit of the fraction for some rounding decisions
1153  const extractbit_exprt rounding_least(fraction, extra_bits);
1154 
1155  // round-to-nearest (ties to even)
1156  const and_exprt round_to_even(
1157  rounding_bit, or_exprt(rounding_least, sticky_bit));
1158 
1159  // round up
1160  const and_exprt round_to_plus_inf(
1161  not_exprt(sign), or_exprt(rounding_bit, sticky_bit));
1162 
1163  // round down
1164  const and_exprt round_to_minus_inf(sign, or_exprt(rounding_bit, sticky_bit));
1165 
1166  // round to zero
1167  false_exprt round_to_zero;
1168 
1169  // now select appropriate one
1170  return if_exprt(rounding_mode_bits.round_to_even, round_to_even,
1171  if_exprt(rounding_mode_bits.round_to_plus_inf, round_to_plus_inf,
1172  if_exprt(rounding_mode_bits.round_to_minus_inf, round_to_minus_inf,
1173  if_exprt(rounding_mode_bits.round_to_zero, round_to_zero,
1174  false_exprt())))); // otherwise zero
1175 }

1177 void float_bvt::round_fraction(
1178  unbiased_floatt &result,
1179  const rounding_mode_bitst &rounding_mode_bits,
1180  const ieee_float_spect &spec)
1181 {
1182  std::size_t fraction_size=spec.f+1;
1183  std::size_t result_fraction_size=
1184  to_unsignedbv_type(result.fraction.type()).get_width();
1185 
1186  // do we need to enlarge the fraction?
1187  if(result_fraction_size<fraction_size)
1188  {
1189  // pad with zeros at bottom
1190  std::size_t padding=fraction_size-result_fraction_size;
1191 
1192  result.fraction=concatenation_exprt(
1193  result.fraction,
1194  unsignedbv_typet(padding).zero_expr(),
1195  unsignedbv_typet(fraction_size));
1196  }
1197  else if(result_fraction_size==fraction_size) // it stays
1198  {
1199  // do nothing
1200  }
1201  else // fraction gets smaller -- rounding
1202  {
1203  std::size_t extra_bits=result_fraction_size-fraction_size;
1204  INVARIANT(
1205  extra_bits >= 1, "the extra bits include at least the rounding bit");
1206 
1207  // this computes the rounding decision
1208  exprt increment=fraction_rounding_decision(
1209  fraction_size, result.sign, result.fraction, rounding_mode_bits);
1210 
1211  // chop off all the extra bits
1212  result.fraction=extractbits_exprt(
1213  result.fraction, result_fraction_size-1, extra_bits,
1214  unsignedbv_typet(fraction_size));
1215 
1216 #if 0
1217  // *** does not catch when the overflow goes subnormal -> normal ***
1218  // incrementing the fraction might result in an overflow
1219  result.fraction=
1220  bv_utils.zero_extension(result.fraction, result.fraction.size()+1);
1221 
1222  result.fraction=bv_utils.incrementer(result.fraction, increment);
1223 
1224  exprt overflow=result.fraction.back();
1225 
1226  // In case of an overflow, the exponent has to be incremented.
1227  // "Post normalization" is then required.
1228  result.exponent=
1229  bv_utils.incrementer(result.exponent, overflow);
1230 
1231  // post normalization of the fraction
1232  exprt integer_part1=result.fraction.back();
1233  exprt integer_part0=result.fraction[result.fraction.size()-2];
1234  const or_exprt new_integer_part(integer_part1, integer_part0);
1235 
1236  result.fraction.resize(result.fraction.size()-1);
1237  result.fraction.back()=new_integer_part;
1238 
1239 #else
1240  // When incrementing due to rounding there are two edge
1241  // cases we need to be aware of:
1242  // 1. If the number is normal, the increment can overflow.
1243  // In this case we need to increment the exponent and
1244  // set the MSB of the fraction to 1.
1245  // 2. If the number is the largest subnormal, the increment
1246  // can change the MSB making it normal. Thus the exponent
1247  // must be incremented but the fraction will be OK.
1248  const extractbit_exprt old_msb(result.fraction, fraction_size - 1);
1249 
1250  // increment if 'increment' is true
1251  result.fraction=
1252  plus_exprt(result.fraction,
1253  typecast_exprt(increment, result.fraction.type()));
1254 
1255  // Normal overflow when old MSB == 1 and new MSB == 0
1256  const extractbit_exprt new_msb(result.fraction, fraction_size - 1);
1257  const and_exprt overflow(old_msb, not_exprt(new_msb));
1258 
1259  // Subnormal to normal transition when old MSB == 0 and new MSB == 1
1260  const and_exprt subnormal_to_normal(not_exprt(old_msb), new_msb);
1261 
1262  // In case of an overflow or subnormal to normal conversion,
1263  // the exponent has to be incremented.
1264  result.exponent=
1265  plus_exprt(
1266  result.exponent,
1267  if_exprt(
1268  or_exprt(overflow, subnormal_to_normal),
1269  from_integer(1, result.exponent.type()),
1270  from_integer(0, result.exponent.type())));
1271 
1272  // post normalization of the fraction
1273  // In the case of overflow, set the MSB to 1
1274  // The subnormal case will have (only) the MSB set to 1
