32 typet struct_encodingt::encode(typet type) const
33 {
34  std::queue<typet *> work_queue;
35  work_queue.push(&type);
36  while(!work_queue.empty())
37  {
38  typet &current = *work_queue.front();
39  work_queue.pop();
40  if(const auto struct_tag = type_try_dynamic_cast<struct_tag_typet>(current))
41  {
42  auto width = (*boolbv_width)(*struct_tag);
43  // The bit vector theory of SMT disallows zero bit length length bit
44  // vectors. C++ gives a minimum size for a struct (even an empty struct)
45  // as being one byte; in order to ensure that structs have unique memory
46  // locations. Therefore encoding empty structs as having 8 bits / 1 byte
47  // is a reasonable solution in this case.
48  if(width == 0)
49  width = 8;
50  current = bv_typet{width};
51  }
52  if(const auto array = type_try_dynamic_cast<array_typet>(current))
53  {
54  work_queue.push(&array->element_type());
55  }
56  }
57  return type;
58 }

91 static exprt encode(const with_exprt &with, const namespacet &ns)
92 {
93  const auto tag_type = type_checked_cast<struct_tag_typet>(with.type());
94  const auto struct_type =
95  type_checked_cast<struct_typet>(ns.follow(with.type()));
96  const auto updates = extricate_updates(with);
97  const auto components =
98  make_range(struct_type.components())
99  .map([&](const struct_union_typet::componentt &component) -> exprt {
100  const auto &update = updates.find(component.get_name());
101  if(update != updates.end())
102  return update->second;
103  else
104  return member_exprt{
105  with.old(), component.get_name(), component.type()};
106  })
107  .collect<exprt::operandst>();
108  return struct_exprt{components, tag_type};
109 }

169 exprt struct_encodingt::encode(exprt expr) const
170 {
171  std::queue<exprt *> work_queue;
172  work_queue.push(&expr);
173  while(!work_queue.empty())
174  {
175  exprt &current = *work_queue.front();
176  work_queue.pop();
177  // Note that "with" expressions are handled before type encoding in order to
178  // facilitate checking that they are applied to structs rather than arrays.
179  if(const auto with_expr = expr_try_dynamic_cast<with_exprt>(current))
180  if(can_cast_type<struct_tag_typet>(current.type()))
181  current = ::encode(*with_expr, ns);
182  current.type() = encode(current.type());
183  if(const auto struct_expr = expr_try_dynamic_cast<struct_exprt>(current))
184  current = ::encode(*struct_expr);
185  if(const auto member_expr = expr_try_dynamic_cast<member_exprt>(current))
186  current = encode_member(*member_expr);
187  for(auto &operand : current.operands())
188  work_queue.push(&operand);
189  }
190  return expr;
191 }

193 exprt struct_encodingt::decode(
194  const exprt &encoded,
195  const struct_tag_typet &original_type) const
196 {
197  // The algorithm below works by extracting each of the separate fields from
198  // the combined encoded value using a `member_exprt` which is itself encoded
199  // into a `bit_extract_exprt`. These separated fields can then be combined
200  // using a `struct_exprt`. This is expected to duplicate the input encoded
201  // expression for each of the fields. However for the case where the input
202  // encoded expression is a `constant_exprt`, expression simplification will be
203  // able simplify away the duplicated portions of the constant and the bit
204  // extraction expressions. This yields a clean struct expression where each
205  // field is a separate constant containing the data solely relevant to that
206  // field.
207  INVARIANT(
208  can_cast_type<bv_typet>(encoded.type()),
209  "Structs are expected to be encoded into bit vectors.");
210  const struct_typet definition = ns.get().follow_tag(original_type);
211  exprt::operandst encoded_fields;
212  for(const auto &component : definition.components())
213  {
214  encoded_fields.push_back(typecast_exprt::conditional_cast(
215  encode(member_exprt{typecast_exprt{encoded, original_type}, component}),
216  component.type()));
217  }
218  return simplify_expr(struct_exprt{encoded_fields, original_type}, ns);
219 }
