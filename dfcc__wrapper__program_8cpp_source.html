282 void dfcc_wrapper_programt::encode_requires_write_set()
283 {
284  // call write_set_create(
285  // requires_write_set,
286  // assigns_size = 0,
287  // frees_size = 0,
288  // replacement_mode = false,
289  // assume_requires_ctx = contract_mode == check,
290  // assert_requires_ctx = contract_mode != check,
291  // assume_ensures_ctx = false,
292  // assert_ensures_ctx = false,
293  // )
294  const auto write_set = requires_write_set;
295  preamble.add(goto_programt::make_decl(write_set, wrapper_sl));
296 
297  const auto address_of_write_set = addr_of_requires_write_set;
298  preamble.add(goto_programt::make_decl(address_of_write_set, wrapper_sl));
299  preamble.add(goto_programt::make_assignment(
300  address_of_write_set, address_of_exprt(write_set), wrapper_sl));
301 
302  auto function_symbol =
303  library.dfcc_fun_symbol[dfcc_funt::WRITE_SET_CREATE].symbol_expr();
304  code_function_callt call(function_symbol);
305  auto &arguments = call.arguments();
306 
307  // write set
308  arguments.emplace_back(address_of_write_set);
309 
310  // max assigns clause size
311  arguments.emplace_back(from_integer(0, size_type()));
312 
313  // max frees clause size
314  arguments.emplace_back(from_integer(0, size_type()));
315 
316  // replacement mode
317  arguments.push_back(false_exprt());
318 
319  if(contract_mode == dfcc_contract_modet::CHECK)
320  {
321  // assume_requires_ctx
322  arguments.push_back(true_exprt());
323 
324  // assert_requires_ctx
325  arguments.push_back(false_exprt());
326  }
327  else
328  {
329  // assume_requires_ctx
330  arguments.push_back(false_exprt());
331 
332  // assert_requires_ctx mode
333  arguments.push_back(true_exprt());
334  }
335 
336  // assume_ensures_ctx mode
337  arguments.push_back(false_exprt());
338 
339  // assert_ensures_ctx mode
340  arguments.push_back(false_exprt());
341 
342  preamble.add(goto_programt::make_function_call(call, wrapper_sl));
343 
344  // check for absence of allocation/deallocation in requires clauses
345  // ```c
346  // DECL __check_no_alloc: bool;
347  // CALL __check_no_alloc = check_empty_alloc_dealloc(write_set);
348  // ASSERT __check_no_alloc;
349  // DEAD __check_no_alloc: bool;
350  // ```
351  auto check_var = get_fresh_aux_symbol(
352  bool_typet(),
353  id2string(wrapper_symbol.name),
354  "__no_alloc_dealloc_in_requires",
355  wrapper_sl,
356  wrapper_symbol.mode,
357  goto_model.symbol_table)
358  .symbol_expr();
359 
360  postamble.add(goto_programt::make_decl(check_var, wrapper_sl));
361 
362  postamble.add(goto_programt::make_function_call(
363  code_function_callt{
364  check_var,
365  library
366  .dfcc_fun_symbol
367  [dfcc_funt::WRITE_SET_CHECK_ALLOCATED_DEALLOCATED_IS_EMPTY]
368  .symbol_expr(),
369  {address_of_write_set}},
370  wrapper_sl));
371 
372  source_locationt check_sl(wrapper_sl);
373  check_sl.set_function(wrapper_symbol.name);
374  check_sl.set_property_class("no_alloc_dealloc_in_requires");
375  check_sl.set_comment(
376  "Check that requires do not allocate or deallocate memory");
377  postamble.add(goto_programt::make_assertion(check_var, check_sl));
378  postamble.add(goto_programt::make_dead(check_var, wrapper_sl));
379 
380  // generate write set release and DEAD instructions
381  {
382  code_function_callt call(
383  library.dfcc_fun_symbol[dfcc_funt::WRITE_SET_RELEASE].symbol_expr(),
384  {address_of_write_set});
385  postamble.add(goto_programt::make_function_call(call, wrapper_sl));
386  postamble.add(goto_programt::make_dead(write_set, wrapper_sl));
387  }
388 }

513 void dfcc_wrapper_programt::encode_contract_write_set()
514 {
515  const auto write_set = contract_write_set;
516  preamble.add(goto_programt::make_decl(write_set, wrapper_sl));
517 
518  const auto address_of_contract_write_set = addr_of_contract_write_set;
519  preamble.add(
520  goto_programt::make_decl(address_of_contract_write_set, wrapper_sl));
521  preamble.add(goto_programt::make_assignment(
522  address_of_contract_write_set, address_of_exprt(write_set), wrapper_sl));
523 
524  // We use the empty write set used to check ensures for side effects
525  // to check for side effects in the assigns and frees functions as well
526  const auto address_of_requires_write_set = addr_of_requires_write_set;
527 
528  // call write_set_create
529  {
530  auto function_symbol =
531  library.dfcc_fun_symbol[dfcc_funt::WRITE_SET_CREATE].symbol_expr();
532 
533  code_function_callt call(function_symbol);
534  auto &arguments = call.arguments();
535 
536  // write set
537  arguments.emplace_back(address_of_contract_write_set);
538 
539  // max assigns clause size
540  arguments.emplace_back(
541  from_integer(contract_functions.get_nof_assigns_targets(), size_type()));
542 
543  // max frees clause size
544  arguments.emplace_back(
545  from_integer(contract_functions.get_nof_frees_targets(), size_type()));
546 
547  // activate replace mode
548  const bool replace = contract_mode == dfcc_contract_modet::REPLACE;
549  arguments.emplace_back(
550  (replace ? (exprt)true_exprt() : (exprt)false_exprt()));
551 
552  // not analysing ensures or requires clauses, set all context flags to false
553 
554  // assume_requires_ctx
555  arguments.push_back(false_exprt());
556 
557  // assert_requires_ctx
558  arguments.push_back(false_exprt());
559 
560  // assume_ensures_ctx
561  arguments.push_back(false_exprt());
562 
563  // assert_ensures_ctx
564  arguments.push_back(false_exprt());
565 
566  write_set_checks.add(goto_programt::make_function_call(call));
567  }
568 
569  // build arguments for assigns and frees clauses functions
570  exprt::operandst wrapper_arguments;
571  const auto &wrapper_code_type = to_code_type(wrapper_symbol.type);
572  for(const auto &parameter : wrapper_code_type.parameter_identifiers())
573  {
574  const symbolt &parameter_symbol = ns.lookup(parameter);
575  wrapper_arguments.push_back(parameter_symbol.symbol_expr());
576  }
577 
578  // call spec_assigns to build the contract write set
579  {
580  code_function_callt call(
581  contract_functions.get_spec_assigns_function_symbol().symbol_expr());
582 
583  auto &arguments = call.arguments();
584 
585  // forward wrapper arguments
586  for(const auto &arg : wrapper_arguments)
587  arguments.push_back(arg);
588 
589  // pass write set to populate
590  arguments.emplace_back(address_of_contract_write_set);
591 
592  // pass the empty write set to check side effects against
593  arguments.emplace_back(address_of_requires_write_set);
594 
595  write_set_checks.add(goto_programt::make_function_call(call, wrapper_sl));
596  }
597 
598  // call spec_frees to build the contract write set
599  {
600  code_function_callt call(
601  contract_functions.get_spec_frees_function_symbol().symbol_expr());
602  auto &arguments = call.arguments();
603 
604  // forward wrapper arguments
605  for(const auto &arg : wrapper_arguments)
606  arguments.push_back(arg);
607 
608  // pass write set to populate
609  arguments.emplace_back(address_of_contract_write_set);
610 
611  // pass the empty write set to check side effects against
612  arguments.emplace_back(address_of_requires_write_set);
613 
614  write_set_checks.add(goto_programt::make_function_call(call, wrapper_sl));
615  }
616 
617  // generate write set release and DEAD instructions
618  {
619  code_function_callt call(
620  library.dfcc_fun_symbol[dfcc_funt::WRITE_SET_RELEASE].symbol_expr(),
621  {address_of_contract_write_set});
622  postamble.add(goto_programt::make_function_call(call, wrapper_sl));
623  postamble.add(goto_programt::make_dead(write_set, wrapper_sl));
624  }
625 }

670 void dfcc_wrapper_programt::encode_requires_clauses()
671 {
672  // we use this empty requires write set to check for the absence of side
673  // effects in the requires clauses
674  const auto &wrapper_id = wrapper_symbol.name;
675  const auto &language_mode = utils.get_function_symbol(wrapper_id).mode;
676 
677  // if in replacement mode, check that assertions hold in the current context,
678  // otherwise assume
679  const auto &statement_type =
680  (contract_mode == dfcc_contract_modet::REPLACE) ? ID_assert : ID_assume;
681 
682  // goto program where all requires are added
683  goto_programt requires_program;
684 
685  // translate each requires clause
686  for(const auto &r : contract_code_type.requires())
687  {
688  exprt requires = to_lambda_expr(r).application(contract_lambda_parameters);
689  requires.add_source_location() = r.source_location();
690  if(has_subexpr(requires, ID_exists) || has_subexpr(requires, ID_forall))
691  add_quantified_variable(goto_model.symbol_table, requires, language_mode);
692 
693  source_locationt sl(r.source_location());
694  if(statement_type == ID_assert)
695  {
696  sl.set_property_class(ID_precondition);
697  sl.set_comment(
698  "Check requires clause of contract " + id2string(contract_symbol.name) +
699  " for function " + id2string(wrapper_id));
700  }
701  codet requires_statement(statement_type, {std::move(requires)}, sl);
702  converter.goto_convert(requires_statement, requires_program, language_mode);
703  }
704  const auto address_of_requires_write_set = addr_of_requires_write_set;
705 
706  // fix calls to user-defined memory predicates
707  memory_predicates.fix_calls(requires_program);
708 
709  // instrument for side effects
710  instrument.instrument_goto_program(
711  wrapper_id,
712  requires_program,
713  address_of_requires_write_set,
714  function_pointer_contracts);
715 
716  // append resulting program to preconditions section
717  preconditions.destructive_append(requires_program);
718 }

720 void dfcc_wrapper_programt::encode_ensures_clauses()
721 {
722  const auto &language_mode = wrapper_symbol.mode;
723  const auto &wrapper_id = wrapper_symbol.name;
724  const auto &statement_type =
725  (contract_mode == dfcc_contract_modet::CHECK) ? ID_assert : ID_assume;
726 
727  // goto program where all history variable snapshots are added
728  goto_programt history_snapshot_program;
729 
730  // goto program where all requires are added
731  goto_programt ensures_program;
732 
733  // translate each ensures clause
734  for(const auto &e : contract_code_type.ensures())
735  {
736  exprt ensures = to_lambda_expr(e).application(contract_lambda_parameters);
737  ensures.add_source_location() = e.source_location();
738 
739  if(has_subexpr(ensures, ID_exists) || has_subexpr(ensures, ID_forall))
740  add_quantified_variable(goto_model.symbol_table, ensures, language_mode);
741 
742  // this also rewrites ID_old expressions to fresh variables
743  generate_history_variables_initialization(
744  goto_model.symbol_table,
745  ensures,
746  language_mode,
747  history_snapshot_program);
748 
749  source_locationt sl(e.source_location());
750  if(statement_type == ID_assert)
751  {
752  sl.set_property_class(ID_postcondition);
753  sl.set_comment(
754  "Check ensures clause of contract " + id2string(contract_symbol.name) +
755  " for function " + id2string(wrapper_id));
756  }
757 
758  codet ensures_statement(statement_type, {std::move(ensures)}, sl);
759  converter.goto_convert(ensures_statement, ensures_program, language_mode);
760  }
761 
762  const auto address_of_ensures_write_set = addr_of_ensures_write_set;
763 
764  // When checking an ensures clause we link the contract write set to the
765  // ensures write set to know what was deallocated by the function so that
766  // the was_freed predicate can perform its checks
767  code_function_callt call(
768  library.dfcc_fun_symbol[dfcc_funt::LINK_DEALLOCATED].symbol_expr());
769  auto &arguments = call.arguments();
770  arguments.emplace_back(address_of_ensures_write_set);
771  arguments.emplace_back(addr_of_contract_write_set);
772  link_deallocated_contract.add(
773  goto_programt::make_function_call(call, wrapper_sl));
774 
775  // fix calls to user-defined user-defined memory predicates
776  memory_predicates.fix_calls(ensures_program);
777 
778  // instrument for side effects
779  instrument.instrument_goto_program(
780  wrapper_id,
781  ensures_program,
782  address_of_ensures_write_set,
783  function_pointer_contracts);
784 
785  // add the snapshot program in the history section
786  history.destructive_append(history_snapshot_program);
787 
788  // add the ensures program to the postconditions section
789  postconditions.destructive_append(ensures_program);
790 }

801 {

803  code_function_callt call(wrapped_symbol.symbol_expr());
804 
805  if(return_value_opt)
806  {
807  symbol_exprt return_value = return_value_opt.value();

813  postamble.add(

816  // DEAD
817  postamble.add(goto_programt::make_dead(return_value, wrapper_sl));
818  }

822

827  call.arguments().push_back(parameter_symbol.symbol_expr());

830  // pass write set to check against
831  call.arguments().push_back(addr_of_contract_write_set);

833  function_call.add(goto_programt::make_function_call(call, wrapper_sl));

836 void dfcc_wrapper_programt::encode_havoced_function_call()
837 {
838  // generate local write set and add as parameter to the call
839  exprt::operandst write_set_arguments;
840  for(const auto &parameter :
841  to_code_type(wrapper_symbol.type).parameter_identifiers())
842  {
843  PRECONDITION(!parameter.empty());
844  const symbolt &parameter_symbol = ns.lookup(parameter);
845  write_set_arguments.push_back(parameter_symbol.symbol_expr());
846  }
847 
848  auto address_of_contract_write_set = addr_of_contract_write_set;
849 
850  // check assigns clause inclusion
851  // IF __caller_write_set==NULL GOTO skip_target;
852  // DECL check: bool;
853  // CALL check = __CPROVER_contracts_write_set_check_assigns_clause_inclusion(
854  // __caller_write_set, &__local_write_set);
855  // ASSERT check;
856  // CALL check = __CPROVER_contracts_write_set_check_frees_clause_inclusion(
857  // __caller_write_set, &__local_write_set);
858  // ASSERT check;
859  // DEAD check;
860  // skip_target: skip;
861 
862  auto goto_instruction =
863  write_set_checks.add(goto_programt::make_incomplete_goto(
864  utils.make_null_check_expr(caller_write_set), wrapper_sl));
865 
866  {
867  // assigns clause inclusion
868  auto check_var = get_fresh_aux_symbol(
869  bool_typet(),
870  id2string(wrapper_symbol.name),
871  "__check_assigns_clause_incl",
872  wrapper_sl,
873  wrapper_symbol.mode,
874  goto_model.symbol_table)
875  .symbol_expr();
876 
877  write_set_checks.add(goto_programt::make_decl(check_var, wrapper_sl));
878 
879  code_function_callt check_incl_call(
880  check_var,
881  library
882  .dfcc_fun_symbol[dfcc_funt::WRITE_SET_CHECK_ASSIGNS_CLAUSE_INCLUSION]
883  .symbol_expr(),
884  {caller_write_set, address_of_contract_write_set});
885 
886  write_set_checks.add(
887  goto_programt::make_function_call(check_incl_call, wrapper_sl));
888  source_locationt check_sl(wrapper_sl);
889  check_sl.set_function(wrapper_symbol.name);
890  check_sl.set_property_class("assigns");
891  check_sl.set_comment(
892  "Check that the assigns clause of " + id2string(contract_symbol.name) +
893  " is included in the caller's assigns clause");
894  write_set_checks.add(goto_programt::make_assertion(check_var, check_sl));
895  write_set_checks.add(goto_programt::make_dead(check_var, wrapper_sl));
896  }
897 
898  {
899  // frees clause inclusion
900  auto check_var = get_fresh_aux_symbol(
901  bool_typet(),
902  id2string(wrapper_symbol.name),
903  "__check_frees_clause_incl",
904  wrapper_sl,
905  wrapper_symbol.mode,
906  goto_model.symbol_table)
907  .symbol_expr();
908 
909  write_set_checks.add(goto_programt::make_decl(check_var, wrapper_sl));
910 
911  code_function_callt check_incl_call(
912  check_var,
913  library.dfcc_fun_symbol[dfcc_funt::WRITE_SET_CHECK_FREES_CLAUSE_INCLUSION]
914  .symbol_expr(),
915  {caller_write_set, address_of_contract_write_set});
916 
917  write_set_checks.add(
918  goto_programt::make_function_call(check_incl_call, wrapper_sl));
919 
920  source_locationt check_sl(wrapper_sl);
921  check_sl.set_function(wrapper_symbol.name);
922  check_sl.set_property_class("frees");
923  check_sl.set_comment(
924  "Check that the frees clause of " + id2string(contract_symbol.name) +
925  " is included in the caller's frees clause");
926  write_set_checks.add(goto_programt::make_assertion(check_var, check_sl));
927  write_set_checks.add(goto_programt::make_dead(check_var, wrapper_sl));
928  }
929 
930  auto label_instruction = write_set_checks.add(goto_programt::make_skip());
931  goto_instruction->complete_goto(label_instruction);
932 
933  code_function_callt havoc_call(
934  contract_functions.get_spec_assigns_havoc_function_symbol().symbol_expr(),
935  {address_of_contract_write_set});
936 
937  function_call.add(goto_programt::make_function_call(havoc_call, wrapper_sl));
938 
939  // assign nondet to the return value
940  if(return_value_opt.has_value())
941  {
942  symbol_exprt return_value = return_value_opt.value();
943 
944  // DECL
945  preamble.add(goto_programt::make_decl(return_value, wrapper_sl));
946 
947  // ASSIGN NONDET
948  function_call.add(goto_programt::make_assignment(
949  return_value,
950  side_effect_expr_nondett(return_value.type(), wrapper_sl),
951  wrapper_sl));
952 
953  // SET RETURN VALUE
954  postamble.add(
955  goto_programt::make_set_return_value(return_value, wrapper_sl));
956 
957  // DEAD
958  postamble.add(goto_programt::make_dead(return_value, wrapper_sl));
959  }
960 
961  // nondet free the freeable set, record effects in both the contract write
962  // set and the caller write set
963  code_function_callt deallocate_call(
964  library.dfcc_fun_symbol[dfcc_funt::WRITE_SET_DEALLOCATE_FREEABLE]
965  .symbol_expr(),
966  {address_of_contract_write_set, caller_write_set});
967  function_call.add(
968  goto_programt::make_function_call(deallocate_call, wrapper_sl));
969 }
