219  goto_modelt &goto_model,

221 {

223 
224  int ret=0;
225

227  // of some symbol in the symbol table, and pointerize them too
228  for(const auto &pair : goto_model.symbol_table.symbols)
229  {
230  std::list<symbol_exprt> to_pointerize;
231  symbols_to_pointerize(linker_values, pair.second.value, to_pointerize);
232 
233  if(to_pointerize.empty())
234  continue;

236  << pair.first << messaget::eom;

238  goto_model.symbol_table.get_writeable_ref(pair.first).value,
239  to_pointerize,
240  linker_values);

242  continue;
243  ret=1;
244  for(const auto &sym : to_pointerize)
245  {
246  log.error() << " Could not pointerize '" << sym.get_identifier()
247  << "' in symbol table entry " << pair.first << ". Pretty:\n"
248  << sym.pretty() << "\n";
249  }
250  log.error() << messaget::eom;
251  }
252 
253  // Finally, pointerize all occurrences of linker-defined symbols in the
254  // goto program
255  for(auto &gf : goto_model.goto_functions.function_map)
256  {
257  goto_programt &program=gf.second.body;
258  for(auto &instruction : program.instructions)
259  {

261  if(instruction.has_condition())
262  expressions.push_back(&instruction.condition_nonconst());
263 
264  for(exprt *insts : expressions)
265  {
266  std::list<symbol_exprt> to_pointerize;
267  symbols_to_pointerize(linker_values, *insts, to_pointerize);
268  if(to_pointerize.empty())
269  continue;
270  log.debug() << "Pointerizing a program expression..." << messaget::eom;
271  int fail = pointerize_subexprs_of(*insts, to_pointerize, linker_values);
272  if(to_pointerize.empty() && fail==0)
273  continue;

277  log.error() << " Could not pointerize '" << sym.get_identifier()

280  log.error().source_location = instruction.source_location();
281  }
282  log.error() << messaget::eom;
283  }
284  }
285  }
286  return ret;

311 int linker_script_merget::pointerize_subexprs_of(
312  exprt &expr,
313  std::list<symbol_exprt> &to_pointerize,
314  const linker_valuest &linker_values)
315 {
316  int fail=0, tmp=0;
317  for(auto const &pair : linker_values)
318  for(auto const &pattern : replacement_predicates)
319  {
320  if(!pattern.match(expr))
321  continue;
322  // take a copy, expr will be changed below
323  const symbol_exprt inner_symbol=pattern.inner_symbol(expr);
324  if(pair.first!=inner_symbol.get_identifier())
325  continue;
326  tmp=replace_expr(expr, linker_values, inner_symbol, pair.first,
327  pattern.description());
328  fail=tmp?tmp:fail;
329  auto result=std::find(to_pointerize.begin(), to_pointerize.end(),
330  inner_symbol);
331  if(result==to_pointerize.end())
332  {
333  fail=1;
334  log.error() << "Too many removals of '" << inner_symbol.get_identifier()
335  << "'" << messaget::eom;
336  }
337  else
338  to_pointerize.erase(result);
339  // If we get here, we've just pointerized this expression. That expression
340  // will be a symbol possibly wrapped in some unary junk, but won't contain
341  // other symbols as subexpressions that might need to be pointerized. So
342  // it's safe to bail out here.
343  return fail;
344  }
345 
346  for(auto &op : expr.operands())
347  {
348  tmp = pointerize_subexprs_of(op, to_pointerize, linker_values);
349  fail=tmp?tmp:fail;
350  }
351  return fail;
352 }

355  const linker_valuest &linker_values,
