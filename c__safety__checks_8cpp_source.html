78 void c_safety_checks_rec(
79  goto_functionst::function_mapt::value_type &f,
80  const exprt::operandst &guards,
81  const exprt &expr,
82  access_typet access_type,
83  const namespacet &ns,
84  goto_programt &dest)
85 {
86  if(expr.id() == ID_address_of)
87  {
88  c_safety_checks_address_rec(
89  f, guards, to_address_of_expr(expr).object(), ns, dest);
90  return;
91  }
92  else if(expr.id() == ID_and)
93  {
94  auto new_guards = guards;
95  for(auto &op : expr.operands())
96  {
97  c_safety_checks_rec(
98  f, new_guards, op, access_type, ns, dest); // rec. call
99  new_guards.push_back(op);
100  }
101  return;
102  }
103  else if(expr.id() == ID_or)
104  {
105  auto new_guards = guards;
106  for(auto &op : expr.operands())
107  {
108  c_safety_checks_rec(
109  f, new_guards, op, access_type, ns, dest); // rec. call
110  new_guards.push_back(not_exprt(op));
111  }
112  return;
113  }
114  else if(expr.id() == ID_if)
115  {
116  const auto &if_expr = to_if_expr(expr);
117  c_safety_checks_rec(
118  f, guards, if_expr.cond(), access_typet::R, ns, dest); // rec. call
119  auto new_guards = guards;
120  new_guards.push_back(if_expr.cond());
121  c_safety_checks_rec(
122  f, new_guards, if_expr.true_case(), access_type, ns, dest); // rec. call
123  new_guards.pop_back();
124  new_guards.push_back(not_exprt(if_expr.cond()));
125  c_safety_checks_rec(
126  f, new_guards, if_expr.false_case(), access_type, ns, dest); // rec. call
127  return;
128  }
129  else if(expr.id() == ID_forall || expr.id() == ID_exists)
130  {
131  return;
132  }
133 
134  // now do operands
135  for(auto &op : expr.operands())
136  c_safety_checks_rec(f, guards, op, access_type, ns, dest); // rec. call
137 
138  if(expr.id() == ID_dereference)
139  {
140  if(expr.type().id() == ID_code)
141  {
142  // dealt with elsewhere
143  }
144  else
145  {
146  auto size_opt = pointer_offset_size(expr.type(), ns);
147  if(size_opt.has_value())
148  {
149  auto size = from_integer(*size_opt, size_type());
150  auto pointer = to_dereference_expr(expr).pointer();
151  auto condition = r_or_w_ok_exprt(
152  access_type == access_typet::W ? ID_w_ok : ID_r_ok, pointer, size);
153  condition.add_source_location() = expr.source_location();
154  auto assertion_source_location = expr.source_location();
155  assertion_source_location.set_property_class("pointer");
156  auto pointer_text = expr2c(pointer, ns);
157  assertion_source_location.set_comment(
158  "pointer " + pointer_text + " safe");
159  dest.add(goto_programt::make_assertion(
160  guard(guards, condition), assertion_source_location));
161  }
162  }
163  }
164  else if(expr.id() == ID_div)
165  {
166  const auto &div_expr = to_div_expr(expr);
167  if(
168  div_expr.divisor().is_constant() &&
169  !to_constant_expr(div_expr.divisor()).is_zero())
170  {
171  }
172  else
173  {
174  auto zero = from_integer(0, div_expr.type());
175  auto condition = implies_exprt(
176  conjunction(guards), notequal_exprt(div_expr.divisor(), zero));
177  auto source_location = expr.source_location();
178  condition.add_source_location() = expr.source_location();
179  source_location.set_property_class("division-by-zero");
180  source_location.set_comment("division by zero in " + expr2c(expr, ns));
181  dest.add(goto_programt::make_assertion(
182  guard(guards, condition), source_location));
183  }
184  }
185  else if(expr.id() == ID_mod)
186  {
187  const auto &mod_expr = to_mod_expr(expr);
188  if(
189  mod_expr.divisor().is_constant() &&
190  !to_constant_expr(mod_expr.divisor()).is_zero())
191  {
192  }
193  else
194  {
195  auto zero = from_integer(0, mod_expr.type());
196  auto condition = notequal_exprt(mod_expr.divisor(), zero);
197  auto source_location = expr.source_location();
198  condition.add_source_location() = expr.source_location();
199  source_location.set_property_class("division-by-zero");
200  source_location.set_comment("division by zero in " + expr2c(expr, ns));
201  dest.add(goto_programt::make_assertion(
202  guard(guards, condition), source_location));
203  }
204  }
205  else if(expr.id() == ID_index)
206  {
207  const auto &index_expr = to_index_expr(expr);
208  auto zero = from_integer(0, index_expr.index().type());
209  auto size = typecast_exprt::conditional_cast(
210  index_array_size(index_expr.array().type()), index_expr.index().type());
211  auto condition = and_exprt(
212  binary_relation_exprt(zero, ID_le, index_expr.index()),
213  binary_relation_exprt(size, ID_gt, index_expr.index()));
214  // 'index' may not have a source location, e.g., when implicitly
215  // taking the address of an array.
216  auto source_location = expr.find_source_location();
217  condition.add_source_location() = expr.source_location();
218  source_location.set_property_class("array bounds");
219  source_location.set_comment("array bounds in " + expr2c(expr, ns));
220  dest.add(
221  goto_programt::make_assertion(guard(guards, condition), source_location));
222  }
223 }

242 void c_safety_checks(
243  goto_functionst::function_mapt::value_type &f,
244  const namespacet &ns)
245 {
246  auto &body = f.second.body;
247  goto_programt dest;
248 
249  for(auto it = body.instructions.begin(); it != body.instructions.end(); it++)
250  {
251  if(it->is_assign())
252  {
253  c_safety_checks(f, it->assign_lhs(), access_typet::W, ns, dest);
254  c_safety_checks(f, it->assign_rhs(), access_typet::R, ns, dest);
255  }
256  else if(it->is_function_call())
257  {
258  c_safety_checks(f, it->call_lhs(), access_typet::W, ns, dest);
259  c_safety_checks(f, it->call_function(), access_typet::R, ns, dest);
260  for(const auto &argument : it->call_arguments())
261  c_safety_checks(f, argument, access_typet::R, ns, dest);
262  }
263  else
264  {
265  it->apply([&f, &ns, &dest](const exprt &expr) {
266  c_safety_checks(f, expr, access_typet::R, ns, dest);
267  });
268  }
269 
270  if(it->is_function_call())
271  {
272  const auto &function = it->call_function();
273  if(function.id() == ID_symbol)
274  {
275  const auto &identifier = to_symbol_expr(function).get_identifier();
276  if(identifier == "free")
277  {
278  if(
279  it->call_arguments().size() == 1 &&
280  it->call_arguments()[0].type().id() == ID_pointer)
281  {
282  // Must be 1) dynamically allocated, 2) still alive,
283  // 3) have offset zero, or, alternatively, NULL.
284  const auto &pointer = it->call_arguments()[0];
285  auto condition = or_exprt(
286  equal_exprt(
287  pointer, null_pointer_exprt(to_pointer_type(pointer.type()))),
288  and_exprt(
289  live_object_exprt(pointer),
290  is_dynamic_object_exprt(pointer),
291  offset_is_zero(pointer)));
292  auto source_location = it->source_location();
293  source_location.set_property_class("free");
294  source_location.set_comment(
295  "free argument must be valid dynamic object");
296  dest.add(goto_programt::make_assertion(condition, source_location));
297  }
298  }
299  else if(identifier == "realloc")
300  {
301  if(
302  it->call_arguments().size() == 2 &&
303  it->call_arguments()[0].type().id() == ID_pointer)
304  {
305  // Must be 1) dynamically allocated, 2) still alive,
306  // 3) have offset zero, or, alternatively, NULL.
307  const auto &pointer = it->call_arguments()[0];
308  auto condition = or_exprt(
309  equal_exprt(
310  pointer, null_pointer_exprt(to_pointer_type(pointer.type()))),
311  and_exprt(
312  live_object_exprt(pointer),
313  is_dynamic_object_exprt(pointer),
314  offset_is_zero(pointer)));
315  auto source_location = it->source_location();
316  source_location.set_property_class("realloc");
317  source_location.set_comment(
318  "realloc argument must be valid dynamic object");
319  dest.add(goto_programt::make_assertion(condition, source_location));
320  }
321  }
322  else if(identifier == "memcmp")
323  {
324  if(
325  it->call_arguments().size() == 3 &&
326  it->call_arguments()[0].type().id() == ID_pointer &&
327  it->call_arguments()[1].type().id() == ID_pointer &&
328  it->call_arguments()[2].type().id() == ID_unsignedbv)
329  {
330  // int memcmp(const void *s1, const void *s2, size_t n);
331  const auto &p1 = it->call_arguments()[0];
332  const auto &p2 = it->call_arguments()[1];
333  const auto &size = it->call_arguments()[2];
334  auto condition =
335  and_exprt(r_ok_exprt(p1, size), r_ok_exprt(p2, size));
336  auto source_location = it->source_location();
337  source_location.set_property_class("memcmp");
338  source_location.set_comment("memcmp regions must be valid");
339  dest.add(goto_programt::make_assertion(condition, source_location));
340  }
341  }
342  else if(identifier == "memchr")
343  {
344  if(
345  it->call_arguments().size() == 3 &&
346  it->call_arguments()[0].type().id() == ID_pointer &&
347  it->call_arguments()[2].type().id() == ID_unsignedbv)
348  {
349  // void *memchr(const void *, int, size_t);
350  const auto &p = it->call_arguments()[0];
351  const auto &size = it->call_arguments()[2];
352  auto condition = r_ok_exprt(p, size);
353  auto source_location = it->source_location();
354  source_location.set_property_class("memchr");
355  source_location.set_comment("memchr source must be valid");
356  dest.add(goto_programt::make_assertion(condition, source_location));
357  }
358  }
359  else if(identifier == "memset")
360  {
361  if(
362  it->call_arguments().size() == 3 &&
363  it->call_arguments()[0].type().id() == ID_pointer &&
364  it->call_arguments()[2].type().id() == ID_unsignedbv)
365  {
366  // void *memset(void *b, int c, size_t len);
367  const auto &pointer = it->call_arguments()[0];
368  const auto &size = it->call_arguments()[2];
369  auto condition = w_ok_exprt(pointer, size);
370  auto source_location = it->source_location();
371  source_location.set_property_class("memset");
372  source_location.set_comment("memset destination must be valid");
373  dest.add(goto_programt::make_assertion(condition, source_location));
374  }
375  }
376  else if(identifier == "strlen")
377  {
378  if(
379  it->call_arguments().size() == 1 &&
380  it->call_arguments()[0].type().id() == ID_pointer)
381  {
382  const auto &address = it->call_arguments()[0];
383  auto condition = is_cstring_exprt(address);
384  auto source_location = it->source_location();
385  source_location.set_property_class("strlen");
386  source_location.set_comment("strlen argument must be C string");
387  dest.add(goto_programt::make_assertion(condition, source_location));
388  }
389  }
390  else if(identifier == "__builtin___memset_chk") // clang variant
391  {
392  if(
393  it->call_arguments().size() == 4 &&
394  it->call_arguments()[0].type().id() == ID_pointer &&
395  it->call_arguments()[2].type().id() == ID_unsignedbv)
396  {
397  const auto &pointer = it->call_arguments()[0];
398  const auto &size = it->call_arguments()[2];
399  auto condition = w_ok_exprt(pointer, size);
400  auto source_location = it->source_location();
401  source_location.set_property_class("memset");
402  source_location.set_comment("memset destination must be valid");
403  dest.add(goto_programt::make_assertion(condition, source_location));
404  }
405  }
406  }
407  }
408 
409  std::size_t n = dest.instructions.size();
410  if(n != 0)
411  {
412  body.insert_before_swap(it, dest);
413  dest.clear();
414  it = std::next(it, n);
415  }
416  }
417 }
