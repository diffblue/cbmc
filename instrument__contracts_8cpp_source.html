110 exprt assigns_match(const exprt &assigns, const exprt &lhs)
111 {
112  if(is_symbol_member(lhs) && assigns == lhs)
113  return true_exprt(); // trivial match
114 
115  if(lhs.id() == ID_member)
116  {
117  if(assigns_match(assigns, to_member_expr(lhs).struct_op()).is_true())
118  return true_exprt();
119  }
120  else if(lhs.id() == ID_index)
121  {
122  if(assigns_match(assigns, to_index_expr(lhs).array()).is_true())
123  return true_exprt();
124  }

127  auto lhs_address = make_address(lhs);
128 
129  if(lhs.type() == assigns.type())

131  return equal_exprt(assigns_address, lhs_address);
132  }
133  else
134  {
135  // need to compare offset ranges

146 static exprt make_assigns_assertion(
147  irep_idt function_identifier,
148  const exprt::operandst &assigns,
149  const exprt &lhs)
150 {
151  exprt::operandst disjuncts;
152 
153  for(auto &assigns_clause : assigns)
154  {
155  auto a = instantiate_contract_lambda(assigns_clause);
156 
157  if(a.id() == ID_conditional_target_group)
158  {
159  auto &condition = to_binary_expr(a).op0();
160  auto &targets = to_multi_ary_expr(to_binary_expr(a).op1());
161  for(auto &target : targets.operands())
162  {
163  auto target_address = make_address(target);
164  auto lhs_address = make_address(lhs);
165  lhs_address =
166  typecast_exprt::conditional_cast(lhs_address, target_address.type());
167  disjuncts.push_back(
168  and_exprt(condition, equal_exprt(target_address, lhs_address)));
169  }
170  }
171  else
172  {
173  auto match = assigns_match(a, lhs);
174 
175  // trivial?
176  if(match.is_true())
177  return true_exprt();
178 
179  disjuncts.push_back(std::move(match));
180  }
181  }
182 
183  return disjunction(disjuncts);

271 void instrument_contract_checks(
272  goto_functionst::function_mapt::value_type &f,
273  const namespacet &ns)
274 {
275  // contracts are in a separate symbol, with prefix "contract::"
276  auto contract_identifier = "contract::" + id2string(f.first);
277  const symbolt *symbol_ptr;
278  if(ns.lookup(contract_identifier, symbol_ptr))
279  return; // nothing to check
280 
281  auto &contract = to_code_with_contract_type(symbol_ptr->type);
282 
283  auto &body = f.second.body;
284 
285  if(body.instructions.empty())
286  return; // nothing to check
287 
288  // new instructions to add at the beginning of the function
289  goto_programt add_at_beginning;
290 
291  // precondition?
292  if(!contract.requires().empty())
293  {
294  // stick these in as assumptions, preserving the ordering
295  goto_programt dest;
296  for(auto &assumption : contract.requires())
297  {
298  exprt assumption_instance = instantiate_contract_lambda(assumption);
299  auto fixed_assumption = add_function(f.first, assumption_instance);
300  add_at_beginning.add(goto_programt::make_assumption(
301  fixed_assumption, fixed_assumption.source_location()));
302  }
303  }
304 
305  // record "old(...)" expressions.
306  std::vector<std::pair<symbol_exprt, exprt>> old_exprs;
307  const auto old_prefix = "old::" + id2string(f.first);
308 
309  // postcondition?
310  if(!contract.ensures().empty())
311  {
312  // Stick the postconditions in as assertions at the end
313  auto last = body.instructions.end();
314  if(std::prev(last)->is_end_function())
315  last = std::prev(last);
316 
317  for(auto &assertion : contract.ensures())
318  {
319  exprt assertion_instance = instantiate_contract_lambda(assertion);
320 
321  std::string comment = "postcondition";
322  if(contract.ensures().size() >= 2)
323  comment += " " + expr2text(assertion_instance, ns);
324 
325  auto location = assertion.source_location();
326  location.set_function(f.first); // seems to be missing
327  location.set_property_class(ID_postcondition);
328  location.set_comment(comment);
329 
330  auto replaced_assertion =
331  replace_old(assertion_instance, old_prefix, old_exprs);
332 
333  auto fixed_assertion = add_function(f.first, replaced_assertion);
334 
335  auto assertion_instruction =
336  goto_programt::make_assertion(fixed_assertion, std::move(location));
337 
338  body.insert_before_swap(last, assertion_instruction);
339  }
340  }
341 
342  // do 'old' in the body
343  if(
344  !contract.assigns().empty() || !contract.requires().empty() ||
345  !contract.ensures().empty())
346  {
347  for(auto &instruction : body.instructions)
348  instruction.transform(
349  [&old_prefix, &old_exprs](exprt expr) -> optionalt<exprt> {
350  return replace_old(expr, old_prefix, old_exprs);
351  });
352  }
353 
354  // Add assignments to 'old' symbols at the beginning of the function.
355  {
356  auto tmp =
357  old_assignments(old_exprs, add_function(f.first, symbol_ptr->location));
358  add_at_beginning.destructive_append(tmp);
359  }
360 
361  body.destructive_insert(body.instructions.begin(), add_at_beginning);
362 
363  // assigns?
364  if(
365  !contract.assigns().empty() || !contract.requires().empty() ||
366  !contract.ensures().empty())
367  {
368  for(auto it = body.instructions.begin(); it != body.instructions.end();
369  it++)
370  {
371  if(it->is_assign())
372  {
373  const auto &lhs = it->assign_lhs();
374 
375  // Parameter or local or old? Ignore.
376  if(is_procedure_local(f.first, lhs))
377  continue; // ok
378 
379  if(is_old(lhs))
380  continue; // ok
381 
382  // maybe not ok
383  auto assigns_assertion =
384  make_assigns_assertion(f.first, contract.assigns(), lhs);
385  auto location = it->source_location();
386  location.set_property_class("assigns");
387  location.set_comment("assigns clause");
388  auto assertion_instruction = goto_programt::make_assertion(
389  std::move(assigns_assertion), std::move(location));
390  body.insert_before_swap(it, assertion_instruction);
391  it++; // skip over the assertion we have just generated
392  }
393  }
394  }
395 }

397 void replace_function_calls_by_contracts(
398  goto_functionst::function_mapt::value_type &f,
399  const goto_modelt &goto_model)
400 {
401  auto &body = f.second.body;
402  const namespacet ns(goto_model.symbol_table);
403 
404  std::size_t call_site_counter = 0;
405 
406  for(auto it = body.instructions.begin(); it != body.instructions.end(); it++)
407  {
408  if(it->is_function_call())
409  {
410  const auto &function = it->call_function();
411  if(function.id() == ID_symbol)
412  {
413  const auto &symbol = ns.lookup(to_symbol_expr(function));
414 
415  const auto contract_opt = get_contract(symbol.name, ns);
416 
417  if(!contract_opt.has_value())
418  continue;
419 
420  auto &contract = contract_opt.value();
421 
422  // record "old(...)" expressions.
423  std::vector<std::pair<symbol_exprt, exprt>> old_exprs;
424  const auto old_prefix = "old::" + id2string(f.first) + "::call-site-" +
425  std::to_string(++call_site_counter) + "::";
426 
427  // need to substitute parameters
428  const auto f_it =
429  goto_model.goto_functions.function_map.find(symbol.name);
430 
431  if(f_it == goto_model.goto_functions.function_map.end())
432  DATA_INVARIANT(false, "failed to find function in function_map");
433 
434  replace_symbolt replace_symbol;
435  const auto &parameters = to_code_type(symbol.type).parameters();
436  const auto &arguments = it->call_arguments();
437 
438  for(std::size_t p = 0; p < f_it->second.parameter_identifiers.size();
439  p++)
440  {
441  auto p_symbol = symbol_exprt(
442  f_it->second.parameter_identifiers[p], parameters[p].type());
443  replace_symbol.insert(p_symbol, arguments[p]);
444  }
445 
446  // replace __CPROVER_return_value by the lhs of the call
447  const auto &call_lhs = it->call_lhs();
448  replace_symbol.insert(
449  symbol_exprt(CPROVER_PREFIX "return_value", call_lhs.type()),
450  call_lhs);
451 
452  goto_programt dest;
453 
454  // assert the preconditions
455  for(auto &precondition : contract.requires())
456  {
457  auto instantiated_precondition =
458  instantiate_contract_lambda(precondition);
459 
460  auto location = it->source_location();
461  location.set_property_class(ID_precondition);
462  location.set_comment(
463  id2string(symbol.display_name()) + " precondition " +
464  expr2text(instantiated_precondition, ns));
465 
466  auto replaced_precondition = instantiated_precondition;
467  replace_symbol(replaced_precondition);
468 
469  dest.add(
470  goto_programt::make_assertion(replaced_precondition, location));
471  }
472 
473  // havoc the 'assigned' variables
474  for(auto &assigns_clause_lambda : contract.assigns())
475  {
476  auto location = it->source_location();
477 
478  auto assigns_clause =
479  instantiate_contract_lambda(assigns_clause_lambda);
480 
481  if(assigns_clause.id() == ID_conditional_target_group)
482  {
483  const auto &condition = to_binary_expr(assigns_clause).op0();
484  auto replaced_condition = condition;
485  replace_symbol(replaced_condition);
486 
487  const auto &targets =
488  to_multi_ary_expr(to_binary_expr(assigns_clause).op1())
489  .operands();
490 
491  for(auto &target : targets)
492  {
493  auto rhs = side_effect_expr_nondett(target.type(), location);
494 
495  auto replaced_lhs = target;
496  replace_symbol(replaced_lhs);
497 
498  auto goto_instruction =
499  dest.add(goto_programt::make_incomplete_goto(
500  not_exprt(replaced_condition), location));
501 
502  dest.add(
503  goto_programt::make_assignment(replaced_lhs, rhs, location));
504 
505  auto skip_instruction =
506  dest.add(goto_programt::make_skip(location));
507 
508  goto_instruction->complete_goto(skip_instruction);
509  }
510  }
511  else
512  {
513  const auto &lhs = assigns_clause;
514  auto rhs = side_effect_expr_nondett(lhs.type(), location);
515 
516  auto replaced_lhs = lhs;
517  replace_symbol(replaced_lhs);
518  auto fixed_lhs = replace_source_location(replaced_lhs, location);
519 
520  dest.add(goto_programt::make_assignment(fixed_lhs, rhs, location));
521  }
522  }
523 
524  // assume the postconditions
525  for(auto &postcondition : contract.ensures())
526  {
527  auto &location = it->source_location();
528 
529  auto replaced_postcondition1 =
530  instantiate_contract_lambda(postcondition);
531  replace_symbol(replaced_postcondition1);
532 
533  auto replaced_postcondition2 =
534  replace_old(replaced_postcondition1, old_prefix, old_exprs);
535 
536  dest.add(
537  goto_programt::make_assumption(replaced_postcondition2, location));
538  }
539 
540  // now insert the assignents to old::... at the beginning
541  // of 'dest'
542  {
543  auto tmp = old_assignments(old_exprs, it->source_location());
544  dest.destructive_insert(dest.instructions.begin(), tmp);
545  }
546 
547  // remove the function call
548  it->turn_into_skip();
549 
550  // insert after 'it' to preserve branches to the call
551  body.destructive_insert(std::next(it), dest);
552  }
553  }
554  }
