75 static std::pair<irep_idt, irep_idt>
76 parse_function_contract_pair(const irep_idt &cli_flag)
77 {
78  auto const correct_format_message =
79  "the format for function and contract pairs is "
80  "`<function_name>[/<contract_name>]`";
81 
82  std::string cli_flag_str = id2string(cli_flag);
83 
84  auto split = split_string(cli_flag_str, '/', true, false);
85 
86  if(split.size() == 1)
87  {
88  return std::make_pair(cli_flag, cli_flag);
89  }
90  else if(split.size() == 2)
91  {
92  auto function_name = split[0];
93  if(function_name.empty())
94  {
95  throw invalid_function_contract_pair_exceptiont{
96  "couldn't find function name before '/' in '" + cli_flag_str + "'",
97  correct_format_message};
98  }
99  auto contract_name = split[1];
100  if(contract_name.empty())
101  {
102  throw invalid_function_contract_pair_exceptiont{
103  "couldn't find contract name after '/' in '" + cli_flag_str + "'",
104  correct_format_message};
105  }
106  return std::make_pair(function_name, contract_name);
107  }
108  else
109  {
110  throw invalid_function_contract_pair_exceptiont{
111  "couldn't parse '" + cli_flag_str + "'", correct_format_message};
112  }
113 }

194 void dfcct::check_transform_goto_model_preconditions()
195 {
196  // check that harness function exists
197  PRECONDITION_WITH_DIAGNOSTICS(
198  dfcc_utilst::function_symbol_with_body_exists(goto_model, harness_id),
199  "Harness function '" + id2string(harness_id) +
200  "' either not found or has no body");
201 
202  if(to_check.has_value())
203  {
204  auto pair = to_check.value();
205  PRECONDITION_WITH_DIAGNOSTICS(
206  dfcc_utilst::function_symbol_with_body_exists(goto_model, pair.first),
207  "Function to check '" + id2string(pair.first) +
208  "' either not found or has no body");
209 
210  // triggers signature compatibility checking
211  contract_handler.get_pure_contract_symbol(pair.second, pair.first);
212 
213  PRECONDITION_WITH_DIAGNOSTICS(
214  pair.first != harness_id,
215  "Function '" + id2string(pair.first) +
216  "' cannot be both be checked against a contract and be the harness");
217 
218  PRECONDITION_WITH_DIAGNOSTICS(
219  pair.second != harness_id,
220  "Function '" + id2string(pair.first) +
221  "' cannot be both the contract to check and be the harness");
222 
223  PRECONDITION_WITH_DIAGNOSTICS(
224  to_replace.find(pair.first) == to_replace.end(),
225  "Function '" + id2string(pair.first) +
226  "' cannot be both checked against contract and replaced by a contract");
227 
228  PRECONDITION_WITH_DIAGNOSTICS(
229  !instrument.do_not_instrument(pair.first),
230  "CPROVER function or builtin '" + id2string(pair.first) +
231  "' cannot be checked against a contract");
232  }
233 
234  for(const auto &pair : to_replace)
235  {
236  // for functions to replace with contracts we don't require the replaced
237  // function to have a body because only the contract is actually used
238  PRECONDITION_WITH_DIAGNOSTICS(
239  dfcc_utilst::function_symbol_exists(goto_model, pair.first),
240  "Function to replace '" + id2string(pair.first) + "' not found");
241 
242  // triggers signature compatibility checking
243  contract_handler.get_pure_contract_symbol(pair.second, pair.first);
244 
245  PRECONDITION_WITH_DIAGNOSTICS(
246  pair.first != harness_id,
247  "Function '" + id2string(pair.first) +
248  "' cannot both be replaced with a contract and be the harness");
249 
250  PRECONDITION_WITH_DIAGNOSTICS(
251  pair.second != harness_id,
252  "Function '" + id2string(pair.first) +
253  "' cannot both be the contract to use for replacement and be the "
254  "harness");

259  std::set<irep_idt> &contract_symbols,
260  std::set<irep_idt> &other_symbols)
261 {
262  // collect contract and other symbols
263  for(auto &entry : goto_model.symbol_table)
264  {
265  const symbolt &symbol = entry.second;
266 
267  // not a function symbol
268  if(symbol.type.id() != ID_code || symbol.is_macro)
269  continue;
270 
271  // is it a pure contract ?
272  const irep_idt &sym_name = symbol.name;
273  if(symbol.is_property && has_prefix(id2string(sym_name), "contract::"))
274  {
275  contract_symbols.insert(sym_name);
276  }
277  else
278  {
279  // it is not a contract
280  other_symbols.insert(sym_name);
281  }
282  }

378 void dfcct::wrap_discovered_function_pointer_contracts()
379 {
380  std::set<irep_idt> swapped;
381  while(!function_pointer_contracts.empty())
382  {
383  std::set<irep_idt> new_contracts;
384  // swap-and-wrap function pointer contracts with themselves
385  for(const auto &fp_contract : function_pointer_contracts)
386  {
387  if(swapped.find(fp_contract) != swapped.end())
388  continue;
389 
390  // contracts for function pointers must be replaced with themselves
391  // so we need to check that:
392  // - the symbol exists as a function symbol
393  // - the symbol exists as a pure contract symbol
394  // - the function symbol is not already swapped for contract checking
395  // - the function symbol is not already swapped with another contract for
396  // replacement
397 
398  const auto str = id2string(fp_contract);
399 
400  // Is it already swapped with another function for contract checking ?
401  PRECONDITION_WITH_DIAGNOSTICS(
402  !to_check.has_value() || to_check.value().first != str,
403  "Function '" + str +
404  "' used as contract for function pointer cannot be itself the object "
405  "of a contract check.");
406 
407  // Is it already swapped with another function for contract checking ?
408  auto found = to_replace.find(str);
409  if(found != to_replace.end())
410  {
411  PRECONDITION_WITH_DIAGNOSTICS(
412  found->first == found->second,
413  "Function '" + str +
414  "' used as contract for function pointer already the object of a "
415  "contract replacement with '" +
416  id2string(found->second) + "'");
417  log.status() << "Function pointer contract '" << fp_contract
418  << "' already wrapped with itself in REPLACE mode"
419  << messaget::eom;
420  }
421  else
422  {
423  // we need to swap it with itself
424  PRECONDITION_WITH_DIAGNOSTICS(
425  dfcc_utilst::function_symbol_exists(goto_model, str),
426  "Function pointer contract '" + str + "' not found.");
427 
428  // triggers signature compatibility checking
429  contract_handler.get_pure_contract_symbol(str);
430 
431  log.status() << "Wrapping function pointer contract '" << fp_contract
432  << "' with itself in REPLACE mode" << messaget::eom;
433 
434  swap_and_wrap.swap_and_wrap_replace(
435  fp_contract, fp_contract, new_contracts);
436  swapped.insert(fp_contract);
437 
438  // remove it from the set of symbols to process
439  if(other_symbols.find(fp_contract) != other_symbols.end())
440  other_symbols.erase(fp_contract);
441  }
442  }
443  // process newly discovered contracts
444  function_pointer_contracts = new_contracts;
445  }
446 }

468 void dfcct::transform_goto_model()
469 {
470  check_transform_goto_model_preconditions();
471  link_model_and_load_dfcc_library();
472  lift_memory_predicates();
473  instrument_harness_function();
474  wrap_checked_function();
475  wrap_replaced_functions();
476  wrap_discovered_function_pointer_contracts();
477  instrument_other_functions();
478 
479  // take the max of function of loop contracts assigns clauses
480  auto assigns_clause_size = instrument.get_max_assigns_clause_size();
481  if(assigns_clause_size > max_assigns_clause_size)
482  max_assigns_clause_size = assigns_clause_size;
483 
484  log.status() << "Specializing cprover_contracts functions for assigns "
485  "clauses of at most "
486  << max_assigns_clause_size << " targets" << messaget::eom;
487  library.specialize(max_assigns_clause_size);
488 
489  library.inhibit_front_end_builtins();
490 
491  // TODO implement a means to inhibit unreachable functions (possibly via the
492  // code that implements drop-unused-functions followed by
493  // generate-function-bodies):
494  // Traverse the call tree from the given entry point to identify
495  // functions symbols that are effectively called in the model,
496  // Then goes over all functions of the model and turns the bodies of all
497  // functions that are not in the used function set into:
498  // ```c
499  // assert(false, "function identified as unreachable");
500  // assume(false);
501  // ```
502  // That way, if the analysis mistakenly pruned some functions, assertions
503  // will be violated and the analysis will fail.
504  // TODO: add a command line flag to tell the instrumentation to not prune
505  // a function.
506  goto_model.goto_functions.update();
507 
508  remove_skip(goto_model);
509  goto_model.goto_functions.update();
510 
511  log.status() << "Removing unused functions" << messaget::eom;
512 
513  // This can prune too many functions if function pointers have not been
514  // yet been removed or if the entry point is not defined.
515  // Another solution would be to rewrite the bodies of functions that seem to
516  // be unreachable into assert(false);assume(false)
517  remove_unused_functions(goto_model, message_handler);
518  goto_model.goto_functions.update();
519 
520  reinitialize_model();
521 }

523 void dfcct::reinitialize_model()
524 {
525  // collect set of functions which are now instrumented
526  std::set<irep_idt> instrumented_functions;
527  instrument.get_instrumented_functions(instrumented_functions);
528  swap_and_wrap.get_swapped_functions(instrumented_functions);
529 
530  log.status() << "Updating init function" << messaget::eom;
531  if(goto_model.can_produce_function(INITIALIZE_FUNCTION))
532  recreate_initialize_function(goto_model, message_handler);
533  nondet_static(goto_model, to_exclude_from_nondet_static);
534 
535  // Initialize the map of instrumented functions by adding extra instructions
536  // to the harness function
537  auto &init_function = goto_model.goto_functions.function_map[harness_id];
538  auto &body = init_function.body;
539  auto begin = body.instructions.begin();
540  goto_programt payload;
541  library.add_instrumented_functions_map_init_instructions(
542  instrumented_functions, begin->source_location(), payload);
543  body.destructive_insert(begin, payload);
544 
545  // Define harness as the entry point, overriding any preexisting one.
546  log.status() << "Setting entry point to " << harness_id << messaget::eom;
547  // remove the CPROVER start function
548  goto_model.symbol_table.erase(
549  goto_model.symbol_table.symbols.find(goto_functionst::entry_point()));
550  // regenerate the CPROVER start function
551  generate_ansi_c_start_function(
552  dfcc_utilst::get_function_symbol(goto_model.symbol_table, harness_id),
553  goto_model.symbol_table,
554  message_handler,
555  c_object_factory_parameterst(options));
556 
557  goto_model.goto_functions.update();
558 }
