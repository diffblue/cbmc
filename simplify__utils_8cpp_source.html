136 static bool sort_and_join(exprt &expr, bool do_sort)
137 {
138  bool no_change = true;
139 
140  if(!expr.has_operands())
141  return true;
142 
143  const struct saj_tablet &saj_entry =
144  get_sort_and_join_table_entry(expr.id(), as_const(expr).type().id());
145  if(saj_entry.id.empty())
146  return true;
147 
148  // check operand types
149 
150  forall_operands(it, expr)
151  if(!is_associative_and_commutative_for_type(saj_entry, it->type().id()))
152  return true;
153 
154  // join expressions
155 
156  exprt::operandst new_ops;
157  new_ops.reserve(as_const(expr).operands().size());
158 
159  forall_operands(it, expr)
160  {
161  if(it->id()==expr.id())
162  {
163  new_ops.reserve(new_ops.capacity()+it->operands().size()-1);
164 
165  forall_operands(it2, *it)
166  new_ops.push_back(*it2);
167 
168  no_change = false;
169  }
170  else
171  new_ops.push_back(*it);
172  }
173 
174  // sort it
175  if(do_sort)
176  no_change = sort_operands(new_ops) && no_change;
177 
178  if(!no_change)
179  expr.operands().swap(new_ops);
180 
181  return no_change;
182 }

194 optionalt<exprt> bits2expr(
195  const std::string &bits,
196  const typet &type,
197  bool little_endian,
198  const namespacet &ns)
199 {
200  // bits start at lowest memory address
201  auto type_bits = pointer_offset_bits(type, ns);
202 
203  if(
204  !type_bits.has_value() ||
205  (type.id() != ID_union && type.id() != ID_union_tag &&
206  *type_bits != bits.size()) ||
207  ((type.id() == ID_union || type.id() == ID_union_tag) &&
208  *type_bits < bits.size()))
209  {
210  return {};
211  }
212 
213  if(
214  type.id() == ID_unsignedbv || type.id() == ID_signedbv ||
215  type.id() == ID_floatbv || type.id() == ID_fixedbv ||
216  type.id() == ID_c_bit_field || type.id() == ID_pointer ||
217  type.id() == ID_bv || type.id() == ID_c_bool)
218  {
219  if(
220  type.id() == ID_pointer && config.ansi_c.NULL_is_zero &&
221  bits.find('1') == std::string::npos)
222  {
223  return null_pointer_exprt(to_pointer_type(type));
224  }
225 
226  endianness_mapt map(type, little_endian, ns);
227 
228  std::string tmp = bits;
229  for(std::string::size_type i = 0; i < bits.size(); ++i)
230  tmp[i] = bits[map.map_bit(i)];
231 
232  std::reverse(tmp.begin(), tmp.end());
233 
234  mp_integer i = binary2integer(tmp, false);
235  return constant_exprt(integer2bvrep(i, bits.size()), type);
236  }
237  else if(type.id() == ID_c_enum)
238  {
239  auto val = bits2expr(
240  bits, to_c_enum_type(type).underlying_type(), little_endian, ns);
241  if(val.has_value())
242  {
243  val->type() = type;
244  return *val;
245  }
246  else
247  return {};
248  }
249  else if(type.id() == ID_c_enum_tag)
250  {
251  auto val = bits2expr(
252  bits, ns.follow_tag(to_c_enum_tag_type(type)), little_endian, ns);
253  if(val.has_value())
254  {
255  val->type() = type;
256  return *val;
257  }
258  else
259  return {};
260  }
261  else if(type.id() == ID_union)
262  {
263  // find a suitable member
264  const union_typet &union_type = to_union_type(type);
265  const union_typet::componentst &components = union_type.components();
266 
267  if(components.empty() && bits.empty())
268  return empty_union_exprt{type};
269 
270  for(const auto &component : components)
271  {
272  auto val = bits2expr(bits, component.type(), little_endian, ns);
273  if(!val.has_value())
274  continue;
275 
276  return union_exprt(component.get_name(), *val, type);
277  }
278  }
279  else if(type.id() == ID_union_tag)
280  {
281  auto val = bits2expr(
282  bits, ns.follow_tag(to_union_tag_type(type)), little_endian, ns);
283  if(val.has_value())
284  {
285  val->type() = type;
286  return *val;
287  }
288  else
289  return {};
290  }
291  else if(type.id() == ID_struct)
292  {
293  const struct_typet &struct_type = to_struct_type(type);
294  const struct_typet::componentst &components = struct_type.components();
295 
296  struct_exprt result({}, type);
297  result.reserve_operands(components.size());
298 
299  mp_integer m_offset_bits = 0;
300  for(const auto &component : components)
301  {
302  const auto m_size = pointer_offset_bits(component.type(), ns);
303  CHECK_RETURN(m_size.has_value() && *m_size >= 0);
304 
305  std::string comp_bits = std::string(
306  bits,
307  numeric_cast_v<std::size_t>(m_offset_bits),
308  numeric_cast_v<std::size_t>(*m_size));
309 
310  auto comp = bits2expr(comp_bits, component.type(), little_endian, ns);
311  if(!comp.has_value())
312  return {};
313  result.add_to_operands(std::move(*comp));
314 
315  m_offset_bits += *m_size;
316  }
317 
318  return std::move(result);
319  }
320  else if(type.id() == ID_struct_tag)
321  {
322  auto val = bits2expr(
323  bits, ns.follow_tag(to_struct_tag_type(type)), little_endian, ns);
324  if(val.has_value())
325  {
326  val->type() = type;
327  return *val;
328  }
329  else
330  return {};
331  }
332  else if(type.id() == ID_array)
333  {
334  const array_typet &array_type = to_array_type(type);
335  const auto &size_expr = array_type.size();
336 
337  PRECONDITION(size_expr.is_constant());
338 
339  const std::size_t number_of_elements =
340  numeric_cast_v<std::size_t>(to_constant_expr(size_expr));
341 
342  const auto el_size_opt = pointer_offset_bits(array_type.element_type(), ns);
343  CHECK_RETURN(el_size_opt.has_value() && *el_size_opt > 0);
344 
345  const std::size_t el_size = numeric_cast_v<std::size_t>(*el_size_opt);
346 
347  array_exprt result({}, array_type);
348  result.reserve_operands(number_of_elements);
349 
350  for(std::size_t i = 0; i < number_of_elements; ++i)
351  {
352  std::string el_bits = std::string(bits, i * el_size, el_size);
353  auto el =
354  bits2expr(el_bits, array_type.element_type(), little_endian, ns);
355  if(!el.has_value())
356  return {};
357  result.add_to_operands(std::move(*el));
358  }
359 
360  return std::move(result);
361  }
362  else if(type.id() == ID_vector)
363  {
364  const vector_typet &vector_type = to_vector_type(type);
365 
366  const std::size_t n_el = numeric_cast_v<std::size_t>(vector_type.size());
367 
368  const auto el_size_opt =
369  pointer_offset_bits(vector_type.element_type(), ns);
370  CHECK_RETURN(el_size_opt.has_value() && *el_size_opt > 0);
371 
372  const std::size_t el_size = numeric_cast_v<std::size_t>(*el_size_opt);
373 
374  vector_exprt result({}, vector_type);
375  result.reserve_operands(n_el);
376 
377  for(std::size_t i = 0; i < n_el; ++i)
378  {
379  std::string el_bits = std::string(bits, i * el_size, el_size);
380  auto el =
381  bits2expr(el_bits, vector_type.element_type(), little_endian, ns);
382  if(!el.has_value())
383  return {};
384  result.add_to_operands(std::move(*el));
385  }
386 
387  return std::move(result);
388  }
389  else if(type.id() == ID_complex)
390  {
391  const complex_typet &complex_type = to_complex_type(type);
392 
393  const auto sub_size_opt = pointer_offset_bits(complex_type.subtype(), ns);
394  CHECK_RETURN(sub_size_opt.has_value() && *sub_size_opt > 0);
395 
396  const std::size_t sub_size = numeric_cast_v<std::size_t>(*sub_size_opt);
397 
398  auto real = bits2expr(
399  bits.substr(0, sub_size), complex_type.subtype(), little_endian, ns);
400  auto imag = bits2expr(
401  bits.substr(sub_size), complex_type.subtype(), little_endian, ns);
402  if(!real.has_value() || !imag.has_value())
403  return {};
404 
405  return complex_exprt(*real, *imag, complex_type);
406  }
407 
408  return {};
409 }

411 optionalt<std::string>
412 expr2bits(const exprt &expr, bool little_endian, const namespacet &ns)
413 {
414  // extract bits from lowest to highest memory address
415  const typet &type = expr.type();
416 
417  if(expr.id() == ID_constant)
418  {
419  const auto &value = to_constant_expr(expr).get_value();
420 
421  if(
422  type.id() == ID_unsignedbv || type.id() == ID_signedbv ||
423  type.id() == ID_floatbv || type.id() == ID_fixedbv ||
424  type.id() == ID_c_bit_field || type.id() == ID_bv ||
425  type.id() == ID_c_bool)
426  {
427  const auto width = to_bitvector_type(type).get_width();
428 
429  endianness_mapt map(type, little_endian, ns);
430 
431  std::string result(width, ' ');
432 
433  for(std::string::size_type i = 0; i < width; ++i)
434  result[map.map_bit(i)] = get_bvrep_bit(value, width, i) ? '1' : '0';
435 
436  return result;
437  }
438  else if(type.id() == ID_pointer)
439  {
440  if(config.ansi_c.NULL_is_zero && is_null_pointer(to_constant_expr(expr)))
441  return std::string(to_bitvector_type(type).get_width(), '0');
442  else
443  return {};
444  }
445  else if(type.id() == ID_c_enum_tag)
446  {
447  const auto &c_enum_type = ns.follow_tag(to_c_enum_tag_type(type));
448  return expr2bits(constant_exprt(value, c_enum_type), little_endian, ns);
449  }
450  else if(type.id() == ID_c_enum)
451  {
452  return expr2bits(
453  constant_exprt(value, to_c_enum_type(type).underlying_type()),
454  little_endian,
455  ns);
456  }
457  }
458  else if(expr.id() == ID_string_constant)
459  {
460  return expr2bits(
461  to_string_constant(expr).to_array_expr(), little_endian, ns);
462  }
463  else if(expr.id() == ID_union)
464  {
465  return expr2bits(to_union_expr(expr).op(), little_endian, ns);
466  }
467  else if(
468  expr.id() == ID_struct || expr.id() == ID_array || expr.id() == ID_vector ||
469  expr.id() == ID_complex)
470  {
471  std::string result;
472  forall_operands(it, expr)
473  {
474  auto tmp = expr2bits(*it, little_endian, ns);
475  if(!tmp.has_value())
476  return {}; // failed
477  result += tmp.value();
478  }
479 
480  return result;
481  }
482 
483  return {};
484 }
