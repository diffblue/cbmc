34 void c_typecheck_baset::typecheck_type(typet &type)
35 {
36  // we first convert, and then check
37  {
38  ansi_c_convert_typet ansi_c_convert_type{get_message_handler(), type};
39  ansi_c_convert_type.write(type);
40  }
41 
42  if(type.id()==ID_already_typechecked)
43  {
44  already_typechecked_typet &already_typechecked =
45  to_already_typechecked_type(type);
46 
47  // need to preserve any qualifiers
48  c_qualifierst c_qualifiers(type);
49  c_qualifiers += c_qualifierst(already_typechecked.get_type());
50  bool packed=type.get_bool(ID_C_packed);

62  type.add(ID_C_typedef, _typedef);
63

65  }
66

68  if(type.find(ID_C_alignment).is_not_nil())
69  {
70  exprt &alignment=static_cast<exprt &>(type.add(ID_C_alignment));
71  if(alignment.id()!=ID_default)
72  {
73  typecheck_expr(alignment);
74  make_constant(alignment);
75  }
76  }
77 
78  if(type.id()==ID_code)
79  typecheck_code_type(to_code_type(type));
80  else if(type.id()==ID_array)
81  typecheck_array_type(to_array_type(type));
82  else if(type.id()==ID_pointer)
83  {

85  INVARIANT(
86  to_bitvector_type(type).get_width() > 0, "pointers must have width");
87  }
88  else if(type.id()==ID_struct ||
89  type.id()==ID_union)
90  typecheck_compound_type(to_struct_union_type(type));
91  else if(type.id()==ID_c_enum)
92  typecheck_c_enum_type(type);
93  else if(type.id()==ID_c_enum_tag)
94  typecheck_c_enum_tag_type(to_c_enum_tag_type(type));
95  else if(type.id()==ID_c_bit_field)
96  typecheck_c_bit_field_type(to_c_bit_field_type(type));
97  else if(type.id()==ID_typeof)
98  typecheck_typeof_type(type);
99  else if(type.id() == ID_typedef_type)
100  typecheck_typedef_type(type);
101  else if(type.id() == ID_struct_tag ||
102  type.id() == ID_union_tag)
103  {
104  // nothing to do, these stay as is
105  }
106  else if(type.id()==ID_vector)
107  {
108  // already done
109  }
110  else if(type.id() == ID_frontend_vector)
111  {
112  typecheck_vector_type(type);
113  }
114  else if(type.id()==ID_custom_unsignedbv ||
115  type.id()==ID_custom_signedbv ||
116  type.id()==ID_custom_floatbv ||
117  type.id()==ID_custom_fixedbv)
118  typecheck_custom_type(type);
119  else if(type.id()==ID_gcc_attribute_mode)
120  {
121  // get that mode
122  const irep_idt gcc_attr_mode = type.get(ID_size);
123 
124  // A list of all modes is at
125  // http://www.delorie.com/gnu/docs/gcc/gccint_53.html
126  typecheck_type(to_type_with_subtype(type).subtype());
127 
128  typet underlying_type = to_type_with_subtype(type).subtype();
129 
130  // gcc allows this, but clang doesn't; it's a compiler hint only,
131  // but we'll try to interpret it the GCC way
132  if(underlying_type.id()==ID_c_enum_tag)
133  {
134  underlying_type =
135  follow_tag(to_c_enum_tag_type(underlying_type)).underlying_type();
136 
137  DATA_INVARIANT(
138  underlying_type.id() == ID_signedbv ||
139  underlying_type.id() == ID_unsignedbv,
140  "underlying type must be bitvector");
141  }

143  if(underlying_type.id()==ID_signedbv ||
144  underlying_type.id()==ID_unsignedbv)

147 
148  typet result;
149 
150  if(gcc_attr_mode == "__QI__") // 8 bits
151  {
152  if(is_signed)
153  result=signed_char_type();
154  else
155  result=unsigned_char_type();
156  }
157  else if(gcc_attr_mode == "__byte__") // 8 bits
158  {
159  if(is_signed)
160  result=signed_char_type();
161  else
162  result=unsigned_char_type();
163  }
164  else if(gcc_attr_mode == "__HI__") // 16 bits
165  {
166  if(is_signed)
167  result=signed_short_int_type();
168  else
169  result=unsigned_short_int_type();
170  }
171  else if(gcc_attr_mode == "__SI__") // 32 bits
172  {
173  if(is_signed)
174  result=signed_int_type();
175  else
176  result=unsigned_int_type();
177  }
178  else if(gcc_attr_mode == "__word__") // long int, we think
179  {
180  if(is_signed)
181  result=signed_long_int_type();
182  else
183  result=unsigned_long_int_type();
184  }
185  else if(gcc_attr_mode == "__pointer__") // size_t/ssize_t, we think
186  {
187  if(is_signed)
188  result=signed_size_type();
189  else
190  result=size_type();
191  }
192  else if(gcc_attr_mode == "__DI__") // 64 bits
193  {
194  if(config.ansi_c.long_int_width==64)
195  {
196  if(is_signed)
197  result=signed_long_int_type();
198  else
199  result=unsigned_long_int_type();

201  else

204 
205  if(is_signed)
206  result=signed_long_long_int_type();
207  else
208  result=unsigned_long_long_int_type();
209  }
210  }
211  else if(gcc_attr_mode == "__TI__") // 128 bits
212  {
213  if(is_signed)
214  result=gcc_signed_int128_type();
215  else
216  result=gcc_unsigned_int128_type();
217  }
218  else if(gcc_attr_mode == "__V2SI__") // vector of 2 ints, deprecated
219  {
220  if(is_signed)
221  {
222  result = vector_typet(
223  c_index_type(), signed_int_type(), from_integer(2, size_type()));
224  }
225  else
226  {
227  result = vector_typet(
228  c_index_type(), unsigned_int_type(), from_integer(2, size_type()));
229  }
230  }
231  else if(gcc_attr_mode == "__V4SI__") // vector of 4 ints, deprecated
232  {
233  if(is_signed)
234  {
235  result = vector_typet(
236  c_index_type(), signed_int_type(), from_integer(4, size_type()));
237  }
238  else
239  {
240  result = vector_typet(
241  c_index_type(), unsigned_int_type(), from_integer(4, size_type()));
242  }
243  }
244  else // give up, just use subtype
245  result = to_type_with_subtype(type).subtype();
246 
247  // save the location
248  result.add_source_location()=type.source_location();
249 
250  if(to_type_with_subtype(type).subtype().id() == ID_c_enum_tag)
251  {
252  const irep_idt &tag_name =
253  to_c_enum_tag_type(to_type_with_subtype(type).subtype())
254  .get_identifier();

256  .subtype() = result;
257  }
258 
259  type=result;
260  }
261  else if(underlying_type.id()==ID_floatbv)
262  {
263  typet result;
264 
265  if(gcc_attr_mode == "__SF__") // 32 bits
266  result=float_type();
267  else if(gcc_attr_mode == "__DF__") // 64 bits
268  result=double_type();
269  else if(gcc_attr_mode == "__TF__") // 128 bits
270  result=gcc_float128_type();

272  result = vector_typet(
273  c_index_type(), float_type(), from_integer(2, size_type()));
274  else if(gcc_attr_mode == "__V2DF__") // deprecated vector of 2 doubles
275  result = vector_typet(
276  c_index_type(), double_type(), from_integer(2, size_type()));
277  else if(gcc_attr_mode == "__V4SF__") // deprecated vector of 4 floats
278  result = vector_typet(
279  c_index_type(), float_type(), from_integer(4, size_type()));
280  else if(gcc_attr_mode == "__V4DF__") // deprecated vector of 4 doubles
281  result = vector_typet(
282  c_index_type(), double_type(), from_integer(4, size_type()));
283  else // give up, just use subtype
284  result = to_type_with_subtype(type).subtype();
285 
286  // save the location
287  result.add_source_location()=type.source_location();
288 
289  type=result;
290  }
291  else if(underlying_type.id()==ID_complex)
292  {
293  // gcc allows this, but clang doesn't -- see enums above
294  typet result;
295 
296  if(gcc_attr_mode == "__SC__") // 32 bits
297  result=float_type();
298  else if(gcc_attr_mode == "__DC__") // 64 bits
299  result=double_type();
300  else if(gcc_attr_mode == "__TC__") // 128 bits
301  result=gcc_float128_type();
302  else // give up, just use subtype
303  result = to_type_with_subtype(type).subtype();
304 
305  // save the location
306  result.add_source_location()=type.source_location();
307

309  }
310  else
311  {
312  throw errort().with_location(type.source_location())
313  << "attribute mode '" << gcc_attr_mode
314  << "' applied to inappropriate type '" << to_string(type) << "'";
315  }
316  }
317 
318  // do a mild bit of rule checking
319 
320  if(type.get_bool(ID_C_restricted) &&
321  type.id()!=ID_pointer &&
322  type.id()!=ID_array)
323  {
324  throw errort().with_location(type.source_location())
325  << "only a pointer can be 'restrict'";
326  }

329 void c_typecheck_baset::typecheck_custom_type(typet &type)
330 {
331  // they all have a width
332  exprt size_expr=
333  static_cast<const exprt &>(type.find(ID_size));
334 
335  typecheck_expr(size_expr);
336  source_locationt source_location=size_expr.source_location();
337  make_constant_index(size_expr);
338 
339  mp_integer size_int;
340  if(to_integer(to_constant_expr(size_expr), size_int))
341  {
342  throw errort().with_location(source_location)
343  << "failed to convert bit vector width to constant";
344  }
345 
346  if(size_int<1)
347  {
348  throw errort().with_location(source_location) << "bit vector width invalid";
349  }
350 
351  type.remove(ID_size);
352  type.set(ID_width, integer2string(size_int));
353 
354  // depending on type, there may be a number of fractional bits
355 
356  if(type.id()==ID_custom_unsignedbv)
357  type.id(ID_unsignedbv);
358  else if(type.id()==ID_custom_signedbv)
359  type.id(ID_signedbv);
360  else if(type.id()==ID_custom_fixedbv)
361  {
362  type.id(ID_fixedbv);
363 
364  exprt f_expr=
365  static_cast<const exprt &>(type.find(ID_f));
366 
367  const source_locationt fraction_source_location =
368  f_expr.find_source_location();
369 
370  typecheck_expr(f_expr);
371 
372  make_constant_index(f_expr);
373 
374  mp_integer f_int;
375  if(to_integer(to_constant_expr(f_expr), f_int))
376  {
377  throw errort().with_location(fraction_source_location)
378  << "failed to convert number of fraction bits to constant";
379  }
380 
381  if(f_int<0 || f_int>size_int)
382  {
383  throw errort().with_location(fraction_source_location)
384  << "fixedbv fraction width invalid";
385  }
386 
387  type.remove(ID_f);
388  type.set(ID_integer_bits, integer2string(size_int-f_int));
389  }
390  else if(type.id()==ID_custom_floatbv)
391  {
392  type.id(ID_floatbv);
393 
394  exprt f_expr=
395  static_cast<const exprt &>(type.find(ID_f));
396 
397  const source_locationt fraction_source_location =
398  f_expr.find_source_location();
399 
400  typecheck_expr(f_expr);
401 
402  make_constant_index(f_expr);
403 
404  mp_integer f_int;
405  if(to_integer(to_constant_expr(f_expr), f_int))
406  {
407  throw errort().with_location(fraction_source_location)
408  << "failed to convert number of fraction bits to constant";
409  }
410 
411  if(f_int<1 || f_int+1>=size_int)
412  {
413  throw errort().with_location(fraction_source_location)
414  << "floatbv fraction width invalid";
415  }
416 
417  type.remove(ID_f);
418  type.set(ID_f, integer2string(f_int));
419  }
420  else
421  UNREACHABLE;
422 }

424 void c_typecheck_baset::typecheck_code_type(code_typet &type)
425 {
426  // the return type is still 'subtype()'
427  type.return_type() = to_type_with_subtype(type).subtype();
428  type.remove_subtype();
429 
430  code_typet::parameterst &parameters=type.parameters();
431 
432  // if we don't have any parameters, we assume it's (...)
433  if(parameters.empty())
434  {
435  type.make_ellipsis();
436  }
437  else // we do have parameters
438  {
439  // is the last one ellipsis?
440  if(type.parameters().back().id()==ID_ellipsis)
441  {
442  type.make_ellipsis();
443  type.parameters().pop_back();
444  }
445 
446  parameter_map.clear();
447 
448  for(auto &param : type.parameters())
449  {
450  // turn the declarations into parameters
451  if(param.id()==ID_declaration)
452  {
453  ansi_c_declarationt &declaration=
454  to_ansi_c_declaration(param);
455 
456 
457  // first fix type
458  code_typet::parametert parameter(
459  declaration.full_type(declaration.declarator()));
460  typet &param_type = parameter.type();
461  std::list<codet> tmp_clean_code;
462  tmp_clean_code.swap(clean_code); // ignore side-effects
463  typecheck_type(param_type);
464  tmp_clean_code.swap(clean_code);
465  adjust_function_parameter(param_type);
466 
467  // adjust the identifier
468  irep_idt identifier=declaration.declarator().get_name();
469 
470  // abstract or not?
471  if(identifier.empty())
472  {
473  // abstract
474  parameter.add_source_location()=declaration.type().source_location();
475  }
476  else
477  {
478  // make visible now, later parameters might use it
479  parameter_map[identifier] = param_type;
480  parameter.set_base_name(declaration.declarator().get_base_name());
481  parameter.add_source_location()=
482  declaration.declarator().source_location();
483  }
484 
485  // put the parameter in place of the declaration
486  param.swap(parameter);
487  }
488  }
489 
490  parameter_map.clear();
491 
492  if(parameters.size() == 1 && parameters[0].type().id() == ID_empty)
493  {
494  // if we just have one parameter of type void, remove it
495  parameters.clear();
496  }
497  }
498 
499  typecheck_type(type.return_type());
500 
501  // 6.7.6.3:
502  // "A function declarator shall not specify a return type that
503  // is a function type or an array type."
504 
505  const typet &decl_return_type = type.return_type();
506 
507  if(decl_return_type.id() == ID_array)
508  {
509  throw errort().with_location(type.source_location())
510  << "function must not return array";
511  }
512 
513  if(decl_return_type.id() == ID_code)
514  {
515  throw errort().with_location(type.source_location())
516  << "function must not return function type";
517  }
518 }

520 void c_typecheck_baset::typecheck_array_type(array_typet &type)
521 {
522  // Typecheck the element type.
523  typecheck_type(type.element_type());
524 
525  // We don't allow void as the element type.
526  if(type.element_type().id() == ID_empty)
527  {
528  throw errort().with_location(type.source_location()) << "array of voids";
529  }
530 
531  // We don't allow incomplete structs or unions as element type.
532  const typet &followed_subtype = follow(type.element_type());
533 
534  if(
535  (followed_subtype.id() == ID_struct || followed_subtype.id() == ID_union) &&
536  to_struct_union_type(followed_subtype).is_incomplete())
537  {
538  // ISO/IEC 9899 6.7.5.2
539  throw errort().with_location(type.source_location())
540  << "array has incomplete element type";
541  }
542 
543  // We don't allow functions as element type.
544  if(type.element_type().id() == ID_code)
545  {
546  // ISO/IEC 9899 6.7.5.2
547  throw errort().with_location(type.source_location())
548  << "array of function element type";
549  }
550 
551  // We add the index type.
552  type.index_type_nonconst() = c_index_type();
553 
554  // Check the array size, if given.
555  if(type.is_complete())
556  {
557  exprt &size = type.size();
558  const source_locationt size_source_location = size.find_source_location();
559  typecheck_expr(size);
560  make_index_type(size);
561 
562  // The size need not be a constant!
563  // We simplify it, for the benefit of array initialisation.
564 
565  exprt tmp_size=size;
566  add_rounding_mode(tmp_size);
567  simplify(tmp_size, *this);
568 
569  if(tmp_size.is_constant())
570  {
571  mp_integer s;
572  if(to_integer(to_constant_expr(tmp_size), s))
573  {
574  throw errort().with_location(size_source_location)
575  << "failed to convert constant: " << tmp_size.pretty();
576  }
577 
578  if(s<0)
579  {
580  throw errort().with_location(size_source_location)
581  << "array size must not be negative, "
582  "but got "
583  << s;
584  }
585 
586  size=tmp_size;
587  }
588  else if(tmp_size.id()==ID_infinity)
589  {
590  size=tmp_size;
591  }
592  else if(tmp_size.id()==ID_symbol &&
593  tmp_size.type().get_bool(ID_C_constant))
594  {
595  // We allow a constant variable as array size, assuming
596  // it won't change.
597  // This criterion can be tricked:
598  // Of course we can modify a 'const' symbol, e.g.,
599  // using a pointer type cast. Interestingly,
600  // at least gcc 4.2.1 makes the very same mistake!
601  size=tmp_size;
602  }
603  else
604  {
605  // not a constant and not infinity
606 
607  PRECONDITION(!current_symbol.name.empty());
608 
609  if(current_symbol.is_static_lifetime)
610  {
611  throw errort().with_location(current_symbol.location)
612  << "array size of static symbol '" << current_symbol.base_name
613  << "' is not constant";
614  }
615 
616  symbolt &new_symbol = get_fresh_aux_symbol(
617  size_type(),
618  id2string(current_symbol.name) + "$array_size",
619  id2string(current_symbol.base_name) + "$array_size",
620  size_source_location,
621  mode,
622  symbol_table);
623  new_symbol.type.set(ID_C_constant, true);
624  new_symbol.value = typecast_exprt::conditional_cast(size, size_type());
625 
626  // produce the code that declares and initializes the symbol
627  symbol_exprt symbol_expr = new_symbol.symbol_expr();
628 
629  code_frontend_declt declaration(symbol_expr);
630  declaration.add_source_location() = size_source_location;
631 
632  code_frontend_assignt assignment;
633  assignment.lhs()=symbol_expr;
634  assignment.rhs() = new_symbol.value;
635  assignment.add_source_location() = size_source_location;
636 
637  // store the code
638  clean_code.push_back(declaration);
639  clean_code.push_back(assignment);
640 
641  // fix type
642  size=symbol_expr;
643  }
644  }

647 void c_typecheck_baset::typecheck_vector_type(typet &type)
648 {
649  // This turns the type with ID_frontend_vector into the type
650  // with ID_vector; the difference is that the latter has
651  // a constant as size, which we establish now.
652  exprt size = static_cast<const exprt &>(type.find(ID_size));
653  const source_locationt source_location = size.find_source_location();
654 
655  typecheck_expr(size);
656 
657  typet subtype = to_type_with_subtype(type).subtype();
658  typecheck_type(subtype);
659 
660  // we are willing to combine 'vector' with various
661  // other types, but not everything!
662 
663  if(subtype.id()!=ID_signedbv &&
664  subtype.id()!=ID_unsignedbv &&
665  subtype.id()!=ID_floatbv &&
666  subtype.id()!=ID_fixedbv)
667  {
668  throw errort().with_location(source_location)
669  << "cannot make a vector of subtype " << to_string(subtype);
670  }
671 
672  make_constant_index(size);
673 
674  mp_integer s;
675  if(to_integer(to_constant_expr(size), s))
676  {
677  throw errort().with_location(source_location)
678  << "failed to convert constant: " << size.pretty();
679  }
680 
681  if(s<=0)
682  {
683  throw errort().with_location(source_location)
684  << "vector size must be positive, "
685  "but got "
686  << s;
687  }
688 
689  // the subtype must have constant size
690  auto sub_size_expr_opt = size_of_expr(subtype, *this);
691 
692  if(!sub_size_expr_opt.has_value())
693  {
694  throw errort().with_location(source_location)
695  << "failed to determine size of vector base type '" << to_string(subtype)
696  << "'";
697  }
698 
699  simplify(sub_size_expr_opt.value(), *this);
700 
701  const auto sub_size = numeric_cast<mp_integer>(sub_size_expr_opt.value());
702 
703  if(!sub_size.has_value())
704  {
705  throw errort().with_location(source_location)
706  << "failed to determine size of vector base type '" << to_string(subtype)
707  << "'";
708  }
709 
710  if(*sub_size == 0)
711  {
712  throw errort().with_location(source_location)
713  << "type had size 0: '" << to_string(subtype) << "'";
714  }
715 
716  // adjust by width of base type
717  if(s % *sub_size != 0)
718  {
719  throw errort().with_location(source_location)
720  << "vector size (" << s << ") expected to be multiple of base type size ("
721  << *sub_size << ")";
722  }
723 
724  s /= *sub_size;
725 
726  // produce the type with ID_vector
727  vector_typet new_type(
728  c_index_type(), subtype, from_integer(s, signed_size_type()));
729  new_type.add_source_location() = source_location;
730  new_type.size().add_source_location() = source_location;
731  type = new_type;
732 }

734 void c_typecheck_baset::typecheck_compound_type(struct_union_typet &type)
735 {
736  // These get replaced by symbol types later.
737  irep_idt identifier;
738 
739  bool have_body=type.find(ID_components).is_not_nil();
740 
741  c_qualifierst original_qualifiers(type);
742 
743  // the type symbol, which may get re-used in other declarations, must not
744  // carry any qualifiers (other than transparent_union, which isn't really a
745  // qualifier)
746  c_qualifierst remove_qualifiers;
747  remove_qualifiers.is_transparent_union =
748  original_qualifiers.is_transparent_union;
749  remove_qualifiers.write(type);
750 
751  bool is_packed = type.get_bool(ID_C_packed);
752  irept alignment = type.find(ID_C_alignment);
753 
754  if(type.find(ID_tag).is_nil())
755  {
756  // Anonymous? Must come with body.
757  PRECONDITION(have_body);
758 
759  // produce symbol
760  type_symbolt compound_symbol{irep_idt{}, type, mode};
761  compound_symbol.location=type.source_location();
762 
763  typecheck_compound_body(to_struct_union_type(compound_symbol.type));
764 
765  std::string typestr = type2name(compound_symbol.type, *this);
766  compound_symbol.base_name = "#anon#" + typestr;
767  compound_symbol.name="tag-#anon#"+typestr;
768  identifier=compound_symbol.name;
769 
770  // We might already have the same anonymous union/struct,
771  // and this is simply ok. Note that the C standard treats
772  // these as different types.
773  if(symbol_table.symbols.find(identifier)==symbol_table.symbols.end())
774  {
775  symbolt *new_symbol;
776  move_symbol(compound_symbol, new_symbol);
777  }
778  }
779  else
780  {
781  identifier=type.find(ID_tag).get(ID_identifier);
782 
783  // does it exist already?
784  symbol_table_baset::symbolst::const_iterator s_it =
785  symbol_table.symbols.find(identifier);
786 
787  if(s_it==symbol_table.symbols.end())
788  {
789  // no, add new symbol
790  irep_idt base_name=type.find(ID_tag).get(ID_C_base_name);
791  type.remove(ID_tag);
792  type.set(ID_tag, base_name);
793 
794  type_symbolt compound_symbol{identifier, type, mode};
795  compound_symbol.base_name=base_name;
796  compound_symbol.location=type.source_location();
797  compound_symbol.pretty_name=id2string(type.id())+" "+id2string(base_name);
798 
799  typet new_type=compound_symbol.type;
800 
801  // mark as incomplete
802  to_struct_union_type(compound_symbol.type).make_incomplete();
803 
804  symbolt *new_symbol;
805  move_symbol(compound_symbol, new_symbol);
806 
807  if(have_body)
808  {
809  typecheck_compound_body(to_struct_union_type(new_type));
810 
811  new_symbol->type.swap(new_type);
812  }
813  }
814  else
815  {
816  // yes, it exists already
817  if(
818  s_it->second.type.id() == type.id() &&
819  to_struct_union_type(s_it->second.type).is_incomplete())
820  {
821  // Maybe we got a body now.
822  if(have_body)
823  {
824  irep_idt base_name=type.find(ID_tag).get(ID_C_base_name);
825  type.remove(ID_tag);
826  type.set(ID_tag, base_name);
827 
828  typecheck_compound_body(type);
829  symbol_table.get_writeable_ref(s_it->first).type.swap(type);
830  }
831  }
832  else if(s_it->second.type.id() != type.id())
833  {
834  throw errort().with_location(type.source_location())
835  << "redefinition of '" << s_it->second.pretty_name << "'"
836  << " as different kind of tag";
837  }
838  else if(have_body)
839  {
840  throw errort().with_location(type.source_location())
841  << "redefinition of body of '" << s_it->second.pretty_name << "'";
842  }
843  }
844  }
845 
846  typet tag_type;
847 
848  if(type.id() == ID_union)
849  tag_type = union_tag_typet(identifier);
850  else if(type.id() == ID_struct)
851  tag_type = struct_tag_typet(identifier);
852  else
853  UNREACHABLE;
854 
855  tag_type.add_source_location() = type.source_location();
856  type.swap(tag_type);
857 
858  original_qualifiers.write(type);
859 
860  if(is_packed)
861  type.set(ID_C_packed, true);
862  if(alignment.is_not_nil())
863  type.set(ID_C_alignment, alignment);
864 }

866 void c_typecheck_baset::typecheck_compound_body(
867  struct_union_typet &type)
868 {
869  struct_union_typet::componentst &components=type.components();
870 
871  struct_union_typet::componentst old_components;
872  old_components.swap(components);
873 
874  // We get these as declarations!
875  for(auto &decl : old_components)
876  {
877  // the arguments are member declarations or static assertions
878  PRECONDITION(decl.id() == ID_declaration);
879 
880  ansi_c_declarationt &declaration=
881  to_ansi_c_declaration(static_cast<exprt &>(decl));
882 
883  if(declaration.get_is_static_assert())
884  {
885  struct_union_typet::componentt new_component;
886  new_component.id(ID_static_assert);
887  new_component.add_source_location()=declaration.source_location();
888  PRECONDITION(declaration.operands().size() == 2);
889  new_component.operands().swap(declaration.operands());
890  components.push_back(new_component);
891  }
892  else
893  {
894  // do first half of type
895  typecheck_type(declaration.type());
896  already_typechecked_typet::make_already_typechecked(declaration.type());
897 
898  for(const auto &declarator : declaration.declarators())
899  {
900  struct_union_typet::componentt new_component(
901  declarator.get_base_name(), declaration.full_type(declarator));
902 
903  // There may be a declarator, which we use as location for
904  // the component. Otherwise, use location of the declaration.
905  const source_locationt source_location =
906  declarator.get_name().empty() ? declaration.source_location()
907  : declarator.source_location();
908 
909  new_component.add_source_location() = source_location;
910  new_component.set_pretty_name(declarator.get_base_name());
911 
912  typecheck_type(new_component.type());
913 
914  // the rules for anonymous members depend on the type and the compiler,
915  // just bit fields are accepted everywhere
916  if(
917  new_component.get_name().empty() &&
918  new_component.type().id() != ID_c_bit_field)
919  {
920  if(config.ansi_c.mode == configt::ansi_ct::flavourt::VISUAL_STUDIO)
921  {
922  // Visual Studio rejects anything other than a struct or union
923  // declaration, but also accepts those when they introduce a tag
924  if(
925  new_component.type().id() != ID_struct_tag &&
926  new_component.type().id() != ID_union_tag)
927  {
928  throw errort().with_location(source_location)
929  << "no members defined";
930  }
931  }
932  else
933  {
934  // GCC and Clang ignore anything other than an untagged struct or
935  // union; we could print a warning, but there isn't any ambiguity in
936  // semantics here. Printing a warning could elevate this to an error
937  // when compiling code with goto-cc with -Werror.
938  // Note that our type checking always creates a struct_tag/union_tag
939  // type, but only named struct/union types have an ID_tag member.
940  if(
941  new_component.type().id() == ID_struct_tag &&
942  follow_tag(to_struct_tag_type(new_component.type()))
943  .find(ID_tag)
944  .is_nil())
945  {
946  // ok, anonymous struct
947  }
948  else if(
949  new_component.type().id() == ID_union_tag &&
950  follow_tag(to_union_tag_type(new_component.type()))
951  .find(ID_tag)
952  .is_nil())
953  {
954  // ok, anonymous union
955  }
956  else
957  {
958  continue;
959  }
960  }
961  }
962 
963  if(!is_complete_type(new_component.type()) &&
964  (new_component.type().id()!=ID_array ||
965  !to_array_type(new_component.type()).is_incomplete()))
966  {
967  throw errort().with_location(source_location)
968  << "incomplete type not permitted here";
969  }
970 
971  if(new_component.type().id() == ID_empty)
972  {
973  throw errort().with_location(source_location)
974  << "void-typed member not permitted";
975  }
976 
977  if(
978  new_component.type().id() == ID_c_bit_field &&
979  to_c_bit_field_type(new_component.type()).get_width() == 0 &&
980  !new_component.get_name().empty())
981  {
982  throw errort().with_location(source_location)
983  << "zero-width bit-field with declarator not permitted";
984  }
985 
986  components.push_back(new_component);
987  }
988  }
989  }
990 
991  unsigned anon_member_counter=0;
992 
993  // scan for anonymous members, and name them
994  for(auto &member : components)
995  {
996  if(!member.get_name().empty())
997  continue;
998 
999  member.set_name("$anon"+std::to_string(anon_member_counter++));
1000  member.set_anonymous(true);
1001  }
1002 
1003  // scan for duplicate members
1004 
1005  {
1006  std::unordered_set<irep_idt> members;
1007 
1008  for(const auto &c : components)
1009  {
1010  if(!members.insert(c.get_name()).second)
1011  {
1012  throw errort().with_location(c.source_location())
1013  << "duplicate member '" << c.get_name() << '\'';
1014  }
1015  }
1016  }
1017 
1018  // We allow an incomplete (C99) array as _last_ member!
1019  // Zero-length is allowed everywhere.
1020 
1021  if(type.id()==ID_struct ||
1022  type.id()==ID_union)
1023  {
1024  for(struct_union_typet::componentst::iterator
1025  it=components.begin();
1026  it!=components.end();
1027  it++)
1028  {
1029  typet &c_type=it->type();
1030 
1031  if(c_type.id()==ID_array &&
1032  to_array_type(c_type).is_incomplete())
1033  {
1034  // needs to be last member
1035  if(type.id()==ID_struct && it!=--components.end())
1036  {
1037  throw errort().with_location(it->source_location())
1038  << "flexible struct member must be last member";
1039  }
1040 
1041  // make it zero-length
1042  to_array_type(c_type).size() = from_integer(0, c_index_type());
1043  c_type.set(ID_C_flexible_array_member, true);
1044  }
1045  }
1046  }
1047 
1048  // We may add some minimal padding inside and at
1049  // the end of structs and
1050  // as additional member for unions.
1051 
1052  if(type.id()==ID_struct)
1053  add_padding(to_struct_type(type), *this);
1054  else if(type.id()==ID_union)
1055  add_padding(to_union_type(type), *this);
1056 
1057  // finally, check _Static_assert inside the compound
1058  for(struct_union_typet::componentst::iterator
1059  it=components.begin();
1060  it!=components.end();
1061  ) // no it++
1062  {
1063  if(it->id()==ID_static_assert)
1064  {
1065  if(config.ansi_c.mode == configt::ansi_ct::flavourt::VISUAL_STUDIO)
1066  {
1067  throw errort().with_location(it->source_location())
1068  << "static_assert not supported in compound body";
1069  }
1070 
1071  exprt &assertion = to_binary_expr(*it).op0();
1072  typecheck_expr(assertion);
1073  typecheck_expr(to_binary_expr(*it).op1());
1074  assertion = typecast_exprt(assertion, bool_typet());
1075  make_constant(assertion);
1076 
1077  if(assertion.is_false())
1078  {
1079  throw errort().with_location(it->source_location())
1080  << "failed _Static_assert";
1081  }
1082  else if(!assertion.is_true())
1083  {
1084  // should warn/complain
1085  }
1086 
1087  it=components.erase(it);
1088  }
1089  else
1090  it++;
1091  }
1092 
1093  // Visual Studio strictly follows the C standard and does not permit empty
1094  // struct/union declarations
1095  if(
1096  components.empty() &&
1097  config.ansi_c.mode == configt::ansi_ct::flavourt::VISUAL_STUDIO)
1098  {
1099  throw errort().with_location(type.source_location())
1100  << "C requires that a struct or union has at least one member";
1101  }
1102 }

1194 void c_typecheck_baset::typecheck_c_enum_type(typet &type)
1195 {
1196  // These come with the declarations
1197  // of the enum constants as operands.
1198 
1199  exprt &as_expr=static_cast<exprt &>(static_cast<irept &>(type));
1200  source_locationt source_location=type.source_location();
1201 
1202  // We allow empty enums in the grammar to get better
1203  // error messages.
1204  if(as_expr.operands().empty())
1205  {
1206  throw errort().with_location(source_location) << "empty enum";
1207  }
1208 
1209  const bool have_underlying_type =
1210  type.find_type(ID_enum_underlying_type).is_not_nil();
1211 
1212  if(have_underlying_type)
1213  {
1214  typecheck_type(type.add_type(ID_enum_underlying_type));
1215 
1216  const typet &underlying_type =
1217  static_cast<const typet &>(type.find(ID_enum_underlying_type));
1218 
1219  if(!is_signed_or_unsigned_bitvector(underlying_type))
1220  {
1221  throw errort().with_location(source_location)
1222  << "non-integral type '" << underlying_type.get(ID_C_c_type)
1223  << "' is an invalid underlying type";
1224  }
1225  }
1226 
1227  // enums start at zero;
1228  // we also track min and max to find a nice base type
1229  mp_integer value=0, min_value=0, max_value=0;
1230 
1231  std::vector<c_enum_typet::c_enum_membert> enum_members;
1232  enum_members.reserve(as_expr.operands().size());
1233 
1234  // We need to determine a width, and a signedness
1235  // to obtain an 'underlying type'.
1236  // We just do int, but gcc might pick smaller widths
1237  // if the type is marked as 'packed'.
1238  // gcc/clang may also pick a larger width. Visual Studio doesn't.
1239 
1240  for(auto &op : as_expr.operands())
1241  {
1242  ansi_c_declarationt &declaration=to_ansi_c_declaration(op);
1243  exprt &v=declaration.declarator().value();
1244 
1245  if(v.is_not_nil()) // value given?
1246  {
1247  exprt tmp_v=v;
1248  typecheck_expr(tmp_v);
1249  add_rounding_mode(tmp_v);
1250  simplify(tmp_v, *this);
1251  if(tmp_v.is_true())
1252  value=1;
1253  else if(tmp_v.is_false())
1254  value=0;
1255  else if(
1256  tmp_v.is_constant() && !to_integer(to_constant_expr(tmp_v), value))
1257  {
1258  }
1259  else
1260  {
1261  throw errort().with_location(v.source_location())
1262  << "enum is not a constant";
1263  }
1264  }
1265 
1266  if(value<min_value)
1267  min_value=value;
1268  if(value>max_value)
1269  max_value=value;
1270 
1271  typet constant_type;
1272 
1273  if(have_underlying_type)
1274  {
1275  constant_type = type.find_type(ID_enum_underlying_type);
1276  const auto &tmp = to_integer_bitvector_type(constant_type);
1277 
1278  if(value < tmp.smallest() || value > tmp.largest())
1279  {
1280  throw errort().with_location(v.source_location())
1281  << "enumerator value is not representable in the underlying type '"
1282  << constant_type.get(ID_C_c_type) << "'";
1283  }
1284  }
1285  else
1286  {
1287  constant_type = enum_constant_type(min_value, max_value);
1288  }
1289 
1290  v=from_integer(value, constant_type);
1291 
1292  declaration.type()=constant_type;
1293  typecheck_declaration(declaration);
1294 
1295  irep_idt base_name=
1296  declaration.declarator().get_base_name();
1297 
1298  irep_idt identifier=
1299  declaration.declarator().get_name();
1300 
1301  // store
1302  c_enum_typet::c_enum_membert member;
1303  member.set_identifier(identifier);
1304  member.set_base_name(base_name);
1305  // Note: The value will be correctly set to a bv type when we know
1306  // the width of the bitvector
1307  member.set_value(integer2string(value));
1308  enum_members.push_back(member);
1309 
1310  // produce value for next constant
1311  ++value;
1312  }
1313 
1314  // Remove these now; we add them to the
1315  // c_enum symbol later.
1316  as_expr.operands().clear();
1317 
1318  bool is_packed=type.get_bool(ID_C_packed);
1319 
1320  // We use a subtype to store the underlying type.
1321  bitvector_typet underlying_type(ID_nil);
1322 
1323  if(have_underlying_type)
1324  {
1325  underlying_type =
1326  to_bitvector_type(type.find_type(ID_enum_underlying_type));
1327  }
1328  else
1329  {
1330  underlying_type = enum_underlying_type(min_value, max_value, is_packed);
1331  }
1332 
1333  // Get the width to make the values have a bitvector type
1334  std::size_t width = underlying_type.get_width();
1335  for(auto &member : enum_members)
1336  {
1337  // Note: This is inefficient as it first turns integers to strings
1338  // and then turns them back to bvrep
1339  auto value = string2integer(id2string(member.get_value()));
1340  member.set_value(integer2bvrep(value, width));
1341  }
1342 
1343  // tag?
1344  if(type.find(ID_tag).is_nil())
1345  {
1346  // None, it's anonymous. We generate a tag.
1347  std::string anon_identifier="#anon_enum";
1348
