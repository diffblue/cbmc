95 void dfcc_contract_functionst::gen_spec_assigns_function()
96 {
97  const auto &spec_function_symbol = utils.clone_and_rename_function(
98  pure_contract_symbol.name, spec_assigns_function_id, empty_typet());
99 
100  const auto &spec_function_id = spec_function_symbol.name;
101 
102  auto &spec_code_type = to_code_type(spec_function_symbol.type);
103 
104  exprt::operandst lambda_parameters;
105 
106  if(code_with_contract.return_type().id() != ID_empty)
107  {
108  // use a dummy symbol for __CPROVER_return_value
109  // which does occur in the assigns clause anyway
110  lambda_parameters.push_back(
111  symbol_exprt("dummy_return_value", code_with_contract.return_type()));
112  }
113 
114  for(const auto &param_id : spec_code_type.parameter_identifiers())
115  {
116  lambda_parameters.push_back(ns.lookup(param_id).symbol_expr());
117  }
118 
119  // fetch the goto_function to add instructions to
120  goto_functiont &goto_function =
121  goto_model.goto_functions.function_map.at(spec_function_id);
122  goto_programt &body = goto_function.body;
123 
124  for(const auto &assigns_expr : code_with_contract.assigns())
125  {
126  auto expr = to_lambda_expr(assigns_expr).application(lambda_parameters);
127  expr.add_source_location() = assigns_expr.source_location();
128  if(can_cast_expr<conditional_target_group_exprt>(expr))
129  {
130  encode_assignable_target_group(
131  to_conditional_target_group_expr(expr), body);
132  }
133  else
134  {
135  encode_assignable_target(expr, body);
136  }
137  }
138 
139  body.add(goto_programt::make_end_function(spec_function_symbol.location));
140 
141  goto_model.goto_functions.update();
142 
143  inline_and_check_warnings(spec_function_id);
144 
145  PRECONDITION_WITH_DIAGNOSTICS(
146  utils.has_no_loops(spec_function_id),
147  "loops in assigns clause specification functions must be unwound before "
148  "model instrumentation");
149 
150  goto_model.goto_functions.update();
151 }

153 void dfcc_contract_functionst::encode_assignable_target_group(
154  const conditional_target_group_exprt &group,
155  goto_programt &dest)
156 {
157  const source_locationt &source_location = group.source_location();
158 
159  // clean up side effects from the condition expression if needed
160  cleanert cleaner(goto_model.symbol_table, log.get_message_handler());
161  exprt condition(group.condition());
162  if(has_subexpr(condition, ID_side_effect))
163  cleaner.clean(condition, dest, language_mode);
164 
165  // Jump target if condition is false
166  auto goto_instruction = dest.add(
167  goto_programt::make_incomplete_goto(not_exprt{condition}, source_location));
168 
169  for(const auto &target : group.targets())
170  encode_assignable_target(target, dest);
171 
172  auto label_instruction = dest.add(goto_programt::make_skip(source_location));
173  goto_instruction->complete_goto(label_instruction);
174 }

176 void dfcc_contract_functionst::encode_assignable_target(
177  const exprt &target,
178  goto_programt &dest)
179 {
180  const source_locationt &source_location = target.source_location();
181 
182  if(can_cast_expr<side_effect_expr_function_callt>(target))
183  {
184  // A function call target `foo(params)` becomes `CALL foo(params);`
185  // ```
186  const auto &funcall = to_side_effect_expr_function_call(target);
187  code_function_callt code_function_call(to_symbol_expr(funcall.function()));
188  auto &arguments = code_function_call.arguments();
189  for(auto &arg : funcall.arguments())
190  arguments.emplace_back(arg);
191  dest.add(
192  goto_programt::make_function_call(code_function_call, source_location));
193  }
194  else if(is_assignable(target))
195  {
196  // An lvalue `target` becomes
197  //` CALL __CPROVER_assignable(&target, sizeof(target), is_ptr_to_ptr);`
198  const auto &size =
199  size_of_expr(target.type(), namespacet(goto_model.symbol_table));
200 
201  if(!size.has_value())
202  {
203  throw invalid_source_file_exceptiont{
204  "no definite size for lvalue assigns clause target " +
205  from_expr_using_mode(ns, language_mode, target),
206  target.source_location()};
207  }
208  // we have to build the symbol manually because it might not
209  // be present in the symbol table if the user program does not already
210  // use it.
211  code_function_callt code_function_call(
212  symbol_exprt(CPROVER_PREFIX "assignable", empty_typet()));
213  auto &arguments = code_function_call.arguments();
214 
215  // ptr
216  arguments.emplace_back(typecast_exprt::conditional_cast(
217  address_of_exprt{target}, pointer_type(empty_typet())));
218 
219  // size
220  arguments.emplace_back(size.value());
221 
222  // is_ptr_to_ptr
223  arguments.emplace_back(make_boolean_expr(target.type().id() == ID_pointer));
224 
225  dest.add(
226  goto_programt::make_function_call(code_function_call, source_location));
227  }
228  else
229  {
230  // any other type of target is unsupported
231  throw invalid_source_file_exceptiont(
232  "unsupported assigns clause target " +
233  from_expr_using_mode(ns, language_mode, target),
234  target.source_location());
235  }
236 }

303  const irep_idt &function_id)
304 {
305  std::set<irep_idt> no_body;

307  std::set<irep_idt> recursive_call;
308  std::set<irep_idt> not_enough_arguments;
309 
310  utils.inline_function(
311  function_id,
312  no_body,
313  recursive_call,
314  missing_function,
315  not_enough_arguments);
316 
317  // check that the only no body / missing functions are the cprover builtins
318  for(const auto &id : no_body)
319  {
320  INVARIANT(

322  "no body for '" + id2string(id) + "' when inlining '" +
323  id2string(function_id) + "'");
324  }
325 
326  for(auto it : missing_function)
327  {
328  INVARIANT(
329  library.is_front_end_builtin(it),
330  "missing function '" + id2string(it) + "' when inlining '" +
331  id2string(function_id) + "'");
332  }
333 
334  INVARIANT(
335  recursive_call.size() == 0,
336  "recursive calls when inlining '" + id2string(function_id) + "'");
337 
338  INVARIANT(
339  not_enough_arguments.size() == 0,
340  "not enough arguments when inlining '" + id2string(function_id) + "'");
