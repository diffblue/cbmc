183 bool ansi_c_languaget::to_expr(
184  const std::string &code,
185  const std::string &,
186  exprt &expr,
187  const namespacet &ns)
188 {
189  expr.make_nil();
190 
191  // no preprocessing yet...
192 
193  std::istringstream i_preprocessed(
194  "void __my_expression = (void) (\n"+code+"\n);");
195 
196  // parsing
197 
198  ansi_c_parser.clear();
199  ansi_c_parser.set_file(irep_idt());
200  ansi_c_parser.in=&i_preprocessed;
201  ansi_c_parser.log.set_message_handler(get_message_handler());
202  ansi_c_parser.mode=config.ansi_c.mode;
203  ansi_c_parser.ts_18661_3_Floatn_types=config.ansi_c.ts_18661_3_Floatn_types;
204  ansi_c_scanner_init();
205 
206  bool result=ansi_c_parser.parse();
207 
208  if(ansi_c_parser.parse_tree.items.empty())
209  result=true;
210  else
211  {
212  expr=ansi_c_parser.parse_tree.items.front().declarator().value();
213 
214  // typecheck it
215  result=ansi_c_typecheck(expr, get_message_handler(), ns);
216  }
217 
218  // save some memory
219  ansi_c_parser.clear();
220 
221  // now remove that (void) cast
222  if(expr.id()==ID_typecast &&
223  expr.type().id()==ID_empty &&
224  expr.operands().size()==1)
225  {
226  expr = to_typecast_expr(expr).op();
227  }
228 
229  return result;
230 }
