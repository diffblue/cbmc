61 void interpretert::initialize(bool init)
62 {
63  build_memory_map();
64  // reset the call stack
65  call_stack = call_stackt{};
66 
67  total_steps=0;
68  const goto_functionst::function_mapt::const_iterator
69  main_it=goto_functions.function_map.find(goto_functionst::entry_point());
70 
71  if(main_it==goto_functions.function_map.end())
72  throw "main not found";
73 
74  const goto_functionst::goto_functiont &goto_function=main_it->second;
75 
76  if(!goto_function.body_available())
77  throw "main has no body";
78 
79  pc=goto_function.body.instructions.begin();
80  function=main_it;
81 
82  done=false;
83  if(init)
84  {
85  // execute instructions up to and including __CPROVER_initialize()
86  while(!done && call_stack.size() == 0)
87  {
88  show_state();
89  step();
90  }
91  // initialization
92  while(!done && call_stack.size() > 0)
93  {
94  show_state();
95  step();
96  }
97  // invoke the user entry point
98  while(!done && call_stack.size() == 0)
99  {
100  show_state();
101  step();
102  }
103  clear_input_flags();
104  input_vars.clear();
105  }

128 void interpretert::command()
129 {
130  #define BUFSIZE 100
131  char command[BUFSIZE];
132  if(fgets(command, BUFSIZE-1, stdin)==nullptr)
133  {
134  done=true;
135  return;
136  }
137 
138  char ch=tolower(command[0]);
139  if(ch=='q')
140  done=true;
141  else if(ch=='h')
142  {
143  output.status() << "Interpreter help\n"
144  << "h: display this menu\n"
145  << "j: output json trace\n"
146  << "m: output memory dump\n"
147  << "o: output goto trace\n"
148  << "q: quit\n"
149  << "r: run up to entry point\n"
150  << "s#: step a number of instructions\n"
151  << "sa: step across a function\n"
152  << "so: step out of a function\n"
153  << "se: step until end of program\n"
154  << messaget::eom;
155  }
156  else if(ch=='j')
157  {
158  json_arrayt json_steps;
159  convert<json_arrayt>(ns, steps, json_steps);
160  ch=tolower(command[1]);
161  if(ch==' ')
162  {
163  std::ofstream file;
164  file.open(command+2);
165  if(file.is_open())
166  {
167  json_steps.output(file);
168  file.close();
169  return;
170  }
171  }
172  json_steps.output(output.result());
173  }
174  else if(ch=='m')
175  {
176  ch=tolower(command[1]);
177  print_memory(ch=='i');
178  }
179  else if(ch=='o')
180  {
181  ch=tolower(command[1]);
182  if(ch==' ')
183  {
184  std::ofstream file;
185  file.open(command+2);
186  if(file.is_open())
187  {
188  steps.output(ns, file);
189  file.close();
190  return;
191  }
192  }
193  steps.output(ns, output.result());
194  }
195  else if(ch=='r')
196  {
197  ch=tolower(command[1]);
198  initialize(ch!='0');
199  }
200  else if((ch=='s') || (ch==0))
201  {
202  num_steps=1;
203  std::size_t stack_depth = npos;
204  ch=tolower(command[1]);
205  if(ch=='e')
206  num_steps=npos;
207  else if(ch=='o')
208  stack_depth=call_stack.size();
209  else if(ch=='a')
210  stack_depth=call_stack.size()+1;
211  else
212  {
213  num_steps = unsafe_string2size_t(command + 1);
214  if(num_steps==0)
215  num_steps=1;
216  }
217  while(!done && ((num_steps==npos) || ((num_steps--)>0)))
218  {
219  step();
220  show_state();
221  }
222  while(!done && (stack_depth<=call_stack.size()) && (stack_depth!=npos))
223  {
224  step();
225  show_state();
226  }
227  return;
228  }
229  show_state();
230 }

233 void interpretert::step()
234 {
235  total_steps++;
236  if(pc==function->second.body.instructions.end())
237  {
238  if(call_stack.empty())
239  done=true;
240  else
241  {
242  pc=call_stack.top().return_pc;
243  function=call_stack.top().return_function;
244  // TODO: this increases memory size quite quickly.
245  // Should check alternatives
246  call_stack.pop();
247  }
248 
249  return;
250  }
251 
252  next_pc=pc;
253  next_pc++;
254 
255  steps.add_step(goto_trace_stept());
256  goto_trace_stept &trace_step=steps.get_last_step();
257  trace_step.thread_nr=thread_id;
258  trace_step.pc=pc;
259  switch(pc->type())
260  {
261  case GOTO:
262  trace_step.type=goto_trace_stept::typet::GOTO;
263  execute_goto();
264  break;
265 
266  case ASSUME:
267  trace_step.type=goto_trace_stept::typet::ASSUME;
268  execute_assume();
269  break;
270 
271  case ASSERT:
272  trace_step.type=goto_trace_stept::typet::ASSERT;
273  execute_assert();
274  break;
275 
276  case OTHER:
277  execute_other();
278  break;
279 
280  case DECL:
281  trace_step.type=goto_trace_stept::typet::DECL;
282  execute_decl();
283  break;
284 
285  case SKIP:
286  case LOCATION:
287  trace_step.type=goto_trace_stept::typet::LOCATION;
288  break;
289  case END_FUNCTION:
290  trace_step.type=goto_trace_stept::typet::FUNCTION_RETURN;
291  break;
292 
293  case SET_RETURN_VALUE:
294  trace_step.type=goto_trace_stept::typet::FUNCTION_RETURN;
295  if(call_stack.empty())
296  throw "SET_RETURN_VALUE without call"; // NOLINT(readability/throw)
297 
298  if(call_stack.top().return_value_address != 0)
299  {
300  mp_vectort rhs = evaluate(pc->return_value());
301  assign(call_stack.top().return_value_address, rhs);
302  }
303 
304  next_pc=function->second.body.instructions.end();
305  break;
306 
307  case ASSIGN:
308  trace_step.type=goto_trace_stept::typet::ASSIGNMENT;
309  execute_assign();
310  break;
311 
312  case FUNCTION_CALL:
313  trace_step.type=goto_trace_stept::typet::FUNCTION_CALL;
314  execute_function_call();
315  break;
316 
317  case START_THREAD:
318  trace_step.type=goto_trace_stept::typet::SPAWN;
319  throw "START_THREAD not yet implemented"; // NOLINT(readability/throw)
320 
321  case END_THREAD:
322  throw "END_THREAD not yet implemented"; // NOLINT(readability/throw)
323  break;
324 
325  case ATOMIC_BEGIN:
326  trace_step.type=goto_trace_stept::typet::ATOMIC_BEGIN;
327  throw "ATOMIC_BEGIN not yet implemented"; // NOLINT(readability/throw)
328 
329  case ATOMIC_END:
330  trace_step.type=goto_trace_stept::typet::ATOMIC_END;
331  throw "ATOMIC_END not yet implemented"; // NOLINT(readability/throw)
332 
333  case DEAD:
334  trace_step.type=goto_trace_stept::typet::DEAD;
335  break;
336  case THROW:
337  trace_step.type=goto_trace_stept::typet::GOTO;
338  while(!done && (pc->type() != CATCH))
339  {
340  if(pc==function->second.body.instructions.end())
341  {
342  if(call_stack.empty())
343  done=true;
344  else
345  {
346  pc=call_stack.top().return_pc;
347  function=call_stack.top().return_function;
348  call_stack.pop();
349  }
350  }
351  else
352  {
353  next_pc=pc;
354  next_pc++;
355  }
356  }
357  break;
358  case CATCH:
359  break;
360  case INCOMPLETE_GOTO:
361  case NO_INSTRUCTION_TYPE:
362  throw "encountered instruction with undefined instruction type";
363  }
364  pc=next_pc;
365 }

383 void interpretert::execute_other()
384 {
385  const irep_idt &statement = pc->get_other().get_statement();
386  if(statement==ID_expression)
387  {
388  DATA_INVARIANT(
389  pc->code().operands().size() == 1,
390  "expression statement expected to have one operand");
391  mp_vectort rhs = evaluate(pc->code().op0());
392  }
393  else if(statement==ID_array_set)
394  {
395  mp_vectort tmp = evaluate(pc->code().op1());
396  mp_integer address = evaluate_address(pc->code().op0());
397  mp_integer size = get_size(pc->code().op0().type());
398  mp_vectort rhs;
399  while(rhs.size()<size) rhs.insert(rhs.end(), tmp.begin(), tmp.end());
400  if(size!=rhs.size())
401  output.error() << "!! failed to obtain rhs (" << rhs.size() << " vs. "
402  << size << ")\n"
403  << messaget::eom;
404  else
405  {
406  assign(address, rhs);
407  }
408  }
409  else if(can_cast_expr<code_outputt>(pc->get_other()))
410  {
411  return;
412  }
413  else
414  throw "unexpected OTHER statement: "+id2string(statement);
415 }

457 exprt interpretert::get_value(
458  const typet &type,
459  const mp_integer &offset,
460  bool use_non_det)
461 {
462  const typet real_type=ns.follow(type);
463  if(real_type.id()==ID_struct)
464  {
465  struct_exprt result({}, real_type);
466  const struct_typet &struct_type=to_struct_type(real_type);
467  const struct_typet::componentst &components=struct_type.components();
468 
469  // Retrieve the values for the individual members
470  result.reserve_operands(components.size());
471 
472  mp_integer tmp_offset=offset;
473 
474  for(const auto &c : components)
475  {
476  mp_integer size=get_size(c.type());
477  const exprt operand=get_value(c.type(), tmp_offset);
478  tmp_offset+=size;
479  result.copy_to_operands(operand);
480  }
481 
482  return std::move(result);
483  }
484  else if(real_type.id()==ID_array)
485  {
486  // Get size of array
487  array_exprt result({}, to_array_type(real_type));
488  const exprt &size_expr = to_array_type(type).size();
489  mp_integer subtype_size = get_size(to_array_type(type).element_type());
490  mp_integer count;
491  if(!size_expr.is_constant())
492  {
493  count=base_address_to_actual_size(offset)/subtype_size;
494  }
495  else
496  {
497  count = numeric_cast_v<mp_integer>(to_constant_expr(size_expr));
498  }
499 
500  // Retrieve the value for each member in the array
501  result.reserve_operands(numeric_cast_v<std::size_t>(count));
502  for(mp_integer i=0; i<count; ++i)
503  {
504  const exprt operand = get_value(
505  to_array_type(type).element_type(), offset + i * subtype_size);
506  result.copy_to_operands(operand);
507  }
508  return std::move(result);
509  }
510  if(
511  use_non_det && memory[numeric_cast_v<std::size_t>(offset)].initialized !=
512  memory_cellt::initializedt::WRITTEN_BEFORE_READ)
513  {
514  return side_effect_expr_nondett(type, source_locationt());
515  }
516  mp_vectort rhs;
517  rhs.push_back(memory[numeric_cast_v<std::size_t>(offset)].value);
518  return get_value(type, rhs);
519 }

523 exprt interpretert::get_value(
524  const typet &type,
525  mp_vectort &rhs,
526  const mp_integer &offset)
527 {
528  const typet real_type=ns.follow(type);
529  PRECONDITION(!rhs.empty());
530 
531  if(real_type.id()==ID_struct)
532  {
533  struct_exprt result({}, real_type);
534  const struct_typet &struct_type=to_struct_type(real_type);
535  const struct_typet::componentst &components=struct_type.components();
536 
537  // Retrieve the values for the individual members
538  result.reserve_operands(components.size());
539  mp_integer tmp_offset=offset;
540 
541  for(const struct_union_typet::componentt &expr : components)
542  {
543  mp_integer size=get_size(expr.type());
544  const exprt operand=get_value(expr.type(), rhs, tmp_offset);
545  tmp_offset+=size;
546  result.copy_to_operands(operand);
547  }
548  return std::move(result);
549  }
550  else if(real_type.id()==ID_array)
551  {
552  array_exprt result({}, to_array_type(real_type));
553  const exprt &size_expr = to_array_type(real_type).size();
554 
555  // Get size of array
556  mp_integer subtype_size = get_size(to_array_type(type).element_type());
557 
558  mp_integer count;
559  if(unbounded_size(type))
560  {
561  count=base_address_to_actual_size(offset)/subtype_size;
562  }
563  else
564  {
565  count = numeric_cast_v<mp_integer>(to_constant_expr(size_expr));
566  }
567 
568  // Retrieve the value for each member in the array
569  result.reserve_operands(numeric_cast_v<std::size_t>(count));
570  for(mp_integer i=0; i<count; ++i)
571  {
572  const exprt operand = get_value(
573  to_array_type(type).element_type(), rhs, offset + i * subtype_size);
574  result.copy_to_operands(operand);
575  }
576  return std::move(result);
577  }
578  else if(real_type.id()==ID_floatbv)
579  {
580  ieee_floatt f(to_floatbv_type(type));
581  f.unpack(rhs[numeric_cast_v<std::size_t>(offset)]);
582  return f.to_expr();
583  }
584  else if(real_type.id()==ID_fixedbv)
585  {
586  fixedbvt f;
587  f.from_integer(rhs[numeric_cast_v<std::size_t>(offset)]);
588  return f.to_expr();
589  }
590  else if(real_type.id()==ID_bool)
591  {
592  if(rhs[numeric_cast_v<std::size_t>(offset)] != 0)
593  return true_exprt();
594  else
595  false_exprt();
596  }
597  else if(real_type.id()==ID_c_bool)
598  {
599  return from_integer(
600  rhs[numeric_cast_v<std::size_t>(offset)] != 0 ? 1 : 0, type);
601  }
602  else if(real_type.id() == ID_pointer)
603  {
604  if(rhs[numeric_cast_v<std::size_t>(offset)] == 0)
605  return null_pointer_exprt(to_pointer_type(real_type)); // NULL pointer
606 
607  if(rhs[numeric_cast_v<std::size_t>(offset)] < memory.size())
608  {
609  // We want the symbol pointed to
610  mp_integer address = rhs[numeric_cast_v<std::size_t>(offset)];
611  const symbol_exprt &symbol_expr = address_to_symbol(address);
612  mp_integer offset_from_address = address_to_offset(address);
613 
614  if(offset_from_address == 0)
615  return address_of_exprt(symbol_expr);
616 
617  if(
618  symbol_expr.type().id() == ID_struct ||
619  symbol_expr.type().id() == ID_struct_tag)
620  {
621  const auto c = get_component(symbol_expr.type(), offset_from_address);
622  member_exprt member_expr(symbol_expr, c);
623  return address_of_exprt(member_expr);
624  }
625 
626  return index_exprt(
627  symbol_expr, from_integer(offset_from_address, integer_typet()));
628  }
629 
630  output.error() << "interpreter: invalid pointer "
631  << rhs[numeric_cast_v<std::size_t>(offset)]
632  << " > object count " << memory.size() << messaget::eom;
633 
634  throw "interpreter: reading from invalid pointer";
635  }
636  else if(real_type.id()==ID_string)
637  {
638  // Strings are currently encoded by their irep_idt ID.
639  return constant_exprt(
640  get_string_container().get_string(
641  numeric_cast_v<std::size_t>(rhs[numeric_cast_v<std::size_t>(offset)])),
642  type);
643  }
644 
645  // Retrieve value of basic data type
646  return from_integer(rhs[numeric_cast_v<std::size_t>(offset)], type);
647 }

650 void interpretert::execute_assign()
651 {
652  const exprt &assign_lhs = pc->assign_lhs();
653  const exprt &assign_rhs = pc->assign_rhs();
654 
655  mp_vectort rhs = evaluate(assign_rhs);
656 
657  if(!rhs.empty())
658  {
659  mp_integer address = evaluate_address(assign_lhs);
660  mp_integer size = get_size(assign_lhs.type());
661 
662  if(size!=rhs.size())
663  output.error() << "!! failed to obtain rhs (" << rhs.size() << " vs. "
664  << size << ")\n"
665  << messaget::eom;
666  else
667  {
668  goto_trace_stept &trace_step=steps.get_last_step();
669  assign(address, rhs);
670  trace_step.full_lhs = assign_lhs;
671  trace_step.full_lhs_value=get_value(trace_step.full_lhs.type(), rhs);
672  }
673  }
674  else if(assign_rhs.id() == ID_side_effect)
675  {
676  side_effect_exprt side_effect = to_side_effect_expr(assign_rhs);
677  if(side_effect.get_statement()==ID_nondet)
678  {
679  mp_integer address =
680  numeric_cast_v<std::size_t>(evaluate_address(assign_lhs));
681 
682  mp_integer size = get_size(assign_lhs.type());
683 
684  for(mp_integer i=0; i<size; ++i)
685  {
686  memory[numeric_cast_v<std::size_t>(address + i)].initialized =
687  memory_cellt::initializedt::READ_BEFORE_WRITTEN;
688  }
689  }
690  }

735 void interpretert::execute_function_call()
736 {
737  const auto &call_lhs = pc->call_lhs();
738  const auto &call_function = pc->call_function();
739  const auto &call_arguments = pc->call_arguments();
740 
741  // function to be called
742  mp_integer address = evaluate_address(call_function);
743

746  else if(address>=memory.size())

964 mp_integer interpretert::get_size(const typet &type)
965 {
966  if(unbounded_size(type))
967  return mp_integer(2) << 32;
968 
969  if(type.id()==ID_struct)
970  {
971  const struct_typet::componentst &components=
972  to_struct_type(type).components();
973 
974  mp_integer sum=0;
975 
976  for(const auto &comp : components)
977  {
978  DATA_INVARIANT(
979  comp.type().id() != ID_code, "struct member must not be of code type");
980  sum += get_size(comp.type());
981  }
982 
983  return sum;
984  }
985  else if(type.id()==ID_union)
986  {
987  const union_typet::componentst &components=
988  to_union_type(type).components();
989 
990  mp_integer max_size=0;
991 
992  for(const auto &comp : components)
993  {
994  DATA_INVARIANT(
995  comp.type().id() != ID_code, "union member must not be of code type");
996  max_size = std::max(max_size, get_size(comp.type()));
997  }
998 
999  return max_size;
1000  }
1001  else if(type.id()==ID_array)
1002  {
1003  const exprt &size_expr = to_array_type(type).size();
1004 
1005  mp_integer subtype_size = get_size(to_array_type(type).element_type());
1006 
1007  mp_vectort i = evaluate(size_expr);
1008  if(i.size()==1)
1009  {
1010  // Go via the binary representation to reproduce any
1011  // overflow behaviour.
1012  const constant_exprt size_const = from_integer(i[0], size_expr.type());
1013  const mp_integer size_mp = numeric_cast_v<mp_integer>(size_const);
1014  return subtype_size*size_mp;
1015  }
1016  return subtype_size;
1017  }
1018  else if(type.id() == ID_struct_tag)
1019  {
1020  return get_size(ns.follow_tag(to_struct_tag_type(type)));
1021  }
1022 
1023  return 1;
1024 }
