44 static void run_symtab2gb(
45  const std::vector<std::string> &symtab_filenames,
46  const std::string &gb_filename,
47  const std::string &cmdline_verbosity)
48 {
49  // try opening all the files first to make sure we can
50  // even read/write what we want
51  std::ofstream out_file{gb_filename, std::ios::binary};
52  if(!out_file.is_open())
53  {
54  throw system_exceptiont{"couldn't open output file '" + gb_filename + "'"};
55  }
56  std::vector<std::ifstream> symtab_files;
57  for(auto const &symtab_filename : symtab_filenames)
58  {
59  std::ifstream symtab_file{symtab_filename};
60  if(!symtab_file.is_open())
61  {
62  throw system_exceptiont{"couldn't open input file '" + symtab_filename +
63  "'"};
64  }
65  symtab_files.emplace_back(std::move(symtab_file));
66  }
67 
68  stream_message_handlert message_handler{std::cerr};
69  messaget::eval_verbosity(
70  cmdline_verbosity, messaget::M_STATISTICS, message_handler);
71 
72  auto const symtab_language = new_json_symtab_language();
73  symtab_language->set_message_handler(message_handler);
74 
75  symbol_tablet linked_symbol_table;
76 
77  for(std::size_t ix = 0; ix < symtab_files.size(); ++ix)
78  {
79  auto const &symtab_filename = symtab_filenames[ix];
80  auto &symtab_file = symtab_files[ix];
81  if(failed(symtab_language->parse(symtab_file, symtab_filename)))
82  {
83  source_locationt source_location;
84  source_location.set_file(symtab_filename);
85  throw invalid_source_file_exceptiont{
86  "failed to parse symbol table", source_location};
87  }
88  symbol_tablet symtab{};
89  if(failed(symtab_language->typecheck(symtab, "<unused>")))
90  {
91  source_locationt source_location;
92  source_location.set_file(symtab_filename);
93  throw invalid_source_file_exceptiont{
94  "failed to typecheck symbol table", source_location};
95  }
96  config.set_from_symbol_table(symtab);
97 
98  if(failed(linking(linked_symbol_table, symtab, message_handler)))
99  {
100  throw invalid_input_exceptiont{
101  "failed to link `" + symtab_filename + "'"};
102  }
103  }
104 
105  goto_modelt linked_goto_model;
106  linked_goto_model.symbol_table.swap(linked_symbol_table);
107  goto_convert(linked_goto_model, message_handler);
108 
109  if(failed(write_goto_binary(out_file, linked_goto_model)))
110  {
111  throw system_exceptiont{"failed to write goto binary to " + gb_filename};
112  }
113 }

124 int symtab2gb_parse_optionst::doit()
125 {
126  if(cmdline.isset("version"))
127  {
128  log.status() << CBMC_VERSION << '\n';
129  return CPROVER_EXIT_SUCCESS;
130  }
131  if(cmdline.args.empty())
132  {
133  throw invalid_command_line_argument_exceptiont{
134  "expect at least one input file", "<json-symtab-file>"};
135  }
136  std::vector<std::string> symtab_filenames = cmdline.args;
137  std::string gb_filename = "a.out";
138  if(cmdline.isset(SYMTAB2GB_OUT_FILE_OPT))
139  {
140  gb_filename = cmdline.get_value(SYMTAB2GB_OUT_FILE_OPT);
141  }
142  register_languages();
143  config.set(cmdline);
144  run_symtab2gb(symtab_filenames, gb_filename, cmdline.get_value("verbosity"));
145  return CPROVER_EXIT_SUCCESS;
146 }
