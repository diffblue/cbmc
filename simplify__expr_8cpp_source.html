372 static simplify_exprt::resultt<> simplify_string_index_of(
373  const function_application_exprt &expr,
374  const namespacet &ns,
375  const bool search_from_end)
376 {
377  std::size_t starting_index = 0;
378 
379  // Determine starting index for the comparison (if given)
380  if(expr.arguments().size() == 3)
381  {
382  auto &starting_index_expr = expr.arguments().at(2);
383 
384  if(starting_index_expr.id() != ID_constant)
385  {
386  return simplify_exprt::unchanged(expr);
387  }
388 
389  const mp_integer idx =
390  numeric_cast_v<mp_integer>(to_constant_expr(starting_index_expr));
391 
392  // Negative indices are treated like 0
393  if(idx > 0)
394  {
395  starting_index = numeric_cast_v<std::size_t>(idx);
396  }
397  }
398 
399  const refined_string_exprt &s1 = to_string_expr(expr.arguments().at(0));
400 
401  const auto s1_data_opt = try_get_string_data_array(s1.content(), ns);
402 
403  if(!s1_data_opt)
404  {
405  return simplify_exprt::unchanged(expr);
406  }
407 
408  const array_exprt &s1_data = s1_data_opt->get();
409 
410  const auto search_string_size = s1_data.operands().size();
411  if(starting_index >= search_string_size)
412  {
413  return from_integer(-1, expr.type());
414  }
415 
416  unsigned long starting_offset =
417  starting_index > 0 ? (search_string_size - 1) - starting_index : 0;
418  if(can_cast_expr<refined_string_exprt>(expr.arguments().at(1)))
419  {
420  // Second argument is a string
421 
422  const refined_string_exprt &s2 = to_string_expr(expr.arguments().at(1));
423 
424  const auto s2_data_opt = try_get_string_data_array(s2.content(), ns);
425 
426  if(!s2_data_opt)
427  {
428  return simplify_exprt::unchanged(expr);
429  }
430 
431  const array_exprt &s2_data = s2_data_opt->get();
432 
433  // Searching for empty string is a special case and is simply the
434  // "always found at the first searched position. This needs to take into
435  // account starting position and if you're starting from the beginning or
436  // end.
437  if(s2_data.operands().empty())
438  return from_integer(
439  search_from_end
440  ? starting_index > 0 ? starting_index : search_string_size
441  : 0,
442  expr.type());
443 
444  if(search_from_end)
445  {
446  auto end = std::prev(s1_data.operands().end(), starting_offset);
447  auto it = std::find_end(
448  s1_data.operands().begin(),
449  end,
450  s2_data.operands().begin(),
451  s2_data.operands().end());
452  if(it != end)
453  return from_integer(
454  std::distance(s1_data.operands().begin(), it), expr.type());
455  }
456  else
457  {
458  auto it = std::search(
459  std::next(s1_data.operands().begin(), starting_index),
460  s1_data.operands().end(),
461  s2_data.operands().begin(),
462  s2_data.operands().end());
463 
464  if(it != s1_data.operands().end())
465  return from_integer(
466  std::distance(s1_data.operands().begin(), it), expr.type());
467  }
468  }
469  else if(expr.arguments().at(1).id() == ID_constant)
470  {
471  // Second argument is a constant character
472 
473  const constant_exprt &c1 = to_constant_expr(expr.arguments().at(1));
474  const auto c1_val = numeric_cast_v<mp_integer>(c1);
475 
476  auto pred = [&](const exprt &c2) {
477  const auto c2_val = numeric_cast_v<mp_integer>(to_constant_expr(c2));
478 
479  return c1_val == c2_val;
480  };
481 
482  if(search_from_end)
483  {
484  auto it = std::find_if(
485  std::next(s1_data.operands().rbegin(), starting_offset),
486  s1_data.operands().rend(),
487  pred);
488  if(it != s1_data.operands().rend())
489  return from_integer(
490  std::distance(s1_data.operands().begin(), it.base() - 1),
491  expr.type());
492  }
493  else
494  {
495  auto it = std::find_if(
496  std::next(s1_data.operands().begin(), starting_index),
497  s1_data.operands().end(),
498  pred);
499  if(it != s1_data.operands().end())
500  return from_integer(
501  std::distance(s1_data.operands().begin(), it), expr.type());
502  }
503  }
504  else
505  {
506  return simplify_exprt::unchanged(expr);
507  }
508 
509  return from_integer(-1, expr.type());
510 }

756  const typet &op_type = expr.op().type();
757 
758  // eliminate casts of infinity
759  if(expr.op().id() == ID_infinity)
760  {
761  typet new_type=expr.type();
762  exprt tmp = expr.op();
763  tmp.type()=new_type;
764  return std::move(tmp);
765  }
766

768  if(
769  op_type.id() == ID_pointer && expr.op().is_constant() &&
770  to_constant_expr(expr.op()).get_value() == ID_NULL &&
771  (expr_type.id() == ID_unsignedbv || expr_type.id() == ID_signedbv) &&
772  config.ansi_c.NULL_is_zero)
773  {
774  return from_integer(0, expr_type);
775  }
776 
777  // casts from pointer to integer
778  // where width of integer >= width of pointer
779  // (void*)(intX)expr -> (void*)expr
780  if(
781  expr_type.id() == ID_pointer && expr.op().id() == ID_typecast &&
782  (op_type.id() == ID_signedbv || op_type.id() == ID_unsignedbv ||
783  op_type.id() == ID_bv) &&
784  to_bitvector_type(op_type).get_width() >=
785  to_bitvector_type(expr_type).get_width())
786  {
787  auto new_expr = expr;
788  new_expr.op() = to_typecast_expr(expr.op()).op();

790  }

793  if(expr.type() == expr.op().type())
794  {
795  return expr.op();
796  }

799  if(expr_type.id()==ID_bool)
800  {

803  expr.op(),
804  op_type.id() == ID_floatbv ? ID_ieee_float_notequal : ID_notequal,
805  from_integer(0, op_type));

814  expr_type.id() == ID_c_bool || expr_type.id() == ID_c_bit_field))

818  auto zero = from_integer(0, expr_type);

1560 simplify_exprt::resultt<> simplify_exprt::simplify_object(const exprt &expr)
1561 {
1562  if(expr.id()==ID_plus)
1563  {
1564  if(expr.type().id()==ID_pointer)
1565  {
1566  // kill integers from sum
1567  for(auto &op : expr.operands())
1568  if(op.type().id() == ID_pointer)
1569  return changed(simplify_object(op)); // recursive call
1570  }
1571  }
1572  else if(expr.id()==ID_typecast)
1573  {
1574  auto const &typecast_expr = to_typecast_expr(expr);
1575  const typet &op_type = typecast_expr.op().type();
1576 
1577  if(op_type.id()==ID_pointer)
1578  {
1579  // cast from pointer to pointer
1580  return changed(simplify_object(typecast_expr.op())); // recursive call
1581  }
1582  else if(op_type.id()==ID_signedbv || op_type.id()==ID_unsignedbv)
1583  {
1584  // cast from integer to pointer
1585 
1586  // We do a bit of special treatment for (TYPE *)(a+(int)&o) and
1587  // (TYPE *)(a+(int)((T*)&o+x)), which are re-written to '&o'.
1588 
1589  const exprt &casted_expr = typecast_expr.op();
1590  if(casted_expr.id() == ID_plus && casted_expr.operands().size() == 2)
1591  {
1592  const auto &plus_expr = to_plus_expr(casted_expr);
1593 
1594  const exprt &cand = plus_expr.op0().id() == ID_typecast
1595  ? plus_expr.op0()
1596  : plus_expr.op1();
1597 
1598  if(cand.id() == ID_typecast)
1599  {
1600  const auto &typecast_op = to_typecast_expr(cand).op();
1601 
1602  if(typecast_op.id() == ID_address_of)
1603  {
1604  return typecast_op;
1605  }
1606  else if(
1607  typecast_op.id() == ID_plus && typecast_op.operands().size() == 2 &&
1608  to_plus_expr(typecast_op).op0().id() == ID_typecast &&
1609  to_typecast_expr(to_plus_expr(typecast_op).op0()).op().id() ==
1610  ID_address_of)
1611  {
1612  return to_typecast_expr(to_plus_expr(typecast_op).op0()).op();
1613  }
1614  }
1615  }
1616  }
1617  }
1618  else if(expr.id()==ID_address_of)
1619  {
1620  const auto &object = to_address_of_expr(expr).object();
1621 
1622  if(object.id() == ID_index)
1623  {
1624  // &some[i] -> &some
1625  address_of_exprt new_expr(to_index_expr(object).array());
1626  return changed(simplify_object(new_expr)); // recursion
1627  }
1628  else if(object.id() == ID_member)
1629  {
1630  // &some.f -> &some
1631  address_of_exprt new_expr(to_member_expr(object).compound());
1632  return changed(simplify_object(new_expr)); // recursion
1633  }
1634  }
1635 
1636  return unchanged(expr);
1637 }

1639 simplify_exprt::resultt<>
1640 simplify_exprt::simplify_byte_extract(const byte_extract_exprt &expr)
1641 {
1642  // lift up any ID_if on the object
1643  if(expr.op().id()==ID_if)
1644  {
1645  if_exprt if_expr=lift_if(expr, 0);
1646  if_expr.true_case() =
1647  simplify_byte_extract(to_byte_extract_expr(if_expr.true_case()));
1648  if_expr.false_case() =
1649  simplify_byte_extract(to_byte_extract_expr(if_expr.false_case()));
1650  return changed(simplify_if(if_expr));
1651  }
1652 
1653  const auto el_size = pointer_offset_bits(expr.type(), ns);
1654  if(el_size.has_value() && *el_size < 0)
1655  return unchanged(expr);
1656 
1657  // byte_extract(byte_extract(root, offset1), offset2) =>
1658  // byte_extract(root, offset1+offset2)
1659  if(expr.op().id()==expr.id())
1660  {
1661  auto tmp = expr;
1662 
1663  tmp.offset() = simplify_plus(
1664  plus_exprt(to_byte_extract_expr(expr.op()).offset(), expr.offset()));
1665  tmp.op() = to_byte_extract_expr(expr.op()).op();
1666 
1667  return changed(simplify_byte_extract(tmp)); // recursive call
1668  }
1669 
1670  // byte_extract(byte_update(root, offset, value), offset) =>
1671  // value
1672  if(
1673  ((expr.id() == ID_byte_extract_big_endian &&
1674  expr.op().id() == ID_byte_update_big_endian) ||
1675  (expr.id() == ID_byte_extract_little_endian &&
1676  expr.op().id() == ID_byte_update_little_endian)) &&
1677  expr.offset() == to_byte_update_expr(as_const(expr).op()).offset())
1678  {
1679  const auto &op_byte_update = to_byte_update_expr(expr.op());
1680 
1681  if(expr.type() == op_byte_update.value().type())
1682  {
1683  return op_byte_update.value();
1684  }
1685  else if(el_size.has_value())
1686  {
1687  const auto update_bits_opt =
1688  pointer_offset_bits(op_byte_update.value().type(), ns);
1689 
1690  if(update_bits_opt.has_value() && *el_size <= *update_bits_opt)
1691  {
1692  auto tmp = expr;
1693  tmp.op() = op_byte_update.value();
1694  tmp.offset() = from_integer(0, expr.offset().type());
1695 
1696  return changed(simplify_byte_extract(tmp)); // recursive call
1697  }
1698  }
1699  }
1700 
1701  // the following require a constant offset
1702  auto offset = numeric_cast<mp_integer>(expr.offset());
1703  if(!offset.has_value() || *offset < 0)
1704  return unchanged(expr);
1705 
1706  // byte_extract(byte_update(root, offset_u, value), offset_e) so that the
1707  // update does not affect what is being extracted simplifies to
1708  // byte_extract(root, offset_e)
1709  if(
1710  expr.op().id() == ID_byte_update_big_endian ||
1711  expr.op().id() == ID_byte_update_little_endian)
1712  {
1713  const byte_update_exprt &bu = to_byte_update_expr(expr.op());
1714  const auto update_offset = numeric_cast<mp_integer>(bu.offset());
1715  if(el_size.has_value() && update_offset.has_value())
1716  {
1717  if(
1718  *offset * expr.get_bits_per_byte() + *el_size <=
1719  *update_offset * bu.get_bits_per_byte())
1720  {
1721  auto tmp = expr;
1722  tmp.op() = bu.op();
1723  return changed(simplify_byte_extract(tmp)); // recursive call
1724  }
1725  else
1726  {
1727  const auto update_size = pointer_offset_bits(bu.value().type(), ns);
1728  if(
1729  update_size.has_value() &&
1730  *offset >= *update_offset * bu.get_bits_per_byte() + *update_size)
1731  {
1732  auto tmp = expr;
1733  tmp.op() = bu.op();
1734  return changed(simplify_byte_extract(tmp)); // recursive call
1735  }
1736  }
1737  }
1738  }
1739 
1740  // don't do any of the following if endianness doesn't match, as
1741  // bytes need to be swapped
1742  if(
1743  *offset == 0 && ((expr.id() == ID_byte_extract_little_endian &&
1744  config.ansi_c.endianness ==
1745  configt::ansi_ct::endiannesst::IS_LITTLE_ENDIAN) ||
1746  (expr.id() == ID_byte_extract_big_endian &&
1747  config.ansi_c.endianness ==
1748  configt::ansi_ct::endiannesst::IS_BIG_ENDIAN)))
1749  {
1750  // byte extract of full object is object
1751  if(expr.type() == expr.op().type())
1752  {
1753  return expr.op();
1754  }
1755  else if(
1756  expr.type().id() == ID_pointer && expr.op().type().id() == ID_pointer)
1757  {
1758  return typecast_exprt(expr.op(), expr.type());
1759  }
1760  }
1761 
1762  if(
1763  (expr.type().id() == ID_union || expr.type().id() == ID_union_tag) &&
1764  to_union_type(ns.follow(expr.type())).components().empty())
1765  {
1766  return empty_union_exprt{expr.type()};
1767  }
1768  else if(
1769  (expr.type().id() == ID_struct || expr.type().id() == ID_struct_tag) &&
1770  to_struct_type(ns.follow(expr.type())).components().empty())
1771  {
1772  return struct_exprt{{}, expr.type()};
1773  }
1774 
1775  // no proper simplification for expr.type()==void
1776  // or types of unknown size
1777  if(!el_size.has_value() || *el_size == 0)
1778  return unchanged(expr);
1779 
1780  if(expr.op().id()==ID_array_of &&
1781  to_array_of_expr(expr.op()).op().id()==ID_constant)
1782  {
1783  const auto const_bits_opt = expr2bits(
1784  to_array_of_expr(expr.op()).op(),
1785  config.ansi_c.endianness ==
1786  configt::ansi_ct::endiannesst::IS_LITTLE_ENDIAN,
1787  ns);

1793

1798  *offset * expr.get_bits_per_byte() + *el_size)
1799  {

1804  const_bits,

1806  numeric_cast_v<std::size_t>(*el_size));

1817  expr.op().id() == ID_array_of &&
1818  (*offset * expr.get_bits_per_byte()) % (*el_size) == 0 &&
1819  *el_size <=
1820  pointer_offset_bits(to_array_of_expr(expr.op()).what().type(), ns))

1888  !component_bits.has_value() || *component_bits == 0 ||
