37 incremental_goto_checkert::resultt single_path_symex_only_checkert::
38 operator()(propertiest &properties)
39 {
40  resultt result(resultt::progresst::DONE);
41 
42  initialize_worklist();
43 
44  while(!has_finished_exploration(properties))
45  {
46  path_storaget::patht &path = worklist->peek();
47 
48  (void)resume_path(path);
49 
50  update_properties(properties, result.updated_properties, path.equation);
51 
52  worklist->pop();
53  }
54 
55  log.status() << "Runtime Symex: " << symex_runtime.count() << "s"
56  << messaget::eom;
57 
58  final_update_properties(properties, result.updated_properties);
59 
60  return result;
61 }

93 bool single_path_symex_only_checkert::resume_path(path_storaget::patht &path)
94 {
95  const auto symex_start = std::chrono::steady_clock::now();
96 
97  symex_bmct symex(
98  ui_message_handler,
99  goto_model.get_symbol_table(),
100  path.equation,
101  options,
102  *worklist,
103  guard_manager,
104  unwindset);
105  setup_symex(symex);
106 
107  symex_symbol_table = symex.resume_symex_from_saved_state(
108  goto_symext::get_goto_function(goto_model), path.state, &path.equation);
109 
110  const auto symex_stop = std::chrono::steady_clock::now();
111  symex_runtime += std::chrono::duration<double>(symex_stop - symex_start);
112 
113  postprocess_equation(symex, path.equation, options, ns, ui_message_handler);
114 
115  equation_output(symex, path.equation);
116 
117  return is_ready_to_decide(symex, path);
118 }
