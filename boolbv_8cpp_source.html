39 const bvt &boolbvt::convert_bv(
40  const exprt &expr,
41  std::optional<std::size_t> expected_width)
42 {
43  // check cache first
44  std::pair<bv_cachet::iterator, bool> cache_result=
45  bv_cache.insert(std::make_pair(expr, bvt()));
46 
47  // get a reference to the cache entry
48  auto &cache_entry = cache_result.first->second;
49 
50  if(!cache_result.second)
51  {
52  // Found in cache
53  return cache_entry;
54  }
55 
56  // Iterators into hash_maps do not remain valid when inserting
57  // more elements recursively. C++11 §23.2.5/13
58  // However, the _reference_ to the entry does!
59  cache_entry = convert_bitvector(expr);
60 
61  INVARIANT_WITH_DIAGNOSTICS(
62  !expected_width || cache_entry.size() == *expected_width,
63  "bitvector width shall match the indicated expected width",
64  expr.find_source_location(),
65  irep_pretty_diagnosticst(expr));
66 
67  // check
68  for(const auto &literal : cache_entry)
69  {
70  if(freeze_all && !literal.is_constant())
71  prop.set_frozen(literal);
72 
73  INVARIANT_WITH_DIAGNOSTICS(
74  literal.var_no() != literalt::unused_var_no(),
75  "variable number must be different from the unused variable number",
76  expr.find_source_location(),
77  irep_pretty_diagnosticst(expr));
78  }
79 
80  return cache_entry;
81 }

262 bvt boolbvt::convert_array_comprehension(const array_comprehension_exprt &expr)
263 {
264  std::size_t width=boolbv_width(expr.type());
265 
266  const exprt &array_size = expr.type().size();
267 
268  const auto size = numeric_cast_v<mp_integer>(to_constant_expr(array_size));
269 
270  typet counter_type = expr.arg().type();
271 
272  bvt bv;
273  bv.resize(width);
274 
275  for(mp_integer i = 0; i < size; ++i)
276  {
277  exprt counter=from_integer(i, counter_type);
278 
279  exprt body = expr.instantiate({counter});
280 
281  const bvt &tmp = convert_bv(body);
282 
283  INVARIANT(
284  size * tmp.size() == width,
285  "total bitvector width shall equal the number of operands times the size "
286  "per operand");
287 
288  std::size_t offset = numeric_cast_v<std::size_t>(i * tmp.size());
289 
290  for(std::size_t j=0; j<tmp.size(); j++)
291  bv[offset+j]=tmp[j];
292  }
293 
294  return bv;
295 }

332 literalt boolbvt::convert_rest(const exprt &expr)
333 {
334  PRECONDITION(expr.is_boolean());
335 
336  if(expr.id()==ID_typecast)
337  return convert_typecast(to_typecast_expr(expr));
338  else if(expr.id()==ID_equal)
339  return convert_equality(to_equal_expr(expr));
340  else if(expr.id()==ID_verilog_case_equality ||
341  expr.id()==ID_verilog_case_inequality)
342  return convert_verilog_case_equality(to_binary_relation_expr(expr));
343  else if(expr.id()==ID_notequal)
344  {
345  const auto &notequal_expr = to_notequal_expr(expr);
346  return !convert_equality(
347  equal_exprt(notequal_expr.lhs(), notequal_expr.rhs()));
348  }
349  else if(expr.id()==ID_ieee_float_equal ||
350  expr.id()==ID_ieee_float_notequal)
351  {
352  return convert_ieee_float_rel(to_binary_relation_expr(expr));
353  }
354  else if(expr.id()==ID_le || expr.id()==ID_ge ||
355  expr.id()==ID_lt || expr.id()==ID_gt)
356  {
357  return convert_bv_rel(to_binary_relation_expr(expr));
358  }
359  else if(expr.id()==ID_extractbit)
360  return convert_extractbit(to_extractbit_expr(expr));
361  else if(expr.id()==ID_forall)
362  return convert_quantifier(to_quantifier_expr(expr));
363  else if(expr.id()==ID_exists)
364  return convert_quantifier(to_quantifier_expr(expr));
365  else if(expr.id()==ID_let)
366  {
367  bvt bv=convert_let(to_let_expr(expr));
368 
369  DATA_INVARIANT(bv.size()==1,
370  "convert_let must return 1-bit vector for boolean let");
371 
372  return bv[0];
373  }
374  else if(expr.id()==ID_index)
375  {
376  bvt bv=convert_index(to_index_expr(expr));
377  CHECK_RETURN(bv.size() == 1);
378  return bv[0];
379  }
380  else if(expr.id()==ID_member)
381  {
382  bvt bv=convert_member(to_member_expr(expr));
383  CHECK_RETURN(bv.size() == 1);
384  return bv[0];
385  }
386  else if(expr.id()==ID_case)
387  {
388  bvt bv=convert_case(expr);
389  CHECK_RETURN(bv.size() == 1);
390  return bv[0];
391  }
392  else if(expr.id()==ID_cond)
393  {
394  bvt bv = convert_cond(to_cond_expr(expr));
395  CHECK_RETURN(bv.size() == 1);
396  return bv[0];
397  }
398  else if(expr.id()==ID_sign)
399  {
400  const auto &op = to_sign_expr(expr).op();
401  const bvt &bv = convert_bv(op);
402  CHECK_RETURN(!bv.empty());
403  const irep_idt type_id = op.type().id();
404  if(type_id == ID_signedbv || type_id == ID_fixedbv || type_id == ID_floatbv)
405  return bv[bv.size()-1];
406  if(type_id == ID_unsignedbv)
407  return const_literal(false);
408  }
409  else if(expr.id()==ID_reduction_or || expr.id()==ID_reduction_and ||
410  expr.id()==ID_reduction_nor || expr.id()==ID_reduction_nand ||
411  expr.id()==ID_reduction_xor || expr.id()==ID_reduction_xnor)
412  return convert_reduction(to_unary_expr(expr));
413  else if(expr.id()==ID_onehot || expr.id()==ID_onehot0)
414  return convert_onehot(to_unary_expr(expr));
415  else if(
416  const auto binary_overflow =
417  expr_try_dynamic_cast<binary_overflow_exprt>(expr))
418  {
419  return convert_binary_overflow(*binary_overflow);
420  }
421  else if(
422  const auto unary_overflow =
423  expr_try_dynamic_cast<unary_overflow_exprt>(expr))
424  {
425  return convert_unary_overflow(*unary_overflow);
426  }
427  else if(expr.id()==ID_isnan)
428  {
429  const auto &op = to_unary_expr(expr).op();
430  const bvt &bv = convert_bv(op);
431 
432  if(op.type().id() == ID_floatbv)
433  {
434  float_utilst float_utils(prop, to_floatbv_type(op.type()));
435  return float_utils.is_NaN(bv);
436  }
437  else if(op.type().id() == ID_fixedbv)
438  return const_literal(false);
439  }
440  else if(expr.id()==ID_isfinite)
441  {
442  const auto &op = to_unary_expr(expr).op();
443  const bvt &bv = convert_bv(op);
444 
445  if(op.type().id() == ID_floatbv)
446  {
447  float_utilst float_utils(prop, to_floatbv_type(op.type()));
448  return prop.land(
449  !float_utils.is_infinity(bv),
450  !float_utils.is_NaN(bv));
451  }
452  else if(op.id() == ID_fixedbv)
453  return const_literal(true);
454  }
455  else if(expr.id()==ID_isinf)
456  {
457  const auto &op = to_unary_expr(expr).op();
458  const bvt &bv = convert_bv(op);
459 
460  if(op.type().id() == ID_floatbv)
461  {
462  float_utilst float_utils(prop, to_floatbv_type(op.type()));
463  return float_utils.is_infinity(bv);
464  }
465  else if(op.type().id() == ID_fixedbv)
466  return const_literal(false);
467  }
468  else if(expr.id()==ID_isnormal)
469  {
470  const auto &op = to_unary_expr(expr).op();
471 
472  if(op.type().id() == ID_floatbv)
473  {
474  const bvt &bv = convert_bv(op);
475  float_utilst float_utils(prop, to_floatbv_type(op.type()));
476  return float_utils.is_normal(bv);
477  }
478  else if(op.type().id() == ID_fixedbv)
479  return const_literal(true);
480  }
481  else if(expr.id() == ID_function_application)
482  {
483  functions.record(to_function_application_expr(expr));
484  return prop.new_variable();
485  }
486 
487  return SUB::convert_rest(expr);
488 }
