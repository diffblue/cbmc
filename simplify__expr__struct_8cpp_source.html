23 simplify_exprt::simplify_member(const member_exprt &expr)
24 {
25  const irep_idt &component_name=
26  to_member_expr(expr).get_component_name();
27 
28  const exprt &op = expr.compound();
29  const typet &op_type=ns.follow(op.type());
30 
31  if(op.id()==ID_with)
32  {
33  // the following optimization only works on structs,
34  // and not on unions
35 
36  if(op.operands().size()>=3 &&
37  op_type.id()==ID_struct)
38  {
39  exprt::operandst new_operands = op.operands();
40 
41  while(new_operands.size() > 1)
42  {
43  exprt &op1 = new_operands[new_operands.size() - 2];
44  exprt &op2 = new_operands[new_operands.size() - 1];
45 
46  if(op1.get(ID_component_name)==component_name)
47  {
48  // found it!
49  DATA_INVARIANT(
50  op2.type() == expr.type(),
51  "member expression type must match component type");
52 
53  return op2;
54  }
55  else // something else, get rid of it
56  new_operands.resize(new_operands.size() - 2);
57  }
58 
59  DATA_INVARIANT(new_operands.size() == 1, "post-condition of loop");
60 
61  auto new_member_expr = expr;
62  new_member_expr.struct_op() = new_operands.front();
63  // do this recursively
64  return changed(simplify_member(new_member_expr));
65  }
66  else if(op_type.id()==ID_union)
67  {
68  const with_exprt &with_expr=to_with_expr(op);
69 
70  if(with_expr.where().get(ID_component_name)==component_name)
71  {
72  // WITH(s, .m, v).m -> v
73  return with_expr.new_value();
74  }
75  }
76  }
77  else if(op.id()==ID_update)
78  {
79  if(op.operands().size()==3 &&
80  op_type.id()==ID_struct)
81  {
82  const update_exprt &update_expr=to_update_expr(op);
83  const exprt::operandst &designator=update_expr.designator();
84 
85  // look at very first designator
86  if(designator.size()==1 &&
87  designator.front().id()==ID_member_designator)
88  {
89  if(designator.front().get(ID_component_name)==component_name)
90  {
91  // UPDATE(s, .m, v).m -> v
92  return update_expr.new_value();
93  }
94  // the following optimization only works on structs,
95  // and not on unions
96  else if(op_type.id()==ID_struct)
97  {
98  // UPDATE(s, .m1, v).m2 -> s.m2
99  auto new_expr = expr;
100  new_expr.struct_op() = update_expr.old();
101 
102  // do this recursively
103  return changed(simplify_member(new_expr));
104  }
105  }
106  }
107  }
108  else if(op.id() == ID_struct)
109  {
110  // pull out the right member
111  const struct_typet &struct_type = to_struct_type(op_type);
112  if(struct_type.has_component(component_name))
113  {
114  std::size_t number = struct_type.component_number(component_name);
115  DATA_INVARIANT(
116  op.operands().size() > number,
117  "struct expression must have sufficiently many operands");
118  DATA_INVARIANT(
119  op.operands()[number].type() == expr.type(),
120  "member expression type must match component type");
121  return op.operands()[number];
122  }
123  }
124  else if(op.id()==ID_byte_extract_little_endian ||
125  op.id()==ID_byte_extract_big_endian)
126  {
127  const auto &byte_extract_expr = to_byte_extract_expr(op);
128 
129  if(op_type.id()==ID_struct)
130  {
131  // This rewrites byte_extract(s, o, struct_type).member
132  // to byte_extract(s, o+member_offset, member_type)
133 
134  const struct_typet &struct_type=to_struct_type(op_type);
135  const struct_typet::componentt &component=
136  struct_type.get_component(component_name);
137 
138  if(
139  component.is_nil() || component.type().id() == ID_c_bit_field ||
140  component.is_boolean())
141  {
142  return unchanged(expr);
143  }
144 
145  // add member offset to index
146  auto offset_int = member_offset(struct_type, component_name, ns);
147  if(!offset_int.has_value())
148  return unchanged(expr);
149 
150  const exprt &struct_offset = byte_extract_expr.offset();
151  exprt member_offset = from_integer(*offset_int, struct_offset.type());
152  exprt final_offset =
153  simplify_plus(plus_exprt(struct_offset, member_offset));
154 
155  byte_extract_exprt result(
156  op.id(),
157  byte_extract_expr.op(),
158  final_offset,
159  byte_extract_expr.get_bits_per_byte(),
160  expr.type());
161 
162  return changed(simplify_byte_extract(result)); // recursive call
163  }
164  else if(op_type.id() == ID_union)
165  {
166  // rewrite byte_extract(X, 0).member to X
167  // if the type of X is that of the member
168  if(byte_extract_expr.offset().is_zero())
169  {
170  const union_typet &union_type = to_union_type(op_type);
171  const typet &subtype = union_type.component_type(component_name);
172 
173  if(subtype == byte_extract_expr.op().type())
174  return byte_extract_expr.op();
175  }
176  }
177  }
178  else if(op.id()==ID_union && op_type.id()==ID_union)
179  {
180  // trivial?
181  if(to_union_expr(op).op().type() == expr.type())
182  return to_union_expr(op).op();
183 
184  // need to convert!
185  auto target_size = pointer_offset_size(expr.type(), ns);
186 
187  if(target_size.has_value())
188  {
189  mp_integer target_bits = target_size.value() * config.ansi_c.char_width;
190  const auto bits = expr2bits(op, true, ns);
191 
192  if(bits.has_value() &&
193  mp_integer(bits->size())>=target_bits)
194  {
195  std::string bits_cut =
196  std::string(*bits, 0, numeric_cast_v<std::size_t>(target_bits));
197 
198  auto tmp = bits2expr(bits_cut, expr.type(), true, ns);
199 
200  if(tmp.has_value())
201  return std::move(*tmp);
202  }
203  }
204  }
205  else if(op.id() == ID_typecast)
206  {
207  const auto &typecast_expr = to_typecast_expr(op);
208 
209  // Try to look through member(cast(x)) if the cast is between structurally
210  // identical types:
211  if(op_type == typecast_expr.op().type())
212  {
213  auto new_expr = expr;
214  new_expr.struct_op() = typecast_expr.op();
215  return changed(simplify_member(new_expr));
216  }
217 
218  // Try to translate into an equivalent member (perhaps nested) of the type
219  // being cast (for example, this might turn ((struct A)x).field1 into
220  // x.substruct.othersubstruct.field2, if field1 and field2 have the same
221  // type and offset with respect to x.
222  if(op_type.id() == ID_struct)
223  {
224  std::optional<mp_integer> requested_offset =
225  member_offset(to_struct_type(op_type), component_name, ns);
226  if(requested_offset.has_value())
227  {
228  auto equivalent_member = get_subexpression_at_offset(
229  typecast_expr.op(), *requested_offset, expr.type(), ns);
230 
231  // Guess: turning this into a byte-extract operation is not really an
232  // optimisation.
233  if(
234  equivalent_member.has_value() &&
235  equivalent_member.value().id() != ID_byte_extract_little_endian &&
236  equivalent_member.value().id() != ID_byte_extract_big_endian)
237  {
238  auto tmp = equivalent_member.value();
239  return changed(simplify_rec(tmp));
240  }
241  }
242  }
243  }
244  else if(op.id() == ID_let)
245  {
246  // Push a member operator inside a let binding, to avoid the let bisecting
247  // structures we otherwise know how to analyse, such as
248  // (let x = 1 in ({x, x})).field1 --> let x = 1 in ({x, x}.field1) -->
249  // let x = 1 in x
250  member_exprt pushed_in_member = to_member_expr(expr);
251  pushed_in_member.op() = to_let_expr(op).where();
252  auto new_expr = op;
253  to_let_expr(new_expr).where() = pushed_in_member;
254  to_let_expr(new_expr).type() = to_let_expr(new_expr).where().type();
255 
256  return changed(simplify_rec(new_expr));
257  }
258 
259  return unchanged(expr);
260 }

262 simplify_exprt::resultt<>
263 simplify_exprt::simplify_member_preorder(const member_exprt &expr)
264 {
265  const exprt &compound = expr.compound();
266 
267  if(compound.id() == ID_if)
268  {
269  if_exprt if_expr = lift_if(expr, 0);
270  return changed(simplify_if_preorder(if_expr));
271  }
272  else
273  {
274  auto r_it = simplify_rec(compound); // recursive call
275  if(r_it.has_changed())
276  {
277  auto tmp = expr;
278  tmp.compound() = r_it.expr;
279  return tmp;
280  }
281  }
282 
283  return unchanged(expr);
284 }
