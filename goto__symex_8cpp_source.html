40 void goto_symext::symex_assign(
41  statet &state,
42  const exprt &o_lhs,
43  const exprt &o_rhs)
44 {
45  exprt lhs = clean_expr(o_lhs, state, true);
46  exprt rhs = clean_expr(o_rhs, state, false);
47 
48  DATA_INVARIANT_WITH_DIAGNOSTICS(
49  lhs.type() == rhs.type(),
50  "assignments must be type consistent, got",
51  lhs.type().pretty(),
52  rhs.type().pretty(),
53  state.source.pc->source_location());
54 
55  log.conditional_output(
56  log.debug(), [this, &lhs](messaget::mstreamt &mstream) {
57  mstream << "Assignment to " << format(lhs) << " ["
58  << pointer_offset_bits(lhs.type(), ns).value_or(0) << " bits]"
59  << messaget::eom;
60  });
61 
62  // rvalues present within the lhs (for example, "some_array[this_rvalue]" or
63  // "byte_extract <type> from an_lvalue offset this_rvalue") can affect whether
64  // we use field-sensitive symbols or not, so L2-rename them up front:
65  lhs = state.l2_rename_rvalues(lhs, ns);
66  do_simplify(lhs);
67  lhs = state.field_sensitivity.apply(ns, state, std::move(lhs), true);
68 
69  if(rhs.id() == ID_side_effect)
70  {
71  const side_effect_exprt &side_effect_expr = to_side_effect_expr(rhs);
72  const irep_idt &statement = side_effect_expr.get_statement();
73 
74  if(
75  statement == ID_cpp_new || statement == ID_cpp_new_array ||
76  statement == ID_java_new_array_data)
77  {
78  symex_cpp_new(state, lhs, side_effect_expr);
79  }
80  else if(statement == ID_allocate)
81  symex_allocate(state, lhs, side_effect_expr);
82  else if(statement == ID_va_start)
83  symex_va_start(state, lhs, side_effect_expr);
84  else
85  UNREACHABLE;
86  }
87  else
88  {
89  assignment_typet assignment_type = symex_targett::assignment_typet::STATE;
90 
91  // Let's hide return value assignments.
92  if(
93  lhs.id() == ID_symbol &&
94  id2string(to_symbol_expr(lhs).get_identifier()).find("#return_value!") !=
95  std::string::npos)
96  {
97  assignment_type = symex_targett::assignment_typet::HIDDEN;
98  }
99 
100  // We hide if we are in a hidden function.
101  if(state.call_stack().top().hidden_function)
102  assignment_type = symex_targett::assignment_typet::HIDDEN;
103 
104  // We hide if we are executing a hidden instruction.
105  if(state.source.pc->source_location().get_hide())
106  assignment_type = symex_targett::assignment_typet::HIDDEN;
107 
108  symex_assignt symex_assign{
109  state, assignment_type, ns, symex_config, target};
110 
111  // Try to constant propagate potential side effects of the assignment, when
112  // simplification is turned on and there is one thread only. Constant
113  // propagation is only sound for sequential code as here we do not take into
114  // account potential writes from other threads when propagating the values.
115  if(symex_config.simplify_opt && state.threads.size() <= 1)
116  {
117  if(constant_propagate_assignment_with_side_effects(
118  state, symex_assign, lhs, rhs))
119  return;
120  }
121 
122  exprt::operandst lhs_if_then_else_conditions;
123  symex_assign.assign_rec(
124  lhs, expr_skeletont{}, rhs, lhs_if_then_else_conditions);
125  }
126 }

135 static std::string get_alnum_string(const array_exprt &char_array)
136 {
137  const auto &ibv_type =
138  to_integer_bitvector_type(to_array_type(char_array.type()).element_type());
139 
140  const std::size_t n_bits = ibv_type.get_width();
141  CHECK_RETURN(n_bits % 8 == 0);
142 
143  static_assert(CHAR_BIT == 8, "bitwidth of char assumed to be 8");
144 
145  const std::size_t n_chars = n_bits / 8;
146 
147  INVARIANT(
148  sizeof(std::size_t) >= n_chars,
149  "size_t shall be large enough to represent a character");
150 
151  std::string result;
152 
153  for(const auto &c : char_array.operands())
154  {
155  const std::size_t c_val = numeric_cast_v<std::size_t>(to_constant_expr(c));
156 
157  for(std::size_t i = 0; i < n_chars; i++)
158  {
159  const char c_chunk = static_cast<char>((c_val >> (i * 8)) & 0xff);
160  result.push_back(c_chunk);
161  }
162  }
163 
164  return escape_non_alnum(result);
165 }

167 bool goto_symext::constant_propagate_assignment_with_side_effects(
168  statet &state,
169  symex_assignt &symex_assign,
170  const exprt &lhs,
171  const exprt &rhs)
172 {
173  if(rhs.id() == ID_function_application)
174  {
175  const function_application_exprt &f_l1 = to_function_application_expr(rhs);
176 
177  if(f_l1.function().id() == ID_symbol)
178  {
179  const irep_idt &func_id =
180  to_symbol_expr(f_l1.function()).get_identifier();
181 
182  if(func_id == ID_cprover_string_concat_func)
183  {
184  return constant_propagate_string_concat(state, symex_assign, f_l1);
185  }
186  else if(func_id == ID_cprover_string_empty_string_func)
187  {
188  // constant propagating the empty string always succeeds as it does
189  // not depend on potentially non-constant inputs
190  constant_propagate_empty_string(state, symex_assign, f_l1);
191  return true;
192  }
193  else if(func_id == ID_cprover_string_substring_func)
194  {
195  return constant_propagate_string_substring(state, symex_assign, f_l1);
196  }
197  else if(
198  func_id == ID_cprover_string_of_int_func ||
199  func_id == ID_cprover_string_of_long_func)
200  {
201  return constant_propagate_integer_to_string(state, symex_assign, f_l1);
202  }
203  else if(func_id == ID_cprover_string_delete_char_at_func)
204  {
205  return constant_propagate_delete_char_at(state, symex_assign, f_l1);
206  }
207  else if(func_id == ID_cprover_string_delete_func)
208  {
209  return constant_propagate_delete(state, symex_assign, f_l1);
210  }
211  else if(func_id == ID_cprover_string_set_length_func)
212  {
213  return constant_propagate_set_length(state, symex_assign, f_l1);
214  }
215  else if(func_id == ID_cprover_string_char_set_func)
216  {
217  return constant_propagate_set_char_at(state, symex_assign, f_l1);
218  }
219  else if(func_id == ID_cprover_string_trim_func)
220  {
221  return constant_propagate_trim(state, symex_assign, f_l1);
222  }
223  else if(func_id == ID_cprover_string_to_lower_case_func)
224  {
225  return constant_propagate_case_change(state, symex_assign, f_l1, false);
226  }
227  else if(func_id == ID_cprover_string_to_upper_case_func)
228  {
229  return constant_propagate_case_change(state, symex_assign, f_l1, true);
230  }
231  else if(func_id == ID_cprover_string_replace_func)
232  {
233  return constant_propagate_replace(state, symex_assign, f_l1);
234  }
235  }
236  }
237 
238  return false;
239 }

468  statet &state,
469  symex_assignt &symex_assign,
470  const function_application_exprt &f_l1)
471 {
472  const std::size_t num_operands = f_l1.arguments().size();
473 
474  PRECONDITION(num_operands >= 4);
475  PRECONDITION(num_operands <= 5);
476 
477  const auto &f_type = f_l1.function_type();
478  const auto &length_type = f_type.domain().at(0);
479  const auto &char_type = to_pointer_type(f_type.domain().at(1)).base_type();
480 
481  const refined_string_exprt &s = to_string_expr(f_l1.arguments().at(2));
482  const auto s_data_opt = try_evaluate_constant_string(state, s.content());
483 
484  if(!s_data_opt)
485  return false;
486 
487  const array_exprt &s_data = s_data_opt->get();
488 
489  mp_integer end_index;
490 
491  if(num_operands == 5)
492  {
493  const auto end_index_expr_opt =
494  try_evaluate_constant(state, f_l1.arguments().at(4));
495 
496  if(!end_index_expr_opt)
497  {
498  return false;
499  }
500 
501  end_index =
502  numeric_cast_v<mp_integer>(to_constant_expr(*end_index_expr_opt));
503 
504  if(end_index < 0 || end_index > s_data.operands().size())
505  {
506  return false;
507  }
508  }
509  else
510  {
511  end_index = s_data.operands().size();
512  }
513 
514  const auto start_index_expr_opt =
515  try_evaluate_constant(state, f_l1.arguments().at(3));
516 
517  if(!start_index_expr_opt)
518  {
519  return false;
520  }
521 
522  const mp_integer start_index =
523  numeric_cast_v<mp_integer>(to_constant_expr(*start_index_expr_opt));
524 
525  if(start_index < 0 || start_index > end_index)
526  {
527  return false;
528  }
529 
530  const constant_exprt new_char_array_length =
531  from_integer(end_index - start_index, length_type);
532 
533  const array_typet new_char_array_type(char_type, new_char_array_length);
534 
535  exprt::operandst operands(
536  std::next(
537  s_data.operands().begin(), numeric_cast_v<std::size_t>(start_index)),
538  std::next(
539  s_data.operands().begin(), numeric_cast_v<std::size_t>(end_index)));
540 
541  const array_exprt new_char_array(std::move(operands), new_char_array_type);
542 
543  assign_string_constant(
544  state,
545  symex_assign,
546  to_ssa_expr(f_l1.arguments().at(0)),
547  new_char_array_length,
548  to_ssa_expr(f_l1.arguments().at(1)),
549  new_char_array);
550 
551  return true;
552 }

554 bool goto_symext::constant_propagate_integer_to_string(
555  statet &state,
556  symex_assignt &symex_assign,
557  const function_application_exprt &f_l1)
558 {
559  // The function application expression f_l1 takes the following arguments:
560  // - result string length (output parameter)
561  // - result string data array (output parameter)
562  // - integer to convert to a string
563  // - radix (optional, default 10)
564  const std::size_t num_operands = f_l1.arguments().size();
565 
566  PRECONDITION(num_operands >= 3);
567  PRECONDITION(num_operands <= 4);
568 
569  const auto &f_type = f_l1.function_type();
570  const auto &length_type = f_type.domain().at(0);
571  const auto &char_type = to_pointer_type(f_type.domain().at(1)).base_type();
572 
573  const auto &integer_opt =
574  try_evaluate_constant(state, f_l1.arguments().at(2));
575 
576  if(!integer_opt)
577  {
578  return false;
579  }
580 
581  const mp_integer integer = numeric_cast_v<mp_integer>(integer_opt->get());
582 
583  unsigned base = 10;
584 
585  if(num_operands == 4)
586  {
587  const auto &base_constant_opt =
588  try_evaluate_constant(state, f_l1.arguments().at(3));
589 
590  if(!base_constant_opt)
591  {
592  return false;
593  }
594 
595  const auto base_opt = numeric_cast<unsigned>(base_constant_opt->get());
596 
597  if(!base_opt)
598  {
599  return false;
600  }
601 
602  base = *base_opt;
603  }
604 
605  std::string s = integer2string(integer, base);
606 
607  const constant_exprt new_char_array_length =
608  from_integer(s.length(), length_type);
609 
610  const array_typet new_char_array_type(char_type, new_char_array_length);
611 
612  exprt::operandst operands;
613 
614  std::transform(
615  s.begin(),
616  s.end(),
617  std::back_inserter(operands),
618  [&char_type](const char c) { return from_integer(tolower(c), char_type); });
619 
620  const array_exprt new_char_array(std::move(operands), new_char_array_type);
621 
622  assign_string_constant(
623  state,
624  symex_assign,
625  to_ssa_expr(f_l1.arguments().at(0)),
626  new_char_array_length,
627  to_ssa_expr(f_l1.arguments().at(1)),
628  new_char_array);
629 
630  return true;
631 }

633 bool goto_symext::constant_propagate_delete_char_at(
634  statet &state,
635  symex_assignt &symex_assign,
636  const function_application_exprt &f_l1)
637 {
638  // The function application expression f_l1 takes the following arguments:
639  // - result string length (output parameter)
640  // - result string data array (output parameter)
641  // - string to delete char from
642  // - index of char to delete
643  PRECONDITION(f_l1.arguments().size() == 4);
644 
645  const auto &f_type = f_l1.function_type();
646  const auto &length_type = f_type.domain().at(0);
647  const auto &char_type = to_pointer_type(f_type.domain().at(1)).base_type();
648 
649  const refined_string_exprt &s = to_string_expr(f_l1.arguments().at(2));
650  const auto s_data_opt = try_evaluate_constant_string(state, s.content());
651 
652  if(!s_data_opt)
653  {
654  return false;
655  }
656 
657  const array_exprt &s_data = s_data_opt->get();
658 
659  const auto &index_opt = try_evaluate_constant(state, f_l1.arguments().at(3));
660 
661  if(!index_opt)
662  {
663  return false;
664  }
665 
666  const mp_integer index = numeric_cast_v<mp_integer>(index_opt->get());
667 
668  if(index < 0 || index >= s_data.operands().size())
