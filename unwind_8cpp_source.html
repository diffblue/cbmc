26 void goto_unwindt::copy_segment(
27  const goto_programt::const_targett start,
28  const goto_programt::const_targett end, // exclusive
29  goto_programt &goto_program) // result
30 {
31  assert(start->location_number<end->location_number);
32  assert(goto_program.empty());
33 
34  // build map for branch targets inside the loop
35  typedef std::map<goto_programt::const_targett, unsigned> target_mapt;
36  target_mapt target_map;
37 
38  unsigned i=0;
39 
40  for(goto_programt::const_targett t=start; t!=end; t++, i++)
41  target_map[t]=i;
42 
43  // make a copy
44  std::vector<goto_programt::targett> target_vector;
45  target_vector.reserve(target_map.size());
46  assert(target_vector.empty());
47 
48  for(goto_programt::const_targett t=start; t!=end; t++)
49  {
50  // copy the instruction
51  goto_programt::targett t_new =
52  goto_program.add(goto_programt::instructiont(*t));
53  unwind_log.insert(t_new, t->location_number);
54  target_vector.push_back(t_new); // store copied instruction
55  }
56 
57  assert(goto_program.instructions.size()==target_vector.size());
58 
59  // adjust intra-segment gotos
60  for(std::size_t target_index = 0; target_index < target_vector.size();
61  target_index++)
62  {
63  goto_programt::targett t = target_vector[target_index];
64 
65  if(!t->is_goto())
66  continue;
67 
68  goto_programt::const_targett tgt=t->get_target();
69 
70  target_mapt::const_iterator m_it=target_map.find(tgt);
71 
72  if(m_it!=target_map.end())
73  {
74  unsigned j=m_it->second;
75 
76  assert(j<target_vector.size());
77  t->set_target(target_vector[j]);
78  }
79  }
80 }

101 void goto_unwindt::unwind(
102  const irep_idt &function_id,
103  goto_programt &goto_program,
104  const goto_programt::const_targett loop_head,
105  const goto_programt::const_targett loop_exit,
106  const unsigned k,
107  const unwind_strategyt unwind_strategy,
108  std::vector<goto_programt::targett> &iteration_points)
109 {
110  assert(iteration_points.empty());
111  assert(loop_head->location_number<loop_exit->location_number);
112 
113  // rest program after unwound part
114  goto_programt rest_program;
115 
116  if(unwind_strategy==unwind_strategyt::PARTIAL)
117  {
118  goto_programt::targett t =
119  rest_program.add(goto_programt::make_skip(loop_head->source_location()));
120 
121  t->location_number=loop_head->location_number;
122 
123  unwind_log.insert(t, loop_head->location_number);
124  }
125  else if(unwind_strategy==unwind_strategyt::CONTINUE)
126  {
127  copy_segment(loop_head, loop_exit, rest_program);
128  }
129  else
130  {
131  PRECONDITION(
132  unwind_strategy == unwind_strategyt::ASSERT_ASSUME ||
133  unwind_strategy == unwind_strategyt::ASSERT_PARTIAL ||
134  unwind_strategy == unwind_strategyt::ASSUME);
135 
136  goto_programt::const_targett t=loop_exit;
137  t--;
138  assert(t->is_backwards_goto());
139 
140  exprt exit_cond = false_exprt(); // default is false
141 
142  if(!t->condition().is_true()) // cond in backedge
143  {
144  exit_cond = boolean_negate(t->condition());
145  }
146  else if(loop_head->is_goto())
147  {
148  if(loop_head->get_target()==loop_exit) // cond in forward edge
149  exit_cond = loop_head->condition();
150  }
151 
152  if(
153  unwind_strategy == unwind_strategyt::ASSERT_ASSUME ||
154  unwind_strategy == unwind_strategyt::ASSERT_PARTIAL)
155  {
156  const std::string loop_number = std::to_string(t->loop_number);
157  source_locationt source_location_annotated = loop_head->source_location();
158  source_location_annotated.set_property_class("unwind");
159  source_location_annotated.set_comment(
160  "unwinding assertion loop " + loop_number);
161  goto_programt::targett assertion = rest_program.add(
162  goto_programt::make_assertion(exit_cond, source_location_annotated));
163  unwind_log.insert(assertion, loop_head->location_number);
164  }
165 
166  if(
167  unwind_strategy == unwind_strategyt::ASSUME ||
168  unwind_strategy == unwind_strategyt::ASSERT_ASSUME)
169  {
170  goto_programt::targett assumption =
171  rest_program.add(goto_programt::make_assumption(
172  exit_cond, loop_head->source_location()));
173  unwind_log.insert(assumption, loop_head->location_number);
174  }
175 
176  }
177 
178  assert(!rest_program.empty());
179 
180  // to be filled with copies of the loop body
181  goto_programt copies;
182 
183  if(k!=0)
184  {
185  iteration_points.resize(k);
186 
187  // add a goto before the loop exit to guard against 'fall-out'
188 
189  goto_programt::const_targett t_before=loop_exit;
190  t_before--;
191 
192  if(!t_before->is_goto() || !t_before->condition().is_true())
193  {
194  goto_programt::targett t_goto = goto_program.insert_before(
195  loop_exit,
196  goto_programt::make_goto(
197  goto_program.const_cast_target(loop_exit),
198  true_exprt(),
199  loop_exit->source_location()));
200  t_goto->location_number=loop_exit->location_number;
201 
202  unwind_log.insert(t_goto, loop_exit->location_number);
203  }
204 
205  // add a skip before the loop exit
206 
207  goto_programt::targett t_skip = goto_program.insert_before(
208  loop_exit, goto_programt::make_skip(loop_head->source_location()));
209  t_skip->location_number=loop_head->location_number;
210 
211  unwind_log.insert(t_skip, loop_exit->location_number);
212 
213  // where to go for the next iteration
214  goto_programt::targett loop_iter=t_skip;
215 
216  // record the exit point of first iteration
217  iteration_points[0]=loop_iter;
218 
219  // re-direct any branches that go to loop_head to loop_iter
220 
221  for(goto_programt::targett
222  t=goto_program.const_cast_target(loop_head);
223  t!=loop_iter; t++)
224  {
225  if(!t->is_goto())
226  continue;
227 
228  if(t->get_target()==loop_head)
229  t->set_target(loop_iter);
230  }
231 
232  // k-1 additional copies
233  for(unsigned i=1; i<k; i++)
234  {
235  goto_programt tmp_program;
236  copy_segment(loop_head, loop_exit, tmp_program);
237  assert(!tmp_program.instructions.empty());
238 
239  iteration_points[i]=--tmp_program.instructions.end();
240 
241  copies.destructive_append(tmp_program);
242  }
243  }
244  else
245  {
246  // insert skip for loop body
247 
248  goto_programt::targett t_skip = goto_program.insert_before(
249  loop_head, goto_programt::make_skip(loop_head->source_location()));
250  t_skip->location_number=loop_head->location_number;
251 
252  unwind_log.insert(t_skip, loop_head->location_number);
253 
254  // redirect gotos into loop body
255  for(auto &instruction : goto_program.instructions)
256  {
257  if(!instruction.is_goto())
258  continue;
259 
260  goto_programt::const_targett t = instruction.get_target();
261 
262  if(t->location_number>=loop_head->location_number &&
263  t->location_number<loop_exit->location_number)
264  {
265  instruction.set_target(t_skip);
266  }
267  }
268 
269  // delete loop body
270  goto_program.instructions.erase(loop_head, loop_exit);
271  }
272 
273  // after unwound part
274  copies.destructive_append(rest_program);
275 
276  // now insert copies before loop_exit
277  goto_program.destructive_insert(loop_exit, copies);
278 }

280 void goto_unwindt::unwind(
281  const irep_idt &function_id,
282  goto_programt &goto_program,
283  const unwindsett &unwindset,
284  const unwind_strategyt unwind_strategy)
285 {
286  for(goto_programt::const_targett i_it=goto_program.instructions.begin();
287  i_it!=goto_program.instructions.end();)
288  {
289 #ifdef DEBUG
290  std::cout << "Instruction:\n";
291  i_it->output(std::cout);
292 #endif
293 
294  if(!i_it->is_backwards_goto())
295  {
296  i_it++;
297  continue;
298  }
299 
300  PRECONDITION(!function_id.empty());
301  const irep_idt loop_id = goto_programt::loop_id(function_id, *i_it);
302 
303  auto limit=unwindset.get_limit(loop_id, 0);
304 
305  if(!limit.has_value())
306  {
307  // no unwinding given
308  i_it++;
309  continue;
310  }
311 
312  goto_programt::const_targett loop_head=i_it->get_target();
313  goto_programt::const_targett loop_exit=i_it;
314  loop_exit++;
315  assert(loop_exit!=goto_program.instructions.end());
316 
317  unwind(
318  function_id, goto_program, loop_head, loop_exit, *limit, unwind_strategy);
319 
320  // necessary as we change the goto program in the previous line
321  i_it=loop_exit;
322  }
323 }
