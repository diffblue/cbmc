155 void smt2_incremental_decision_proceduret::define_dependent_functions(
156  const exprt &expr)
157 {
158  std::unordered_set<exprt, irep_hash> seen_expressions =
159  make_range(expression_identifiers)
160  .map([](const std::pair<exprt, smt_identifier_termt> &expr_identifier) {
161  return expr_identifier.first;
162  });
163  std::stack<exprt> to_be_defined;
164  const auto push_dependencies_needed = [&](const exprt &expr) {
165  bool result = false;
166  for(const auto &dependency : gather_dependent_expressions(expr))
167  {
168  if(!seen_expressions.insert(dependency).second)
169  continue;
170  result = true;
171  to_be_defined.push(dependency);
172  }
173  return result;
174  };
175  push_dependencies_needed(expr);
176  while(!to_be_defined.empty())
177  {
178  const exprt current = to_be_defined.top();
179  if(push_dependencies_needed(current))
180  continue;
181  if(const auto symbol_expr = expr_try_dynamic_cast<symbol_exprt>(current))
182  {
183  const irep_idt &identifier = symbol_expr->get_identifier();
184  const symbolt *symbol = nullptr;
185  if(ns.lookup(identifier, symbol) || symbol->value.is_nil())
186  {
187  send_function_definition(
188  *symbol_expr,
189  symbol_expr->get_identifier(),
190  solver_process,
191  expression_identifiers,
192  identifier_table);
193  }
194  else
195  {
196  if(push_dependencies_needed(symbol->value))
197  continue;
198  const smt_define_function_commandt function{
199  symbol->name, {}, convert_expr_to_smt(symbol->value)};
200  expression_identifiers.emplace(*symbol_expr, function.identifier());
201  identifier_table.emplace(identifier, function.identifier());
202  solver_process->send(function);
203  }
204  }
205  else if(const auto array_expr = expr_try_dynamic_cast<array_exprt>(current))
206  define_array_function(*array_expr);
207  else if(
208  const auto array_of_expr = expr_try_dynamic_cast<array_of_exprt>(current))
209  {
210  define_array_function(*array_of_expr);
211  }
212  else if(
213  const auto nondet_symbol =
214  expr_try_dynamic_cast<nondet_symbol_exprt>(current))
215  {
216  send_function_definition(
217  *nondet_symbol,
218  nondet_symbol->get_identifier(),
219  solver_process,
220  expression_identifiers,
221  identifier_table);
222  }
223  to_be_defined.pop();
224  }
225 }

414 exprt smt2_incremental_decision_proceduret::get(const exprt &expr) const
415 {
416  log.conditional_output(log.debug(), [&](messaget::mstreamt &debug) {
417  debug << "`get` - \n " + expr.pretty(2, 0) << messaget::eom;
418  });
419  auto descriptor = [&]() -> optionalt<smt_termt> {
420  if(const auto index_expr = expr_try_dynamic_cast<index_exprt>(expr))
421  {
422  const auto array = get_identifier(
423  index_expr->array(),
424  expression_handle_identifiers,
425  expression_identifiers);
426  const auto index = get_identifier(
427  index_expr->index(),
428  expression_handle_identifiers,
429  expression_identifiers);
430  if(!array || !index)
431  return {};
432  return smt_array_theoryt::select(*array, *index);
433  }
434  return get_identifier(
435  expr, expression_handle_identifiers, expression_identifiers);
436  }();
437  if(!descriptor)
438  {
439  if(gather_dependent_expressions(expr).empty())
440  {
441  INVARIANT(
442  objects_are_already_tracked(expr, object_map),
443  "Objects in expressions being read should already be tracked from "
444  "point of being set/handled.");
445  descriptor = ::convert_expr_to_smt(
446  expr,
447  object_map,
448  pointer_sizes_map,
449  object_size_function.make_application,
450  is_dynamic_object_function.make_application);
451  }
452  else
453  {
454  const auto symbol_expr = expr_try_dynamic_cast<symbol_exprt>(expr);
455  INVARIANT(
456  symbol_expr, "Unhandled expressions are expected to be symbols");
457  // Note this case is currently expected to be encountered during trace
458  // generation for -
459  // * Steps which were removed via --slice-formula.
460  // * Getting concurrency clock values.
461  // The below implementation which returns the given expression was chosen
462  // based on the implementation of `smt2_convt::get` in the non-incremental
463  // smt2 decision procedure.
464  log.warning()
465  << "`get` attempted for unknown symbol, with identifier - \n"
466  << symbol_expr->get_identifier() << messaget::eom;
467  return expr;
468  }
469  }
470  if(const auto array_type = type_try_dynamic_cast<array_typet>(expr.type()))
471  {
472  if(array_type->is_incomplete())
473  return expr;
474  return get_expr(*descriptor, *array_type);
475  }
476  return get_expr(*descriptor, expr.type());
