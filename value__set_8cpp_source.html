432  to_struct_union_type(followed_type);

455 void value_sett::get_value_set_rec(
456  const exprt &expr,
457  object_mapt &dest,
458  const std::string &suffix,
459  const typet &original_type,
460  const namespacet &ns) const
461 {

464  std::cout << "GET_VALUE_SET_REC SUFFIX: " << suffix << '\n';
465 #endif
466 
467  const typet &expr_type=ns.follow(expr.type());
468 
469  if(expr.id()==ID_unknown || expr.id()==ID_invalid)
470  {
471  insert(dest, exprt(ID_unknown, original_type));
472  }
473  else if(expr.id()==ID_index)
474  {
475  const typet &type = to_index_expr(expr).array().type();
476 
477  DATA_INVARIANT(
478  type.id() == ID_array, "operand 0 of index expression must be an array");
479 
480  get_value_set_rec(
481  to_index_expr(expr).array(), dest, "[]" + suffix, original_type, ns);
482  }
483  else if(expr.id()==ID_member)
484  {
485  const typet &type = ns.follow(to_member_expr(expr).compound().type());
486 
487  DATA_INVARIANT(
488  type.id() == ID_struct || type.id() == ID_union,
489  "compound of member expression must be struct or union");
490 
491  const std::string &component_name=
492  expr.get_string(ID_component_name);
493 
494  get_value_set_rec(
495  to_member_expr(expr).compound(),
496  dest,
497  "." + component_name + suffix,
498  original_type,
499  ns);
500  }
501  else if(expr.id()==ID_symbol)
502  {
503  auto entry_index = get_index_of_symbol(
504  to_symbol_expr(expr).get_identifier(), expr_type, suffix, ns);
505 
506  if(entry_index.has_value())
507  make_union(dest, find_entry(*entry_index)->object_map);
508  else
509  insert(dest, exprt(ID_unknown, original_type));
510  }
511  else if(expr.id() == ID_nondet_symbol)
512  {
513  if(expr.type().id() == ID_pointer)
514  {
515  // we'll take the union of all objects we see, with unspecified offsets
516  values.iterate([this, &dest](const irep_idt &key, const entryt &value) {
517  for(const auto &object : value.object_map.read())
518  insert(dest, object.first, offsett());
519  });
520 
521  // we'll add null, in case it's not there yet
522  insert(
523  dest,
524  exprt(ID_null_object, to_pointer_type(expr_type).base_type()),
525  offsett());
526  }
527  else
528  insert(dest, exprt(ID_unknown, original_type));
529  }
530  else if(expr.id()==ID_if)
531  {
532  get_value_set_rec(
533  to_if_expr(expr).true_case(), dest, suffix, original_type, ns);
534  get_value_set_rec(
535  to_if_expr(expr).false_case(), dest, suffix, original_type, ns);
536  }
537  else if(expr.id()==ID_address_of)
538  {
539  get_reference_set(to_address_of_expr(expr).object(), dest, ns);
540  }
541  else if(expr.id()==ID_dereference)
542  {
543  object_mapt reference_set;
544  get_reference_set(expr, reference_set, ns);
545  const object_map_dt &object_map=reference_set.read();
546

548  insert(dest, exprt(ID_unknown, original_type));

552  it1=object_map.begin();
553  it1!=object_map.end();
554  it1++)

558  const exprt object=object_numbering[it1->first];

560  }
561  }
562  }
563  else if(expr.is_constant())
564  {
565  // check if NULL
566  if(is_null_pointer(to_constant_expr(expr)))
567  {
568  insert(

581  }

585

587  const typet &op_type = op.type();
588

628  }
629  else

634  expr.id() == ID_bitand || expr.id() == ID_bitxor ||

756  it!=pointer_expr_set.read().end();
757  it++)
758  {
759  offsett offset = it->second;
760 
761  // kill any offset
762  offset.reset();
763 
764  insert(dest, it->first, offset);
765  }
766  }
767  else if(expr.id() == ID_lshr || expr.id() == ID_ashr || expr.id() == ID_shl)
768  {
769  const binary_exprt &binary_expr = to_binary_expr(expr);
770 
771  object_mapt pointer_expr_set;
772  get_value_set_rec(
773  binary_expr.op0(), pointer_expr_set, "", binary_expr.op0().type(), ns);
774 
775  for(const auto &object_map_entry : pointer_expr_set.read())
776  {
777  offsett offset = object_map_entry.second;
778 
779  // kill any offset
780  offset.reset();
781 
782  insert(dest, object_map_entry.first, offset);
783  }
784  }
785  else if(expr.id()==ID_side_effect)
786  {
787  const irep_idt &statement = to_side_effect_expr(expr).get_statement();
788 
789  if(statement==ID_function_call)
790  {
791  // these should be gone
792  throw "unexpected function_call sideeffect";
793  }
794  else if(statement==ID_allocate)
795  {
796  PRECONDITION(suffix.empty());
797 
798  const typet &dynamic_type=
799  static_cast<const typet &>(expr.find(ID_C_cxx_alloc_type));
800 
801  dynamic_object_exprt dynamic_object(dynamic_type);
802  dynamic_object.set_instance(location_number);
803  dynamic_object.valid()=true_exprt();
804 
805  insert(dest, dynamic_object, mp_integer{0});
806  }
807  else if(statement==ID_cpp_new ||
808  statement==ID_cpp_new_array)
809  {
810  PRECONDITION(suffix.empty());
811  PRECONDITION(expr_type.id() == ID_pointer);
812 
813  dynamic_object_exprt dynamic_object(
814  to_pointer_type(expr_type).base_type());
815  dynamic_object.set_instance(location_number);
816  dynamic_object.valid()=true_exprt();
817 
818  insert(dest, dynamic_object, mp_integer{0});
819  }
820  else
821  insert(dest, exprt(ID_unknown, original_type));
822  }
823  else if(expr.id()==ID_struct)
824  {
825  const auto &struct_components = to_struct_type(expr_type).components();
826  INVARIANT(
827  struct_components.size() == expr.operands().size(),
828  "struct expression should have an operand per component");
829  auto component_iter = struct_components.begin();
830  bool found_component = false;
831 
832  // a struct constructor, which may contain addresses
833 
834  for(const auto &op : expr.operands())
835  {
836  const std::string &component_name =
837  id2string(component_iter->get_name());
838  if(suffix_starts_with_field(suffix, component_name))
839  {
840  std::string remaining_suffix =
841  strip_first_field_from_suffix(suffix, component_name);
842  get_value_set_rec(op, dest, remaining_suffix, original_type, ns);
843  found_component = true;
844  }
845  ++component_iter;
846  }
847 
848  if(!found_component)
849  {
850  // Struct field doesn't appear as expected -- this has probably been
851  // cast from an incompatible type. Conservatively assume all fields may
852  // be of interest.
853  for(const auto &op : expr.operands())
854  get_value_set_rec(op, dest, suffix, original_type, ns);
855  }
856  }
857  else if(expr.id() == ID_union)
858  {
859  get_value_set_rec(
860  to_union_expr(expr).op(), dest, suffix, original_type, ns);
861  }
862  else if(expr.id()==ID_with)
863  {
864  const with_exprt &with_expr = to_with_expr(expr);
865 
866  // If the suffix is empty we're looking for the whole struct:
867  // default to combining both options as below.
868  if(expr_type.id() == ID_struct && !suffix.empty())
869  {
870  irep_idt component_name = with_expr.where().get(ID_component_name);
871  if(suffix_starts_with_field(suffix, id2string(component_name)))
872  {
873  // Looking for the member overwritten by this WITH expression
874  std::string remaining_suffix =
875  strip_first_field_from_suffix(suffix, id2string(component_name));
876  get_value_set_rec(
877  with_expr.new_value(), dest, remaining_suffix, original_type, ns);
878  }
879  else if(to_struct_type(expr_type).has_component(component_name))
880  {
881  // Looking for a non-overwritten member, look through this expression
882  get_value_set_rec(with_expr.old(), dest, suffix, original_type, ns);
883  }
884  else
885  {
886  // Member we're looking for is not defined in this struct -- this
887  // must be a reinterpret cast of some sort. Default to conservatively
888  // assuming either operand might be involved.
889  get_value_set_rec(with_expr.old(), dest, suffix, original_type, ns);
890  get_value_set_rec(with_expr.new_value(), dest, "", original_type, ns);
891  }
892  }
893  else if(expr_type.id() == ID_array && !suffix.empty())
894  {
895  std::string new_value_suffix;
896  if(has_prefix(suffix, "[]"))
897  new_value_suffix = suffix.substr(2);
898 
899  // Otherwise use a blank suffix on the assumption anything involved with
900  // the new value might be interesting.
901 
902  get_value_set_rec(with_expr.old(), dest, suffix, original_type, ns);
903  get_value_set_rec(
904  with_expr.new_value(), dest, new_value_suffix, original_type, ns);
905  }
906  else
907  {
908  // Something else-- the suffixes used here are a rough guess at best,
909  // so this is imprecise.
910  get_value_set_rec(with_expr.old(), dest, suffix, original_type, ns);
911  get_value_set_rec(with_expr.new_value(), dest, "", original_type, ns);
912  }
913  }
914  else if(expr.id()==ID_array)
915  {
916  // an array constructor, possibly containing addresses
917  std::string new_suffix = suffix;
918  if(has_prefix(suffix, "[]"))
919  new_suffix = suffix.substr(2);
920  // Otherwise we're probably reinterpreting some other type -- try persisting
921  // with the current suffix for want of a better idea.
922 
923  for(const auto &op : expr.operands())
924  get_value_set_rec(op, dest, new_suffix, original_type, ns);
925  }
926  else if(expr.id()==ID_array_of)
927  {
928  // an array constructor, possibly containing an address
929  std::string new_suffix = suffix;
930  if(has_prefix(suffix, "[]"))
931  new_suffix = suffix.substr(2);
932  // Otherwise we're probably reinterpreting some other type -- try persisting
933  // with the current suffix for want of a better idea.
934 
935  get_value_set_rec(
936  to_array_of_expr(expr).what(), dest, new_suffix, original_type, ns);
937  }
938  else if(expr.id()==ID_dynamic_object)
939  {
940  const dynamic_object_exprt &dynamic_object=
941  to_dynamic_object_expr(expr);
942 
943  const std::string prefix=
944  "value_set::dynamic_object"+
945  std::to_string(dynamic_object.get_instance());
946 
947  // first try with suffix
948  const std::string full_name=prefix+suffix;
949 
950  // look it up
951  const entryt *entry = find_entry(full_name);
952 
953  // not found? try without suffix
954  if(!entry)
955  entry = find_entry(prefix);
956 
957  if(!entry)
958  insert(dest, exprt(ID_unknown, original_type));
959  else
960  make_union(dest, entry->object_map);
961  }
962  else if(expr.id()==ID_byte_extract_little_endian ||
963  expr.id()==ID_byte_extract_big_endian)
964  {
965  const auto &byte_extract_expr = to_byte_extract_expr(expr);
966 
967  bool found=false;
968 
969  const typet &op_type = ns.follow(byte_extract_expr.op().type());
970  if(op_type.id() == ID_struct)
971  {
972  exprt offset = byte_extract_expr.offset();
973  if(eval_pointer_offset(offset, ns))
974  simplify(offset, ns);
975 
976  const auto offset_int = numeric_cast<mp_integer>(offset);
977  const auto type_size = pointer_offset_size(expr.type(), ns);
978 
979  const struct_typet &struct_type = to_struct_type(op_type);
980 
981  for(const auto &c : struct_type.components())
982  {
983  const irep_idt &name = c.get_name();
984 
985  if(offset_int.has_value())
986  {
987  auto comp_offset = member_offset(struct_type, name, ns);
988  if(comp_offset.has_value())
989  {
990  if(
991  type_size.has_value() && *offset_int + *type_size <= *comp_offset)
992  {
993  break;
994  }
995 
996  auto member_size = pointer_offset_size(c.type(), ns);
997  if(
998  member_size.has_value() &&
999  *offset_int >= *comp_offset + *member_size)
1000  {
1001  continue;
1002  }
1003  }
1004  }
1005 
1006  found=true;
1007 
1008  member_exprt member(byte_extract_expr.op(), c);
1009  get_value_set_rec(member, dest, suffix, original_type, ns);
1010  }
1011  }
1012 
1013  if(op_type.id() == ID_union)
1014  {
1015  // just collect them all
1016  for(const auto &c : to_union_type(op_type).components())
1017  {
1018  const irep_idt &name = c.get_name();
1019  member_exprt member(byte_extract_expr.op(), name, c.type());
1020  get_value_set_rec(member, dest, suffix, original_type, ns);
1021  }
1022  }
1023 
1024  if(!found)
1025  // we just pass through
1026  get_value_set_rec(
1027  byte_extract_expr.op(), dest, suffix, original_type, ns);
1028  }
1029  else if(expr.id()==ID_byte_update_little_endian ||
1030  expr.id()==ID_byte_update_big_endian)
1031  {
1032  const auto &byte_update_expr = to_byte_update_expr(expr);
1033 
1034  // we just pass through
1035  get_value_set_rec(byte_update_expr.op(), dest, suffix, original_type, ns);
1036  get_value_set_rec(
1037  byte_update_expr.value(), dest, suffix, original_type, ns);
1038 
1039  // we could have checked object size to be more precise
1040  }
1041  else if(expr.id() == ID_let)
1042  {
1043  const auto &let_expr = to_let_expr(expr);
1044  // This depends on copying `value_sett` being cheap -- which it is, because
1045  // our backing store is sharing_mapt.
1046  value_sett value_set_with_local_definition = *this;
1047  value_set_with_local_definition.assign(
1048  let_expr.symbol(), let_expr.value(), ns, false, false);
1049 
1050  value_set_with_local_definition.get_value_set_rec(
1051  let_expr.where(), dest, suffix, original_type, ns);
1052  }
1053  else if(auto eb = expr_try_dynamic_cast<extractbits_exprt>(expr))
1054  {
1055  object_mapt pointer_expr_set;
1056  get_value_set_rec(eb->src(), pointer_expr_set, "", eb->src().type(), ns);
1057 
1058  for(const auto &object_map_entry : pointer_expr_set.read())
1059  {
1060  offsett offset = object_map_entry.second;
1061 
1062  // kill any offset
1063  offset.reset();
1064 
1065  insert(dest, object_map_entry.first, offset);
1066  }
1067  }
1068  else
1069  {
1070  object_mapt pointer_expr_set;
1071  for(const auto &op : expr.operands())
1072  get_value_set_rec(op, pointer_expr_set, "", original_type, ns);
1073 
1074  for(const auto &object_map_entry : pointer_expr_set.read())
1075  {
1076  offsett offset = object_map_entry.second;
1077 
1078  // kill any offset
1079  offset.reset();
1080 
1081  insert(dest, object_map_entry.first, offset);
1082  }
1083 
1084  if(pointer_expr_set.read().empty())
1085  insert(dest, exprt(ID_unknown, original_type));
1086  }
1087 
1088  #ifdef DEBUG
1089  std::cout << "GET_VALUE_SET_REC RESULT:\n";
1090  for(const auto &obj : dest.read())
1091  {
1092  const exprt &e=to_expr(obj);
1093  std::cout << " " << format(e) << "\n";
1094  }
1095  std::cout << "\n";
1096  #endif
1097 }

1137 #endif
1138 
1139  if(expr.id()==ID_symbol ||
1140  expr.id()==ID_dynamic_object ||
1141  expr.id()==ID_string_constant ||
1142  expr.id()==ID_array)
1143  {

1146  to_array_type(expr.type()).element_type().id() == ID_array)

1150

1152  }

1160  for(const auto &map_entry : dest.read())
1161  {
1162  std::cout << "VALUE_SET: " << format(object_numbering[map_entry.first])
1163  << '\n';
1164  }
1165 #endif
1166 
1167  return;
1168  }
1169  else if(expr.id()==ID_index)
1170  {
1171  if(expr.operands().size()!=2)
1172  throw "index expected to have two operands";
1173

1175  const exprt &array=index_expr.array();
1176  const exprt &offset=index_expr.index();
1177 
1178  DATA_INVARIANT(
1179  array.type().id() == ID_array, "index takes array-typed operand");
1180

1182

1184  get_reference_set(array, array_references, ns);
1185

1189  a_it=object_map.begin();
1190  a_it!=object_map.end();
1191  a_it++)
1192  {
1193  const exprt &object=object_numbering[a_it->first];
1194 
1195  if(object.id()==ID_unknown)
1196  insert(dest, exprt(ID_unknown, expr.type()));
1197  else

1201  from_integer(0, c_index_type()));

1205

1208  }

1212

1215  else

1217  }
1218  else
1219  o.reset();
1220

1222  }
1223  }
1224 
1225  return;
1226  }

1230

1246  insert(dest, exprt(ID_unknown, expr.type()));

1248  object.type().id() == ID_struct ||
1249  object.type().id() == ID_struct_tag || object.type().id() == ID_union ||
1250  object.type().id() == ID_union_tag)

1252  offsett o = it->second;
1253

1292 void value_sett::assign(
1293  const exprt &lhs,
1294  const exprt &rhs,
1295  const namespacet &ns,
1296  bool is_simplified,
1297  bool add_to_sets)
1298 {
1299 #ifdef DEBUG
1300  std::cout << "ASSIGN LHS: " << format(lhs) << " : "
1301  << format(lhs.type()) << '\n';
1302  std::cout << "ASSIGN RHS: " << format(rhs) << " : "
1303  << format(rhs.type()) << '\n';
1304  std::cout << "--------------------------------------------\n";
1305  output(std::cout);
1306 #endif
1307 
1308  const typet &type=ns.follow(lhs.type());
1309 
1310  if(type.id() == ID_struct)
1311  {
1312  for(const auto &c : to_struct_type(type).components())
1313  {
1314  const typet &subtype = c.type();
1315  const irep_idt &name = c.get_name();
1316 
1317  // ignore padding
1318  DATA_INVARIANT(
1319  subtype.id() != ID_code, "struct member must not be of code type");
1320  if(c.get_is_padding())
1321  continue;
1322 
1323  member_exprt lhs_member(lhs, name, subtype);
1324

1326 
1327  if(rhs.id()==ID_unknown ||
1328  rhs.id()==ID_invalid)
1329  {
1330  // this branch is deemed dead as otherwise we'd be missing assignments
1331  // that never happened in this branch
1332  UNREACHABLE;
1333  rhs_member=exprt(rhs.id(), subtype);
1334  }
1335  else
1336  {
1337  DATA_INVARIANT(
1338  rhs.type() == lhs.type(),
1339  "value_sett::assign types should match, got: "
1340  "rhs.type():\n" +
1341  rhs.type().pretty() + "\n" + "lhs.type():\n" + lhs.type().pretty());
1342

1344

1346  {
1347  const struct_union_typet &rhs_struct_union_type =
1348  to_struct_union_type(followed);
1349 
1350  const typet &rhs_subtype = rhs_struct_union_type.component_type(name);
1351  rhs_member = simplify_expr(member_exprt{rhs, name, rhs_subtype}, ns);
1352 
1353  assign(lhs_member, rhs_member, ns, true, add_to_sets);
1354  }
1355  }
1356  }
1357  }
1358  else if(type.id()==ID_array)
1359  {
1360  const index_exprt lhs_index(
1361  lhs,
1362  exprt(ID_unknown, c_index_type()),
1363  to_array_type(type).element_type());
1364 
1365  if(rhs.id()==ID_unknown ||
1366  rhs.id()==ID_invalid)
1367  {
1368  assign(
1369  lhs_index,
1370  exprt(rhs.id(), to_array_type(type).element_type()),
1371  ns,
1372  is_simplified,
1373  add_to_sets);
1374  }
1375  else
1376  {
1377  DATA_INVARIANT(
1378  rhs.type() == type,
1379  "value_sett::assign types should match, got: "
1380  "rhs.type():\n" +
1381  rhs.type().pretty() + "\n" + "type:\n" + type.pretty());
1382 
1383  if(rhs.id()==ID_array_of)
1384  {
1385  assign(
1386  lhs_index,
1387  to_array_of_expr(rhs).what(),
1388  ns,
1389  is_simplified,
1390  add_to_sets);
1391  }
1392  else if(rhs.id() == ID_array || rhs.is_constant())
1393  {
1394  for(const auto &op : rhs.operands())
1395  {
1396  assign(lhs_index, op, ns, is_simplified, add_to_sets);
1397  add_to_sets=true;
1398  }
1399  }
1400  else if(rhs.id()==ID_with)
1401  {
1402  const index_exprt op0_index(
1403  to_with_expr(rhs).old(),
1404  exprt(ID_unknown, c_index_type()),
1405  to_array_type(type).element_type());
1406 
1407  assign(lhs_index, op0_index, ns, is_simplified, add_to_sets);
1408  assign(
1409  lhs_index, to_with_expr(rhs).new_value(), ns, is_simplified, true);
1410  }
1411  else
1412  {
1413  const index_exprt rhs_index(
1414  rhs,
