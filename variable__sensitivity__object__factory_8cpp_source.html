101 ABSTRACT_OBJECT_TYPET
102 variable_sensitivity_object_factoryt::get_abstract_object_type(
103  const typet &type) const
104 {
105  ABSTRACT_OBJECT_TYPET abstract_object_type = TWO_VALUE;
106 
107  if(
108  type.id() == ID_signedbv || type.id() == ID_unsignedbv ||
109  type.id() == ID_fixedbv || type.id() == ID_c_bool || type.id() == ID_bool ||
110  type.id() == ID_integer || type.id() == ID_c_bit_field)
111  {
112  return configuration.value_abstract_type;
113  }
114  else if(type.id() == ID_floatbv)
115  {
116  auto float_type = configuration.value_abstract_type;
117  return (float_type == INTERVAL) ? CONSTANT : float_type;
118  }
119  else if(type.id() == ID_array)
120  {
121  return configuration.array_abstract_type;
122  }
123  else if(type.id() == ID_pointer)
124  {
125  return configuration.pointer_abstract_type;
126  }
127  else if(type.id() == ID_struct)
128  {
129  return configuration.struct_abstract_type;
130  }
131  else if(type.id() == ID_union)
132  {
133  return configuration.union_abstract_type;
134  }
135  else if(type.id() == ID_dynamic_object)
136  {
137  return HEAP_ALLOCATION;
138  }
139 
140  return abstract_object_type;
141 }

143 abstract_object_pointert
144 variable_sensitivity_object_factoryt::get_abstract_object(
145  const typet &type,
146  bool top,
147  bool bottom,
148  const exprt &e,
149  const abstract_environmentt &environment,
150  const namespacet &ns) const
151 {
152  const typet &followed_type = ns.follow(type);
153  ABSTRACT_OBJECT_TYPET abstract_object_type =
154  get_abstract_object_type(followed_type);
155 
156  switch(abstract_object_type)
157  {
158  case TWO_VALUE:
159  return initialize_abstract_object<abstract_objectt>(
160  followed_type, top, bottom, e, environment, ns, configuration);
161  case CONSTANT:
162  return initialize_abstract_object<constant_abstract_valuet>(
163  followed_type, top, bottom, e, environment, ns, configuration);
164  case INTERVAL:
165  return initialize_abstract_object<interval_abstract_valuet>(
166  followed_type, top, bottom, e, environment, ns, configuration);
167  case VALUE_SET:
168  return initialize_abstract_object<value_set_abstract_objectt>(
169  followed_type, top, bottom, e, environment, ns, configuration);
170 
171  case ARRAY_INSENSITIVE:
172  return initialize_abstract_object<two_value_array_abstract_objectt>(
173  followed_type, top, bottom, e, environment, ns, configuration);
174  case ARRAY_SENSITIVE:
175  return initialize_abstract_object<full_array_abstract_objectt>(
176  followed_type, top, bottom, e, environment, ns, configuration);
177 
178  case POINTER_INSENSITIVE:
179  return initialize_abstract_object<two_value_pointer_abstract_objectt>(
180  followed_type, top, bottom, e, environment, ns, configuration);
181  case POINTER_SENSITIVE:
182  return initialize_abstract_object<constant_pointer_abstract_objectt>(
183  followed_type, top, bottom, e, environment, ns, configuration);
184  case VALUE_SET_OF_POINTERS:
185  return initialize_abstract_object<value_set_pointer_abstract_objectt>(
186  followed_type, top, bottom, e, environment, ns, configuration);
187 
188  case STRUCT_INSENSITIVE:
189  return initialize_abstract_object<two_value_struct_abstract_objectt>(
190  followed_type, top, bottom, e, environment, ns, configuration);
191  case STRUCT_SENSITIVE:
192  return initialize_abstract_object<full_struct_abstract_objectt>(
193  followed_type, top, bottom, e, environment, ns, configuration);
194 
195  case UNION_INSENSITIVE:
196  return initialize_abstract_object<two_value_union_abstract_objectt>(
197  followed_type, top, bottom, e, environment, ns, configuration);
198 
199  case HEAP_ALLOCATION:
200  {
201  auto dynamic_object = exprt(ID_dynamic_object);
202  dynamic_object.set(
203  ID_identifier, "heap-allocation-" + std::to_string(heap_allocations++));
204  auto heap_symbol = unary_exprt(ID_address_of, dynamic_object, e.type());
205  auto heap_pointer =
206  get_abstract_object(e.type(), false, false, heap_symbol, environment, ns);
207  return heap_pointer;
208  }
209 
210  default:
211  UNREACHABLE;
212  return initialize_abstract_object<abstract_objectt>(
213  followed_type, top, bottom, e, environment, ns, configuration);
214  }
