127 void api_sessiont::verify_model() const
128 {
129  PRECONDITION(implementation->model);
130 
131  // Remove inline assembler; this needs to happen before adding the library.
132  remove_asm(*implementation->model);
133 
134  // add the library
135  messaget log{*implementation->message_handler};
136  log.status() << "Adding CPROVER library (" << config.ansi_c.arch << ")"
137  << messaget::eom;
138  link_to_library(
139  *implementation->model,
140  *implementation->message_handler,
141  cprover_c_library_factory);
142 
143  // Common removal of types and complex constructs
144  if(::process_goto_program(
145  *implementation->model, *implementation->options, log))
146  {
147  return;
148  }
149 
150  // add failed symbols
151  // needs to be done before pointer analysis
152  add_failed_symbols(implementation->model->symbol_table);
153 
154  // label the assertions
155  // This must be done after adding assertions and
156  // before using the argument of the "property" option.
157  // Do not re-label after using the property slicer because
158  // this would cause the property identifiers to change.
159  label_properties(*implementation->model);
160 
161  remove_skip(*implementation->model);
162 
163  ui_message_handlert ui_message_handler(*implementation->message_handler);
164  all_properties_verifier_with_trace_storaget<multi_path_symex_checkert>
165  verifier(
166  *implementation->options, ui_message_handler, *implementation->model);
167  (void)verifier();
168  verifier.report();
169 }
