212 std::set<irep_idt> dfcc_libraryt::get_missing_funs()
213 {
214  std::set<irep_idt> missing;
215 
216  // add `malloc` since it is needed used by the `is_fresh` function
217  missing.insert("malloc");
218 
219  // add `free` and `__CPROVER_deallocate` since they are used by the
220  // `write_set_deallocate_freeable`
221  missing.insert("free");
222 
223  // used by `write_set_release`
224  missing.insert(CPROVER_PREFIX "deallocate");
225 
226  // Make sure all front end functions are loaded
227  missing.insert(CPROVER_PREFIX "assignable");
228  missing.insert(CPROVER_PREFIX "object_from");
229  missing.insert(CPROVER_PREFIX "object_upto");
230  missing.insert(CPROVER_PREFIX "object_whole");
231  missing.insert(CPROVER_PREFIX "freeable");
232 
233  // go over all library functions
234  for(const auto &pair : dfcc_fun_to_name)
235  {
236  symbol_tablet::symbolst::const_iterator found =
237  goto_model.symbol_table.symbols.find(pair.second);
238 
239  if(
240  found == goto_model.symbol_table.symbols.end() ||
241  found->second.value.is_nil())
242  {
243  missing.insert(pair.second);
244  }
245  }
246  return missing;
247 }

251 void dfcc_libraryt::load(std::set<irep_idt> &to_instrument)
252 {
253  PRECONDITION_WITH_DIAGNOSTICS(
254  !loaded, "the cprover_contracts library can only be loaded once");
255  loaded = true;
256 
257  log.status() << "Adding the cprover_contracts library (" << config.ansi_c.arch
258  << ")" << messaget::eom;
259 
260  // these will need to get instrumented as well
261  to_instrument.insert("malloc");
262  to_instrument.insert("free");
263  to_instrument.insert(CPROVER_PREFIX "deallocate");
264 
265  std::set<irep_idt> to_load;
266 
267  // add the whole library
268  to_load.insert(CPROVER_PREFIX "contracts_library");
269 
270  // add front end functions
271  to_load.insert(CPROVER_PREFIX "assignable");
272  to_load.insert(CPROVER_PREFIX "object_from");
273  to_load.insert(CPROVER_PREFIX "object_upto");
274  to_load.insert(CPROVER_PREFIX "object_whole");
275  to_load.insert(CPROVER_PREFIX "freeable");
276 
277  // add stdlib dependences
278  to_load.insert("malloc");
279  to_load.insert("free");
280  to_load.insert(CPROVER_PREFIX "deallocate");
281 
282  symbol_tablet tmp_symbol_table;
283  cprover_c_library_factory_force_load(
284  to_load, tmp_symbol_table, message_handler);
285 
286  // compute missing library functions before modifying the symbol table
287  std::set<irep_idt> missing = get_missing_funs();
288 
289  // copy all loaded symbols to the main symbol table
290  for(const auto &symbol_pair : tmp_symbol_table.symbols)
291  {
292  const auto &sym = symbol_pair.first;
293  if(!goto_model.symbol_table.has_symbol(sym))
294  goto_model.symbol_table.insert(symbol_pair.second);
295  }
296 
297  // compile all missing library functions to GOTO
298  for(const auto &id : missing)
299  {
300  goto_convert(
301  id, goto_model.symbol_table, goto_model.goto_functions, message_handler);
302  }
303 
304  // check that all symbols have a goto_implementation
305  // and populate symbol maps
306  namespacet ns(goto_model.symbol_table);
307  for(const auto &pair : dfcc_fun_to_name)
308  {
309  const auto &found =
310  goto_model.goto_functions.function_map.find(pair.second);
311 
312  INVARIANT(
313  found != goto_model.goto_functions.function_map.end() &&
314  found->second.body_available(),
315  "The body of DFCC library function " + id2string(pair.second) +
316  " could not be found");
317 
318  dfcc_fun_symbol[pair.first] = ns.lookup(pair.second);
319  }
320 
321  // populate symbol maps for easy access to symbols during translation
322  for(const auto &pair : dfcc_type_to_name)
323  {
324  dfcc_type[pair.first] = ns.lookup(pair.second).type;
325  }
326 
327  // fix malloc and free calls
328  fix_malloc_free_calls();
329 
330  // inline the functions that need to be inlined for perf reasons
331  inline_functions();
332 
333  // hide all instructions in counter example traces
334  set_hide(true);
335 }
