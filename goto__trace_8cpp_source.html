33 static optionalt<symbol_exprt> get_object_rec(const exprt &src)
34 {
35  if(src.id()==ID_symbol)

163 static std::string numeric_representation(
164  const constant_exprt &expr,
165  const namespacet &ns,
166  const trace_optionst &options)
167 {
168  std::string result;
169  std::string prefix;
170 
171  const typet &expr_type = expr.type();
172 
173  const typet &underlying_type =
174  expr_type.id() == ID_c_enum_tag
175  ? ns.follow_tag(to_c_enum_tag_type(expr_type)).underlying_type()
176  : expr_type;
177 
178  const irep_idt &value = expr.get_value();
179 
180  const auto width = to_bitvector_type(underlying_type).get_width();
181 
182  const mp_integer value_int = bvrep2integer(id2string(value), width, false);
183 
184  if(options.hex_representation)
185  {
186  result = integer2string(value_int, 16);
187  prefix = "0x";
188  }
189  else
190  {
191  result = integer2binary(value_int, width);
192  prefix = "0b";
193  }
194 
195  std::ostringstream oss;
196  std::string::size_type i = 0;
197  for(const auto c : result)
198  {
199  oss << c;
200  if(++i % config.ansi_c.char_width == 0 && result.size() != i)
201  oss << ' ';
202  }
203  if(options.base_prefix)
204  return prefix + oss.str();
205  else
206  return oss.str();
207 }

209 std::string trace_numeric_value(
210  const exprt &expr,
211  const namespacet &ns,
212  const trace_optionst &options)
213 {
214  const typet &type = expr.type();
215 
216  if(expr.id()==ID_constant)
217  {
218  if(type.id()==ID_unsignedbv ||
219  type.id()==ID_signedbv ||
220  type.id()==ID_bv ||
221  type.id()==ID_fixedbv ||
222  type.id()==ID_floatbv ||
223  type.id()==ID_pointer ||
224  type.id()==ID_c_bit_field ||
225  type.id()==ID_c_bool ||
226  type.id()==ID_c_enum ||
227  type.id()==ID_c_enum_tag)
228  {
229  return numeric_representation(to_constant_expr(expr), ns, options);
230  }
231  else if(type.id()==ID_bool)
232  {
233  return expr.is_true()?"1":"0";
234  }
235  else if(type.id()==ID_integer)
236  {
237  const auto i = numeric_cast<mp_integer>(expr);
238  if(i.has_value() && *i >= 0)
239  {
240  if(options.hex_representation)
241  return "0x" + integer2string(*i, 16);
242  else
243  return "0b" + integer2string(*i, 2);
244  }
245  }
246  }
247  else if(expr.id() == ID_annotated_pointer_constant)
248  {
249  const auto &annotated_pointer_constant =
250  to_annotated_pointer_constant_expr(expr);
251  auto width = to_pointer_type(expr.type()).get_width();
252  auto &value = annotated_pointer_constant.get_value();
253  auto c = constant_exprt(value, unsignedbv_typet(width));
254  return numeric_representation(c, ns, options);
255  }
256  else if(expr.id()==ID_array)
257  {
258  std::string result;
259 
260  forall_operands(it, expr)
261  {
262  if(result.empty())
263  result="{ ";
264  else
265  result+=", ";
266  result+=trace_numeric_value(*it, ns, options);
267  }
268 
269  return result+" }";
270  }
271  else if(expr.id()==ID_struct)
272  {
273  std::string result="{ ";
274 
275  forall_operands(it, expr)
276  {
277  if(it!=expr.operands().begin())
278  result+=", ";
279  result+=trace_numeric_value(*it, ns, options);
280  }
281 
282  return result+" }";
283  }
284  else if(expr.id()==ID_union)
285  {
286  return trace_numeric_value(to_union_expr(expr).op(), ns, options);
287  }
288 
289  return "?";
290 }

322 state_location(const goto_trace_stept &state, const namespacet &ns)
323 {
324  std::string result;
325 
326  const auto &source_location = state.pc->source_location();
327 
328  if(!source_location.get_file().empty())
329  result += "file " + id2string(source_location.get_file());
330 
331  if(!state.function_id.empty())
332  {
333  const symbolt &symbol = ns.lookup(state.function_id);
334  if(!result.empty())
335  result += ' ';
336  result += "function " + id2string(symbol.display_name());

341  if(!result.empty())

394 void show_compact_goto_trace(
395  messaget::mstreamt &out,
396  const namespacet &ns,
397  const goto_tracet &goto_trace,
398  const trace_optionst &options)
399 {
400  std::size_t function_depth = 0;
401 
402  for(const auto &step : goto_trace.steps)
403  {
404  if(step.is_function_call())
405  function_depth++;
406  else if(step.is_function_return())
407  function_depth--;
408 
409  // hide the hidden ones
410  if(step.hidden)
411  continue;
412 
413  switch(step.type)
414  {
415  case goto_trace_stept::typet::ASSERT:
416  if(!step.cond_value)
417  {
418  out << '\n';
419  out << messaget::red << "Violated property:" << messaget::reset << '\n';
420  if(!step.pc->source_location().is_nil())
421  out << " " << state_location(step, ns) << '\n';
422 
423  out << " " << messaget::red << step.comment << messaget::reset << '\n';
424 
425  if(step.pc->is_assert())
426  {
427  out << " "
428  << from_expr(ns, step.function_id, step.original_condition)
429  << '\n';
430  }
431 
432  out << '\n';
433  }
434  break;
435 
436  case goto_trace_stept::typet::ASSIGNMENT:
437  if(
438  step.assignment_type ==
439  goto_trace_stept::assignment_typet::ACTUAL_PARAMETER)
440  {
441  break;
442  }
443 
444  out << " ";
445 
446  if(!step.pc->source_location().get_line().empty())
447  {
448  out << messaget::faint << step.pc->source_location().get_line() << ':'
449  << messaget::reset << ' ';
450  }
451 
452  trace_value(
453  out,
454  ns,
455  step.get_lhs_object(),
456  step.full_lhs,
457  step.full_lhs_value,
458  options);
459  break;
460 
461  case goto_trace_stept::typet::FUNCTION_CALL:
462  // downwards arrow
463  out << '\n' << messaget::faint << u8"\u21b3" << messaget::reset << ' ';
464  if(!step.pc->source_location().get_file().empty())
465  {
466  out << messaget::faint << step.pc->source_location().get_file();
467 
468  if(!step.pc->source_location().get_line().empty())
469  {
470  out << messaget::faint << ':'
471  << step.pc->source_location().get_line();
472  }
473 
474  out << messaget::reset << ' ';
475  }
476 
477  {
478  // show the display name
479  const auto &f_symbol = ns.lookup(step.called_function);
480  out << f_symbol.display_name();
481  }
482 
483  {
484  auto arg_strings = make_range(step.function_arguments)
485  .map([&ns, &step](const exprt &arg) {
486  return from_expr(ns, step.function_id, arg);
487  });
488 
489  out << '(';
490  join_strings(out, arg_strings.begin(), arg_strings.end(), ", ");
491  out << ")\n";
492  }
493 
494  break;
495 
496  case goto_trace_stept::typet::FUNCTION_RETURN:
497  // upwards arrow
498  out << messaget::faint << u8"\u21b5" << messaget::reset << '\n';
499  break;
500 
501  case goto_trace_stept::typet::ASSUME:
502  case goto_trace_stept::typet::LOCATION:
503  case goto_trace_stept::typet::GOTO:
504  case goto_trace_stept::typet::DECL:
505  case goto_trace_stept::typet::OUTPUT:
506  case goto_trace_stept::typet::INPUT:
507  case goto_trace_stept::typet::SPAWN:
508  case goto_trace_stept::typet::MEMORY_BARRIER:
509  case goto_trace_stept::typet::ATOMIC_BEGIN:
510  case goto_trace_stept::typet::ATOMIC_END:
511  case goto_trace_stept::typet::DEAD:
512  case goto_trace_stept::typet::CONSTRAINT:
513  case goto_trace_stept::typet::SHARED_READ:
514  case goto_trace_stept::typet::SHARED_WRITE:
515  break;
516 
517  case goto_trace_stept::typet::NONE:
518  UNREACHABLE;
519  }
520  }

523 void show_full_goto_trace(
524  messaget::mstreamt &out,
525  const namespacet &ns,
526  const goto_tracet &goto_trace,
527  const trace_optionst &options)
528 {
529  unsigned prev_step_nr=0;
530  bool first_step=true;
531  std::size_t function_depth=0;
532 
533  for(const auto &step : goto_trace.steps)
534  {
535  // hide the hidden ones
536  if(step.hidden)
537  continue;
538 
539  switch(step.type)
540  {
541  case goto_trace_stept::typet::ASSERT:
542  if(!step.cond_value)
543  {
544  out << '\n';
545  out << messaget::red << "Violated property:" << messaget::reset << '\n';
546  if(!step.pc->source_location().is_nil())
547  {
548  out << " " << state_location(step, ns) << '\n';
549  }
550 
551  out << " " << messaget::red << step.comment << messaget::reset << '\n';
552 
553  if(step.pc->is_assert())
554  {
555  out << " "
556  << from_expr(ns, step.function_id, step.original_condition)
557  << '\n';
558  }
559 
560  out << '\n';
561  }
562  break;
563 
564  case goto_trace_stept::typet::ASSUME:
565  if(step.cond_value && step.pc->is_assume())
566  {
567  out << "\n";
568  out << "Assumption:\n";
569 
570  if(!step.pc->source_location().is_nil())
571  out << " " << step.pc->source_location() << '\n';
572 
573  out << " " << from_expr(ns, step.function_id, step.original_condition)
574  << '\n';
575  }
576  break;
577 
578  case goto_trace_stept::typet::LOCATION:
579  break;
580 
581  case goto_trace_stept::typet::GOTO:
582  break;
583 
584  case goto_trace_stept::typet::ASSIGNMENT:
585  if(
586  step.pc->is_assign() ||
587  step.pc->is_set_return_value() || // returns have a lhs!
588  step.pc->is_function_call() ||
589  (step.pc->is_other() && step.full_lhs.is_not_nil()))
590  {
591  if(prev_step_nr!=step.step_nr || first_step)
592  {
593  first_step=false;
594  prev_step_nr=step.step_nr;
595  show_state_header(out, ns, step, step.step_nr, options);
596  }
597 
598  out << " ";
599  trace_value(
600  out,
601  ns,
602  step.get_lhs_object(),
603  step.full_lhs,
604  step.full_lhs_value,
605  options);
606  }
607  break;
608 
609  case goto_trace_stept::typet::DECL:
610  if(prev_step_nr!=step.step_nr || first_step)
611  {
612  first_step=false;
613  prev_step_nr=step.step_nr;
614  show_state_header(out, ns, step, step.step_nr, options);
615  }
616 
617  out << " ";
618  trace_value(
619  out, ns, step.get_lhs_object(), step.full_lhs, step.full_lhs_value, options);
620  break;
621 
622  case goto_trace_stept::typet::OUTPUT:
623  if(step.formatted)
624  {
625  printf_formattert printf_formatter(ns);
626  printf_formatter(id2string(step.format_string), step.io_args);
627  printf_formatter.print(out);
628  out << '\n';
629  }
630  else
631  {
632  show_state_header(out, ns, step, step.step_nr, options);
633  out << " OUTPUT " << step.io_id << ':';
634 
635  for(std::list<exprt>::const_iterator
636  l_it=step.io_args.begin();
637  l_it!=step.io_args.end();
638  l_it++)
639  {
640  if(l_it!=step.io_args.begin())
641  out << ';';
642 
643  out << ' ' << from_expr(ns, step.function_id, *l_it);
644 
645  // the binary representation
646  out << " (" << trace_numeric_value(*l_it, ns, options) << ')';
647  }
648 
649  out << '\n';
650  }
651  break;
652 
653  case goto_trace_stept::typet::INPUT:
654  show_state_header(out, ns, step, step.step_nr, options);
655  out << " INPUT " << step.io_id << ':';
656 
657  for(std::list<exprt>::const_iterator
658  l_it=step.io_args.begin();
659  l_it!=step.io_args.end();
660  l_it++)
661  {
662  if(l_it!=step.io_args.begin())
663  out << ';';
664 
665  out << ' ' << from_expr(ns, step.function_id, *l_it);
666 
667  // the binary representation
668  out << " (" << trace_numeric_value(*l_it, ns, options) << ')';
669  }
670 
671  out << '\n';
672  break;
673 
674  case goto_trace_stept::typet::FUNCTION_CALL:
675  function_depth++;
676  if(options.show_function_calls)
677  {
678  out << "\n#### Function call: " << step.called_function;
679  out << '(';
680 
681  bool first = true;
682  for(auto &arg : step.function_arguments)
683  {
684  if(first)
685  first = false;
686  else
687  out << ", ";
688 
689  out << from_expr(ns, step.function_id, arg);
690  }
691 
692  out << ") (depth " << function_depth << ") ####\n";
693  }
694  break;
695 
696  case goto_trace_stept::typet::FUNCTION_RETURN:
697  function_depth--;
698  if(options.show_function_calls)
699  {
700  out << "\n#### Function return from " << step.function_id << " (depth "
701  << function_depth << ") ####\n";
702  }
703  break;
704 
705  case goto_trace_stept::typet::SPAWN:
706  case goto_trace_stept::typet::MEMORY_BARRIER:
707  case goto_trace_stept::typet::ATOMIC_BEGIN:
708  case goto_trace_stept::typet::ATOMIC_END:
709  case goto_trace_stept::typet::DEAD:
710  break;
711 
712  case goto_trace_stept::typet::CONSTRAINT:
713  case goto_trace_stept::typet::SHARED_READ:
714  case goto_trace_stept::typet::SHARED_WRITE:
715  case goto_trace_stept::typet::NONE:
716  UNREACHABLE;
717  }
718  }
719 }

721 static void show_goto_stack_trace(
722  messaget::mstreamt &out,
723  const namespacet &ns,
724  const goto_tracet &goto_trace)
725 {
726  // map from thread number to a call stack
727  std::map<unsigned, std::vector<goto_tracet::stepst::const_iterator>>
728  call_stacks;
729 
730  // by default, we show thread 0
731  unsigned thread_to_show = 0;
732 
733  for(auto s_it = goto_trace.steps.begin(); s_it != goto_trace.steps.end();
734  s_it++)
735  {
736  const auto &step = *s_it;
737  auto &stack = call_stacks[step.thread_nr];
738 
739  if(step.is_assert())
740  {
741  if(!step.cond_value)
742  {
743  stack.push_back(s_it);
744  thread_to_show = step.thread_nr;
745  }
746  }
