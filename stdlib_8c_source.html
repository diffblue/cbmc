345 long strtol(const char *nptr, char **endptr, int base)
346 {
347  __CPROVER_HIDE:;
348  #ifdef __CPROVER_STRING_ABSTRACTION
349  __CPROVER_precondition(__CPROVER_is_zero_string(nptr),
350  "zero-termination of argument of strtol");
351  #endif
352 
353  if(base==1 || base<0 || base>36)
354  {
355  errno=EINVAL;
356  return 0;
357  }
358 
359  long res=0;
360  _Bool in_number=0;
361  char sign=0;
362 
363  // 32 chars is an arbitrarily chosen limit
364  int i=0;
365  for( ; i<31; ++i)
366  {
367  char ch=nptr[i];
368  char sub=0;
369  if(ch==0)
370  break;
371  else if((base==0 || base==16) && !in_number &&
372  ch=='0' && (nptr[i+1]=='x' || nptr[i+1]=='X'))
373  {
374  base=16;
375  in_number=1;
376  ++i;
377  continue;
378  }
379  else if(base==0 && !in_number && ch=='0')
380  {
381  base=8;
382  in_number=1;
383  continue;
384  }
385  else if(!in_number && !sign && isspace(ch))
386  continue;
387  else if(!in_number && !sign && (ch=='-' || ch=='+'))
388  {
389  sign=ch;
390  continue;
391  }
392  else if(base>10 && ch>='a' && ch-'a'<base-10)
393  sub='a'-10;
394  else if(base>10 && ch>='A' && ch-'A'<base-10)
395  sub='A'-10;
396  else if(isdigit(ch))
397  {
398  sub='0';
399  base=base==0 ? 10 : base;
400  }
401  else
402  break;
403 
404  in_number=1;
405  _Bool overflow = __builtin_mul_overflow(res, (long)base, &res);
406  if(overflow || __builtin_add_overflow(res, (long)(ch - sub), &res))
407  {
408  errno=ERANGE;
409  if(sign=='-')
410  return LONG_MIN;
411  else
412  return LONG_MAX;
413  }
414  }
415 
416  if(endptr!=0)
417  *endptr=(char*)nptr+i;
418 
419  if(sign=='-')
420  res*=-1;
421 
422  return res;
423 }
