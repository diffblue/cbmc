63 void dfcc_swap_and_wrapt::swap_and_wrap(
64  const dfcc_contract_modet contract_mode,
65  const irep_idt &function_id,
66  const irep_idt &contract_id,
67  std::set<irep_idt> &function_pointer_contracts,
68  bool allow_recursive_calls)
69 {
70  auto pair = cache.insert({function_id, {contract_id, contract_mode}});
71  auto inserted = pair.second;
72 
73  if(!inserted)
74  {
75  auto old_contract_id = pair.first->second.first;
76  auto old_contract_mode = pair.first->second.second;
77 
78  // different swapp already performed, abort
79  if(old_contract_id != contract_id || old_contract_mode != contract_mode)
80  {
81  auto mode1 = (old_contract_mode == dfcc_contract_modet::REPLACE)
82  ? "REPLACE"
83  : "CHECK";
84  auto mode2 =
85  (contract_mode == dfcc_contract_modet::REPLACE) ? "REPLACE" : "CHECK)";
86 
87  std::ostringstream err_msg;
88  err_msg << "DFCC: multiple attempts to swap and wrap function '"
89  << function_id << "':\n";
90  err_msg << "- with '" << old_contract_id << "' in mode " << mode1 << "\n";
91  err_msg << "- with '" << contract_id << "' in mode " << mode2 << "\n";
92  throw invalid_input_exceptiont(err_msg.str());
93  }
94  // same swap already performed
95  return;
96  }
97 
98  // actually perform the translation
99  switch(contract_mode)
100  {
101  case dfcc_contract_modet::CHECK:
102  {
103  check_contract(
104  function_id,
105  contract_id,
106  function_pointer_contracts,
107  allow_recursive_calls);
108  break;
109  }
110  case dfcc_contract_modet::REPLACE:
111  {
112  replace_with_contract(function_id, contract_id, function_pointer_contracts);
113  break;
114  }
115  }
116 }

150 void dfcc_swap_and_wrapt::check_contract(
151  const irep_idt &function_id,
152  const irep_idt &contract_id,
153  std::set<irep_idt> &function_pointer_contracts,
154  bool allow_recursive_calls)
155 {
156  // all code generation needs to run on functions with unmodified signatures
157  const irep_idt &wrapper_id = function_id;
158  const irep_idt wrapped_id =
159  id2string(wrapper_id) + "_wrapped_for_contract_checking";
160  utils.wrap_function(wrapper_id, wrapped_id);
161 
162  // wrapper body
163  goto_programt body;
164 
165  const auto &wrapper_symbol = utils.get_function_symbol(wrapper_id);
166 
167  auto check_started = utils
168  .create_static_symbol(
169  bool_typet(),
170  id2string(function_id),
171  "__contract_check_in_progress",
172  wrapper_symbol.location,
173  wrapper_symbol.mode,
174  wrapper_symbol.module,
175  false_exprt())
176  .symbol_expr();
177 
178  auto check_completed = utils
179  .create_static_symbol(
180  bool_typet(),
181  id2string(function_id),
182  "__contract_checked_once",
183  wrapper_symbol.location,
184  wrapper_symbol.mode,
185  wrapper_symbol.module,
186  false_exprt())
187  .symbol_expr();
188 
189  auto check_started_goto = body.add(goto_programt::make_incomplete_goto(
190  check_started, wrapper_symbol.location));
191 
192  // At most a single top level call to the checked function in any execution
193 
194  // Recursive calls within a contract check correspond to
195  // `check_started && !check_completed` and are allowed.
196 
197  // Any other call occuring with `check_completed` true is forbidden.
198  source_locationt sl(wrapper_symbol.location);
199  sl.set_function(wrapper_symbol.name);
200  sl.set_property_class("single_top_level_call");
201  sl.set_comment(
202  "Only a single top-level call to function " + id2string(function_id) +
203  " when checking contract " + id2string(contract_id));
204  body.add(goto_programt::make_assertion(not_exprt(check_completed), sl));
205  body.add(goto_programt::make_assignment(
206  check_started, true_exprt(), wrapper_symbol.location));
207 
208  const auto write_set_symbol = utils.create_new_parameter_symbol(
209  function_id,
210  "__write_set_to_check",
211  library.dfcc_type[dfcc_typet::CAR_SET_PTR]);
212 
213  contract_handler.add_contract_instructions(
214  dfcc_contract_modet::CHECK,
215  wrapper_id,
216  wrapped_id,
217  contract_id,
218  write_set_symbol,
219  body,
220  function_pointer_contracts);
221 
222  body.add(goto_programt::make_assignment(
223  check_completed, true_exprt(), wrapper_symbol.location));
224  body.add(goto_programt::make_assignment(
225  check_started, false_exprt(), wrapper_symbol.location));
226 
227  // unconditionally Jump to the end after the check
228  auto goto_end_function =
229  body.add(goto_programt::make_incomplete_goto(wrapper_symbol.location));
230 
231  // Jump to the replacement section if check already in progress
232  auto contract_replacement_label =
233  body.add(goto_programt::make_skip(wrapper_symbol.location));
234  check_started_goto->complete_goto(contract_replacement_label);
235 
236  if(allow_recursive_calls)
237  {
238  contract_handler.add_contract_instructions(
239  dfcc_contract_modet::REPLACE,
240  wrapper_id,
241  wrapped_id,
242  contract_id,
243  write_set_symbol,
244  body,
245  function_pointer_contracts);
246  }
247  else
248  {
249  source_locationt sl(wrapper_symbol.location);
250  sl.set_function(wrapper_symbol.name);
251  sl.set_property_class("no_recursive_call");
252  sl.set_comment(
253  "No recursive call to function " + id2string(function_id) +
254  " when checking contract " + id2string(contract_id));
255  body.add(goto_programt::make_assertion(false_exprt(), sl));
256  body.add(
257  goto_programt::make_assumption(false_exprt(), wrapper_symbol.location));
258  }
259 
260  auto end_function_label =
261  body.add(goto_programt::make_end_function(wrapper_symbol.location));
262  goto_end_function->complete_goto(end_function_label);
263 
264  // write the body to the GOTO function
265  goto_model.goto_functions.function_map.at(function_id).body.swap(body);
266 
267  // extend the signature of the wrapper function with the write set parameter
268  utils.add_parameter(write_set_symbol, function_id);
269 
270  utils.set_hide(wrapper_id, true);
271 
272  // instrument the wrapped function
273  instrument.instrument_wrapped_function(wrapped_id, wrapper_id);
274 
275  goto_model.goto_functions.update();
276 }
