52 void c_typecheck_baset::typecheck_symbol(symbolt &symbol)
53 {
54  bool is_function=symbol.type.id()==ID_code;
55 
56  const typet &final_type=follow(symbol.type);
57 
58  // set a few flags
59  symbol.is_lvalue=!symbol.is_type && !symbol.is_macro;
60 
61  irep_idt root_name=symbol.base_name;
62  irep_idt new_name=symbol.name;
63 
64  if(symbol.is_file_local)
65  {
66  // file-local stuff -- stays as is
67  // collisions are resolved during linking
68  }
69  else if(symbol.is_extern && !is_function)
70  {
71  // variables marked as "extern" go into the global namespace
72  // and have static lifetime
73  new_name=root_name;
74  symbol.is_static_lifetime=true;
75 
76  if(symbol.value.is_not_nil() && !symbol.is_macro)
77  {
78  // According to the C standard this should be an error, but at least some
79  // versions of Visual Studio insist to use this in their C library, and
80  // GCC just warns as well.
81  warning().source_location = symbol.value.find_source_location();
82  warning() << "'extern' symbol '" << new_name
83  << "' should not have an initializer" << eom;
84  }
85  }
86  else if(!is_function && symbol.value.id()==ID_code)
87  {
88  error().source_location=symbol.value.find_source_location();
89  error() << "only functions can have a function body" << eom;
90  throw 0;
91  }
92 
93  // set the pretty name
94  if(symbol.is_type && final_type.id() == ID_struct)
95  {
96  symbol.pretty_name="struct "+id2string(symbol.base_name);
97  }
98  else if(symbol.is_type && final_type.id() == ID_union)
99  {
100  symbol.pretty_name="union "+id2string(symbol.base_name);
101  }
102  else if(symbol.is_type && final_type.id() == ID_c_enum)
103  {
104  symbol.pretty_name="enum "+id2string(symbol.base_name);
105  }
106  else
107  {
108  symbol.pretty_name=new_name;
109  }
110 
111  if(!symbol.is_type && !symbol.is_extern && symbol.type.id() == ID_empty)
112  {
113  error().source_location = symbol.location;
114  error() << "void-typed symbol not permitted" << eom;
115  throw 0;
116  }
117 
118  // see if we have it already
119  symbol_table_baset::symbolst::const_iterator old_it =
120  symbol_table.symbols.find(symbol.name);
121 
122  if(old_it==symbol_table.symbols.end())
123  {
124  // just put into symbol_table
125  symbolt *new_symbol;
126  move_symbol(symbol, new_symbol);
127 
128  typecheck_new_symbol(*new_symbol);
129  }
130  else
131  {
132  if(old_it->second.is_type!=symbol.is_type)
133  {
134  error().source_location=symbol.location;
135  error() << "redeclaration of '" << symbol.display_name()
136  << "' as a different kind of symbol" << eom;
137  throw 0;
138  }
139 
140  symbolt &existing_symbol = symbol_table.get_writeable_ref(symbol.name);
141  if(symbol.is_type)
142  typecheck_redefinition_type(existing_symbol, symbol);
143  else
144  typecheck_redefinition_non_type(existing_symbol, symbol);
145  }

178 void c_typecheck_baset::typecheck_redefinition_type(
179  symbolt &old_symbol,
180  symbolt &new_symbol)
181 {
182  const typet &final_old=follow(old_symbol.type);
183  const typet &final_new=follow(new_symbol.type);
184 
185  // see if we had something incomplete before
186  if(
187  (final_old.id() == ID_struct &&
188  to_struct_type(final_old).is_incomplete()) ||
189  (final_old.id() == ID_union && to_union_type(final_old).is_incomplete()) ||
190  (final_old.id() == ID_c_enum && to_c_enum_type(final_old).is_incomplete()))
191  {
192  // is the new one complete?
193  if(
194  final_new.id() == final_old.id() &&
195  ((final_new.id() == ID_struct &&
196  !to_struct_type(final_new).is_incomplete()) ||
197  (final_new.id() == ID_union &&
198  !to_union_type(final_new).is_incomplete()) ||
199  (final_new.id() == ID_c_enum &&
200  !to_c_enum_type(final_new).is_incomplete())))
201  {
202  // overwrite location
203  old_symbol.location=new_symbol.location;
204 
205  // move body
206  old_symbol.type.swap(new_symbol.type);
207  }
208  else if(new_symbol.type.id()==old_symbol.type.id())
209  return; // ignore
210  else
211  {
212  error().source_location=new_symbol.location;
213  error() << "conflicting definition of type symbol '"
214  << new_symbol.display_name() << '\'' << eom;
215  throw 0;
216  }
217  }
218  else
219  {
220  // see if new one is just a tag
221  if(
222  (final_new.id() == ID_struct &&
223  to_struct_type(final_new).is_incomplete()) ||
224  (final_new.id() == ID_union &&
225  to_union_type(final_new).is_incomplete()) ||
226  (final_new.id() == ID_c_enum &&
227  to_c_enum_type(final_new).is_incomplete()))
228  {
229  if(final_old.id() == final_new.id())
230  {
231  // just ignore silently
232  }
233  else
234  {
235  // arg! new tag type
236  error().source_location=new_symbol.location;
237  error() << "conflicting definition of tag symbol '"
238  << new_symbol.display_name() << '\'' << eom;
239  throw 0;
240  }
241  }
242  else if(config.ansi_c.os==configt::ansi_ct::ost::OS_WIN &&
243  final_new.id()==ID_c_enum && final_old.id()==ID_c_enum)
244  {
245  // under Windows, ignore this silently;
246  // MSC doesn't think this is a problem, but GCC complains.
247  }
248  else if(
249  config.ansi_c.os == configt::ansi_ct::ost::OS_WIN &&
250  final_new.id() == ID_pointer && final_old.id() == ID_pointer &&
251  follow(to_pointer_type(final_new).base_type()).id() == ID_c_enum &&
252  follow(to_pointer_type(final_old).base_type()).id() == ID_c_enum)
253  {
254  // under Windows, ignore this silently;
255  // MSC doesn't think this is a problem, but GCC complains.
256  }
257  else
258  {
259  // see if it changed
260  if(follow(new_symbol.type)!=follow(old_symbol.type))
261  {
262  error().source_location=new_symbol.location;
263  error() << "type symbol '" << new_symbol.display_name()
264  << "' defined twice:\n"
265  << "Original: " << to_string(old_symbol.type) << "\n"
266  << " New: " << to_string(new_symbol.type) << eom;
267  throw 0;
268  }
269  }
270  }

273 static bool is_instantiation_of_flexible_array(
274  const struct_typet &old_type,
275  const struct_typet &new_type)
276 {
277  const struct_typet::componentst &old_components = old_type.components();
278  const struct_typet::componentst &new_components = new_type.components();
279 
280  if(old_components.size() != new_components.size())
281  return false;
282 
283  if(old_components.empty())
284  return false;
285 
286  for(std::size_t i = 0; i < old_components.size() - 1; ++i)
287  {
288  if(old_components[i].type() != new_components[i].type())
289  return false;
290  }
291 
292  if(
293  old_components.back().type().id() != ID_array ||
294  new_components.back().type().id() != ID_array)
295  {
296  return false;
297  }
298 
299  const auto &old_array_type = to_array_type(old_components.back().type());
300  const auto &new_array_type = to_array_type(new_components.back().type());
301 
302  return old_array_type.element_type() == new_array_type.element_type() &&
303  old_array_type.get_bool(ID_C_flexible_array_member) &&
304  new_array_type.get_bool(ID_C_flexible_array_member) &&
305  (old_array_type.size().is_nil() || old_array_type.size().is_zero());
306 }

308 void c_typecheck_baset::typecheck_redefinition_non_type(
309  symbolt &old_symbol,
310  symbolt &new_symbol)
311 {
312  const typet &final_old=follow(old_symbol.type);
313  const typet &initial_new=follow(new_symbol.type);
314 
315  if(
316  final_old.id() == ID_array &&
317  to_array_type(final_old).size().is_not_nil() &&
318  initial_new.id() == ID_array &&
319  to_array_type(initial_new).size().is_nil() &&
320  to_array_type(final_old).element_type() ==
321  to_array_type(initial_new).element_type())
322  {
323  // this is ok, just use old type
324  new_symbol.type=old_symbol.type;
325  }
326  else if(
327  final_old.id() == ID_array && to_array_type(final_old).size().is_nil() &&
328  initial_new.id() == ID_array &&
329  to_array_type(initial_new).size().is_not_nil() &&
330  to_array_type(final_old).element_type() ==
331  to_array_type(initial_new).element_type())
332  {
333  // update the type to enable the use of sizeof(x) on the
334  // right-hand side of a definition of x
335  old_symbol.type=new_symbol.type;
336  }
337 
338  // do initializer, this may change the type
339  if(new_symbol.type.id() != ID_code && !new_symbol.is_macro)
340  do_initializer(new_symbol);
341 
342  const typet &final_new=follow(new_symbol.type);
343 
344  // K&R stuff?
345  if(old_symbol.type.id()==ID_KnR)
346  {
347  // check the type
348  if(final_new.id()==ID_code)
349  {
350  error().source_location=new_symbol.location;
351  error() << "function type not allowed for K&R function parameter"
352  << eom;
353  throw 0;
354  }
355 
356  // fix up old symbol -- we now got the type
357  old_symbol.type=new_symbol.type;
358  return;
359  }
360 
361  if(final_new.id()==ID_code)
362  {
363  if(final_old.id()!=ID_code)
364  {
365  error().source_location=new_symbol.location;
366  error() << "error: function symbol '" << new_symbol.display_name()
367  << "' redefined with a different type:\n"
368  << "Original: " << to_string(old_symbol.type) << "\n"
369  << " New: " << to_string(new_symbol.type) << eom;
370  throw 0;
371  }
372 
373  code_typet &old_ct=to_code_type(old_symbol.type);
374  code_typet &new_ct=to_code_type(new_symbol.type);
375 
376  if(
377  old_ct.return_type() != new_ct.return_type() &&
378  !old_ct.get_bool(ID_C_incomplete))
379  {
380  throw invalid_source_file_exceptiont{
381  "function symbol '" + id2string(new_symbol.display_name()) +
382  "' redefined with a different type:\n" +
383  "Original: " + to_string(old_symbol.type) + "\n" +
384  " New: " + to_string(new_symbol.type),
385  new_symbol.location};
386  }
387  const bool inlined = old_ct.get_inlined() || new_ct.get_inlined();
388 
389  if(old_ct.has_ellipsis() && !new_ct.has_ellipsis())
390  old_ct=new_ct;
391  else if(!old_ct.has_ellipsis() && new_ct.has_ellipsis())
392  {
393  if(to_code_with_contract_type(new_ct).has_contract())
394  {
395  error().source_location = new_symbol.location;
396  error() << "code contract on incomplete function re-declaration" << eom;
397  throw 0;
398  }
399  new_ct=old_ct;
400  }
401 
402  if(inlined)
403  {
404  old_ct.set_inlined(true);
405  new_ct.set_inlined(true);
406  }
407 
408  // do body
409 
410  if(new_symbol.value.is_not_nil())
411  {
412  if(old_symbol.value.is_not_nil())
413  {
414  // gcc allows re-definition if the first
415  // definition is marked as "extern inline"
416 
417  if(
418  old_ct.get_inlined() &&
419  (config.ansi_c.mode == configt::ansi_ct::flavourt::GCC ||
420  config.ansi_c.mode == configt::ansi_ct::flavourt::CLANG ||
421  config.ansi_c.mode == configt::ansi_ct::flavourt::ARM ||
422  config.ansi_c.mode == configt::ansi_ct::flavourt::VISUAL_STUDIO))
423  {
424  // overwrite "extern inline" properties
425  old_symbol.is_extern=new_symbol.is_extern;
426  old_symbol.is_file_local=new_symbol.is_file_local;
427 
428  // remove parameter declarations to avoid conflicts
429  for(const auto &old_parameter : old_ct.parameters())
430  {
431  const irep_idt &identifier = old_parameter.get_identifier();
432 
433  symbol_table_baset::symbolst::const_iterator p_s_it =
434  symbol_table.symbols.find(identifier);
435  if(p_s_it!=symbol_table.symbols.end())
436  symbol_table.erase(p_s_it);
437  }
438  }
439  else
440  {
441  error().source_location=new_symbol.location;
442  error() << "function body '" << new_symbol.display_name()
443  << "' defined twice" << eom;
444  throw 0;
445  }
446  }
447  else if(inlined)
448  {
449  // preserve "extern inline" properties
450  old_symbol.is_extern=new_symbol.is_extern;
451  old_symbol.is_file_local=new_symbol.is_file_local;
452  }
453  else if(new_symbol.is_weak)
454  {
455  // weak symbols
456  old_symbol.is_weak=true;
457  }
458 
459  if(new_symbol.is_macro)
460  old_symbol.is_macro=true;
461  else
462  typecheck_function_body(new_symbol);
463 
464  // overwrite location
465  old_symbol.location=new_symbol.location;
466 
467  // move body
468  old_symbol.value.swap(new_symbol.value);
469 
470  // overwrite type (because of parameter names)
471  old_symbol.type=new_symbol.type;
472  }
473  else if(to_code_with_contract_type(new_ct).has_contract())
474  {
475  // overwrite type to add contract, but keep the old parameter identifiers
476  // (if any)
477  auto new_parameters_it = new_ct.parameters().begin();
478  for(auto &p : old_ct.parameters())
479  {
480  if(new_parameters_it != new_ct.parameters().end())
481  {
482  new_parameters_it->set_identifier(p.get_identifier());
483  ++new_parameters_it;
484  }
485  }
486 
487  old_symbol.type = new_symbol.type;
488  }
489 
490  return;
491  }
492 
493  if(final_old!=final_new)
494  {
495  if(
496  final_old.id() == ID_array && to_array_type(final_old).size().is_nil() &&
497  final_new.id() == ID_array &&
498  to_array_type(final_new).size().is_not_nil() &&
499  to_array_type(final_old).element_type() ==
500  to_array_type(final_new).element_type())
501  {
502  old_symbol.type=new_symbol.type;
503  }
504  else if(
505  final_old.id() == ID_pointer &&
506  to_pointer_type(final_old).base_type().id() == ID_code &&
507  to_code_type(to_pointer_type(final_old).base_type()).has_ellipsis() &&
508  final_new.id() == ID_pointer &&
509  to_pointer_type(final_new).base_type().id() == ID_code)
510  {
511  // to allow
512  // int (*f) ();
513  // int (*f) (int)=0;
514  old_symbol.type=new_symbol.type;
515  }
516  else if(
517  final_old.id() == ID_pointer &&
518  to_pointer_type(final_old).base_type().id() == ID_code &&
519  final_new.id() == ID_pointer &&
520  to_pointer_type(final_new).base_type().id() == ID_code &&
521  to_code_type(to_pointer_type(final_new).base_type()).has_ellipsis())
522  {
523  // to allow
524  // int (*f) (int)=0;
525  // int (*f) ();
526  }
527  else if(
528  final_old.id() == ID_struct && final_new.id() == ID_struct &&
529  is_instantiation_of_flexible_array(
530  to_struct_type(final_old), to_struct_type(final_new)))
531  {
532  old_symbol.type = new_symbol.type;
533  }
534  else
535  {
536  error().source_location=new_symbol.location;
537  error() << "symbol '" << new_symbol.display_name()
538  << "' redefined with a different type:\n"
539  << "Original: " << to_string(old_symbol.type) << "\n"
540  << " New: " << to_string(new_symbol.type) << eom;
541  throw 0;
542  }
543  }
544  else // finals are equal
545  {
546  }
547 
548  // do value
549  if(new_symbol.value.is_not_nil())
550  {
551  // see if we already have one
552  if(old_symbol.value.is_not_nil())
553  {
554  if(
555  new_symbol.is_macro && final_new.id() == ID_c_enum &&
556  old_symbol.value.is_constant() && new_symbol.value.is_constant() &&
557  old_symbol.value.get(ID_value) == new_symbol.value.get(ID_value))
558  {
559  // ignore
560  }
561  else
562  {
563  warning().source_location = new_symbol.value.find_source_location();
564  warning() << "symbol '" << new_symbol.display_name()
565  << "' already has an initial value" << eom;
566  }
567  }
568  else
569  {
570  old_symbol.value=new_symbol.value;
571  old_symbol.type=new_symbol.type;
572  old_symbol.is_macro=new_symbol.is_macro;
573  }
574  }
575 
576  // take care of some flags
577  if(old_symbol.is_extern && !new_symbol.is_extern)
578  old_symbol.location=new_symbol.location;
579 
580  old_symbol.is_extern=old_symbol.is_extern && new_symbol.is_extern;
581 
582  // We should likely check is_volatile and
583  // is_thread_local for consistency. GCC complains if these
584  // mismatch.
585 }

587 void c_typecheck_baset::typecheck_function_body(symbolt &symbol)
588 {
589  if(symbol.value.id() != ID_code)
590  {
591  error().source_location = symbol.location;
592  error() << "function '" << symbol.name << "' is initialized with "
593  << symbol.value.id() << eom;
594  throw 0;
595  }
596 
597  code_typet &code_type = to_code_type(symbol.type);
598 
599  // reset labels
600  labels_used.clear();
601  labels_defined.clear();
602 
603  // A function declaration int foo(); does not specify the parameter list, but
604  // a function definition int foo() { ... } does fix the parameter list to be
605  // empty.
606  if(code_type.parameters().empty() && code_type.has_ellipsis())
607  code_type.remove_ellipsis();
608 
609  // fix type
610  symbol.value.type()=code_type;
611 
612  // set return type
613  return_type=code_type.return_type();
614 
615  // Add the parameter declarations into the symbol table
616  add_parameters_to_symbol_table(symbol);
617 
618  // typecheck the body code
619  typecheck_code(to_code(symbol.value));
620 
621  // check the labels
622  for(const auto &label : labels_used)
623  {
624  if(labels_defined.find(label.first) == labels_defined.end())
625  {
626  error().source_location = label.second;
627  error() << "branching label '" << label.first
628  << "' is not defined in function" << eom;
629  throw 0;
630  }
631  }

634 void c_typecheck_baset::apply_asm_label(
635  const irep_idt &asm_label,
636  symbolt &symbol)
637 {
638  const irep_idt orig_name=symbol.name;
639 
640  // restrict renaming to functions and global variables;
641  // procedure-local ones would require fixing the scope, as we
642  // do for parameters below
643  if(!asm_label.empty() &&
644  !symbol.is_type &&
645  (symbol.type.id()==ID_code || symbol.is_static_lifetime))
646  {
647  symbol.name=asm_label;
648  symbol.base_name=asm_label;
649  }
650 
651  if(symbol.name!=orig_name)
652  {
653  if(!asm_label_map.insert(
654  std::make_pair(orig_name, asm_label)).second)
655  {
656  if(asm_label_map[orig_name]!=asm_label)
657  {
658  error().source_location=symbol.location;
659  error() << "error: replacing asm renaming "
660  << asm_label_map[orig_name] << " by "
661  << asm_label << eom;
662  throw 0;
663  }
664  }
665  }
666  else if(asm_label.empty())
667  {
668  asm_label_mapt::const_iterator entry=
669  asm_label_map.find(symbol.name);
670  if(entry!=asm_label_map.end())
671  {
672  symbol.name=entry->second;
673  symbol.base_name=entry->second;
674  }
675  }
676 
677  if(symbol.name!=orig_name &&
678  symbol.type.id()==ID_code &&

749 void c_typecheck_baset::typecheck_declaration(
750  ansi_c_declarationt &declaration)
751 {
752  if(declaration.get_is_static_assert())
753  {
754  codet code(ID_static_assert);
755  code.add_source_location() = declaration.source_location();
756  code.operands().swap(declaration.operands());
757  typecheck_code(code);
758  }
759  else
760  {
761  // get storage spec
762  c_storage_spect c_storage_spec(declaration.type());
763 
764  // first typecheck the type of the declaration
765  typecheck_type(declaration.type());
766 
767  // mark as 'already typechecked'
768  already_typechecked_typet::make_already_typechecked(declaration.type());
769 
770  // Now do declarators, if any.
771  for(auto &declarator : declaration.declarators())
772  {
773  c_storage_spect full_spec(declaration.full_type(declarator));
774  full_spec|=c_storage_spec;
775 
776  declaration.set_is_inline(full_spec.is_inline);
777  declaration.set_is_static(full_spec.is_static);
778  declaration.set_is_extern(full_spec.is_extern);
779  declaration.set_is_thread_local(full_spec.is_thread_local);
780  declaration.set_is_register(full_spec.is_register);
781  declaration.set_is_typedef(full_spec.is_typedef);
782  declaration.set_is_weak(full_spec.is_weak);
783 
784  symbolt symbol = declaration.to_symbol(declarator);
785  current_symbol=symbol;
786 
787  // now check other half of type
788  typecheck_type(symbol.type);
789 
790  if(!full_spec.alias.empty())
791  {
792  if(symbol.value.is_not_nil())
793  {
794  error().source_location=symbol.location;
795  error() << "alias attribute cannot be used with a body"
796  << eom;
797  throw 0;
798  }
799 
800  // alias function need not have been declared yet, thus
801  // can't lookup
802  // also cater for renaming/placement in sections
803  const auto &renaming_entry = asm_label_map.find(full_spec.alias);
804  if(renaming_entry == asm_label_map.end())
805  symbol.value = symbol_exprt::typeless(full_spec.alias);
806  else
807  symbol.value = symbol_exprt::typeless(renaming_entry->second);
808  symbol.is_macro=true;
809  }
810 
811  if(full_spec.is_used && symbol.is_file_local)
812  {
813  // GCC __attribute__((__used__)) - do not treat those as file-local, but
814  // make sure the name is unique
815  symbol.is_file_local = false;
816 
817  symbolt symbol_for_renaming = symbol;
818  if(!full_spec.asm_label.empty())
819  symbol_for_renaming.name = full_spec.asm_label;
820  full_spec.asm_label = djb_manglert{}(
821  symbol_for_renaming,
822  id2string(symbol_for_renaming.location.get_file()));
823  }
824 
825  if(full_spec.section.empty())
826  apply_asm_label(full_spec.asm_label, symbol);
827  else
828  {
829  // section name is not empty, do a bit of parsing
830  std::string asm_name = id2string(full_spec.section);
831 
832  if(asm_name[0] == '.')
833  {
834  std::string::size_type primary_section = asm_name.find('.', 1);
835 
836  if(primary_section != std::string::npos)
837  asm_name.resize(primary_section);
838  }
839 
840  asm_name += "$$";
841 
842  if(!full_spec.asm_label.empty())
843  asm_name+=id2string(full_spec.asm_label);
844  else
845  asm_name+=id2string(symbol.name);
846 
847  apply_asm_label(asm_name, symbol);
848  }
849 
850  irep_idt identifier=symbol.name;
851  declarator.set_name(identifier);
852 
853  typecheck_symbol(symbol);
854 
855  // check the contract, if any
856  symbolt &new_symbol = symbol_table.get_writeable_ref(identifier);
857  if(
858  new_symbol.type.id() == ID_code &&
859  to_code_with_contract_type(new_symbol.type).has_contract())
860  {
861  code_with_contract_typet code_type =
862  to_code_with_contract_type(new_symbol.type);
863 
864  // ensure parameter declarations are available for type checking to
865  // succeed
866  binding_exprt::variablest temporary_parameter_symbols;
867 
868  const auto &return_type = code_type.return_type();
869  if(return_type.id() != ID_empty)
870  {
871  parameter_map[CPROVER_PREFIX "return_value"] = return_type;
872  temporary_parameter_symbols.emplace_back(
873  CPROVER_PREFIX "return_value", return_type);
874  }
875 
876  std::size_t anon_counter = 0;
877  for(auto &p : code_type.parameters())
878  {
879  // may be anonymous
880  if(p.get_base_name().empty())
881  p.set_base_name("#anon" + std::to_string(anon_counter++));
882 
883  // produce identifier
884  const irep_idt &base_name = p.get_base_name();
885  irep_idt parameter_identifier =
886  id2string(identifier) + "::" + id2string(base_name);
887 
888  p.set_identifier(parameter_identifier);
889 
890  PRECONDITION(
891  parameter_map.find(parameter_identifier) == parameter_map.end());
892  parameter_map[parameter_identifier] = p.type();
893  temporary_parameter_symbols.emplace_back(
894  parameter_identifier, p.type());
895  }
896 
897  for(auto &requires : code_type.requires())
898  {
899  typecheck_expr(requires);
900  implicit_typecast_bool(requires);
901  std::string clause_type = "preconditions";
902  check_history_expr_return_value(requires, clause_type);
903  check_was_freed(requires, clause_type);
904  lambda_exprt lambda{temporary_parameter_symbols, requires};
905  lambda.add_source_location() = requires.source_location();
906  requires.swap(lambda);
907  }
908 
909  typecheck_spec_assigns(code_type.assigns());
910  for(auto &assigns : code_type.assigns())
911  {
912  std::string clause_type = "assigns clauses";
913  check_history_expr_return_value(assigns, clause_type);
914  lambda_exprt lambda{temporary_parameter_symbols, assigns};
915  lambda.add_source_location() = assigns.source_location();
916  assigns.swap(lambda);
917  }
918 
919  typecheck_spec_frees(code_type.frees());
920  for(auto &frees : code_type.frees())
921  {
922  lambda_exprt lambda{temporary_parameter_symbols, frees};
923  lambda.add_source_location() = frees.source_location();
924  frees.swap(lambda);
925  }
926 
927  for(auto &ensures : code_type.ensures())
928  {
929  typecheck_expr(ensures);
930  implicit_typecast_bool(ensures);
931  disallow_subexpr_by_id(
932  ensures,
933  ID_loop_entry,
934  CPROVER_PREFIX "loop_entry is not allowed in postconditions.");
935  lambda_exprt lambda{temporary_parameter_symbols, ensures};
936  lambda.add_source_location() = ensures.source_location();
937  ensures.swap(lambda);
938  }
939 
940  for(const auto &parameter_sym : temporary_parameter_symbols)
941  parameter_map.erase(parameter_sym.get_identifier());
942 
943  // create a contract symbol
944  symbolt contract;
945  contract.name = "contract::" + id2string(new_symbol.name);
946  contract.base_name = new_symbol.base_name;
947  contract.pretty_name = new_symbol.pretty_name;
948  contract.is_property = true;
949  contract.type = code_type;
950  contract.mode = new_symbol.mode;
951  contract.module = module;
952  contract.location = new_symbol.location;
953 
954  auto entry = symbol_table.insert(std::move(contract));
955  if(!entry.second)
956  {
957  error().source_location = new_symbol.location;
958  error() << "contract '" << new_symbol.display_name()
959  << "' already set at " << entry.first.location.as_string()
960  << eom;
961  throw 0;
962  }
963 
964  // Remove the contract from the original symbol as we have created a
965  // dedicated contract symbol.
966  new_symbol.type.remove(ID_C_spec_assigns);
967  new_symbol.type.remove(ID_C_spec_frees);
968  new_symbol.type.remove(ID_C_spec_ensures);
969  new_symbol.type.remove(ID_C_spec_requires);
970  }
971  }
972  }
973 }
