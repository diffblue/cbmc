52 void c_typecheck_baset::typecheck_symbol(symbolt &symbol)
53 {
54  bool is_function=symbol.type.id()==ID_code;
55 
56  const typet &final_type=follow(symbol.type);
57 
58  // set a few flags
59  symbol.is_lvalue=!symbol.is_type && !symbol.is_macro;
60 
61  irep_idt root_name=symbol.base_name;
62  irep_idt new_name=symbol.name;
63 
64  if(symbol.is_file_local)
65  {
66  // file-local stuff -- stays as is
67  // collisions are resolved during linking
68  }
69  else if(symbol.is_extern && !is_function)
70  {
71  // variables marked as "extern" go into the global namespace
72  // and have static lifetime
73  new_name=root_name;
74  symbol.is_static_lifetime=true;
75 
76  if(symbol.value.is_not_nil() && !symbol.is_macro)
77  {
78  // According to the C standard this should be an error, but at least some
79  // versions of Visual Studio insist to use this in their C library, and
80  // GCC just warns as well.
81  warning().source_location = symbol.value.find_source_location();
82  warning() << "'extern' symbol '" << new_name
83  << "' should not have an initializer" << eom;
84  }
85  }
86  else if(!is_function && symbol.value.id()==ID_code)
87  {
88  error().source_location=symbol.value.find_source_location();
89  error() << "only functions can have a function body" << eom;
90  throw 0;
91  }
92 
93  // set the pretty name
94  if(symbol.is_type && final_type.id() == ID_struct)
95  {
96  symbol.pretty_name="struct "+id2string(symbol.base_name);
97  }
98  else if(symbol.is_type && final_type.id() == ID_union)
99  {
100  symbol.pretty_name="union "+id2string(symbol.base_name);
101  }
102  else if(symbol.is_type && final_type.id() == ID_c_enum)
103  {
104  symbol.pretty_name="enum "+id2string(symbol.base_name);
105  }
106  else
107  {
108  symbol.pretty_name=new_name;
109  }
110 
111  if(!symbol.is_type && !symbol.is_extern && symbol.type.id() == ID_empty)
112  {
113  error().source_location = symbol.location;
114  error() << "void-typed symbol not permitted" << eom;
115  throw 0;
116  }
117 
118  // see if we have it already
119  symbol_table_baset::symbolst::const_iterator old_it =
120  symbol_table.symbols.find(symbol.name);
121 
122  if(old_it==symbol_table.symbols.end())
123  {
124  // just put into symbol_table
125  symbolt *new_symbol;
126  move_symbol(symbol, new_symbol);
127 
128  typecheck_new_symbol(*new_symbol);
129  }
130  else
131  {
132  if(old_it->second.is_type!=symbol.is_type)
133  {
134  error().source_location=symbol.location;
135  error() << "redeclaration of '" << symbol.display_name()
136  << "' as a different kind of symbol" << eom;
137  throw 0;
138  }
139 
140  symbolt &existing_symbol = symbol_table.get_writeable_ref(symbol.name);
141  if(symbol.is_type)
142  typecheck_redefinition_type(existing_symbol, symbol);
143  else
144  {
145  if(
146  (!old_it->second.is_static_lifetime || !symbol.is_static_lifetime) &&
147  symbol.type.id() != ID_code)
148  {
149  error().source_location = symbol.location;
150  error() << "redeclaration of '" << symbol.display_name()
151  << "' with no linkage" << eom;
152  throw 0;
153  }
154 
155  typecheck_redefinition_non_type(existing_symbol, symbol);
156  }
157  }

190 void c_typecheck_baset::typecheck_redefinition_type(
191  symbolt &old_symbol,
192  symbolt &new_symbol)
193 {
194  const typet &final_old=follow(old_symbol.type);
195  const typet &final_new=follow(new_symbol.type);
196 
197  // see if we had something incomplete before
198  if(
199  (final_old.id() == ID_struct &&
200  to_struct_type(final_old).is_incomplete()) ||
201  (final_old.id() == ID_union && to_union_type(final_old).is_incomplete()) ||
202  (final_old.id() == ID_c_enum && to_c_enum_type(final_old).is_incomplete()))
203  {
204  // is the new one complete?
205  if(
206  final_new.id() == final_old.id() &&
207  ((final_new.id() == ID_struct &&
208  !to_struct_type(final_new).is_incomplete()) ||
209  (final_new.id() == ID_union &&
210  !to_union_type(final_new).is_incomplete()) ||
211  (final_new.id() == ID_c_enum &&
212  !to_c_enum_type(final_new).is_incomplete())))
213  {
214  // overwrite location
215  old_symbol.location=new_symbol.location;
216 
217  // move body
218  old_symbol.type.swap(new_symbol.type);
219  }
220  else if(new_symbol.type.id()==old_symbol.type.id())
221  return; // ignore
222  else
223  {
224  error().source_location=new_symbol.location;
225  error() << "conflicting definition of type symbol '"
226  << new_symbol.display_name() << '\'' << eom;
227  throw 0;
228  }
229  }
230  else
231  {
232  // see if new one is just a tag
233  if(
234  (final_new.id() == ID_struct &&
235  to_struct_type(final_new).is_incomplete()) ||
236  (final_new.id() == ID_union &&
237  to_union_type(final_new).is_incomplete()) ||
238  (final_new.id() == ID_c_enum &&
239  to_c_enum_type(final_new).is_incomplete()))
240  {
241  if(final_old.id() == final_new.id())
242  {
243  // just ignore silently
244  }
245  else
246  {
247  // arg! new tag type
248  error().source_location=new_symbol.location;
249  error() << "conflicting definition of tag symbol '"
250  << new_symbol.display_name() << '\'' << eom;
251  throw 0;
252  }
253  }
254  else if(config.ansi_c.os==configt::ansi_ct::ost::OS_WIN &&
255  final_new.id()==ID_c_enum && final_old.id()==ID_c_enum)
256  {
257  // under Windows, ignore this silently;
258  // MSC doesn't think this is a problem, but GCC complains.
259  }
260  else if(
261  config.ansi_c.os == configt::ansi_ct::ost::OS_WIN &&
262  final_new.id() == ID_pointer && final_old.id() == ID_pointer &&
263  follow(to_pointer_type(final_new).base_type()).id() == ID_c_enum &&
264  follow(to_pointer_type(final_old).base_type()).id() == ID_c_enum)
265  {
266  // under Windows, ignore this silently;
267  // MSC doesn't think this is a problem, but GCC complains.
268  }
269  else
270  {
271  // see if it changed
272  if(follow(new_symbol.type)!=follow(old_symbol.type))
273  {
274  error().source_location=new_symbol.location;
275  error() << "type symbol '" << new_symbol.display_name()
276  << "' defined twice:\n"
277  << "Original: " << to_string(old_symbol.type) << "\n"
278  << " New: " << to_string(new_symbol.type) << eom;
279  throw 0;
280  }
281  }
282  }

285 static bool is_instantiation_of_flexible_array(
286  const struct_typet &old_type,
287  const struct_typet &new_type)
288 {
289  const struct_typet::componentst &old_components = old_type.components();
290  const struct_typet::componentst &new_components = new_type.components();
291 
292  if(old_components.size() != new_components.size())
293  return false;
294 
295  if(old_components.empty())
296  return false;
297 
298  for(std::size_t i = 0; i < old_components.size() - 1; ++i)
299  {
300  if(old_components[i].type() != new_components[i].type())
301  return false;
302  }
303 
304  if(
305  old_components.back().type().id() != ID_array ||
306  new_components.back().type().id() != ID_array)
307  {
308  return false;
309  }
310 
311  const auto &old_array_type = to_array_type(old_components.back().type());
312  const auto &new_array_type = to_array_type(new_components.back().type());
313 
314  return old_array_type.element_type() == new_array_type.element_type() &&
315  old_array_type.get_bool(ID_C_flexible_array_member) &&
316  new_array_type.get_bool(ID_C_flexible_array_member) &&
317  (old_array_type.size().is_nil() || old_array_type.size().is_zero());
318 }

320 void c_typecheck_baset::typecheck_redefinition_non_type(
321  symbolt &old_symbol,
322  symbolt &new_symbol)
323 {
324  const typet &final_old=follow(old_symbol.type);
325  const typet &initial_new=follow(new_symbol.type);
326 
327  if(
328  final_old.id() == ID_array &&
329  to_array_type(final_old).size().is_not_nil() &&
330  initial_new.id() == ID_array &&
331  to_array_type(initial_new).size().is_nil() &&
332  to_array_type(final_old).element_type() ==
333  to_array_type(initial_new).element_type())
334  {
335  // this is ok, just use old type
336  new_symbol.type=old_symbol.type;
337  }
338  else if(
339  final_old.id() == ID_array && to_array_type(final_old).size().is_nil() &&
340  initial_new.id() == ID_array &&
341  to_array_type(initial_new).size().is_not_nil() &&
342  to_array_type(final_old).element_type() ==
343  to_array_type(initial_new).element_type())
344  {
345  // update the type to enable the use of sizeof(x) on the
346  // right-hand side of a definition of x
347  old_symbol.type=new_symbol.type;
348  }
349 
350  // do initializer, this may change the type
351  if(new_symbol.type.id() != ID_code && !new_symbol.is_macro)
352  do_initializer(new_symbol);
353 
354  const typet &final_new=follow(new_symbol.type);
355 
356  // K&R stuff?
357  if(old_symbol.type.id()==ID_KnR)
358  {
359  // check the type
360  if(final_new.id()==ID_code)
361  {
362  error().source_location=new_symbol.location;
363  error() << "function type not allowed for K&R function parameter"
364  << eom;
365  throw 0;
366  }
367 
368  // fix up old symbol -- we now got the type
369  old_symbol.type=new_symbol.type;
370  return;
371  }
372 
373  if(final_new.id()==ID_code)
374  {
375  if(final_old.id()!=ID_code)
376  {
377  error().source_location=new_symbol.location;
378  error() << "error: function symbol '" << new_symbol.display_name()
379  << "' redefined with a different type:\n"
380  << "Original: " << to_string(old_symbol.type) << "\n"
381  << " New: " << to_string(new_symbol.type) << eom;
382  throw 0;
383  }
384 
385  code_typet &old_ct=to_code_type(old_symbol.type);
386  code_typet &new_ct=to_code_type(new_symbol.type);
387 
388  if(
389  old_ct.return_type() != new_ct.return_type() &&
390  !old_ct.get_bool(ID_C_incomplete) &&
391  new_ct.return_type().id() != ID_constructor &&
392  new_ct.return_type().id() != ID_destructor)
393  {
394  if(
395  old_ct.return_type().id() == ID_constructor ||
396  old_ct.return_type().id() == ID_destructor)
397  {
398  new_ct = old_ct;
399  }
400  else
401  {
402  throw invalid_source_file_exceptiont{
403  "function symbol '" + id2string(new_symbol.display_name()) +
404  "' redefined with a different type:\n" +
405  "Original: " + to_string(old_symbol.type) + "\n" +
406  " New: " + to_string(new_symbol.type),
407  new_symbol.location};
408  }
409  }
410  const bool inlined = old_ct.get_inlined() || new_ct.get_inlined();
411 
412  if(old_ct.has_ellipsis() && !new_ct.has_ellipsis())
413  old_ct=new_ct;
414  else if(!old_ct.has_ellipsis() && new_ct.has_ellipsis())
415  {
416  if(to_code_with_contract_type(new_ct).has_contract())
417  {
418  error().source_location = new_symbol.location;
419  error() << "code contract on incomplete function re-declaration" << eom;
420  throw 0;
421  }
422  new_ct=old_ct;
423  }
424 
425  if(inlined)
426  {
427  old_ct.set_inlined(true);
428  new_ct.set_inlined(true);
429  }
430 
431  // do body
432 
433  if(new_symbol.value.is_not_nil())
434  {
435  if(old_symbol.value.is_not_nil())
436  {
437  // gcc allows re-definition if the first
438  // definition is marked as "extern inline"
439 
440  if(
441  old_ct.get_inlined() &&
442  (config.ansi_c.mode == configt::ansi_ct::flavourt::GCC ||
443  config.ansi_c.mode == configt::ansi_ct::flavourt::CLANG ||
444  config.ansi_c.mode == configt::ansi_ct::flavourt::ARM ||
445  config.ansi_c.mode == configt::ansi_ct::flavourt::VISUAL_STUDIO))
446  {
447  // overwrite "extern inline" properties
448  old_symbol.is_extern=new_symbol.is_extern;
449  old_symbol.is_file_local=new_symbol.is_file_local;
450 
451  // remove parameter declarations to avoid conflicts
452  for(const auto &old_parameter : old_ct.parameters())
453  {
454  const irep_idt &identifier = old_parameter.get_identifier();
455 
456  symbol_table_baset::symbolst::const_iterator p_s_it =
457  symbol_table.symbols.find(identifier);
458  if(p_s_it!=symbol_table.symbols.end())
459  symbol_table.erase(p_s_it);
460  }
461  }
462  else
463  {
464  error().source_location=new_symbol.location;
465  error() << "function body '" << new_symbol.display_name()
466  << "' defined twice" << eom;
467  throw 0;
468  }
469  }
470  else if(inlined)
471  {
472  // preserve "extern inline" properties
473  old_symbol.is_extern=new_symbol.is_extern;
474  old_symbol.is_file_local=new_symbol.is_file_local;
475  }
476  else if(new_symbol.is_weak)
477  {
478  // weak symbols
479  old_symbol.is_weak=true;
480  }
481 
482  if(new_symbol.is_macro)
483  old_symbol.is_macro=true;
484  else
485  typecheck_function_body(new_symbol);
486 
487  // overwrite location
488  old_symbol.location=new_symbol.location;
489 
490  // move body
491  old_symbol.value.swap(new_symbol.value);
492 
493  // overwrite type (because of parameter names)
494  old_symbol.type=new_symbol.type;
495  }
496  else if(to_code_with_contract_type(new_ct).has_contract())
497  {
498  // overwrite type to add contract, but keep the old parameter identifiers
499  // (if any)
500  auto new_parameters_it = new_ct.parameters().begin();
501  for(auto &p : old_ct.parameters())
502  {
503  if(new_parameters_it != new_ct.parameters().end())
504  {
505  new_parameters_it->set_identifier(p.get_identifier());
506  ++new_parameters_it;
507  }
508  }
509 
510  old_symbol.type = new_symbol.type;
511  }
512 
513  return;
514  }
515 
516  if(final_old!=final_new)
517  {
518  if(
519  final_old.id() == ID_array && to_array_type(final_old).size().is_nil() &&
520  final_new.id() == ID_array &&
521  to_array_type(final_new).size().is_not_nil() &&
522  to_array_type(final_old).element_type() ==
523  to_array_type(final_new).element_type())
524  {
525  old_symbol.type=new_symbol.type;
526  }
527  else if(
528  final_old.id() == ID_pointer &&
529  to_pointer_type(final_old).base_type().id() == ID_code &&
530  to_code_type(to_pointer_type(final_old).base_type()).has_ellipsis() &&
531  final_new.id() == ID_pointer &&
532  to_pointer_type(final_new).base_type().id() == ID_code)
533  {
534  // to allow
535  // int (*f) ();
536  // int (*f) (int)=0;
537  old_symbol.type=new_symbol.type;
538  }
539  else if(
540  final_old.id() == ID_pointer &&
541  to_pointer_type(final_old).base_type().id() == ID_code &&
542  final_new.id() == ID_pointer &&
543  to_pointer_type(final_new).base_type().id() == ID_code &&
544  to_code_type(to_pointer_type(final_new).base_type()).has_ellipsis())
545  {
546  // to allow
547  // int (*f) (int)=0;
548  // int (*f) ();
549  }
550  else if(
551  final_old.id() == ID_struct && final_new.id() == ID_struct &&
552  is_instantiation_of_flexible_array(
553  to_struct_type(final_old), to_struct_type(final_new)))
554  {
555  old_symbol.type = new_symbol.type;
556  }
557  else
558  {
559  error().source_location=new_symbol.location;
560  error() << "symbol '" << new_symbol.display_name()
561  << "' redefined with a different type:\n"
562  << "Original: " << to_string(old_symbol.type) << "\n"
563  << " New: " << to_string(new_symbol.type) << eom;
564  throw 0;
565  }
566  }
567  else // finals are equal
568  {
569  }
570 
571  // do value
572  if(new_symbol.value.is_not_nil())
573  {
574  // see if we already have one
575  if(old_symbol.value.is_not_nil())
576  {
577  if(
578  new_symbol.is_macro && final_new.id() == ID_c_enum &&
579  old_symbol.value.is_constant() && new_symbol.value.is_constant() &&
580  old_symbol.value.get(ID_value) == new_symbol.value.get(ID_value))
581  {
582  // ignore
583  }
584  else
585  {
586  warning().source_location = new_symbol.value.find_source_location();
587  warning() << "symbol '" << new_symbol.display_name()
588  << "' already has an initial value" << eom;
589  }
590  }
591  else
592  {
593  old_symbol.value=new_symbol.value;
594  old_symbol.type=new_symbol.type;
595  old_symbol.is_macro=new_symbol.is_macro;
596  }
597  }
598 
599  // take care of some flags
600  if(old_symbol.is_extern && !new_symbol.is_extern)
601  old_symbol.location=new_symbol.location;
602 
603  old_symbol.is_extern=old_symbol.is_extern && new_symbol.is_extern;
604 
605  // We should likely check is_volatile and
606  // is_thread_local for consistency. GCC complains if these
607  // mismatch.
608 }

610 void c_typecheck_baset::typecheck_function_body(symbolt &symbol)
611 {
612  if(symbol.value.id() != ID_code)
613  {
614  error().source_location = symbol.location;
615  error() << "function '" << symbol.name << "' is initialized with "
616  << symbol.value.id() << eom;
617  throw 0;
618  }
619 
620  code_typet &code_type = to_code_type(symbol.type);
621 
622  // reset labels
623  labels_used.clear();
624  labels_defined.clear();
625 
626  // A function declaration int foo(); does not specify the parameter list, but
627  // a function definition int foo() { ... } does fix the parameter list to be
628  // empty.
629  if(code_type.parameters().empty() && code_type.has_ellipsis())
630  code_type.remove_ellipsis();
631 
632  // fix type
633  symbol.value.type()=code_type;
634 
635  // set return type
636  return_type=code_type.return_type();
637 
638  // Add the parameter declarations into the symbol table
639  add_parameters_to_symbol_table(symbol);
640 
641  // typecheck the body code
642  typecheck_code(to_code(symbol.value));
643 
644  // check the labels
645  for(const auto &label : labels_used)
646  {
647  if(labels_defined.find(label.first) == labels_defined.end())
648  {
649  error().source_location = label.second;
650  error() << "branching label '" << label.first
651  << "' is not defined in function" << eom;
652  throw 0;
653  }
654  }

657 void c_typecheck_baset::apply_asm_label(
658  const irep_idt &asm_label,
659  symbolt &symbol)
660 {
661  const irep_idt orig_name=symbol.name;
662 
663  // restrict renaming to functions and global variables;
664  // procedure-local ones would require fixing the scope, as we
665  // do for parameters below
666  if(!asm_label.empty() &&
667  !symbol.is_type &&
668  (symbol.type.id()==ID_code || symbol.is_static_lifetime))
669  {
670  symbol.name=asm_label;
671  symbol.base_name=asm_label;
672  }
673 
674  if(symbol.name!=orig_name)
675  {
676  if(!asm_label_map.insert(
677  std::make_pair(orig_name, asm_label)).second)
678  {

772 void c_typecheck_baset::typecheck_declaration(
773  ansi_c_declarationt &declaration)
774 {
775  if(declaration.get_is_static_assert())
776  {
777  codet code(ID_static_assert);
778  code.add_source_location() = declaration.source_location();
779  code.operands().swap(declaration.operands());
780  typecheck_code(code);
781  }
782  else
783  {
784  // get storage spec
785  c_storage_spect c_storage_spec(declaration.type());
786 
787  // first typecheck the type of the declaration
788  typecheck_type(declaration.type());
789 
790  // mark as 'already typechecked'
791  already_typechecked_typet::make_already_typechecked(declaration.type());
792 
793  // Now do declarators, if any.
794  for(auto &declarator : declaration.declarators())
795  {
796  c_storage_spect full_spec(declaration.full_type(declarator));
797  full_spec|=c_storage_spec;
798 
799  declaration.set_is_inline(full_spec.is_inline);
800  declaration.set_is_static(full_spec.is_static);
801  declaration.set_is_extern(full_spec.is_extern);
802  declaration.set_is_thread_local(full_spec.is_thread_local);
803  declaration.set_is_register(full_spec.is_register);
804  declaration.set_is_typedef(full_spec.is_typedef);
805  declaration.set_is_weak(full_spec.is_weak);
806 
807  symbolt symbol = declaration.to_symbol(declarator);
808  current_symbol=symbol;
809 
810  // now check other half of type
811  typecheck_type(symbol.type);
812 
813  if(!full_spec.alias.empty())
814  {
815  if(symbol.value.is_not_nil())
816  {
817  error().source_location=symbol.location;
818  error() << "alias attribute cannot be used with a body"
819  << eom;
820  throw 0;
821  }
822 
823  // alias function need not have been declared yet, thus
824  // can't lookup
825  // also cater for renaming/placement in sections
826  const auto &renaming_entry = asm_label_map.find(full_spec.alias);
827  if(renaming_entry == asm_label_map.end())
828  symbol.value = symbol_exprt::typeless(full_spec.alias);
829  else
830  symbol.value = symbol_exprt::typeless(renaming_entry->second);
831  symbol.is_macro=true;
832  }
833 
834  if(full_spec.is_used && symbol.is_file_local)
835  {
836  // GCC __attribute__((__used__)) - do not treat those as file-local, but
837  // make sure the name is unique
838  symbol.is_file_local = false;
839 
840  symbolt symbol_for_renaming = symbol;
841  if(!full_spec.asm_label.empty())
842  symbol_for_renaming.name = full_spec.asm_label;
843  full_spec.asm_label = djb_manglert{}(
844  symbol_for_renaming,
845  id2string(symbol_for_renaming.location.get_file()));
846  }
847 
848  if(full_spec.section.empty())
849  apply_asm_label(full_spec.asm_label, symbol);
850  else
851  {
852  // section name is not empty, do a bit of parsing
853  std::string asm_name = id2string(full_spec.section);
854 
855  if(asm_name[0] == '.')
856  {
857  std::string::size_type primary_section = asm_name.find('.', 1);
858 
859  if(primary_section != std::string::npos)
860  asm_name.resize(primary_section);
861  }
862 
863  asm_name += "$$";
864 
865  if(!full_spec.asm_label.empty())
866  asm_name+=id2string(full_spec.asm_label);
867  else
868  asm_name+=id2string(symbol.name);
869 
870  apply_asm_label(asm_name, symbol);
871  }
872 
873  irep_idt identifier=symbol.name;
874  declarator.set_name(identifier);
875 
876  typecheck_symbol(symbol);
877 
878  // check the contract, if any
879  symbolt &new_symbol = symbol_table.get_writeable_ref(identifier);
880  if(
881  new_symbol.type.id() == ID_code &&
882  to_code_with_contract_type(new_symbol.type).has_contract())
883  {
884  code_with_contract_typet code_type =
885  to_code_with_contract_type(new_symbol.type);
886 
887  // ensure parameter declarations are available for type checking to
888  // succeed
889  binding_exprt::variablest temporary_parameter_symbols;
890 
891  const auto &return_type = code_type.return_type();
892  if(return_type.id() != ID_empty)
893  {
894  parameter_map[CPROVER_PREFIX "return_value"] = return_type;
895  temporary_parameter_symbols.emplace_back(
896  CPROVER_PREFIX "return_value", return_type);
897  }
898 
899  std::size_t anon_counter = 0;
900  for(auto &p : code_type.parameters())
901  {
902  // may be anonymous
903  if(p.get_base_name().empty())
904  p.set_base_name("#anon" + std::to_string(anon_counter++));
905 
906  // produce identifier
907  const irep_idt &base_name = p.get_base_name();
908  irep_idt parameter_identifier =
909  id2string(identifier) + "::" + id2string(base_name);
910 
911  p.set_identifier(parameter_identifier);
912 
913  PRECONDITION(
914  parameter_map.find(parameter_identifier) == parameter_map.end());
915  parameter_map[parameter_identifier] = p.type();
916  temporary_parameter_symbols.emplace_back(
917  parameter_identifier, p.type());
918  }
919 
920  for(auto &c_requires : code_type.c_requires())
921  {
922  typecheck_expr(c_requires);
923  implicit_typecast_bool(c_requires);
924  std::string clause_type = "preconditions";
925  check_history_expr_return_value(c_requires, clause_type);
926  check_was_freed(c_requires, clause_type);
927  lambda_exprt lambda{temporary_parameter_symbols, c_requires};
928  lambda.add_source_location() = c_requires.source_location();
929  c_requires.swap(lambda);
930  }
931 
932  typecheck_spec_assigns(code_type.c_assigns());
933  for(auto &assigns : code_type.c_assigns())
934  {
935  std::string clause_type = "assigns clauses";
936  check_history_expr_return_value(assigns, clause_type);
937  lambda_exprt lambda{temporary_parameter_symbols, assigns};
938  lambda.add_source_location() = assigns.source_location();
939  assigns.swap(lambda);
940  }
941 
942  typecheck_spec_frees(code_type.c_frees());
943  for(auto &frees : code_type.c_frees())
944  {
945  lambda_exprt lambda{temporary_parameter_symbols, frees};
946  lambda.add_source_location() = frees.source_location();
947  frees.swap(lambda);
948  }
949 
950  for(auto &ensures : code_type.c_ensures())
951  {
952  typecheck_expr(ensures);
953  implicit_typecast_bool(ensures);
954  disallow_subexpr_by_id(
955  ensures,
956  ID_loop_entry,
957  CPROVER_PREFIX "loop_entry is not allowed in postconditions.");
958  lambda_exprt lambda{temporary_parameter_symbols, ensures};
959  lambda.add_source_location() = ensures.source_location();
960  ensures.swap(lambda);
961  }
962 
963  for(const auto &parameter_sym : temporary_parameter_symbols)
964  parameter_map.erase(parameter_sym.get_identifier());
965 
966  // create a contract symbol
967  symbolt contract;
968  contract.name = "contract::" + id2string(new_symbol.name);
969  contract.base_name = new_symbol.base_name;
970  contract.pretty_name = new_symbol.pretty_name;
971  contract.is_property = true;
972  contract.type = code_type;
973  contract.mode = new_symbol.mode;
974  contract.module = module;
975  contract.location = new_symbol.location;
976 
977  auto entry = symbol_table.insert(std::move(contract));
978  if(!entry.second)
979  {
980  error().source_location = new_symbol.location;
981  error() << "contract '" << new_symbol.display_name()
982  << "' already set at " << entry.first.location.as_string()
983  << eom;
984  throw 0;
985  }
986 
987  // Remove the contract from the original symbol as we have created a
988  // dedicated contract symbol.
989  new_symbol.type.remove(ID_C_spec_assigns);
990  new_symbol.type.remove(ID_C_spec_frees);
991  new_symbol.type.remove(ID_C_spec_ensures);
992  new_symbol.type.remove(ID_C_spec_requires);
993  }
994  }
995  }
