226 void remove_exceptionst::instrument_exception_handler(
227  goto_programt &goto_program,
228  const goto_programt::targett &instr_it,
229  bool may_catch)
230 {
231  PRECONDITION(instr_it->type() == CATCH);
232 
233  if(may_catch)
234  {
235  // retrieve the exception variable
236  const exprt &thrown_exception_local =
237  to_code_landingpad(instr_it->code()).catch_expr();
238 
239  const symbol_exprt thrown_global_symbol=
240  get_inflight_exception_global();
241  // next we reset the exceptional return to NULL
242  null_pointer_exprt null_voidptr((pointer_type(java_void_type())));
243 
244  // add the assignment @inflight_exception = NULL
245  goto_program.insert_after(
246  instr_it,
247  goto_programt::make_assignment(
248  code_assignt(thrown_global_symbol, null_voidptr),
249  instr_it->source_location()));
250 
251  // add the assignment exc = @inflight_exception (before the null assignment)
252  goto_program.insert_after(
253  instr_it,
254  goto_programt::make_assignment(
255  code_assignt(
256  thrown_exception_local,
257  typecast_exprt(thrown_global_symbol, thrown_exception_local.type())),
258  instr_it->source_location()));
259  }
260 
261  instr_it->turn_into_skip();

285 goto_programt::targett remove_exceptionst::find_universal_exception(
286  const remove_exceptionst::stack_catcht &stack_catch,
287  goto_programt &goto_program,
288  std::size_t &universal_try,
289  std::size_t &universal_catch)
290 {
291  for(std::size_t i=stack_catch.size(); i>0;)
292  {
293  i--;
294  for(std::size_t j=0; j<stack_catch[i].size(); ++j)
295  {
296  if(stack_catch[i][j].first.empty())
297  {
298  // Record the position of the default behaviour as any further catches
299  // will not capture the throw
300  universal_try=i;
301  universal_catch=j;
302 
303  // Universal handler. Highest on the stack takes
304  // precedence, so overwrite any we've already seen:
305  return stack_catch[i][j].second;
306  }
307  }
308  }
309  // Unless we have a universal exception handler, jump to end of function
310  return goto_program.get_end_function();
311 }

323 void remove_exceptionst::add_exception_dispatch_sequence(
324  const irep_idt &function_identifier,
325  goto_programt &goto_program,
326  const goto_programt::targett &instr_it,
327  const remove_exceptionst::stack_catcht &stack_catch,
328  const std::vector<symbol_exprt> &locals)
329 {
330  // Jump to the universal handler or function end, as appropriate.
331  // This will appear after the GOTO-based dynamic dispatch below
332  goto_programt::targett default_dispatch=goto_program.insert_after(instr_it);
333 
334  // find the symbol corresponding to the caught exceptions
335  symbol_exprt exc_thrown =
336  get_inflight_exception_global();
337 
338  std::size_t default_try=0;
339  std::size_t default_catch=(!stack_catch.empty()) ? stack_catch[0].size() : 0;
340 
341  goto_programt::targett default_target=
342  find_universal_exception(stack_catch, goto_program,
343  default_try, default_catch);
344 
345  // add GOTOs implementing the dynamic dispatch of the
346  // exception handlers.
347  // The first loop is in forward order because the insertion reverses the order
348  // we note that try1{ try2 {} catch2c {} catch2d {}} catch1a() {} catch1b{}
349  // must catch in the following order: 2c 2d 1a 1b hence the numerical index
350  // is reversed whereas the letter ordering remains the same.
351  for(std::size_t i=default_try; i<stack_catch.size(); i++)
352  {
353  for(std::size_t j=(i==default_try) ? default_catch : stack_catch[i].size();
354  j>0;)
355  {
356  j--;
357  goto_programt::targett new_state_pc=
358  stack_catch[i][j].second;
359  if(!stack_catch[i][j].first.empty())
360  {
361  // use instanceof to check that this is the correct handler
362  struct_tag_typet type(stack_catch[i][j].first);
363 
364  java_instanceof_exprt check(exc_thrown, type);
365 
366  // Normal exception handler, make an instanceof check.
367  goto_programt::targett t_exc = goto_program.insert_after(
368  instr_it,
369  goto_programt::make_goto(
370  new_state_pc, check, instr_it->source_location()));
371 
372  if(remove_added_instanceof)
373  {
374  remove_instanceof(
375  function_identifier,
376  t_exc,
377  goto_program,
378  symbol_table,
379  *class_hierarchy,
380  message_handler);
381  }
382  }
383  }
384  }
385 
386  *default_dispatch = goto_programt::make_goto(
387  default_target, true_exprt(), instr_it->source_location());
388 
389  // add dead instructions
390  for(const auto &local : locals)
391  {
392  goto_program.insert_after(
393  instr_it, goto_programt::make_dead(local, instr_it->source_location()));
394  }
395 }

430 remove_exceptionst::instrument_function_call(
431  const irep_idt &function_identifier,
432  goto_programt &goto_program,
433  const goto_programt::targett &instr_it,

448

452  get_inflight_exception_global(),

469

471  goto_program.insert_after(
472  instr_it,
473  goto_programt::make_goto(
474  next_it,

476  instr_it->source_location()));

488 void remove_exceptionst::instrument_exceptions(
489  const irep_idt &function_identifier,
490  goto_programt &goto_program)
491 {
492  stack_catcht stack_catch; // stack of try-catch blocks
493  std::vector<std::vector<symbol_exprt>> stack_locals; // stack of local vars
494  std::vector<symbol_exprt> locals;
495 
496  if(goto_program.empty())
497  return;
498 
499  bool program_or_callees_may_throw =
500  function_or_callees_may_throw(goto_program);
501 
502  bool did_something = false;
503  bool added_goto_instruction = false;
504 
505  Forall_goto_program_instructions(instr_it, goto_program)
506  {
507  if(instr_it->is_decl())
508  {
509  locals.push_back(instr_it->decl_symbol());
510  }
511  // Is it a handler push/pop or catch landing-pad?
512  else if(instr_it->type() == CATCH)
513  {
514  const irep_idt &statement = instr_it->code().get_statement();
515  // Is it an exception landing pad (start of a catch block)?
516  if(statement==ID_exception_landingpad)
517  {
518  instrument_exception_handler(
519  goto_program, instr_it, program_or_callees_may_throw);
520  }
521  // Is it a catch handler pop?
522  else if(statement==ID_pop_catch)
523  {
524  // pop the local vars stack
525  if(!stack_locals.empty())
526  {
527  locals=stack_locals.back();
528  stack_locals.pop_back();
529  }
530  // pop from the stack if possible
531  if(!stack_catch.empty())
532  {
533  stack_catch.pop_back();
534  }
535  else
536  {
537 #ifdef DEBUG
538  std::cout << "Remove exceptions: empty stack\n";
539 #endif
540  }
541  }
542  // Is it a catch handler push?
543  else if(statement==ID_push_catch)
544  {
545  stack_locals.push_back(locals);
546  locals.clear();
547 
548  remove_exceptionst::catch_handlerst exception;
549  stack_catch.push_back(exception);
550  remove_exceptionst::catch_handlerst &last_exception=
551  stack_catch.back();
552 
553  // copy targets
554  const code_push_catcht::exception_listt &exception_list =
555  to_code_push_catch(instr_it->code()).exception_list();
556 
557  // The target list can be empty if `finish_catch_push_targets` found that
558  // the targets were unreachable (in which case no exception can truly
559  // be thrown at runtime)
560  INVARIANT(
561  instr_it->targets.empty() ||
562  exception_list.size()==instr_it->targets.size(),
563  "`exception_list` should contain current instruction's targets");
564 
565  // Fill the map with the catch type and the target
566  unsigned i=0;
567  for(auto target : instr_it->targets)
568  {
569  last_exception.push_back(
570  std::make_pair(exception_list[i].get_tag(), target));
571  i++;
572  }
573  }
574  else
575  {
576  INVARIANT(
577  false,
578  "CATCH opcode should be one of push-catch, pop-catch, landingpad");
579  }
580 
581  instr_it->turn_into_skip();
582  did_something = true;
583  }
584  else if(instr_it->type() == THROW)
585  {
586  did_something = instrument_throw(
587  function_identifier, goto_program, instr_it, stack_catch, locals);
588  }
589  else if(instr_it->type() == FUNCTION_CALL)
590  {
591  instrumentation_resultt result =
592  instrument_function_call(
593  function_identifier, goto_program, instr_it, stack_catch, locals);
594  did_something = result != instrumentation_resultt::DID_NOTHING;
595  added_goto_instruction =
596  result == instrumentation_resultt::ADDED_CODE_WITH_MAY_THROW;
597  }
598  }
599 
600  // INITIALIZE_FUNCTION should not contain any exception handling branches for
601  // two reasons: (1) with symex-driven lazy loading it means that code that
602  // references @inflight_exception might be generated before
603  // @inflight_exception is initialized; (2) symex can analyze
604  // INITIALIZE_FUNCTION much faster if it doesn't contain any branching.
605  INVARIANT(
606  function_identifier != INITIALIZE_FUNCTION || !added_goto_instruction,
607  INITIALIZE_FUNCTION " should not contain any exception handling branches");
608 
609  if(did_something)
610  remove_skip(goto_program);
611 }
