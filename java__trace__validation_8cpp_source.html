95 bool check_struct_structure(const struct_exprt &expr)
96 {
97  if(!expr.has_operands())
98  return false;
99  if(const auto sub_struct = expr_try_dynamic_cast<struct_exprt>(expr.op0()))
100  check_struct_structure(*sub_struct);
101  else if(
102  !can_cast_expr<constant_exprt>(expr.op0()) &&
103  !can_cast_expr<annotated_pointer_constant_exprt>(expr.op0()))
104  {
105  return false;
106  }
107  if(
108  expr.operands().size() > 1 && std::any_of(
109  ++expr.operands().begin(),
110  expr.operands().end(),
111  [&](const exprt &operand) {
112  return operand.id() != ID_constant &&
113  operand.id() !=
114  ID_annotated_pointer_constant;
115  }))
116  {
117  return false;
118  }
119  return true;
120 }

146 static void check_lhs_assumptions(
147  const exprt &lhs,
148  const namespacet &ns,
149  const validation_modet vm)
150 {
151  DATA_CHECK_WITH_DIAGNOSTICS(
152  vm,
153  valid_lhs_expr_high_level(lhs),
154  "LHS",
155  lhs.pretty(),
156  "Unsupported expression.");
157  // check member lhs structure
158  if(const auto member = expr_try_dynamic_cast<member_exprt>(lhs))
159  {
160  DATA_CHECK_WITH_DIAGNOSTICS(
161  vm,
162  check_member_structure(*member),
163  "LHS",
164  lhs.pretty(),
165  "Expecting a member with nested symbol operand.");
166  }
167  // check symbol lhs structure
168  else if(const auto symbol = expr_try_dynamic_cast<symbol_exprt>(lhs))
169  {
170  DATA_CHECK_WITH_DIAGNOSTICS(
171  vm,
172  check_symbol_structure(*symbol),
173  "LHS",
174  lhs.pretty(),
175  "Expecting a symbol with non-empty identifier.");
176  }
177  // check index lhs structure
178  else if(const auto index = expr_try_dynamic_cast<index_exprt>(lhs))
179  {
180  DATA_CHECK_WITH_DIAGNOSTICS(
181  vm,
182  check_index_structure(*index),
183  "LHS",
184  lhs.pretty(),
185  "Expecting an index expression with a symbol array and constant or "
186  "symbol index value.");
187  }
188  // check byte extract lhs structure
189  else if(const auto byte = expr_try_dynamic_cast<byte_extract_exprt>(lhs))
190  {
191  DATA_CHECK_WITH_DIAGNOSTICS(
192  vm,
193  check_index_structure(*byte),
194  "LHS",
195  lhs.pretty(),
196  "Expecting a byte extract expression with a symbol array and constant or "
197  "symbol index value.");
198  }
199  else
200  {
201  DATA_CHECK_WITH_DIAGNOSTICS(
202  vm,
203  false,
204  "LHS",
205  lhs.pretty(),
206  "Expression does not meet any trace assumptions.");
207  }
208 }

210 static void check_rhs_assumptions(
211  const exprt &rhs,
212  const namespacet &ns,
213  const validation_modet vm)
214 {
215  DATA_CHECK_WITH_DIAGNOSTICS(
216  vm,
217  valid_rhs_expr_high_level(rhs),
218  "RHS",
219  rhs.pretty(),
220  "Unsupported expression.");
221  // check address_of rhs structure (String only)
222  if(const auto address = expr_try_dynamic_cast<address_of_exprt>(rhs))
223  {
224  DATA_CHECK_WITH_DIAGNOSTICS(
225  vm,
226  check_address_structure(*address),
227  "RHS",
228  rhs.pretty(),
229  "Expecting an address of with nested symbol.");
230  }
231  // check symbol rhs structure (String only)
232  else if(const auto symbol_expr = expr_try_dynamic_cast<symbol_exprt>(rhs))
233  {
234  DATA_CHECK_WITH_DIAGNOSTICS(
235  vm,
236  check_symbol_structure(*symbol_expr),
237  "RHS",
238  rhs.pretty(),
239  "Expecting a symbol with non-empty identifier.");
240  }
241  // check struct rhs structure
242  else if(const auto struct_expr = expr_try_dynamic_cast<struct_exprt>(rhs))
243  {
244  DATA_CHECK_WITH_DIAGNOSTICS(
245  vm,
246  check_struct_structure(*struct_expr),
247  "RHS",
248  rhs.pretty(),
249  "Expecting all non-base class operands to be constants.");
250  }
251  // check array rhs structure
252  else if(can_cast_expr<array_exprt>(rhs))
253  {
254  // seems no check is required.
255  }
256  // check array rhs structure
257  else if(can_cast_expr<array_list_exprt>(rhs))
258  {
259  // seems no check is required.
260  }
261  // check constant rhs structure
262  else if(const auto constant_expr = expr_try_dynamic_cast<constant_exprt>(rhs))
263  {
264  DATA_CHECK_WITH_DIAGNOSTICS(
265  vm,
266  check_constant_structure(*constant_expr),
267  "RHS",
268  rhs.pretty(),
269  "Expecting a constant expression to have no operands and a non-empty "
270  "value.");
271  }
272  // check pointer constant rhs structure
273  else if(
274  const auto constant_expr =
275  expr_try_dynamic_cast<annotated_pointer_constant_exprt>(rhs))
276  {
277  DATA_CHECK_WITH_DIAGNOSTICS(
278  vm,
279  check_annotated_pointer_constant_structure(*constant_expr),
280  "RHS",
281  rhs.pretty(),
282  "Expecting the operand of an annotated pointer constant expression "
283  "to be a constant, address_of, or plus expression.");
284  }
285  // check byte extract rhs structure
286  else if(const auto byte = expr_try_dynamic_cast<byte_extract_exprt>(rhs))
287  {
288  DATA_CHECK_WITH_DIAGNOSTICS(
289  vm,
290  byte->operands().size() == 2,
291  "RHS",
292  rhs.pretty(),
293  "Expecting a byte extract with two operands.");
294  DATA_CHECK_WITH_DIAGNOSTICS(
295  vm,
296  can_cast_expr<constant_exprt>(simplify_expr(byte->op(), ns)),
297  "RHS",
298  rhs.pretty(),
299  "Expecting a byte extract with constant value.");
300  DATA_CHECK_WITH_DIAGNOSTICS(
301  vm,
302  can_cast_expr<constant_exprt>(simplify_expr(byte->offset(), ns)),
303  "RHS",
304  rhs.pretty(),
305  "Expecting a byte extract with constant index.");
306  }
307  else
308  {
309  DATA_CHECK_WITH_DIAGNOSTICS(
310  vm,
311  false,
312  "RHS",
313  rhs.pretty(),
314  "Expression does not meet any trace assumptions.");
315  }
