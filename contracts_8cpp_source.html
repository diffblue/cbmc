63  const auto loop_head_location = loop_head->source_location();
64 
65  // Vector representing a (possibly multidimensional) decreases clause
66  const auto &decreases_clause_exprs = decreases_clause.operands();
67

69  // at the start of and end of a loop body
70  std::vector<symbol_exprt> old_decreases_vars, new_decreases_vars;
71 
72  // replace bound variables by fresh instances
73  if(has_subexpr(invariant, ID_exists) || has_subexpr(invariant, ID_forall))
74  add_quantified_variable(symbol_table, invariant, mode);
75 
76  // instrument
77  //
78  // ... preamble ...
79  // HEAD:
80  // ... eval guard ...
81  // if (!guard)
82  // goto EXIT;
83  // ... loop body ...
84  // goto HEAD;
85  // EXIT:
86  // ... postamble ...
87  //
88  // to
89  //
90  // ... preamble ...
91  // ,- initialize loop_entry history vars;
92  // | entered_loop = false
93  // loop assigns check | initial_invariant_val = invariant_expr;
94  // - unchecked, temps | in_base_case = true;
95  // func assigns check | snapshot (write_set);
96  // - disabled via pragma | goto HEAD;
97  // | STEP:
98  // --. | assert (initial_invariant_val);
99  // loop assigns check | | in_base_case = false;
100  // - not applicable >======= havoc (assigns_set);
101  // func assigns check | | assume (invariant_expr);
102  // + deferred | `- old_variant_val = decreases_clause_expr;
103  // --' * HEAD:
104  // loop assigns check ,- ... eval guard ...
105  // + assertions added | if (!guard)
106  // func assigns check | goto EXIT;
107  // - disabled via pragma `- ... loop body ...
108  // ,- entered_loop = true
109  // | if (in_base_case)
110  // | goto STEP;
111  // loop assigns check | assert (invariant_expr);
112  // - unchecked, temps | new_variant_val = decreases_clause_expr;
113  // func assigns check | assert (new_variant_val < old_variant_val);
114  // - disabled via pragma | dead old_variant_val, new_variant_val;
115  // | * assume (false);
116  // | * EXIT:
117  // To be inserted at ,~~~|~~~~ assert (entered_loop ==> !in_base_case);
118  // every unique EXIT | | dead loop_entry history vars, in_base_case;
119  // (break, goto etc.) `~~~`- ~~ dead initial_invariant_val, entered_loop;
120  // ... postamble ..
121  //
122  // Asterisks (*) denote anchor points (goto targets) for instrumentations.
123  // We insert generated code above and/below these targets.
124  //
125  // Assertions for assigns clause checking are inserted in the loop body.
126 
127  // an intermediate goto_program to store generated instructions
128  // to be inserted before the loop head
129  goto_programt pre_loop_head_instrs;
130 
131  // Process "loop_entry" history variables.
132  // We find and replace all "__CPROVER_loop_entry" subexpressions in invariant.
133  std::map<exprt, exprt> history_var_map;
134  replace_history_parameter(
135  symbol_table,
136  invariant,
137  history_var_map,
138  loop_head_location,
139  mode,
140  pre_loop_head_instrs,
141  ID_loop_entry);
142 
143  // Create a temporary to track if we entered the loop,
144  // i.e., the loop guard was satisfied.
145  const auto entered_loop =
146  new_tmp_symbol(
147  bool_typet(), loop_head_location, mode, symbol_table, "__entered_loop")
148  .symbol_expr();
149  pre_loop_head_instrs.add(
150  goto_programt::make_decl(entered_loop, loop_head_location));
151  pre_loop_head_instrs.add(
152  goto_programt::make_assignment(entered_loop, false_exprt{}));
153 
154  // Create a snapshot of the invariant so that we can check the base case,
155  // if the loop is not vacuous and must be abstracted with contracts.
156  const auto initial_invariant_val =
157  new_tmp_symbol(
158  bool_typet(), loop_head_location, mode, symbol_table, "__init_invariant")
159  .symbol_expr();
160  pre_loop_head_instrs.add(
161  goto_programt::make_decl(initial_invariant_val, loop_head_location));
162  {
163  // Although the invariant at this point will not have side effects,
164  // it is still a C expression, and needs to be "goto_convert"ed.
165  // Note that this conversion may emit many GOTO instructions.
166  code_assignt initial_invariant_value_assignment{
167  initial_invariant_val, invariant};
168  initial_invariant_value_assignment.add_source_location() =
169  loop_head_location;
170  converter.goto_convert(
171  initial_invariant_value_assignment, pre_loop_head_instrs, mode);
172  }
173 
174  // Create a temporary variable to track base case vs inductive case
175  // instrumentation of the loop.
176  const auto in_base_case =
177  new_tmp_symbol(
178  bool_typet(), loop_head_location, mode, symbol_table, "__in_base_case")
179  .symbol_expr();
180  pre_loop_head_instrs.add(
181  goto_programt::make_decl(in_base_case, loop_head_location));
182  pre_loop_head_instrs.add(
183  goto_programt::make_assignment(in_base_case, true_exprt{}));
184 
185  // CAR snapshot instructions for checking assigns clause
186  goto_programt snapshot_instructions;
187 
188  loop_cfg_infot cfg_info(goto_function, loop);
189 
190  // track static local symbols
191  instrument_spec_assignst instrument_spec_assigns(
192  function_name,
193  goto_functions,
194  cfg_info,
195  symbol_table,
196  log.get_message_handler());
197 
198  instrument_spec_assigns.track_static_locals_between(
199  loop_head, loop_end, snapshot_instructions);
200 
201  // set of targets to havoc
202  assignst to_havoc;
203 
204  if(assigns_clause.is_nil())
205  {
206  // No assigns clause was specified for this loop.
207  // Infer memory locations assigned by the loop from the loop instructions
208  // and the inferred aliasing relation.
209  try
210  {
211  get_assigns(local_may_alias, loop, to_havoc);
212 
213  // remove loop-local symbols from the inferred set
214  cfg_info.erase_locals(to_havoc);
215 
216  // If the set contains pairs (i, a[i]),
217  // we widen them to (i, __CPROVER_POINTER_OBJECT(a))
218  widen_assigns(to_havoc);
219 
220  log.debug() << "No loop assigns clause provided. Inferred targets: {";
221  // Add inferred targets to the loop assigns clause.
222  bool ran_once = false;
223  for(const auto &target : to_havoc)
224  {
225  if(ran_once)
226  log.debug() << ", ";
227  ran_once = true;
228  log.debug() << format(target);
229  instrument_spec_assigns.track_spec_target(
230  target, snapshot_instructions);
231  }
232  log.debug() << "}" << messaget::eom;
233 
234  instrument_spec_assigns.get_static_locals(
235  std::inserter(to_havoc, to_havoc.end()));
236  }
237  catch(const analysis_exceptiont &exc)
238  {
239  log.error() << "Failed to infer variables being modified by the loop at "
240  << loop_head_location
241  << ".\nPlease specify an assigns clause.\nReason:"
242  << messaget::eom;
243  throw exc;
244  }
245  }
246  else
247  {
248  // An assigns clause was specified for this loop.
249  // Add the targets to the set of expressions to havoc.
250  for(const auto &target : assigns_clause.operands())
251  {
252  to_havoc.insert(target);
253  instrument_spec_assigns.track_spec_target(target, snapshot_instructions);
254  }
255  }
256 
257  // Insert instrumentation
258  // This must be done before havocing the write set.
259  // FIXME: this is not true for write set targets that
260  // might depend on other write set targets.
261  pre_loop_head_instrs.destructive_append(snapshot_instructions);
262 
263  // Insert a jump to the loop head
264  // (skipping over the step case initialization code below)
265  pre_loop_head_instrs.add(
266  goto_programt::make_goto(loop_head, true_exprt{}, loop_head_location));
267 
268  // The STEP case instructions follow.
269  // We skip past it initially, because of the unconditional jump above,
270  // but jump back here if we get past the loop guard while in_base_case.
271 
272  const auto step_case_target =
273  pre_loop_head_instrs.add(goto_programt::make_assignment(
274  in_base_case, false_exprt{}, loop_head_location));
275 
276  // If we jump here, then the loop runs at least once,
277  // so add the base case assertion:
278  // assert(initial_invariant_val)
279  // We use a block scope for assertion, since it's immediately goto converted,
280  // and doesn't need to be kept around.
281  {
282  code_assertt assertion{initial_invariant_val};
283  assertion.add_source_location() = loop_head_location;
284  assertion.add_source_location().set_comment(
285  "Check loop invariant before entry");
286  converter.goto_convert(assertion, pre_loop_head_instrs, mode);
287  }
288 
289  // Insert the first block of pre_loop_head_instrs,
290  // with a pragma to disable assigns clause checking.
291  // All of the initialization code so far introduces local temporaries,
292  // which need not be checked against the enclosing scope's assigns clause.
293  goto_function.body.destructive_insert(
294  loop_head, add_pragma_disable_assigns_check(pre_loop_head_instrs));
295 
296  // Generate havocing code for assignment targets.
297  havoc_assigns_targetst havoc_gen(to_havoc, ns);
298  havoc_gen.append_full_havoc_code(loop_head_location, pre_loop_head_instrs);
299 
300  // Insert the second block of pre_loop_head_instrs: the havocing code.
301  // We do not `add_pragma_disable_assigns_check`,
302  // so that the enclosing scope's assigns clause instrumentation
303  // would pick these havocs up for inclusion (subset) checks.
304  goto_function.body.destructive_insert(loop_head, pre_loop_head_instrs);
305 
306  // Generate: assume(invariant) just after havocing
307  // We use a block scope for assumption, since it's immediately goto converted,
308  // and doesn't need to be kept around.
309  {
310  code_assumet assumption{invariant};
311  assumption.add_source_location() = loop_head_location;
312  converter.goto_convert(assumption, pre_loop_head_instrs, mode);
313  }
314 
315  // Create fresh temporary variables that store the multidimensional
316  // decreases clause's value before and after the loop
317  for(const auto &clause : decreases_clause.operands())
318  {
319  const auto old_decreases_var =
320  new_tmp_symbol(clause.type(), loop_head_location, mode, symbol_table)
321  .symbol_expr();
322  pre_loop_head_instrs.add(
323  goto_programt::make_decl(old_decreases_var, loop_head_location));
324  old_decreases_vars.push_back(old_decreases_var);
325 
326  const auto new_decreases_var =
327  new_tmp_symbol(clause.type(), loop_head_location, mode, symbol_table)
328  .symbol_expr();
329  pre_loop_head_instrs.add(
330  goto_programt::make_decl(new_decreases_var, loop_head_location));
331  new_decreases_vars.push_back(new_decreases_var);
332  }
333 
334  // TODO: Fix loop contract handling for do/while loops.
335  if(loop_end->is_goto() && !loop_end->condition().is_true())
336  {
337  log.error() << "Loop contracts are unsupported on do/while loops: "
338  << loop_head_location << messaget::eom;
339  throw 0;
340 
341  // non-deterministically skip the loop if it is a do-while loop.
342  // pre_loop_head_instrs.add(goto_programt::make_goto(
343  // loop_end, side_effect_expr_nondett(bool_typet(), loop_head_location)));
344  }
345 
346  // Generate: assignments to store the multidimensional decreases clause's
347  // value just before the loop_head
348  if(!decreases_clause.is_nil())
349  {
350  for(size_t i = 0; i < old_decreases_vars.size(); i++)
351  {
352  code_assignt old_decreases_assignment{
353  old_decreases_vars[i], decreases_clause_exprs[i]};
354  old_decreases_assignment.add_source_location() = loop_head_location;
355  converter.goto_convert(
356  old_decreases_assignment, pre_loop_head_instrs, mode);
357  }
358 
359  goto_function.body.destructive_insert(
360  loop_head, add_pragma_disable_assigns_check(pre_loop_head_instrs));
361  }
362 
363  // Insert the third and final block of pre_loop_head_instrs,
364  // with a pragma to disable assigns clause checking.
365  // The variables to store old variant value are local temporaries,
366  // which need not be checked against the enclosing scope's assigns clause.
367  goto_function.body.destructive_insert(
368  loop_head, add_pragma_disable_assigns_check(pre_loop_head_instrs));
369 
370  // Perform write set instrumentation on the entire loop.
371  instrument_spec_assigns.instrument_instructions(
372  goto_function.body,
373  loop_head,
374  loop_end,
375  [&loop](const goto_programt::targett &t) { return loop.contains(t); });
376 
377  // Now we begin instrumenting at the loop_end.
378  // `pre_loop_end_instrs` are to be inserted before `loop_end`.
379  goto_programt pre_loop_end_instrs;
380 
381  // Record that we entered the loop.
382  pre_loop_end_instrs.add(
383  goto_programt::make_assignment(entered_loop, true_exprt{}));
384 
385  // Jump back to the step case to havoc the write set, assume the invariant,
386  // and execute an arbitrary iteration.
387  pre_loop_end_instrs.add(goto_programt::make_goto(
388  step_case_target, in_base_case, loop_head_location));
389 
390  // The following code is only reachable in the step case,
391  // i.e., when in_base_case == false,
392  // because of the unconditional jump above.
393 
394  // Generate the inductiveness check:
395  // assert(invariant)
396  // We use a block scope for assertion, since it's immediately goto converted,
397  // and doesn't need to be kept around.
398  {
399  code_assertt assertion{invariant};
400  assertion.add_source_location() = loop_head_location;
401  assertion.add_source_location().set_comment(
402  "Check that loop invariant is preserved");
403  converter.goto_convert(assertion, pre_loop_end_instrs, mode);
404  }
405 
406  // Generate: assignments to store the multidimensional decreases clause's
407  // value after one iteration of the loop
408  if(!decreases_clause.is_nil())
409  {
410  for(size_t i = 0; i < new_decreases_vars.size(); i++)
411  {
412  code_assignt new_decreases_assignment{
413  new_decreases_vars[i], decreases_clause_exprs[i]};
414  new_decreases_assignment.add_source_location() = loop_head_location;
415  converter.goto_convert(
416  new_decreases_assignment, pre_loop_end_instrs, mode);
417  }
418 
419  // Generate: assertion that the multidimensional decreases clause's value
420  // after the loop is lexicographically smaller than its initial value.
421  code_assertt monotonic_decreasing_assertion{
422  generate_lexicographic_less_than_check(
423  new_decreases_vars, old_decreases_vars)};
424  monotonic_decreasing_assertion.add_source_location() = loop_head_location;
425  monotonic_decreasing_assertion.add_source_location().set_comment(
426  "Check decreases clause on loop iteration");
427  converter.goto_convert(
428  monotonic_decreasing_assertion, pre_loop_end_instrs, mode);
429 
430  // Discard the temporary variables that store decreases clause's value
431  for(size_t i = 0; i < old_decreases_vars.size(); i++)
432  {
433  pre_loop_end_instrs.add(
434  goto_programt::make_dead(old_decreases_vars[i], loop_head_location));
435  pre_loop_end_instrs.add(
436  goto_programt::make_dead(new_decreases_vars[i], loop_head_location));
437  }
438  }
439 
440  insert_before_swap_and_advance(
441  goto_function.body,
442  loop_end,
443  add_pragma_disable_assigns_check(pre_loop_end_instrs));
444 
445  // change the back edge into assume(false) or assume(guard)
446  loop_end->turn_into_assume();
447  loop_end->condition_nonconst() = boolean_negate(loop_end->condition());
448 
449  std::set<goto_programt::targett> seen_targets;
450  // Find all exit points of the loop, make temporary variables `DEAD`,
451  // and check that step case was checked for non-vacuous loops.
452  for(const auto &t : loop)
453  {
454  if(!t->is_goto())
455  continue;
456 
457  auto exit_target = t->get_target();
458  if(
459  loop.contains(exit_target) ||
460  seen_targets.find(exit_target) != seen_targets.end())
461  continue;
462 
463  seen_targets.insert(exit_target);
464 
465  goto_programt pre_loop_exit_instrs;
466  // Assertion to check that step case was checked if we entered the loop.
467  source_locationt annotated_location = loop_head_location;
468  annotated_location.set_comment(
469  "Check that loop instrumentation was not truncated");
470  pre_loop_exit_instrs.add(goto_programt::make_assertion(
471  or_exprt{not_exprt{entered_loop}, not_exprt{in_base_case}},
472  annotated_location));
473  // Instructions to make all the temporaries go dead.
474  pre_loop_exit_instrs.add(
475  goto_programt::make_dead(in_base_case, loop_head_location));
476  pre_loop_exit_instrs.add(
477  goto_programt::make_dead(initial_invariant_val, loop_head_location));
478  for(const auto &v : history_var_map)
479  {
480  pre_loop_exit_instrs.add(
481  goto_programt::make_dead(to_symbol_expr(v.second), loop_head_location));
482  }
483  // Insert these instructions, preserving the loop end target.
484  insert_before_swap_and_advance(
485  goto_function.body, exit_target, pre_loop_exit_instrs);
486  }

491 static void generate_contract_constraints(
492  symbol_tablet &symbol_table,
493  goto_convertt &converter,
494  exprt &instantiated_clause,
495  const irep_idt &mode,
496  const std::function<void(goto_programt &)> &is_fresh_update,
497  goto_programt &program,
498  const source_locationt &location)
499 {
500  if(
501  has_subexpr(instantiated_clause, ID_exists) ||
502  has_subexpr(instantiated_clause, ID_forall))
503  {
504  add_quantified_variable(symbol_table, instantiated_clause, mode);
505  }
506 
507  goto_programt constraint;
508  if(location.get_property_class() == ID_assume)
509  {
510  converter.goto_convert(code_assumet(instantiated_clause), constraint, mode);
511  }
512  else
513  {
514  converter.goto_convert(code_assertt(instantiated_clause), constraint, mode);
515  }
516  constraint.instructions.back().source_location_nonconst() = location;
517  is_fresh_update(constraint);
518  program.destructive_append(constraint);
519 }

521 static const code_with_contract_typet &
522 get_contract(const irep_idt &function, const namespacet &ns)
523 {
524  const std::string &function_str = id2string(function);
525  const auto &function_symbol = ns.lookup(function);
526 
527  const symbolt *contract_sym;
528  if(ns.lookup("contract::" + function_str, contract_sym))
529  {
530  // no contract provided in the source or just an empty assigns clause
531  return to_code_with_contract_type(function_symbol.type);
532  }
533 
534  const auto &type = to_code_with_contract_type(contract_sym->type);
535  if(type != function_symbol.type)
536  {
537  throw invalid_input_exceptiont(
538  "Contract of '" + function_str + "' has different signature.");
539  }
540 
541  return type;
542 }

544 void code_contractst::apply_function_contract(
545  const irep_idt &function,
546  const source_locationt &location,
547  goto_programt &function_body,
548  goto_programt::targett &target)
549 {
550  const auto &const_target =
551  static_cast<const goto_programt::targett &>(target);
552  // Return if the function is not named in the call; currently we don't handle
553  // function pointers.
554  PRECONDITION(const_target->call_function().id() == ID_symbol);
555 
556  const irep_idt &target_function =
557  to_symbol_expr(const_target->call_function()).get_identifier();
558  const symbolt &function_symbol = ns.lookup(target_function);
559  const code_typet &function_type = to_code_type(function_symbol.type);
560 
561  // Isolate each component of the contract.
562  const auto &type = get_contract(target_function, ns);
563 
564  // Prepare to instantiate expressions in the callee
565  // with expressions from the call site (e.g. the return value).
566  exprt::operandst instantiation_values;
567 
568  // keep track of the call's return expression to make it nondet later
569  optionalt<exprt> call_ret_opt = {};
570 
571  // if true, the call return variable variable was created during replacement
572  bool call_ret_is_fresh_var = false;
573 
574  if(function_type.return_type() != empty_typet())
575  {
576  // Check whether the function's return value is not disregarded.
577  if(target->call_lhs().is_not_nil())
578  {
579  // If so, have it replaced appropriately.
580  // For example, if foo() ensures that its return value is > 5, then
581  // rewrite calls to foo as follows:
582  // x = foo() -> assume(__CPROVER_return_value > 5) -> assume(x > 5)
583  auto &lhs_expr = const_target->call_lhs();
584  call_ret_opt = lhs_expr;
585  instantiation_values.push_back(lhs_expr);
586  }
587  else
588  {
589  // If the function does return a value, but the return value is
590  // disregarded, check if the postcondition includes the return value.
591  if(std::any_of(
592  type.ensures().begin(), type.ensures().end(), [](const exprt &e) {
593  return has_symbol_expr(
594  to_lambda_expr(e).where(), CPROVER_PREFIX "return_value", true);
595  }))
596  {
597  // The postcondition does mention __CPROVER_return_value, so mint a
598  // fresh variable to replace __CPROVER_return_value with.
599  call_ret_is_fresh_var = true;
600  const symbolt &fresh = get_fresh_aux_symbol(
601  function_type.return_type(),
602  id2string(target_function),
603  "ignored_return_value",
604  const_target->source_location(),
605  symbol_table.lookup_ref(target_function).mode,
606  ns,
607  symbol_table);
608  auto fresh_sym_expr = fresh.symbol_expr();
609  call_ret_opt = fresh_sym_expr;
610  instantiation_values.push_back(fresh_sym_expr);
611  }
612  else
613  {
614  // unused, add a dummy with the right type
615  instantiation_values.push_back(
616  exprt{ID_nil, function_type.return_type()});
617  }
618  }
619  }
620 
621  // Replace formal parameters
622  const auto &arguments = const_target->call_arguments();
623  instantiation_values.insert(
624  instantiation_values.end(), arguments.begin(), arguments.end());
625 
626  const auto &mode = function_symbol.mode;
627 
628  // new program where all contract-derived instructions are added
629  goto_programt new_program;
630 
631  is_fresh_replacet is_fresh(*this, log, target_function);
632  is_fresh.create_declarations();
633  is_fresh.add_memory_map_decl(new_program);
634 
635  // Generate: assert(requires)
636  for(const auto &clause : type.requires())
637  {
638  auto instantiated_clause =
639  to_lambda_expr(clause).application(instantiation_values);
640  source_locationt _location = clause.source_location();
641  _location.set_line(location.get_line());
642  _location.set_comment(std::string("Check requires clause of ")
643  .append(target_function.c_str())
644  .append(" in ")
645  .append(function.c_str()));
646  _location.set_property_class(ID_precondition);
647  generate_contract_constraints(
648  symbol_table,
649  converter,
650  instantiated_clause,
651  mode,
652  [&is_fresh](goto_programt &requires) {
653  is_fresh.update_requires(requires);
654  },
655  new_program,
656  _location);
657  }
658 
659  // Translate requires_contract(ptr, contract) clauses to assertions
660  for(auto &expr : type.requires_contract())
661  {
662  assert_function_pointer_obeys_contract(
663  to_function_pointer_obeys_contract_expr(
664  to_lambda_expr(expr).application(instantiation_values)),
665  ID_precondition,
666  mode,
667  new_program);
668  }
669 
670  // Generate all the instructions required to initialize history variables
671  exprt::operandst instantiated_ensures_clauses;
672  for(auto clause : type.ensures())
673  {
674  auto instantiated_clause =
675  to_lambda_expr(clause).application(instantiation_values);
676  instantiated_clause.add_source_location() = clause.source_location();
677  generate_history_variables_initialization(
678  symbol_table, instantiated_clause, mode, new_program);
679  instantiated_ensures_clauses.push_back(instantiated_clause);
680  }
681 
682  // ASSIGNS clause should not refer to any quantified variables,
683  // and only refer to the common symbols to be replaced.
684  exprt::operandst targets;
685  for(auto &target : type.assigns())
686  targets.push_back(to_lambda_expr(target).application(instantiation_values));
687 
688  // Create a sequence of non-deterministic assignments ...
689 
690  // ... for the assigns clause targets and static locals
691  goto_programt havoc_instructions;
692  function_cfg_infot cfg_info({});
693  havoc_assigns_clause_targetst havocker(
694  target_function,
695  targets,
696  goto_functions,
697  cfg_info,
698  location,
699  symbol_table,
700  log.get_message_handler());
701 
702  havocker.get_instructions(havoc_instructions);
703 
704  // ... for the return value
705  if(call_ret_opt.has_value())
706  {
707  auto &call_ret = call_ret_opt.value();
708  auto &loc = call_ret.source_location();
709  auto &type = call_ret.type();
710 
711  // Declare if fresh
712  if(call_ret_is_fresh_var)
713  havoc_instructions.add(
714  goto_programt::make_decl(to_symbol_expr(call_ret), loc));
715 
716  side_effect_expr_nondett expr(type, location);
717  havoc_instructions.add(goto_programt::make_assignment(
718  code_assignt{call_ret, std::move(expr), loc}, loc));
719  }
720 
721  // Insert havoc instructions immediately before the call site.
722  new_program.destructive_append(havoc_instructions);
723 
724  // Generate: assume(ensures)
725  for(auto &clause : instantiated_ensures_clauses)
726  {
727  if(clause.is_false())
728  {
729  throw invalid_input_exceptiont(
730  std::string("Attempt to assume false at ")
731  .append(clause.source_location().as_string())
732  .append(".\nPlease update ensures clause to avoid vacuity."));
733  }
734  source_locationt _location = clause.source_location();
735  _location.set_comment("Assume ensures clause");
736  _location.set_property_class(ID_assume);
737  generate_contract_constraints(
738  symbol_table,
739  converter,
740  clause,
741  mode,
742  [&is_fresh](goto_programt &ensures) { is_fresh.update_ensures(ensures); },
743  new_program,
744  _location);
745  }
746 
747  // Translate ensures_contract(ptr, contract) clauses to assumptions
748  for(auto &expr : type.ensures_contract())
749  {
750  assume_function_pointer_obeys_contract(
751  to_function_pointer_obeys_contract_expr(
752  to_lambda_expr(expr).application(instantiation_values)),
753  mode,
754  new_program);
755  }
756 
757  // Kill return value variable if fresh
758  if(call_ret_is_fresh_var)
759  {
760  log.conditional_output(
761  log.warning(), [&target](messaget::mstreamt &mstream) {
762  target->output(mstream);
763  mstream << messaget::eom;
764  });
765  auto dead_inst =
766  goto_programt::make_dead(to_symbol_expr(call_ret_opt.value()), location);
767  function_body.insert_before_swap(target, dead_inst);
768  ++target;
769  }
770 
771  is_fresh.add_memory_map_dead(new_program);
772 
773  // Erase original function call
774  *target = goto_programt::make_skip();
775 
776  // insert contract replacement instructions
777  insert_before_swap_and_advance(function_body, target, new_program);
778 
779  // Add this function to the set of replaced functions.
780  summarized.insert(target_function);
781 
782  // restore global goto_program consistency
783  goto_functions.update();
784 }

786 void code_contractst::apply_loop_contract(
787  const irep_idt &function_name,
788  goto_functionst::goto_functiont &goto_function)
789 {
790  const bool may_have_loops = std::any_of(
791  goto_function.body.instructions.begin(),
792  goto_function.body.instructions.end(),
793  [](const goto_programt::instructiont &instruction) {
794  return instruction.is_backwards_goto();
795  });
796 
797  if(!may_have_loops)
798  return;
799 
800  inlining_decoratort decorated(log.get_message_handler());
801  goto_function_inline(
802  goto_functions, function_name, ns, log.get_message_handler());
803 
804  INVARIANT(
805  decorated.get_recursive_call_set().size() == 0,
806  "Recursive functions found during inlining");
807 
808  // restore internal invariants
809  goto_functions.update();
810 
811  local_may_aliast local_may_alias(goto_function);
812  natural_loops_mutablet natural_loops(goto_function.body);
813 
814  // A graph node type that stores information about a loop.
815  // We create a DAG representing nesting of various loops in goto_function,
816  // sort them topologically, and instrument them in the top-sorted order.
817  //
818  // The goal here is not avoid explicit "subset checks" on nested write sets.
819  // When instrumenting in a top-sorted order,
820  // the outer loop would no longer observe the inner loop's write set,
821  // but only corresponding `havoc` statements,
822  // which can be instrumented in the usual way to achieve a "subset check".
823 
824  struct loop_graph_nodet : public graph_nodet<empty_edget>
825  {
826  const typename natural_loops_mutablet::loopt &content;
827  const goto_programt::targett head_target, end_target;
828  exprt assigns_clause, invariant, decreases_clause;
829 
830  loop_graph_nodet(
831  const typename natural_loops_mutablet::loopt &loop,
832  const goto_programt::targett head,
833  const goto_programt::targett end,
834  const exprt &assigns,
835  const exprt &inv,
836  const exprt &decreases)
837  : content(loop),
838  head_target(head),
839  end_target(end),
840  assigns_clause(assigns),
841  invariant(inv),
842  decreases_clause(decreases)
843  {
844  }
845  };
846  grapht<loop_graph_nodet> loop_nesting_graph;
847 
848  std::list<size_t> to_check_contracts_on_children;
849 
850  for(const auto &loop_head_and_content : natural_loops.loop_map)
851  {
852  const auto &loop_content = loop_head_and_content.second;
853  if(loop_content.empty())
854  continue;
855 
856  auto loop_head = loop_head_and_content.first;
857  auto loop_end = loop_head;
858 
859  // Find the last back edge to `loop_head`
860  for(const auto &t : loop_content)
861  {
862  if(
863  t->is_goto() && t->get_target() == loop_head &&
864  t->location_number > loop_end->location_number)
865  loop_end = t;
866  }
867 
868  if(loop_end == loop_head)
869  {
870  log.error() << "Could not find end of the loop starting at: "
871  << loop_head->source_location() << messaget::eom;
872  throw 0;
873  }
874 
875  exprt assigns_clause =
876  static_cast<const exprt &>(loop_end->condition().find(ID_C_spec_assigns));
877  exprt invariant = static_cast<const exprt &>(
878  loop_end->condition().find(ID_C_spec_loop_invariant));
879  exprt decreases_clause = static_cast<const exprt &>(
880  loop_end->condition().find(ID_C_spec_decreases));
881 
882  if(invariant.is_nil())
883  {
884  if(decreases_clause.is_not_nil() || assigns_clause.is_not_nil())
885  {
886  invariant = true_exprt{};
887  // assigns clause is missing; we will try to automatic inference
888  log.warning()
889  << "The loop at " << loop_head->source_location().as_string()
890  << " does not have an invariant in its contract.\n"
891  << "Hence, a default invariant ('true') is being used.\n"
892  << "This choice is sound, but verification may fail"
893  << " if it is be too weak to prove the desired properties."
894  << messaget::eom;
895  }
896  }
897  else
898  {
899  invariant = conjunction(invariant.operands());
900  if(decreases_clause.is_nil())
901  {
902  log.warning() << "The loop at "
903  << loop_head->source_location().as_string()
904  << " does not have a decreases clause in its contract.\n"
905  << "Termination of this loop will not be verified."
906  << messaget::eom;
907  }
908  }
909 
910  const auto idx = loop_nesting_graph.add_node(
911  loop_content,
912  loop_head,
913  loop_end,
914  assigns_clause,
915  invariant,
916  decreases_clause);
917 
918  if(
919  assigns_clause.is_nil() && invariant.is_nil() &&
920  decreases_clause.is_nil())
921  continue;
922 
923  to_check_contracts_on_children.push_back(idx);
924 
925  // By definition the `loop_content` is a set of instructions computed
926  // by `natural_loops` based on the CFG.
927  // Since we perform assigns clause instrumentation by sequentially
928  // traversing instructions from `loop_head` to `loop_end`,
929  // here we ensure that all instructions in `loop_content` belong within
930  // the [loop_head, loop_end] target range
931 
932  // Check 1. (i \in loop_content) ==> loop_head <= i <= loop_end
933  for(const auto &i : loop_content)
934  {
935  if(std::distance(loop_head, i) < 0 || std::distance(i, loop_end) < 0)
936  {
937  log.conditional_output(
938  log.error(), [&i, &loop_head](messaget::mstreamt &mstream) {
939  mstream << "Computed loop at " << loop_head->source_location()
940  << "contains an instruction beyond [loop_head, loop_end]:"
941  << messaget::eom;
942  i->output(mstream);
943  mstream << messaget::eom;
944  });
945  throw 0;
946  }
947  }
948  }
949 
950  for(size_t outer = 0; outer < loop_nesting_graph.size(); ++outer)
951  {
952  for(size_t inner = 0; inner < loop_nesting_graph.size(); ++inner)
953  {
954  if(inner == outer)
955  continue;
956 
957  if(loop_nesting_graph[outer].content.contains(
958  loop_nesting_graph[inner].head_target))
959  {
960  if(!loop_nesting_graph[outer].content.contains(
961  loop_nesting_graph[inner].end_target))
962  {
963  log.error()
964  << "Overlapping loops at:\n"
965  << loop_nesting_graph[outer].head_target->source_location()
966  << "\nand\n"
967  << loop_nesting_graph[inner].head_target->source_location()
968  << "\nLoops must be nested or sequential for contracts to be "
969  "enforced."
970  << messaget::eom;
971  }
972  loop_nesting_graph.add_edge(inner, outer);
973  }
974  }
975  }
976 
977  // make sure all children of a contractified loop also have contracts
978  while(!to_check_contracts_on_children.empty())
979  {
980  const auto loop_idx = to_check_contracts_on_children.front();
981  to_check_contracts_on_children.pop_front();
982 
983  const auto &loop_node = loop_nesting_graph[loop_idx];
984  if(
985  loop_node.assigns_clause.is_nil() && loop_node.invariant.is_nil() &&
986  loop_node.decreases_clause.is_nil())
987  {
988  log.error()
989  << "Inner loop at: " << loop_node.head_target->source_location()
990  << " does not have contracts, but an enclosing loop does.\n"
991  << "Please provide contracts for this loop, or unwind it first."
992  << messaget::eom;
993  throw 0;
994  }
995 
996  for(const auto child_idx : loop_nesting_graph.get_predecessors(loop_idx))
997  to_check_contracts_on_children.push_back(child_idx);
998  }
999 
1000  // Iterate over the (natural) loops in the function, in topo-sorted order,
1001  // and apply any loop contracts that we find.
1002  for(const auto &idx : loop_nesting_graph.topsort())
1003  {
1004  const auto &loop_node = loop_nesting_graph[idx];
1005  if(
1006  loop_node.assigns_clause.is_nil() && loop_node.invariant.is_nil() &&
1007  loop_node.decreases_clause.is_nil())
1008  continue;
1009 
1010  // Computed loop "contents" needs to be refreshed to include any newly
1011  // introduced instrumentation, e.g. havoc instructions for assigns clause,
1012  // on inner loops in to the outer loop's contents.
1013  // Else, during the outer loop instrumentation these instructions will be
1014  // "masked" just as any other instruction not within its "contents."
1015 
1016  goto_functions.update();
1017  natural_loops_mutablet updated_loops(goto_function.body);
1018 
1019  check_apply_loop_contracts(
1020  function_name,
1021  goto_function,
1022  local_may_alias,
1023  loop_node.head_target,
1024  loop_node.end_target,
1025  updated_loops.loop_map[loop_node.head_target],
1026  loop_node.assigns_clause,
1027  loop_node.invariant,
1028  loop_node.decreases_clause,
1029  symbol_table.lookup_ref(function_name).mode);
1030  }
1031 }

1043 void code_contractst::check_frame_conditions_function(const irep_idt &function)
1044 {
1045  // Get the function object before instrumentation.
1046  auto function_obj = goto_functions.function_map.find(function);
1047 
1048  INVARIANT(
1049  function_obj != goto_functions.function_map.end(),
1050  "Function '" + id2string(function) + "'must exist in the goto program");
1051 
1052  const auto &goto_function = function_obj->second;
1053  auto &function_body = function_obj->second.body;
1054 
1055  function_cfg_infot cfg_info(goto_function);
1056 
1057  instrument_spec_assignst instrument_spec_assigns(
1058  function,
1059  goto_functions,
1060  cfg_info,
1061  symbol_table,
1062  log.get_message_handler());
1063 
1064  // Detect and track static local variables before inlining
1065  goto_programt snapshot_static_locals;
1066  instrument_spec_assigns.track_static_locals(snapshot_static_locals);
1067 
1068  // Full inlining of the function body
1069  // Inlining is performed so that function calls to a same function
1070  // occurring under different write sets get instrumented specifically
1071  // for each write set
1072  inlining_decoratort decorated(log.get_message_handler());
1073  goto_function_inline(goto_functions, function, ns, decorated);
1074 
1075  decorated.throw_on_recursive_calls(log, 0);
1076 
1077  // Clean up possible fake loops that are due to `IF 0!=0 GOTO i` instructions
1078  simplify_gotos(function_body, ns);
1079 
1080  // Restore internal coherence in the programs
1081  goto_functions.update();
1082 
1083  INVARIANT(
1084  is_loop_free(function_body, ns, log),
1085  "Loops remain in function '" + id2string(function) +
1086  "', assigns clause checking instrumentation cannot be applied.");
1087 
1088  // Start instrumentation
1089  auto instruction_it = function_body.instructions.begin();
1090 
1091  // Inject local static snapshots
1092  insert_before_swap_and_advance(
1093  function_body, instruction_it, snapshot_static_locals);
1094 
1095  // Track targets mentioned in the specification
1096  const symbolt &function_symbol = ns.lookup(function);
1097  const code_typet &function_type = to_code_type(function_symbol.type);
1098  exprt::operandst instantiation_values;
1099  // assigns clauses cannot refer to the return value, but we still need an
1100  // element in there to apply the lambda function consistently
1101  if(function_type.return_type() != empty_typet())
1102  instantiation_values.push_back(exprt{ID_nil, function_type.return_type()});
1103  for(const auto &param : function_type.parameters())
1104  {
1105  instantiation_values.push_back(
1106  ns.lookup(param.get_identifier()).symbol_expr());
1107  }
1108  for(auto &target : get_contract(function, ns).assigns())
1109  {
1110  goto_programt payload;
1111  instrument_spec_assigns.track_spec_target(
1112  to_lambda_expr(target).application(instantiation_values), payload);
1113  insert_before_swap_and_advance(function_body, instruction_it, payload);
1114  }
1115 
1116  // Track formal parameters
1117  goto_programt snapshot_function_parameters;
1118  for(const auto &param : function_type.parameters())
1119  {
1120  goto_programt payload;
1121  instrument_spec_assigns.track_stack_allocated(
1122  ns.lookup(param.get_identifier()).symbol_expr(), payload);
1123  insert_before_swap_and_advance(function_body, instruction_it, payload);
1124  }
1125 
1126  // Restore internal coherence in the programs
1127  goto_functions.update();
1128 
1129  // Insert write set inclusion checks.
1130  instrument_spec_assigns.instrument_instructions(
1131  function_body, instruction_it, function_body.instructions.end());
1132 }

1226 void code_contractst::add_contract_check(
1227  const irep_idt &wrapper_function,
1228  const irep_idt &mangled_function,
1229  goto_programt &dest)
1230 {
1231  PRECONDITION(!dest.instructions.empty());
1232 
1233  // build:
1234  // decl ret
1235  // decl parameter1 ...
1236  // decl history_parameter1 ... [optional]
1237  // assume(requires) [optional]
1238  // ret=function(parameter1, ...)
1239  // assert(ensures)
1240 
1241  const auto &code_type = get_contract(wrapper_function, ns);
