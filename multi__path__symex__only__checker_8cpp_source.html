73 void multi_path_symex_only_checkert::generate_equation()
74 {
75  const auto symex_start = std::chrono::steady_clock::now();
76 
77  symex.symex_from_entry_point_of(
78  goto_symext::get_goto_function(goto_model), symex_symbol_table);
79 
80  const auto symex_stop = std::chrono::steady_clock::now();
81  std::chrono::duration<double> symex_runtime =
82  std::chrono::duration<double>(symex_stop - symex_start);
83  log.status() << "Runtime Symex: " << symex_runtime.count() << "s"
84  << messaget::eom;
85 
86  postprocess_equation(symex, equation, options, ns, ui_message_handler);
87 }
