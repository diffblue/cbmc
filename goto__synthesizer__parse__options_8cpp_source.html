151 int goto_synthesizer_parse_optionst::get_goto_program()
152 {
153  log.status() << "Reading GOTO program from '" << cmdline.args[0] << "'"
154  << messaget::eom;
155 
156  config.set(cmdline);
157 
158  auto result = read_goto_binary(cmdline.args[0], ui_message_handler);
159 
160  if(!result.has_value())
161  return CPROVER_EXIT_USAGE_ERROR;
162 
163  goto_model = std::move(result.value());
164 
165  config.set_from_symbol_table(goto_model.symbol_table);
166  return CPROVER_EXIT_SUCCESS;
