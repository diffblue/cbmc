21 static optionalt<codet> static_lifetime_init(
22  const irep_idt &identifier,
23  symbol_table_baset &symbol_table)
24 {
25  const namespacet ns(symbol_table);
26  const symbolt &symbol = ns.lookup(identifier);
27 
28  if(!symbol.is_static_lifetime)
29  return {};
30 
31  if(symbol.is_type || symbol.is_macro)
32  return {};
33 
34  // special values
35  if(
36  identifier == CPROVER_PREFIX "constant_infinity_uint" ||
37  identifier == CPROVER_PREFIX "memory" || identifier == "__func__" ||
38  identifier == "__FUNCTION__" || identifier == "__PRETTY_FUNCTION__" ||
39  identifier == "argc'" || identifier == "argv'" || identifier == "envp'" ||
40  identifier == "envp_size'")
41  return {};
42 
43  // just for linking
44  if(has_prefix(id2string(identifier), CPROVER_PREFIX "architecture_"))
45  return {};
46 
47  const typet &type = ns.follow(symbol.type);
48 
49  // check type
50  if(type.id() == ID_code || type.id() == ID_empty)
51  return {};
52 
53  if(type.id() == ID_array && to_array_type(type).size().is_nil())
54  {
55  // C standard 6.9.2, paragraph 5
56  // adjust the type to an array of size 1
57  symbolt &writable_symbol = symbol_table.get_writeable_ref(identifier);
58  writable_symbol.type = type;
59  writable_symbol.type.set(ID_size, from_integer(1, size_type()));
60  }
61 
62  if(
63  (type.id() == ID_struct || type.id() == ID_union) &&
64  to_struct_union_type(type).is_incomplete())
65  {
66  return {}; // do not initialize
67  }
68 
69  exprt rhs;
70 
71  if((symbol.value.is_nil() && symbol.is_extern) ||
72  symbol.value.id() == ID_nondet)
73  {
74  if(symbol.value.get_bool(ID_C_no_nondet_initialization))
75  return {};
76 
77  // Nondet initialise if not linked, or if explicitly requested.
78  // Compilers would usually complain about the unlinked symbol case.
79  const auto nondet = nondet_initializer(symbol.type, symbol.location, ns);
80  CHECK_RETURN(nondet.has_value());
81  rhs = *nondet;
82  }
83  else if(symbol.value.is_nil())
84  {
85  const auto zero = zero_initializer(symbol.type, symbol.location, ns);
86  CHECK_RETURN(zero.has_value());
87  rhs = *zero;
88  }
89  else
90  rhs = symbol.value;
91 
92  return code_frontend_assignt{symbol.symbol_expr(), rhs, symbol.location};
93 }

95 void static_lifetime_init(
96  symbol_table_baset &symbol_table,
97  const source_locationt &source_location)
98 {
99  PRECONDITION(symbol_table.has_symbol(INITIALIZE_FUNCTION));
100 
101  const namespacet ns(symbol_table);
102 
103  symbolt &init_symbol = symbol_table.get_writeable_ref(INITIALIZE_FUNCTION);
104 
105  init_symbol.value=code_blockt();
106  init_symbol.value.add_source_location()=source_location;
107 
108  code_blockt &dest=to_code_block(to_code(init_symbol.value));
109 
110  // add the magic label to hide
111  dest.add(code_labelt(CPROVER_PREFIX "HIDE", code_skipt()));
112 
113  // do assignments based on "value"
114 
115  // sort alphabetically for reproducible results
116  std::set<std::string> symbols;
117 
118  for(const auto &symbol_pair : symbol_table.symbols)
119  {
120  symbols.insert(id2string(symbol_pair.first));
121  }
122 
123  // first do framework variables
124  for(const std::string &id : symbols)
125  if(has_prefix(id, CPROVER_PREFIX))
126  {
127  auto code = static_lifetime_init(id, symbol_table);
128  if(code.has_value())
129  dest.add(std::move(*code));
130  }
131 
132  // now all other variables
133  for(const std::string &id : symbols)
134  if(!has_prefix(id, CPROVER_PREFIX))
135  {
136  auto code = static_lifetime_init(id, symbol_table);
137  if(code.has_value())
138  dest.add(std::move(*code));
139  }
140 
141  // now call designated "initialization" functions
142 
143  for(const std::string &id : symbols)
144  {
145  const symbolt &symbol=ns.lookup(id);
146 
147  if(symbol.type.id() != ID_code)
148  continue;
149 
150  const code_typet &code_type = to_code_type(symbol.type);
151  if(
152  code_type.return_type().id() == ID_constructor &&
153  code_type.parameters().empty())
154  {
155  dest.add(code_expressiont{side_effect_expr_function_callt{
156  symbol.symbol_expr(), {}, code_type.return_type(), source_location}});
157  }
158  }
