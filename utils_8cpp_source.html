105 void havoc_assigns_targetst::append_havoc_code_for_expr(
106  const source_locationt location,
107  const exprt &expr,
108  goto_programt &dest)
109 {
110  if(expr.id() == ID_pointer_object)
111  {
112  // pointer_object is still used internally to support malloc/free
113  append_object_havoc_code_for_expr(
114  location, to_pointer_object_expr(expr).pointer(), dest);
115  return;
116  }
117  else if(can_cast_expr<side_effect_expr_function_callt>(expr))
118  {
119  const auto &funcall = to_side_effect_expr_function_call(expr);
120  // type-checking ensures the function expression is necessarily a symbol
121  const auto &ident = to_symbol_expr(funcall.function()).get_identifier();
122  if(ident == CPROVER_PREFIX "object_whole")
123  {
124  append_object_havoc_code_for_expr(
125  location, funcall.arguments().at(0), dest);
126  }
127  else if(ident == CPROVER_PREFIX "object_from")
128  {
129  const auto ptr = typecast_exprt::conditional_cast(
130  funcall.arguments().at(0), pointer_type(char_type()));
131 
132  exprt obj_size = object_size(ptr);
133  minus_exprt size{
134  obj_size,
135  typecast_exprt::conditional_cast(pointer_offset(ptr), obj_size.type())};
136 
137  append_havoc_slice_code(expr.source_location(), ptr, size, dest);
138  }
139  else if(ident == CPROVER_PREFIX "object_upto")
140  {
141  const auto ptr = typecast_exprt::conditional_cast(
142  funcall.arguments().at(0), pointer_type(char_type()));
143  const auto size = typecast_exprt::conditional_cast(
144  funcall.arguments().at(1), size_type());
145  append_havoc_slice_code(expr.source_location(), ptr, size, dest);
146  }
147  else if(ident == CPROVER_PREFIX "assignable")
148  {
149  const auto &ptr = funcall.arguments().at(0);
150  const auto &size = funcall.arguments().at(1);
151  if(funcall.arguments().at(2).is_true())
152  {
153  append_havoc_pointer_code(expr.source_location(), ptr, dest);
154  }
155  else
156  {
157  append_havoc_slice_code(expr.source_location(), ptr, size, dest);
158  }
159  }
160  else
161  {
162  UNREACHABLE;
163  }
164  }
165  else
166  {
167  // we have an lvalue expression, make nondet assignment
168  havoc_utilst::append_havoc_code_for_expr(location, expr, dest);
169  }
170 }

172 exprt all_dereferences_are_valid(const exprt &expr, const namespacet &ns)
173 {
174  exprt::operandst validity_checks;
175 
176  if(auto deref = expr_try_dynamic_cast<dereference_exprt>(expr))
177  {
178  const auto size_of_expr_opt = size_of_expr(expr.type(), ns);
179  CHECK_RETURN(size_of_expr_opt.has_value());
180 
181  validity_checks.push_back(r_ok_exprt{deref->pointer(), *size_of_expr_opt});
182  }
183 
184  for(const auto &op : expr.operands())
185  validity_checks.push_back(all_dereferences_are_valid(op, ns));
186 
187  return conjunction(validity_checks);
188 }

190 exprt generate_lexicographic_less_than_check(
191  const std::vector<symbol_exprt> &lhs,
192  const std::vector<symbol_exprt> &rhs)
193 {
194  PRECONDITION(lhs.size() == rhs.size());
195 
196  if(lhs.empty())
197  {
198  return false_exprt();
199  }
200 
201  // Store conjunctions of equalities.
202  // For example, suppose that the two input vectors are <s1, s2, s3> and <l1,
203  // l2, l3>.
204  // Then this vector stores <s1 == l1, s1 == l1 && s2 == l2,
205  // s1 == l1 && s2 == l2 && s3 == l3>.
206  // In fact, the last element is unnecessary, so we do not create it.
207  exprt::operandst equality_conjunctions(lhs.size());
208  equality_conjunctions[0] = binary_relation_exprt(lhs[0], ID_equal, rhs[0]);
209  for(size_t i = 1; i < equality_conjunctions.size() - 1; i++)
210  {
211  binary_relation_exprt component_i_equality{lhs[i], ID_equal, rhs[i]};
212  equality_conjunctions[i] =
213  and_exprt(equality_conjunctions[i - 1], component_i_equality);
214  }
215 
216  // Store inequalities between the i-th components of the input vectors
217  // (i.e. lhs and rhs).
218  // For example, suppose that the two input vectors are <s1, s2, s3> and <l1,
219  // l2, l3>.
220  // Then this vector stores <s1 < l1, s1 == l1 && s2 < l2, s1 == l1 &&
221  // s2 == l2 && s3 < l3>.
222  exprt::operandst lexicographic_individual_comparisons(lhs.size());
223  lexicographic_individual_comparisons[0] =
224  binary_relation_exprt(lhs[0], ID_lt, rhs[0]);
225  for(size_t i = 1; i < lexicographic_individual_comparisons.size(); i++)
226  {
227  binary_relation_exprt component_i_less_than{lhs[i], ID_lt, rhs[i]};
228  lexicographic_individual_comparisons[i] =
229  and_exprt(equality_conjunctions[i - 1], component_i_less_than);
230  }
231  return disjunction(lexicographic_individual_comparisons);
232 }

282 bool is_loop_free(
283  const goto_programt &goto_program,
284  namespacet &ns,
285  messaget &log)
286 {
287  // create cfg from instruction list
288  cfg_baset<empty_cfg_nodet> cfg;
289  cfg(goto_program);
290 
291  // check that all nodes are there
292  INVARIANT(
293  goto_program.instructions.size() == cfg.size(),
294  "Instruction list vs CFG size mismatch.");
295 
296  // compute SCCs
297  using idxt = graph_nodet<empty_cfg_nodet>::node_indext;
298  std::vector<idxt> node_to_scc(cfg.size(), -1);
299  auto nof_sccs = cfg.SCCs(node_to_scc);
300 
301  // compute size of each SCC
302  std::vector<int> scc_size(nof_sccs, 0);
303  for(auto scc : node_to_scc)
304  {
305  INVARIANT(
306  0 <= scc && scc < nof_sccs, "Could not determine SCC for instruction");
307  scc_size[scc]++;
308  }
309 
310  // check they are all of size 1
311  for(size_t scc_id = 0; scc_id < nof_sccs; scc_id++)
312  {
313  auto size = scc_size[scc_id];
314  if(size > 1)
315  {
316  log.conditional_output(
317  log.error(),
318  [&cfg, &node_to_scc, &scc_id, &size](messaget::mstreamt &mstream) {
319  mstream << "Found CFG SCC with size " << size << messaget::eom;
320  for(const auto &node_id : node_to_scc)
321  {
322  if(node_to_scc[node_id] == scc_id)
323  {
324  const auto &pc = cfg[node_id].PC;
325  pc->output(mstream);
326  mstream << messaget::eom;
327  }
328  }
329  });
330  return false;
331  }
332  }
333  return true;
334 }

381 void add_quantified_variable(
382  symbol_table_baset &symbol_table,
383  exprt &expression,
384  const irep_idt &mode)
385 {
386  if(expression.id() == ID_not || expression.id() == ID_typecast)
387  {
388  // For unary connectives, recursively check for
389  // nested quantified formulae in the term
390  auto &unary_expression = to_unary_expr(expression);
391  add_quantified_variable(symbol_table, unary_expression.op(), mode);
392  }
393  if(expression.id() == ID_notequal || expression.id() == ID_implies)
394  {
395  // For binary connectives, recursively check for
396  // nested quantified formulae in the left and right terms
397  auto &binary_expression = to_binary_expr(expression);
398  add_quantified_variable(symbol_table, binary_expression.lhs(), mode);
399  add_quantified_variable(symbol_table, binary_expression.rhs(), mode);
400  }
401  if(expression.id() == ID_if)
402  {
403  // For ternary connectives, recursively check for
404  // nested quantified formulae in all three terms
405  auto &if_expression = to_if_expr(expression);
406  add_quantified_variable(symbol_table, if_expression.cond(), mode);
407  add_quantified_variable(symbol_table, if_expression.true_case(), mode);
408  add_quantified_variable(symbol_table, if_expression.false_case(), mode);
409  }
410  if(expression.id() == ID_and || expression.id() == ID_or)
411  {
412  // For multi-ary connectives, recursively check for
413  // nested quantified formulae in all terms
414  auto &multi_ary_expression = to_multi_ary_expr(expression);
415  for(auto &operand : multi_ary_expression.operands())
416  {
417  add_quantified_variable(symbol_table, operand, mode);
418  }
419  }
420  else if(expression.id() == ID_exists || expression.id() == ID_forall)
421  {
422  // When a quantifier expression is found, create a fresh symbol for each
423  // quantified variable and rewrite the expression to use those fresh
424  // symbols.
425  auto &quantifier_expression = to_quantifier_expr(expression);
426  std::vector<symbol_exprt> fresh_variables;
427  fresh_variables.reserve(quantifier_expression.variables().size());
428  for(const auto &quantified_variable : quantifier_expression.variables())
429  {
430  // 1. create fresh symbol
431  symbolt new_symbol = new_tmp_symbol(
432  quantified_variable.type(),
433  quantified_variable.source_location(),
434  mode,
435  symbol_table);
436 
437  // 2. add created fresh symbol to expression map
438  fresh_variables.push_back(new_symbol.symbol_expr());
439  }
440 
441  // use fresh symbols
442  exprt where = quantifier_expression.instantiate(fresh_variables);
443 
444  // recursively check for nested quantified formulae
445  add_quantified_variable(symbol_table, where, mode);
446 
447  // replace previous variables and body
448  quantifier_expression.variables() = fresh_variables;
449  quantifier_expression.where() = std::move(where);
450  }
451 }

453 void replace_history_parameter(
454  symbol_table_baset &symbol_table,
455  exprt &expr,
456  std::map<exprt, exprt> &parameter2history,
457  source_locationt location,
458  const irep_idt &mode,
459  goto_programt &history,
460  const irep_idt &id)
461 {
462  for(auto &op : expr.operands())
463  {
464  replace_history_parameter(
465  symbol_table, op, parameter2history, location, mode, history, id);
466  }
467 
468  if(expr.id() == ID_old || expr.id() == ID_loop_entry)
469  {
470  const auto &parameter = to_history_expr(expr, id).expression();
471 
472  const auto &id = parameter.id();
473  if(
474  id == ID_dereference || id == ID_member || id == ID_symbol ||
475  id == ID_ptrmember || id == ID_constant || id == ID_typecast ||
476  id == ID_index)
477  {
478  auto it = parameter2history.find(parameter);
479 
480  if(it == parameter2history.end())
481  {
482  // 0. Create a skip target to jump to, if the parameter is invalid
483  goto_programt skip_program;
484  const auto skip_target =
485  skip_program.add(goto_programt::make_skip(location));
486 
487  // 1. Create a temporary symbol expression that represents the
488  // history variable
489  symbol_exprt tmp_symbol =
490  new_tmp_symbol(parameter.type(), location, mode, symbol_table)
491  .symbol_expr();
492 
493  // 2. Associate the above temporary variable to it's corresponding
494  // expression
495  parameter2history[parameter] = tmp_symbol;
496 
497  // 3. Add the required instructions to the instructions list
498  // 3.1. Declare the newly created temporary variable
499  history.add(goto_programt::make_decl(tmp_symbol, location));
500 
501  // 3.2. Skip storing the history if the expression is invalid
502  history.add(goto_programt::make_goto(
503  skip_target,
504  not_exprt{
505  all_dereferences_are_valid(parameter, namespacet(symbol_table))},
506  location));
507 
508  // 3.3. Add an assignment such that the value pointed to by the new
509  // temporary variable is equal to the value of the corresponding
510  // parameter
511  history.add(
512  goto_programt::make_assignment(tmp_symbol, parameter, location));
513 
514  // 3.4. Add a skip target
515  history.destructive_append(skip_program);
516  }
517 
518  expr = parameter2history[parameter];
519  }
520  else
521  {
522  throw invalid_source_file_exceptiont(
523  "Tracking history of " + id2string(parameter.id()) +
524  " expressions is not supported yet.",
525  expr.source_location());
526  }
527  }
