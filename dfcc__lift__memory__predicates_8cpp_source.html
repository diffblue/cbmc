78 std::set<irep_idt> dfcc_lift_memory_predicatest::lift_predicates(
79  std::set<irep_idt> &discovered_function_pointer_contracts)
80 {
81  std::set<irep_idt> candidates;
82  for(const auto &pair : goto_model.symbol_table)
83  {
84  if(
85  pair.second.type.id() == ID_code &&
86  !instrument.do_not_instrument(pair.first))
87  {
88  const auto &iter =
89  goto_model.goto_functions.function_map.find(pair.first);
90  if(
91  iter != goto_model.goto_functions.function_map.end() &&
92  iter->second.body_available())
93  {
94  candidates.insert(pair.first);
95  }
96  }
97  }
98 
99  std::set<irep_idt> predicates;
100 
101  // iterate until no new predicate is found
102  bool new_predicates_found = true;
103  while(new_predicates_found)
104  {
105  new_predicates_found = false;
106  for(const auto &function_id : candidates)
107  {
108  if(
109  predicates.find(function_id) == predicates.end() &&
110  calls_memory_predicates(
111  goto_model.goto_functions.function_map[function_id].body, predicates))
112  {
113  predicates.insert(function_id);
114  new_predicates_found = true;
115  }
116  }
117  }
118 
119  if(predicates.empty())
120  return predicates;
121 
122  // some predicates were found, build dependency graph
123  struct dep_graph_nodet : public graph_nodet<empty_edget>
124  {
125  const irep_idt &function_id;
126  explicit dep_graph_nodet(const irep_idt &function_id)
127  : function_id(function_id)
128  {
129  }
130  };
131 
132  grapht<dep_graph_nodet> dep_graph;
133 
134  // inverse mapping from names to dep_graph_nodet identifiers
135  std::map<irep_idt, std::size_t> id_to_node_map;
136  // create all nodes
137  for(auto &predicate : predicates)
138  {
139  id_to_node_map.insert({predicate, dep_graph.add_node(predicate)});
140  }
141 
142  // create all edges
143  for(auto &caller : predicates)
144  {
145  const auto caller_id = id_to_node_map[caller];
146  for(const auto &instruction :
147  goto_model.goto_functions.function_map[caller].body.instructions)
148  {
149  if(
150  instruction.is_function_call() &&
151  instruction.call_function().id() == ID_symbol)
152  {
153  const auto &callee =
154  to_symbol_expr(instruction.call_function()).get_identifier();
155  if(predicates.find(callee) != predicates.end())
156  {
157  log.conditional_output(log.debug(), [&](messaget::mstreamt &mstream) {
158  mstream << "Memory predicate dependency " << caller << " -> "
159  << callee << messaget::eom;
160  });
161  const auto callee_id = id_to_node_map[callee];
162  if(callee != caller) // do not add edges for self-recursive functions
163  dep_graph.add_edge(callee_id, caller_id);
164  }
165  }
166  }
167  }
168 
169  // lift in dependency order
170  auto sorted = dep_graph.topsort();
171  PRECONDITION_WITH_DIAGNOSTICS(
172  !sorted.empty(),
173  "could not determine instrumentation order for memory predicates, most "
174  "likely due to mutual recursion");
175  for(const auto &idx : sorted)
176  {
177  lift_predicate(
178  dep_graph[idx].function_id, discovered_function_pointer_contracts);
179  }
180 
181  return predicates;
182 }

184 // returns an optional rank
185 static optionalt<std::size_t> is_param_expr(
186  const exprt &expr,
187  const std::map<irep_idt, std::size_t> &parameter_rank)
188 {
189  if(expr.id() == ID_typecast)
190  {
191  // ignore typecasts
192  return is_param_expr(to_typecast_expr(expr).op(), parameter_rank);
193  }
194  else if(expr.id() == ID_symbol)
195  {
196  const irep_idt &ident = to_symbol_expr(expr).get_identifier();
197  const auto found = parameter_rank.find(ident);
198  if(found != parameter_rank.end())
199  {
200  return {found->second};
201  }
202  else
203  {
204  return {};
205  }
206  }
207  else
208  {
209  // bail on anything else
210  return {};
211  }

214 void dfcc_lift_memory_predicatest::collect_parameters_to_lift(
215  const irep_idt &function_id)
216 {
217  const symbolt &function_symbol = utils.get_function_symbol(function_id);
218  // map of parameter name to its rank in the signature
219  std::map<irep_idt, std::size_t> parameter_rank;
220  const auto &parameters = to_code_type(function_symbol.type).parameters();
221  for(std::size_t rank = 0; rank < parameters.size(); rank++)
222  {
223  parameter_rank[parameters[rank].get_identifier()] = rank;
224  }
225  const goto_programt &body =
226  goto_model.goto_functions.function_map[function_id].body;
227  for(const auto &it : body.instructions)
228  {
229  // for all function calls, if a parameter of function_id is passed
230  // in a lifted position, add its rank to the set of lifted parameters
231  if(it.is_function_call() && it.call_function().id() == ID_symbol)
232  {
233  const irep_idt &callee_id =
234  to_symbol_expr(it.call_function()).get_identifier();
235  if(callee_id == CPROVER_PREFIX "is_fresh")
236  {
237  auto opt_rank = is_param_expr(it.call_arguments()[0], parameter_rank);
238  if(opt_rank.has_value())
239  {
240  lifted_parameters[function_id].insert(opt_rank.value());
241  }
242  }
243  else if(callee_id == CPROVER_PREFIX "pointer_in_range_dfcc")
244  {
245  auto opt_rank = is_param_expr(it.call_arguments()[1], parameter_rank);
246  if(opt_rank.has_value())
247  {
248  lifted_parameters[function_id].insert(opt_rank.value());
249  }
250  }
251  else if(callee_id == CPROVER_PREFIX "obeys_contract")
252  {
253  auto opt_rank = is_param_expr(it.call_arguments()[0], parameter_rank);
254  if(opt_rank.has_value())
255  {
256  lifted_parameters[function_id].insert(opt_rank.value());
257  }
258  }
259  else if(is_lifted_function(callee_id))
260  {
261  // search wether a parameter of function_id is passed to a lifted
262  // parameter of callee_id
263  for(const std::size_t callee_rank : lifted_parameters[callee_id])
264  {
265  auto opt_rank =
266  is_param_expr(it.call_arguments()[callee_rank], parameter_rank);
267  if(opt_rank.has_value())
268  {
269  lifted_parameters[function_id].insert(opt_rank.value());
270  }
271  }
272  }
273  }
274  }

277 void dfcc_lift_memory_predicatest::add_pointer_type(

289  << messaget::eom;
290  });

292  const auto &old_symbol_expr = old_symbol.symbol_expr();
293  // create new parameter symbol, same everything except type
294  symbolt new_symbol(
295  old_symbol.name, pointer_type(old_symbol.type), old_symbol.mode);
296  new_symbol.base_name = old_symbol.base_name;
297  new_symbol.location = old_symbol.location;
298  new_symbol.module = old_symbol.module;
299  new_symbol.is_lvalue = old_symbol.is_lvalue;
300  new_symbol.is_state_var = old_symbol.is_state_var;
301  new_symbol.is_thread_local = old_symbol.is_thread_local;
302  new_symbol.is_file_local = old_symbol.is_file_local;
303  new_symbol.is_auxiliary = old_symbol.is_auxiliary;
304  new_symbol.is_parameter = old_symbol.is_parameter;
305  goto_model.symbol_table.erase(entry);

311  code_typet::parametert parameter(res.first.type);
312  parameter.set_base_name(res.first.base_name);
313  parameter.set_identifier(res.first.name);
314  parameters[parameter_rank] = parameter;

317 void dfcc_lift_memory_predicatest::lift_parameters_and_update_body(
318  const irep_idt &function_id,
319  std::set<irep_idt> &discovered_function_pointer_contracts)
320 {
321  replace_symbolt replace_lifted_params;
322  // add pointer types to all parameters that require it
323  for(const auto rank : lifted_parameters[function_id])
324  {
325  add_pointer_type(function_id, rank, replace_lifted_params);
326  }
327  auto &body = goto_model.goto_functions.function_map[function_id].body;
328  // update the function body
329  for(auto &instruction : body.instructions)
330  {
331  // rewrite all occurrences of lifted parameters
332  instruction.transform([&replace_lifted_params](exprt expr) {
333  const bool changed = !replace_lifted_params.replace(expr);
334  return changed ? optionalt<exprt>{expr} : nullopt;
335  });
336 
337  // add address-of to all arguments expressions passed in lifted position to
338  // another memory predicate
339  if(
340  instruction.is_function_call() &&
341  instruction.call_function().id() == ID_symbol)
342  {
343  // add address-of to arguments that are passed in lifted position
344  auto &callee_id =
345  to_symbol_expr(instruction.call_function()).get_identifier();
346  if(is_lifted_function(callee_id))
347  {
348  for(const auto &rank : lifted_parameters[callee_id])
349  {
350  const auto arg = instruction.call_arguments()[rank];
351  log.conditional_output(log.debug(), [&](messaget::mstreamt &mstream) {
352  mstream << "Adding address_of to call " << callee_id
353  << ", argument " << format(arg) << messaget::eom;
354  });
355  instruction.call_arguments()[rank] = address_of_exprt(arg);
356  }
357  }
358  }
359  }

362 void dfcc_lift_memory_predicatest::lift_predicate(
363  const irep_idt &function_id,
364  std::set<irep_idt> &discovered_function_pointer_contracts)
365 {
366  // when this function_id gets processed, any memory predicate it calls has
367  // already been processed (except itself if it is recursive);
368 
369  log.status() << "Instrumenting memory predicate '" << function_id << "'"
370  << messaget::eom;
371 
372  // first step: identify which parameters are passed directly to core
373  // predicates of lifted positions in user-defined memory predicates and mark
374  // them as lifted
375  collect_parameters_to_lift(function_id);
376  lift_parameters_and_update_body(
377  function_id, discovered_function_pointer_contracts);
378 
379  log.conditional_output(log.debug(), [&](messaget::mstreamt &mstream) {
380  mstream << "Ranks of lifted parameters for " << function_id << ": ";
381  for(const auto rank : lifted_parameters[function_id])
382  mstream << rank << ", ";
383  mstream << messaget::eom;
384  });
385 
386  // instrument the function for side effects: adds the write_set parameter,
387  // adds checks for side effects, maps core predicates to their implementation.
388  instrument.instrument_function(
389  function_id, discovered_function_pointer_contracts);
390 }
