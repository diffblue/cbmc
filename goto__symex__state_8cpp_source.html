73 renamedt<ssa_exprt, L2> goto_symex_statet::assignment(
74  ssa_exprt lhs, // L0/L1
75  const exprt &rhs, // L2
76  const namespacet &ns,
77  bool rhs_is_simplified,
78  bool record_value,
79  bool allow_pointer_unsoundness)
80 {
81  // identifier should be l0 or l1, make sure it's l1
82  lhs = rename_ssa<L1>(std::move(lhs), ns).get();
83  irep_idt l1_identifier=lhs.get_identifier();
84 
85  // the type might need renaming
86  rename<L2>(lhs.type(), l1_identifier, ns);
87  lhs.update_type();
88  if(run_validation_checks)
89  {
90  DATA_INVARIANT(!check_renaming_l1(lhs), "lhs renaming failed on l1");
91  }
92 
93  // do the l2 renaming
94  level2.increase_generation(l1_identifier, lhs, fresh_l2_name_provider);
95  renamedt<ssa_exprt, L2> l2_lhs = set_indices<L2>(std::move(lhs), ns);
96  lhs = l2_lhs.get();
97 
98  // in case we happen to be multi-threaded, record the memory access
99  bool is_shared=l2_thread_write_encoding(lhs, ns);
100 
101  if(run_validation_checks)
102  {
103  DATA_INVARIANT(!check_renaming(lhs), "lhs renaming failed on l2");
104  DATA_INVARIANT(!check_renaming(rhs), "rhs renaming failed on l2");
105  }
106 
107  // see #305 on GitHub for a simple example and possible discussion
108  if(is_shared && lhs.type().id() == ID_pointer && !allow_pointer_unsoundness)
109  throw unsupported_operation_exceptiont(
110  "pointer handling for concurrency is unsound");
111 
112  // Update constant propagation map -- the RHS is L2
113  if(!is_shared && record_value && goto_symex_is_constantt(ns)(rhs))
114  {
115  const auto propagation_entry = propagation.find(l1_identifier);
116  if(!propagation_entry.has_value())
117  propagation.insert(l1_identifier, rhs);
118  else if(propagation_entry->get() != rhs)
119  propagation.replace(l1_identifier, rhs);
120  }
121  else
122  propagation.erase_if_exists(l1_identifier);
123 
124  {
125  // update value sets
126  exprt l1_rhs(rhs);
127  get_l1_name(l1_rhs);
128 
129  const ssa_exprt l1_lhs = remove_level_2(lhs);
130  if(run_validation_checks)
131  {
132  DATA_INVARIANT(!check_renaming_l1(l1_lhs), "lhs renaming failed on l1");
133  DATA_INVARIANT(!check_renaming_l1(l1_rhs), "rhs renaming failed on l1");
134  }
135 
136  value_set.assign(l1_lhs, l1_rhs, ns, rhs_is_simplified, is_shared);
137  }
138 
139 #ifdef DEBUG
140  std::cout << "Assigning " << l1_identifier << '\n';
141  value_set.output(std::cout);
142  std::cout << "**********************\n";
143 #endif
144 
145  return l2_lhs;
146 }

167 template <levelt level>
168 renamedt<exprt, level>
169 goto_symex_statet::rename(exprt expr, const namespacet &ns)
170 {
171  // rename all the symbols with their last known value
172 
173  static_assert(
174  level == L0 || level == L1 || level == L1_WITH_CONSTANT_PROPAGATION ||
175  level == L2,
176  "must handle all renaming levels");
177 
178  if(is_ssa_expr(expr))
179  {
180  exprt original_expr = expr;
181  ssa_exprt &ssa=to_ssa_expr(expr);
182 
183  if(level == L0)
184  {
185  return renamedt<exprt, level>{
186  std::move(rename_ssa<L0>(std::move(ssa), ns).value())};
187  }
188  else if(level == L1)
189  {
190  return renamedt<exprt, level>{
191  std::move(rename_ssa<L1>(std::move(ssa), ns).value())};
192  }
193  else
194  {
195  ssa = set_indices<L1>(std::move(ssa), ns).get();
196  rename<level>(expr.type(), ssa.get_identifier(), ns);
197  ssa.update_type();
198 
199  // renaming taken care of by l2_thread_encoding, or already at L2
200  if(l2_thread_read_encoding(ssa, ns) || !ssa.get_level_2().empty())
201  {
202  if(level == L1_WITH_CONSTANT_PROPAGATION)
203  {
204  // Don't actually rename to L2 -- we just used `ssa` to check whether
205  // constant-propagation was applicable
206  return renamedt<exprt, level>(std::move(original_expr));
207  }
208  else
209  return renamedt<exprt, level>(std::move(ssa));
210  }
211  else
212  {
213  // We also consider propagation if we go up to L2.
214  // L1 identifiers are used for propagation!
215  auto p_it = propagation.find(ssa.get_identifier());
216 
217  if(p_it.has_value())
218  {
219  return renamedt<exprt, level>(*p_it); // already L2
220  }
221  else
222  {
223  if(level == L2)
224  ssa = set_indices<L2>(std::move(ssa), ns).get();
225  return renamedt<exprt, level>(std::move(ssa));
226  }
227  }
228  }
229  }
230  else if(expr.id()==ID_symbol)
231  {
232  const auto &type = as_const(expr).type();
233 
234  // we never rename function symbols
235  if(type.id() == ID_code || type.id() == ID_mathematical_function)
236  {
237  rename<level>(expr.type(), to_symbol_expr(expr).get_identifier(), ns);
238  return renamedt<exprt, level>{std::move(expr)};
239  }
240  else
241  return rename<level>(ssa_exprt{expr}, ns);
242  }
243  else if(expr.id()==ID_address_of)
244  {
245  auto &address_of_expr = to_address_of_expr(expr);
246  rename_address<level>(address_of_expr.object(), ns);
247  to_pointer_type(expr.type()).base_type() =
248  as_const(address_of_expr).object().type();
249  return renamedt<exprt, level>{std::move(expr)};
250  }
251  else if(expr.is_nil())
252  {
253  return renamedt<exprt, level>{std::move(expr)};
254  }
255  else
256  {
257  rename<level>(expr.type(), irep_idt(), ns);
258 
259  // do this recursively
260  Forall_operands(it, expr)
261  *it = rename<level>(std::move(*it), ns).get();
262 
263  const exprt &c_expr = as_const(expr);
264  INVARIANT_WITH_DIAGNOSTICS(
265  (expr.id() != ID_with ||
266  c_expr.type() == to_with_expr(c_expr).old().type()) &&
267  (expr.id() != ID_if ||
268  (c_expr.type() == to_if_expr(c_expr).true_case().type() &&
269  c_expr.type() == to_if_expr(c_expr).false_case().type())),
270  "Type of renamed expr should be the same as operands for with_exprt and "
271  "if_exprt",
272  irep_pretty_diagnosticst{expr});
273 
274  if(level == L2)
275  expr = field_sensitivity.apply(ns, *this, std::move(expr), false);
276 
277  return renamedt<exprt, level>{std::move(expr)};
278  }
279 }

286 exprt goto_symex_statet::l2_rename_rvalues(exprt lvalue, const namespacet &ns)
287 {
288  rename(lvalue.type(), irep_idt(), ns);
289 
290  if(lvalue.id() == ID_symbol)
291  {
292  // Nothing to do
293  }
294  else if(is_read_only_object(lvalue))
295  {
296  // Ignore apparent writes to 'NULL-object' and similar read-only objects
297  }
298  else if(lvalue.id() == ID_typecast)
299  {
300  auto &typecast_lvalue = to_typecast_expr(lvalue);
301  typecast_lvalue.op() = l2_rename_rvalues(typecast_lvalue.op(), ns);
302  }
303  else if(lvalue.id() == ID_member)
304  {
305  auto &member_lvalue = to_member_expr(lvalue);
306  member_lvalue.compound() = l2_rename_rvalues(member_lvalue.compound(), ns);
307  }
308  else if(lvalue.id() == ID_index)
309  {
310  // The index is an rvalue:
311  auto &index_lvalue = to_index_expr(lvalue);
312  index_lvalue.array() = l2_rename_rvalues(index_lvalue.array(), ns);
313  index_lvalue.index() = rename(index_lvalue.index(), ns).get();
314  }
315  else if(
316  lvalue.id() == ID_byte_extract_little_endian ||
317  lvalue.id() == ID_byte_extract_big_endian)
318  {
319  // The offset is an rvalue:
320  auto &byte_extract_lvalue = to_byte_extract_expr(lvalue);
321  byte_extract_lvalue.op() = l2_rename_rvalues(byte_extract_lvalue.op(), ns);
322  byte_extract_lvalue.offset() = rename(byte_extract_lvalue.offset(), ns);
323  }
324  else if(lvalue.id() == ID_if)
325  {
326  // The condition is an rvalue:
327  auto &if_lvalue = to_if_expr(lvalue);
328  if_lvalue.cond() = rename(if_lvalue.cond(), ns);
329  if(!if_lvalue.cond().is_false())
330  if_lvalue.true_case() = l2_rename_rvalues(if_lvalue.true_case(), ns);
331  if(!if_lvalue.cond().is_true())
332  if_lvalue.false_case() = l2_rename_rvalues(if_lvalue.false_case(), ns);
333  }
334  else if(lvalue.id() == ID_complex_real)
335  {
336  auto &complex_real_lvalue = to_complex_real_expr(lvalue);
337  complex_real_lvalue.op() = l2_rename_rvalues(complex_real_lvalue.op(), ns);
338  }
339  else if(lvalue.id() == ID_complex_imag)
340  {
341  auto &complex_imag_lvalue = to_complex_imag_expr(lvalue);
342  complex_imag_lvalue.op() = l2_rename_rvalues(complex_imag_lvalue.op(), ns);
343  }
344  else
345  {
346  throw unsupported_operation_exceptiont(
347  "l2_rename_rvalues case `" + lvalue.id_string() + "' not handled");
348  }
349 
350  return lvalue;
351 }

357 bool goto_symex_statet::l2_thread_read_encoding(
358  ssa_exprt &expr,
359  const namespacet &ns)
360 {
361  // do we have threads?
362  if(threads.size()<=1)
363  return false;
364 
365  // is it a shared object?
366  PRECONDITION(dirty != nullptr);
367  const irep_idt &obj_identifier=expr.get_object_name();
368  if(
369  obj_identifier == guard_identifier() ||
370  (!ns.lookup(obj_identifier).is_shared() && !(*dirty)(obj_identifier)))
371  {
372  return false;
373  }
374 
375  // only continue if an indivisible object is being accessed
376  if(field_sensitivity.is_divisible(expr, true))
377  return false;
378 
379  const ssa_exprt ssa_l1 = remove_level_2(expr);
380  const irep_idt &l1_identifier=ssa_l1.get_identifier();
381  const exprt guard_as_expr = guard.as_expr();
382 
383  // see whether we are within an atomic section
384  if(atomic_section_id!=0)
385  {
386  guardt write_guard{false_exprt{}, guard_manager};
387 
388  const auto a_s_writes = written_in_atomic_section.find(ssa_l1);
389  if(a_s_writes!=written_in_atomic_section.end())
390  {
391  for(const auto &guard_in_list : a_s_writes->second)
392  {
393  guardt g = guard_in_list;
394  g-=guard;
395  if(g.is_true())
396  // There has already been a write to l1_identifier within this atomic
397  // section under the same guard, or a guard implied by the current
398  // one.
399  return false;
400 
401  write_guard |= guard_in_list;
402  }
403  }
404 
405  not_exprt no_write(write_guard.as_expr());
406 
407  // we cannot determine for sure that there has been a write already
408  // so generate a read even if l1_identifier has been written on
409  // all branches flowing into this read
410  guardt read_guard{false_exprt{}, guard_manager};
411 
412  a_s_r_entryt &a_s_read=read_in_atomic_section[ssa_l1];
413  for(const auto &a_s_read_guard : a_s_read.second)
414  {
415  guardt g = a_s_read_guard; // copy
416  g-=guard;
417  if(g.is_true())
418  // There has already been a read of l1_identifier within this atomic
419  // section under the same guard, or a guard implied by the current one.
420  return false;
421 
422  read_guard |= a_s_read_guard;
423  }
424 
425  guardt cond = read_guard;
426  if(!no_write.op().is_false())
427  cond |= guardt{no_write.op(), guard_manager};
428 
429  // It is safe to perform constant propagation in case we have read or
430  // written this object within the atomic section. We must actually do this,
431  // because goto_state::apply_condition may have placed the latest value in
432  // the propagation map without recording an assignment.
433  auto p_it = propagation.find(ssa_l1.get_identifier());
434  const exprt l2_true_case =
435  p_it.has_value() ? *p_it : set_indices<L2>(ssa_l1, ns).get();
436 
437  if(!cond.is_true())
438  level2.increase_generation(l1_identifier, ssa_l1, fresh_l2_name_provider);
439 
440  if(a_s_read.second.empty())
441  a_s_read.first = level2.latest_index(l1_identifier);
442 
443  const renamedt<ssa_exprt, L2> l2_false_case = set_indices<L2>(ssa_l1, ns);
444 
445  exprt tmp;
446  if(cond.is_false())
447  tmp = l2_false_case.get();
448  else if(cond.is_true())
449  tmp = l2_true_case;
450  else
451  tmp = if_exprt{cond.as_expr(), l2_true_case, l2_false_case.get()};
452 
453  record_events.push(false);
454  ssa_exprt ssa_l2 = assignment(std::move(ssa_l1), tmp, ns, true, true).get();
455  record_events.pop();
456 
457  symex_target->assignment(
458  guard_as_expr,
459  ssa_l2,
460  ssa_l2,
461  ssa_l2.get_original_expr(),
462  tmp,
463  source,
464  symex_targett::assignment_typet::PHI);
465 
466  INVARIANT(!check_renaming(ssa_l2), "expr should be renamed to L2");
467  expr = std::move(ssa_l2);
468 
469  a_s_read.second.push_back(guard);
470  if(!no_write.op().is_false())
471  a_s_read.second.back().add(no_write);
472 
473  return true;
474  }
475 
476  // No event and no fresh index, but avoid constant propagation
477  if(!record_events.top())
478  {
479  expr = set_indices<L2>(std::move(ssa_l1), ns).get();
480  return true;
481  }
482 
483  // produce a fresh L2 name
484  level2.increase_generation(l1_identifier, ssa_l1, fresh_l2_name_provider);
485  expr = set_indices<L2>(std::move(ssa_l1), ns).get();
486 
487  // and record that
488  INVARIANT_STRUCTURED(
489  symex_target!=nullptr, nullptr_exceptiont, "symex_target is null");
490  symex_target->shared_read(guard_as_expr, expr, atomic_section_id, source);
491 
492  return true;
493 }

495 goto_symex_statet::write_is_shared_resultt goto_symex_statet::write_is_shared(
496  const ssa_exprt &expr,
497  const namespacet &ns) const
498 {
499  if(!record_events.top())
500  return write_is_shared_resultt::NOT_SHARED;
501 
502  PRECONDITION(dirty != nullptr);
503  const irep_idt &obj_identifier = expr.get_object_name();
504  if(
505  obj_identifier == guard_identifier() ||
506  (!ns.lookup(obj_identifier).is_shared() && !(*dirty)(obj_identifier)))
507  {
508  return write_is_shared_resultt::NOT_SHARED;
509  }
510 
511  // only continue if an indivisible object is being accessed
512  if(field_sensitivity.is_divisible(expr, true))
513  return write_is_shared_resultt::NOT_SHARED;
514 
515  if(atomic_section_id != 0)
516  return write_is_shared_resultt::IN_ATOMIC_SECTION;
517 
518  return write_is_shared_resultt::SHARED;

523 bool goto_symex_statet::l2_thread_write_encoding(
524  const ssa_exprt &expr,
525  const namespacet &ns)
526 {
527  switch(write_is_shared(expr, ns))
528  {
529  case write_is_shared_resultt::NOT_SHARED:
530  return false;
531  case write_is_shared_resultt::IN_ATOMIC_SECTION:
532  {
533  written_in_atomic_section[remove_level_2(expr)].push_back(guard);
534  return false;
535  }
536  case write_is_shared_resultt::SHARED:
537  break;
538  }
539 
540  // record a shared write
541  symex_target->shared_write(
542  guard.as_expr(),
543  expr,
544  atomic_section_id,
545  source);
546 
547  // do we have threads?
548  return threads.size() > 1;

551 template <levelt level>
552 void goto_symex_statet::rename_address(exprt &expr, const namespacet &ns)
553 {
554  if(is_ssa_expr(expr))
555  {
556  ssa_exprt &ssa=to_ssa_expr(expr);
557 
558  // only do L1!
559  ssa = set_indices<L1>(std::move(ssa), ns).get();
560 
561  rename<level>(expr.type(), ssa.get_identifier(), ns);
562  ssa.update_type();
563  }
564  else if(expr.id()==ID_symbol)
565  {
566  expr=ssa_exprt(expr);
567  rename_address<level>(expr, ns);
568  }
569  else
570  {
571  if(expr.id()==ID_index)
572  {
573  index_exprt &index_expr=to_index_expr(expr);
574 
575  rename_address<level>(index_expr.array(), ns);
576  PRECONDITION(index_expr.array().type().id() == ID_array);
577  expr.type() = to_array_type(index_expr.array().type()).element_type();
578 
579  // the index is not an address
580  index_expr.index() =
581  rename<level>(std::move(index_expr.index()), ns).get();
582  }
583  else if(expr.id()==ID_if)
584  {
585  // the condition is not an address
586  if_exprt &if_expr=to_if_expr(expr);
587  if_expr.cond() = rename<level>(std::move(if_expr.cond()), ns).get();
588  rename_address<level>(if_expr.true_case(), ns);
589  rename_address<level>(if_expr.false_case(), ns);
590 
591  if_expr.type()=if_expr.true_case().type();
592  }
593  else if(expr.id()==ID_member)
594  {
595  member_exprt &member_expr=to_member_expr(expr);
596 
597  rename_address<level>(member_expr.struct_op(), ns);
598 
599  // type might not have been renamed in case of nesting of
600  // structs and pointers/arrays
601  if(
602  member_expr.struct_op().type().id() != ID_struct_tag &&
603  member_expr.struct_op().type().id() != ID_union_tag)
604  {
605  const struct_union_typet &su_type=
606  to_struct_union_type(member_expr.struct_op().type());
607  const struct_union_typet::componentt &comp=
608  su_type.get_component(member_expr.get_component_name());
609  PRECONDITION(comp.is_not_nil());
610  expr.type()=comp.type();
611  }
612  else
613  rename<level>(expr.type(), irep_idt(), ns);
614  }
615  else
616  {
617  // this could go wrong, but we would have to re-typecheck ...
618  rename<level>(expr.type(), irep_idt(), ns);
619 
620  // do this recursively; we assume here
621  // that all the operands are addresses

631 static bool requires_renaming(const typet &type, const namespacet &ns)
632 {
633  if(type.id() == ID_array)
634  {
635  const auto &array_type = to_array_type(type);
636  return requires_renaming(array_type.element_type(), ns) ||
637  !array_type.size().is_constant();
638  }
639  else if(type.id() == ID_struct || type.id() == ID_union)
640  {
641  const struct_union_typet &s_u_type = to_struct_union_type(type);
642  const struct_union_typet::componentst &components = s_u_type.components();
643 
644  for(auto &component : components)
645  {
646  // be careful, or it might get cyclic
647  if(component.type().id() == ID_array)
648  {
649  if(!to_array_type(component.type()).size().is_constant())
650  return true;
651  }
652  else if(
653  component.type().id() != ID_pointer &&
654  requires_renaming(component.type(), ns))
655  {
656  return true;
657  }
658  }
659 
660  return false;
661  }
662  else if(type.id() == ID_pointer)
663  {
664  return requires_renaming(to_pointer_type(type).base_type(), ns);
665  }
666  else if(type.id() == ID_union_tag)
667  {
668  const symbolt &symbol = ns.lookup(to_union_tag_type(type));
669  return requires_renaming(symbol.type, ns);
670  }
671  else if(type.id() == ID_struct_tag)
672  {
673  const symbolt &symbol = ns.lookup(to_struct_tag_type(type));
674  return requires_renaming(symbol.type, ns);
675  }
676 
677  return false;
678 }

680 template <levelt level>
681 void goto_symex_statet::rename(
682  typet &type,
683  const irep_idt &l1_identifier,
684  const namespacet &ns)
685 {
686  // check whether there are symbol expressions in the type; if not, there
687  // is no need to expand the struct/union tags in the type
688  if(!requires_renaming(type, ns))
689  return; // no action
690 
691  // rename all the symbols with their last known value
692  // to the given level
693 
694  std::pair<l1_typest::iterator, bool> l1_type_entry;
695  if(level==L2 &&
696  !l1_identifier.empty())
697  {
698  l1_type_entry=l1_types.insert(std::make_pair(l1_identifier, type));
699 
700  if(!l1_type_entry.second) // was already in map
701  {
702  // do not change a complete array type to an incomplete one
703 
704  const typet &type_prev=l1_type_entry.first->second;
705 
706  if(type.id()!=ID_array ||
707  type_prev.id()!=ID_array ||
708  to_array_type(type).is_incomplete() ||
709  to_array_type(type_prev).is_complete())
710  {
711  type=l1_type_entry.first->second;
712  return;
713  }
714  }
715  }
716 
717  // expand struct and union tag types
718  type = ns.follow(type);
719 
720  if(type.id()==ID_array)
721  {
722  auto &array_type = to_array_type(type);
723  rename<level>(array_type.element_type(), irep_idt(), ns);
724  array_type.size() = rename<level>(std::move(array_type.size()), ns).get();
725  }
726  else if(type.id() == ID_struct || type.id() == ID_union)
727  {
728  struct_union_typet &s_u_type=to_struct_union_type(type);
729  struct_union_typet::componentst &components=s_u_type.components();
730 
731  for(auto &component : components)
732  {
733  // be careful, or it might get cyclic
734  if(component.type().id() == ID_array)
735  {
736  auto &array_type = to_array_type(component.type());
737  array_type.size() =
738  rename<level>(std::move(array_type.size()), ns).get();
739  }
740  else if(component.type().id() != ID_pointer)
741  rename<level>(component.type(), irep_idt(), ns);
742  }
743  }
744  else if(type.id()==ID_pointer)
745  {
746  rename<level>(to_pointer_type(type).base_type(), irep_idt(), ns);
747  }
748 
749  if(level==L2 &&
750  !l1_identifier.empty())
751  l1_type_entry.first->second=type;
752 }

817 {

819

821  rename(ssa.type(), l1_identifier, ns);
822  ssa.update_type();

837  }
838 
839  // L2 renaming
840  exprt fields = field_sensitivity.get_fields(ns, *this, ssa, false);
841  fields.visit_pre([this](const exprt &e) {
842  if(auto l1_symbol = expr_try_dynamic_cast<symbol_exprt>(e))
843  {
844  const ssa_exprt &field_ssa = to_ssa_expr(*l1_symbol);
845  const std::size_t field_generation = level2.increase_generation(
846  l1_symbol->get_identifier(), field_ssa, fresh_l2_name_provider);
847  CHECK_RETURN(field_generation == 1);
848  }
849  else if(auto fs_ssa = expr_try_dynamic_cast<field_sensitive_ssa_exprt>(e))
850  {
851  const ssa_exprt &ssa = fs_ssa->get_object_ssa();
852  const std::size_t field_generation = level2.increase_generation(
853  ssa.get_identifier(), ssa, fresh_l2_name_provider);
854  CHECK_RETURN(field_generation == 1);
855  }
856  });
857 
858  record_events.push(false);
859  exprt expr_l2 = rename(std::move(ssa), ns).get();
860  INVARIANT(
861  is_ssa_expr(expr_l2),
862  "symbol to declare should not be replaced by constant propagation");
863  ssa = to_ssa_expr(expr_l2);
864  record_events.pop();
865 
866  return ssa;
