427 void dfcc_instrumentt::instrument_goto_function(
428  const irep_idt &function_id,
429  goto_functiont &goto_function,
430  const exprt &write_set,
431  const std::set<symbol_exprt> &local_statics,
432  const loop_contract_configt &loop_contract_config,
433  std::set<irep_idt> &function_pointer_contracts)
434 {
435  if(!goto_function.body_available())
436  {
437  // generate a default body `assert(false);assume(false);`
438  std::string options = "assert-false-assume-false";
439  c_object_factory_parameterst object_factory_params;
440  auto generate_function_bodies = generate_function_bodies_factory(
441  options, object_factory_params, goto_model.symbol_table, message_handler);
442  generate_function_bodies->generate_function_body(
443  goto_function, goto_model.symbol_table, function_id);
444  return;
445  }
446 
447  auto &body = goto_function.body;
448 
449  // build control flow graph information
450  dfcc_cfg_infot cfg_info(
451  function_id,
452  goto_function,
453  write_set,
454  loop_contract_config,
455  goto_model.symbol_table,
456  message_handler,
457  library);
458 
459  // instrument instructions
460  instrument_instructions(
461  function_id,
462  body,
463  body.instructions.begin(),
464  body.instructions.end(),
465  cfg_info,
466  function_pointer_contracts);
467 
468  // recalculate numbers, etc.
469  goto_model.goto_functions.update();
470 
471  if(loop_contract_config.apply_loop_contracts)
472  {
473  apply_loop_contracts(
474  function_id,
475  goto_function,
476  cfg_info,
477  loop_contract_config,
478  local_statics,
479  function_pointer_contracts);
480  }
481 
482  // insert add/remove instructions for local statics in the top level write set
483  auto begin = body.instructions.begin();

485

488  {
489  insert_add_decl_call(function_id, write_set, local_static, begin, body);
490  insert_record_dead_call(function_id, write_set, local_static, end, body);
491  }
492

494  dfcc_utilst::get_function_symbol(goto_model.symbol_table, function_id)
495  .type);
496  const auto &top_level_tracked = cfg_info.get_top_level_tracked();
497 
498  // automatically add/remove function parameters that must be tracked in the
499  // function write set (they must be explicitly tracked if they are assigned
500  // in the body of a loop)
501  for(const auto &param : code_type.parameters())
502  {
503  const irep_idt &param_id = param.get_identifier();
504  if(top_level_tracked.find(param_id) != top_level_tracked.end())
505  {
506  symbol_exprt param_symbol{param.get_identifier(), param.type()};

508  insert_record_dead_call(function_id, write_set, param_symbol, end, body);
509  }
510  }
511 
512  remove_skip(body);
513 
514  // recalculate numbers, etc.
515  goto_model.goto_functions.update();
516 }

518 void dfcc_instrumentt::instrument_instructions(
519  const irep_idt &function_id,
520  goto_programt &goto_program,
521  goto_programt::targett first_instruction,
522  const goto_programt::targett &last_instruction,
523  dfcc_cfg_infot &cfg_info,
524  std::set<irep_idt> &function_pointer_contracts)
525 {
526  // rewrite pointer_in_range calls
527  dfcc_pointer_in_ranget pointer_in_range(library, message_handler);
528  pointer_in_range.rewrite_calls(
529  goto_program, first_instruction, last_instruction, cfg_info);
530 
531  // rewrite is_fresh calls
532  dfcc_is_fresht is_fresh(library, message_handler);
533  is_fresh.rewrite_calls(
534  goto_program, first_instruction, last_instruction, cfg_info);
535 
536  // rewrite is_freeable/was_freed calls
537  dfcc_is_freeablet is_freeable(library, message_handler);
538  is_freeable.rewrite_calls(
539  goto_program, first_instruction, last_instruction, cfg_info);
540 
541  // rewrite obeys_contract calls
542  dfcc_obeys_contractt obeys_contract(library, message_handler);
543  obeys_contract.rewrite_calls(
544  goto_program,
545  first_instruction,
546  last_instruction,
547  cfg_info,
548  function_pointer_contracts);
549 
550  const namespacet ns(goto_model.symbol_table);
551  auto &target = first_instruction;
552 
553  // excluding the last
554  while(target != last_instruction)
555  {
556  if(target->is_decl())
557  {
558  instrument_decl(function_id, target, goto_program, cfg_info);
559  }
560  if(target->is_dead())
561  {
562  instrument_dead(function_id, target, goto_program, cfg_info);
563  }
564  else if(target->is_assign())
565  {
566  instrument_assign(function_id, target, goto_program, cfg_info);
567  }
568  else if(target->is_function_call())
569  {
570  instrument_function_call(function_id, target, goto_program, cfg_info);
571  }
572  else if(target->is_other())
573  {
574  instrument_other(function_id, target, goto_program, cfg_info);
575  }
576  // else do nothing
577  target++;
578  }
579 }

761 void dfcc_instrumentt::instrument_assign(
762  const irep_idt &function_id,
763  goto_programt::targett &target,
764  goto_programt &goto_program,
765  dfcc_cfg_infot &cfg_info)
766 {
767  const auto &lhs = target->assign_lhs();
768  const auto &rhs = target->assign_rhs();
769  const auto &target_location = target->source_location();
770  auto &write_set = cfg_info.get_write_set(target);
771 
772  // check the lhs
773  if(cfg_info.must_check_lhs(target))
774  instrument_lhs(function_id, target, lhs, goto_program, cfg_info);
775 
776  // handle dead_object updates (created by __builtin_alloca for instance)
777  // Remark: we do not really need to track this deallocation since the default
778  // CBMC checks are already able to detect writes to DEAD objects
779  const auto dead_ptr = is_dead_object_update(lhs, rhs);
780  if(dead_ptr.has_value())
781  {
782  // ```
783  // ASSIGN dead_object := if_exprt(nondet, ptr, dead_object);
784  // ----
785  // IF !write_set GOTO skip_target;
786  // CALL record_deallocated(write_set, ptr);
787  // skip_target: SKIP;
788  // ```
789 
790  // step over the instruction
791  target++;
792 
793  goto_programt payload;
794 
795  auto goto_instruction = payload.add(goto_programt::make_incomplete_goto(
796  dfcc_utilst::make_null_check_expr(write_set), target_location));
797 
798  payload.add(goto_programt::make_function_call(
799  library.write_set_record_dead_call(
800  write_set, dead_ptr.value(), target_location),
801  target_location));
802 
803  auto label_instruction =
804  payload.add(goto_programt::make_skip(target_location));
805  goto_instruction->complete_goto(label_instruction);
806 
807  insert_before_swap_and_advance(goto_program, target, payload);
808 
809  // step back
810  target--;
811  }
812 
813  // is the rhs expression a side_effect("allocate") expression ?
814  if(rhs.id() == ID_side_effect && rhs.get(ID_statement) == ID_allocate)
815  {
816  // ```
817  // CALL lhs := side_effect(statement = ID_allocate, args = {size, clear});
818  // ----
819  // IF !write_set GOTO skip_target;
820  // CALL add_allocated(write_set, lhs);
821  // skip_target: SKIP;
822  // ```
823 
824  // step over the instruction
825  target++;
826 
827  goto_programt payload;
828  auto goto_instruction = payload.add(goto_programt::make_incomplete_goto(
829  dfcc_utilst::make_null_check_expr(write_set), target_location));
830 
831  payload.add(goto_programt::make_function_call(
832  library.write_set_add_allocated_call(write_set, lhs, target_location),
833  target_location));
834 
835  auto label_instruction =
836  payload.add(goto_programt::make_skip(target_location));
837  goto_instruction->complete_goto(label_instruction);
838 
839  insert_before_swap_and_advance(goto_program, target, payload);
840 
841  // step back
842  target--;
843  }
844 }

846 void dfcc_instrumentt::instrument_fptr_call_instruction_dynamic_lookup(
847  const exprt &write_set,
848  goto_programt::targett &target,
849  goto_programt &goto_program)
850 {
851  // Insert a dynamic lookup in __instrumented_functions_map
852  // and pass the write set only to functions that are known to be able
853  // to accept it.
854  //
855  // ```
856  // IF __instrumented_functions_map[__CPROVER_POINTER_OBJECT(fptr)] != 1
857  // GOTO no_inst;
858  // CALL [lhs] = fptr(params, write_set);
859  // GOTO end;
860  // no_inst:
861  // CALL [lhs] = fptr(params);
862  // end:
863  // SKIP;
864  // ---
865  // SKIP // [lhs] = fptr(params) turned into SKIP
866  // ```
867  const auto &target_location = target->source_location();
868  const auto &callf = target->call_function();
869  auto object_id = pointer_object(
870  (callf.id() == ID_dereference) ? to_dereference_expr(callf).pointer()
871  : address_of_exprt(callf));
872  auto index_expr = index_exprt(
873  library.get_instrumented_functions_map_symbol().symbol_expr(), object_id);
874  auto cond = notequal_exprt(index_expr, from_integer(1, unsigned_char_type()));
875  goto_programt payload;
876  auto goto_no_inst =
877  payload.add(goto_programt::make_incomplete_goto(cond, target_location));
878  code_function_callt call_inst(
879  target->call_lhs(), target->call_function(), target->call_arguments());
880  call_inst.arguments().push_back(write_set);
881  payload.add(goto_programt::make_function_call(call_inst, target_location));
882  auto goto_end_inst = payload.add(
883  goto_programt::make_incomplete_goto(true_exprt(), target_location));
884  auto no_inst_label = payload.add(goto_programt::make_skip(target_location));
885  goto_no_inst->complete_goto(no_inst_label);
886  code_function_callt call_no_inst(
887  target->call_lhs(), target->call_function(), target->call_arguments());
888  payload.add(goto_programt::make_function_call(call_no_inst, target_location));
889  auto end_label = payload.add(goto_programt::make_skip(target_location));
890  goto_end_inst->complete_goto(end_label);
891  // erase the original call
892  target->turn_into_skip();
893  insert_before_swap_and_advance(goto_program, target, payload);
894 }
