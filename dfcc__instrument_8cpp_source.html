355  auto generate_function_bodies = generate_function_bodies_factory(

422 void dfcc_instrumentt::instrument_instructions(
423  const irep_idt &function_id,
424  const exprt &write_set,
425  goto_programt &goto_program,
426  goto_programt::targett first_instruction,
427  const goto_programt::targett &last_instruction,
428  cfg_infot &cfg_info,
429  const std::function<bool(const goto_programt::targett &)> &pred,
430  std::set<irep_idt> &function_pointer_contracts)
431 {
432  // rewrite pointer_in_range calls
433  dfcc_pointer_in_ranget pointer_in_range(library, message_handler);
434  pointer_in_range.rewrite_calls(
435  goto_program, first_instruction, last_instruction, write_set);
436 
437  // rewrite is_fresh calls
438  dfcc_is_fresht is_fresh(library, message_handler);
439  is_fresh.rewrite_calls(
440  goto_program, first_instruction, last_instruction, write_set);
441 
442  // rewrite is_freeable/was_freed calls
443  dfcc_is_freeablet is_freeable(library, message_handler);
444  is_freeable.rewrite_calls(
445  goto_program, first_instruction, last_instruction, write_set);
446 
447  // rewrite obeys_contract calls
448  dfcc_obeys_contractt obeys_contract(library, message_handler);
449  obeys_contract.rewrite_calls(
450  goto_program,
451  first_instruction,
452  last_instruction,
453  write_set,
454  function_pointer_contracts);
455 
456  const namespacet ns(goto_model.symbol_table);
457  auto &target = first_instruction;
458 
459  // excluding the last
460  while(target != last_instruction)
461  {
462  // Skip instructions marked as disabled for assigns clause checking
463  // or rejected by the predicate
464  if(pred && !pred(target))
465  {
466  target++;
467  continue;
468  }
469 
470  if(target->is_decl())
471  {
472  instrument_decl(function_id, write_set, target, goto_program, cfg_info);
473  }
474  if(target->is_dead())
475  {
476  instrument_dead(function_id, write_set, target, goto_program, cfg_info);
477  }
478  else if(target->is_assign())
479  {
480  instrument_assign(function_id, write_set, target, goto_program, cfg_info);
481  }
482  else if(target->is_function_call())
483  {
484  instrument_function_call(
485  function_id, write_set, target, goto_program, cfg_info);
486  }
487  else if(target->is_other())
488  {
489  instrument_other(function_id, write_set, target, goto_program, cfg_info);
490  }
491  // else do nothing
492  target++;
493  }
494 }

604 bool dfcc_instrumentt::must_check_lhs(
605  const source_locationt &lhs_source_location,
606  source_locationt &check_source_location,
607  const irep_idt &language_mode,
608  const exprt &lhs,
609  const cfg_infot &cfg_info)
610 {
611  if(can_cast_expr<symbol_exprt>(lhs))
612  {
613  const auto &symbol_expr = to_symbol_expr(lhs);
614  const auto &id = symbol_expr.get_identifier();
615 
616  check_source_location.set_comment(
617  "Check that " + from_expr_using_mode(ns, language_mode, lhs) +
618  " is assignable");
619 
620  if(cfg_info.is_local(id))
621  return false;
622 
623  // this is a global symbol. Ignore if it is one of the CPROVER globals
624  if(is_cprover_symbol(id))
625  {
626  check_source_location.set_comment(
627  "Check that " + from_expr_using_mode(ns, language_mode, lhs) +
628  " is assignable");
629 
630  return false;
631  }
632 
633  return true;
634  }
635  else
636  {
637  // This is a more complex expression.
638  // Since non-dirty locals are not tracked explicitly in the write set,
639  // we need to skip the check if we can verify that the expression describes
640  // an access to a non-dirty local symbol or an function parameter,
641  // otherwise the check will necessarily fail.
642  // If the expression contains address_of operator,
643  // the assignment gets checked. If the base object of the expression
644  // is a local or a function parameter, it will also be flagged as dirty and
645  // will be tracked explicitly, and the check will pass.
646  // If the expression contains a dereference operation, the assignment gets
647  // checked. If the dereferenced address was computed from a local object,
648  // from a function parameter or returned by a local malloc,
649  // then the object will also be tracked explicitly for being dirty,
650  // and the check will pass.
651  // In all other cases (address of a non-local object, or dereference of
652  // a non-locally computed address) the location must be given explicitly
653  // in the assigns clause to be allowed for assignment and we must check
654  // the assignment.
655  if(cfg_info.is_local_composite_access(lhs))

858 void dfcc_instrumentt::instrument_fptr_call_instruction_dynamic_lookup(
859  const exprt &write_set,
860  goto_programt::targett &target,
861  goto_programt &goto_program)
862 {
863  // Insert a dynamic lookup in __instrumented_functions_map
864  // and pass the write set only to functions that are known to be able
865  // to accept it.
866  //
867  // ```
868  // IF __instrumented_functions_map[__CPROVER_POINTER_OBJECT(fptr)] != 1
869  // GOTO no_inst;
870  // CALL [lhs] = fptr(params, write_set);
871  // GOTO end;
872  // no_inst:
873  // CALL [lhs] = fptr(params);
874  // end:
875  // SKIP;
876  // ---
877  // SKIP // [lhs] = fptr(params) turned into SKIP
878  // ```
879  const auto &target_location = target->source_location();
880  const auto &callf = target->call_function();
881  auto object_id = pointer_object(
882  (callf.id() == ID_dereference) ? to_dereference_expr(callf).pointer()
883  : address_of_exprt(callf));
884  auto index_expr = index_exprt(
885  library.get_instrumented_functions_map_symbol().symbol_expr(), object_id);
886  auto cond = notequal_exprt(index_expr, from_integer(1, unsigned_char_type()));
887  goto_programt payload;
888  auto goto_no_inst =
889  payload.add(goto_programt::make_incomplete_goto(cond, target_location));
890  code_function_callt call_inst(
891  target->call_lhs(), target->call_function(), target->call_arguments());
892  call_inst.arguments().push_back(write_set);
893  payload.add(goto_programt::make_function_call(call_inst, target_location));
894  auto goto_end_inst = payload.add(
895  goto_programt::make_incomplete_goto(true_exprt(), target_location));
896  auto no_inst_label = payload.add(goto_programt::make_skip(target_location));
897  goto_no_inst->complete_goto(no_inst_label);
898  code_function_callt call_no_inst(
899  target->call_lhs(), target->call_function(), target->call_arguments());
900  payload.add(goto_programt::make_function_call(call_no_inst, target_location));
901  auto end_label = payload.add(goto_programt::make_skip(target_location));
902  goto_end_inst->complete_goto(end_label);
903  // erase the original call
904  target->turn_into_skip();
905  insert_before_swap_and_advance(goto_program, target, payload);
906 }
