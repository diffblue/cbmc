53  const propertiest &properties,
54  messaget &message_hander)
55 {
56  const auto any_failures = std::any_of(

58  properties.end(),

69 incremental_goto_checkert::resultt single_loop_incremental_symex_checkert::
70 operator()(propertiest &properties)
71 {
72  resultt result(resultt::progresst::DONE);
73 
74  std::chrono::duration<double> solver_runtime(0);
75 
76  // we haven't got an equation yet
77  if(!initial_equation_generated)
78  {
79  full_equation_generated = !symex.from_entry_point_of(
80  goto_symext::get_goto_function(goto_model), symex_symbol_table);
81 
82  // This might add new properties such as unwinding assertions, for instance.
83  update_properties_status_from_symex_target_equation(
84  properties, result.updated_properties, equation);
85 
86  initial_equation_generated = true;
87  }
88 
89  while(has_properties_to_check(properties))
90  {
91  // There are NOT_CHECKED or UNKNOWN properties.
92 
93  if(count_properties(properties, property_statust::UNKNOWN) > 0)
94  {
95  // We have UNKNOWN properties, i.e. properties that we can check
96  // on the current equation.
97 
98  log.status()
99  << "Passing problem to "
100  << property_decider.get_decision_procedure().decision_procedure_text()
101  << messaget::eom;
102 
103  const auto solver_start = std::chrono::steady_clock::now();
104 
105  if(!current_equation_converted)
106  {
107  postprocess_equation(symex, equation, options, ns, ui_message_handler);
108 
109  log.status() << "converting SSA" << messaget::eom;
110  equation.convert_without_assertions(
111  property_decider.get_decision_procedure());
112 
113  property_decider.update_properties_goals_from_symex_target_equation(
114  properties);
115 
116  // We convert the assertions in a new context.
117  property_decider.get_stack_decision_procedure().push();
118  equation.convert_assertions(
119  property_decider.get_decision_procedure(), false);
120  property_decider.convert_goals();
121 
122  current_equation_converted = true;
123  }
124 
125  property_decider.add_constraint_from_goals(
126  [&properties](const irep_idt &property_id) {
127  return is_property_to_check(properties.at(property_id).status);
128  });
129 
130  log.status()
131  << "Running "
132  << property_decider.get_decision_procedure().decision_procedure_text()
133  << messaget::eom;
134 
135  decision_proceduret::resultt dec_result = property_decider.solve();
136 
137  property_decider.update_properties_status_from_goals(
138  properties, result.updated_properties, dec_result, false);
139 
140  const auto solver_stop = std::chrono::steady_clock::now();
141  solver_runtime +=
142  std::chrono::duration<double>(solver_stop - solver_start);
143  log.status() << "Runtime decision procedure: " << solver_runtime.count()
144  << "s" << messaget::eom;
145 
146  result.progress =
147  dec_result == decision_proceduret::resultt::D_SATISFIABLE
148  ? resultt::progresst::FOUND_FAIL
149  : resultt::progresst::DONE;
150 
151  // We've got a trace to report.
152  if(result.progress == resultt::progresst::FOUND_FAIL)
153  break;
154 
155  // Nothing else to do with the current set of assertions.
156  // Let's pop them.
157  property_decider.get_stack_decision_procedure().pop();
158  }
159 
160  // Now we are finally done.
161  if(full_equation_generated)
162  {
163  // For now, we assume that UNKNOWN properties are PASS.
164  update_status_of_unknown_properties(
165  properties, result.updated_properties);
166 
167  // For now, we assume that NOT_REACHED properties are PASS.
168  update_status_of_not_checked_properties(
169  properties, result.updated_properties);
170 
171  break;
172  }
173 
174  output_incremental_status(properties, log);
175 
176  // We continue symbolic execution
177  full_equation_generated =
178  !symex.resume(goto_symext::get_goto_function(goto_model));
179  revert_slice(equation);
180 
181  // This might add new properties such as unwinding assertions, for instance.
182  update_properties_status_from_symex_target_equation(
183  properties, result.updated_properties, equation);
184 
185  current_equation_converted = false;
186  }
187 
188  return result;
189 }
