71 incremental_goto_checkert::resultt single_loop_incremental_symex_checkert::
72 operator()(propertiest &properties)
73 {
74  resultt result(resultt::progresst::DONE);
75 
76  std::chrono::duration<double> solver_runtime(0);
77 
78  // we haven't got an equation yet
79  if(!initial_equation_generated)
80  {
81  full_equation_generated = !symex.from_entry_point_of(
82  goto_symext::get_goto_function(goto_model), symex_symbol_table);
83 
84  // This might add new properties such as unwinding assertions, for instance.
85  update_properties_status_from_symex_target_equation(
86  properties, result.updated_properties, equation);
87 
88  initial_equation_generated = true;
89  }
90 
91  while(has_properties_to_check(properties))
92  {
93  // There are NOT_CHECKED or UNKNOWN properties.
94 
95  if(count_properties(properties, property_statust::UNKNOWN) > 0)
96  {
97  // We have UNKNOWN properties, i.e. properties that we can check
98  // on the current equation.
99 
100  log.status()
101  << "Passing problem to "
102  << property_decider.get_decision_procedure().decision_procedure_text()
103  << messaget::eom;
104 
105  const auto solver_start = std::chrono::steady_clock::now();
106 
107  if(!current_equation_converted)
108  {
109  postprocess_equation(symex, equation, options, ns, ui_message_handler);
110 
111  log.status() << "converting SSA" << messaget::eom;
112  equation.convert_without_assertions(
113  property_decider.get_decision_procedure());
114 
115  property_decider.update_properties_goals_from_symex_target_equation(
116  properties);
117 
118  // We convert the assertions in a new context.
119  property_decider.get_stack_decision_procedure().push();
120  equation.convert_assertions(
121  property_decider.get_decision_procedure(), false);
122  property_decider.convert_goals();
123 
124  current_equation_converted = true;
125  }
126 
127  property_decider.add_constraint_from_goals(
128  [&properties](const irep_idt &property_id) {
129  return is_property_to_check(properties.at(property_id).status);
130  });
131 
132  log.status()
133  << "Running "
134  << property_decider.get_decision_procedure().decision_procedure_text()
135  << messaget::eom;
136 
137  decision_proceduret::resultt dec_result = property_decider.solve();
138 
139  property_decider.update_properties_status_from_goals(
140  properties, result.updated_properties, dec_result, false);
141 
142  const auto solver_stop = std::chrono::steady_clock::now();
143  solver_runtime +=
144  std::chrono::duration<double>(solver_stop - solver_start);
145  log.status() << "Runtime decision procedure: " << solver_runtime.count()
146  << "s" << messaget::eom;
147 
148  result.progress =
149  dec_result == decision_proceduret::resultt::D_SATISFIABLE
150  ? resultt::progresst::FOUND_FAIL
151  : resultt::progresst::DONE;
152 
153  // We've got a trace to report.
154  if(result.progress == resultt::progresst::FOUND_FAIL)
155  break;
156 
157  // Nothing else to do with the current set of assertions.
158  // Let's pop them.
159  property_decider.get_stack_decision_procedure().pop();
160  }
161 
162  // Now we are finally done.
163  if(full_equation_generated)
164  {
165  // For now, we assume that UNKNOWN properties are PASS.
166  update_status_of_unknown_properties(
167  properties, result.updated_properties);
168 
169  // For now, we assume that NOT_REACHED properties are PASS.
170  update_status_of_not_checked_properties(
171  properties, result.updated_properties);
172 
173  break;
174  }
175 
176  output_incremental_status(properties, log);
177 
178  // We continue symbolic execution
179  full_equation_generated =
180  !symex.resume(goto_symext::get_goto_function(goto_model));
181  revert_slice(equation);
182 
183  // This might add new properties such as unwinding assertions, for instance.
184  update_properties_status_from_symex_target_equation(
185  properties, result.updated_properties, equation);
186 
187  current_equation_converted = false;
188  }
189 
190  return result;
191 }
