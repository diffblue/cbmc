43 {

74 std::string graphml_witnesst::convert_assign_rec(
75  const irep_idt &identifier,
76  const code_assignt &assign)
77 {
78  const auto cit = cache.find({identifier.get_no(), &assign.read()});
79  if(cit != cache.end())
80  return cit->second;
81 
82  std::string result;
83 
84  if(assign.rhs().id() == ID_array_list)
85  {
86  const array_list_exprt &array_list = to_array_list_expr(assign.rhs());
87  const auto &ops = array_list.operands();
88 
89  for(std::size_t listidx = 0; listidx != ops.size(); listidx += 2)
90  {
91  const index_exprt index{assign.lhs(), ops[listidx]};
92  if(!result.empty())
93  result += ' ';
94  result +=
95  convert_assign_rec(identifier, code_assignt{index, ops[listidx + 1]});
96  }
97  }
98  else if(assign.rhs().id() == ID_array)
99  {
100  const array_typet &type = to_array_type(assign.rhs().type());
101 
102  unsigned i=0;
103  for(const auto &op : assign.rhs().operands())
104  {
105  index_exprt index(
106  assign.lhs(), from_integer(i++, c_index_type()), type.element_type());
107  if(!result.empty())
108  result+=' ';
109  result += convert_assign_rec(identifier, code_assignt(index, op));
110  }
111  }
112  else if(assign.rhs().id()==ID_struct ||
113  assign.rhs().id()==ID_union)
114  {
115  // dereferencing may have resulted in an lhs that is the first
116  // struct member; undo this
117  if(
118  assign.lhs().id() == ID_member &&
119  assign.lhs().type() != assign.rhs().type())
120  {
121  code_assignt tmp=assign;
122  tmp.lhs()=to_member_expr(assign.lhs()).struct_op();
123 
124  return convert_assign_rec(identifier, tmp);
125  }
126  else if(assign.lhs().id()==ID_byte_extract_little_endian ||
127  assign.lhs().id()==ID_byte_extract_big_endian)
128  {
129  code_assignt tmp=assign;
130  tmp.lhs()=to_byte_extract_expr(assign.lhs()).op();
131 
132  return convert_assign_rec(identifier, tmp);
133  }
134 
135  const struct_union_typet &type=
136  to_struct_union_type(ns.follow(assign.lhs().type()));
137  const struct_union_typet::componentst &components=
138  type.components();
139 
140  exprt::operandst::const_iterator it=
141  assign.rhs().operands().begin();
142  for(const auto &comp : components)
143  {
144  DATA_INVARIANT(
145  comp.type().id() != ID_code, "struct member must not be of code type");
146  if(
147  comp.get_is_padding() ||
148  // for some reason #is_padding gets lost in *some* cases
149  comp.get_name().starts_with("$pad"))
150  continue;
151 
152  INVARIANT(
153  it != assign.rhs().operands().end(), "expression must have operands");
154 
155  member_exprt member(
156  assign.lhs(),
157  comp.get_name(),
158  it->type());
159  if(!result.empty())
160  result+=' ';
161  result+=convert_assign_rec(identifier, code_assignt(member, *it));
162  ++it;
163 
164  // for unions just assign to the first member
165  if(assign.rhs().id()==ID_union)
166  break;
167  }
168  }
169  else if(assign.rhs().id() == ID_with)
170  {
171  const with_exprt &with_expr = to_with_expr(assign.rhs());
172  const auto &ops = with_expr.operands();
173 
174  for(std::size_t i = 1; i < ops.size(); i += 2)
175  {
176  if(!result.empty())
177  result += ' ';
178 
179  if(ops[i].id() == ID_member_name)
180  {
181  const member_exprt member{
182  assign.lhs(), ops[i].get(ID_component_name), ops[i + 1].type()};
183  result +=
184  convert_assign_rec(identifier, code_assignt(member, ops[i + 1]));
185  }
186  else
187  {
188  const index_exprt index{assign.lhs(), ops[i]};
189  result +=
190  convert_assign_rec(identifier, code_assignt(index, ops[i + 1]));
191  }
192  }
193  }
194  else
195  {
196  exprt clean_rhs=assign.rhs();
197  remove_l0_l1(clean_rhs);
198 
199  exprt clean_lhs = assign.lhs();
200  remove_l0_l1(clean_lhs);
201  std::string lhs = expr_to_string(ns, identifier, clean_lhs);
202 
203  if(
204  lhs.find("#return_value") != std::string::npos ||
205  (lhs.find('$') != std::string::npos &&
206  has_prefix(lhs, "return_value___VERIFIER_nondet_")))
207  {
208  lhs="\\result";
209  }
210 
211  result = lhs + " = " + expr_to_string(ns, identifier, clean_rhs) + ";";
212  }
213 
214  cache.insert({{identifier.get_no(), &assign.read()}, result});
215  return result;
216 }

218 static bool filter_out(
219  const goto_tracet &goto_trace,
220  const goto_tracet::stepst::const_iterator &prev_it,
221  goto_tracet::stepst::const_iterator &it)
222 {
223  if(
224  it->hidden &&
225  (!it->pc->is_assign() || it->pc->assign_rhs().id() != ID_side_effect ||
226  it->pc->assign_rhs().get(ID_statement) != ID_nondet))
227  return true;
228 
229  if(!it->is_assignment() && !it->is_goto() && !it->is_assert())
230  return true;
231 
232  // we filter out steps with the same source location
233  // TODO: if these are assignments we should accumulate them into
234  // a single edge
235  if(
236  prev_it != goto_trace.steps.end() &&
237  prev_it->pc->source_location() == it->pc->source_location())
238  return true;
239 
240  if(it->is_goto() && it->pc->condition().is_true())
241  return true;
242 
243  const source_locationt &source_location = it->pc->source_location();
244 
245  if(source_location.is_nil() ||
246  source_location.get_file().empty() ||
247  source_location.is_built_in() ||
248  source_location.get_line().empty())
249  {
250  const irep_idt id = source_location.get_function();
251  // Do not filter out assertions in functions the name of which starts with
252  // CPROVER_PREFIX, because we need to maintain those as violation nodes:
253  // these are assertions generated, for examples, for memory leaks.
254  if(!id.starts_with(CPROVER_PREFIX) || !it->is_assert())
255  return true;
256  }
257 
258  return false;
259 }

261 static bool contains_symbol_prefix(const exprt &expr, const std::string &prefix)
262 {
263  if(
264  expr.id() == ID_symbol &&
265  to_symbol_expr(expr).get_identifier().starts_with(prefix))
266  {
267  return true;
268  }
269 
270  for(const auto &op : expr.operands())
271  {
272  if(contains_symbol_prefix(op, prefix))
273  return true;
274  }
275  return false;

279 void graphml_witnesst::operator()(const goto_tracet &goto_trace)
280 {
281  unsigned int max_thread_idx = 0;
282  bool trace_has_violation = false;
283  for(goto_tracet::stepst::const_iterator it = goto_trace.steps.begin();
284  it != goto_trace.steps.end();
285  ++it)
286  {
287  if(it->thread_nr > max_thread_idx)
288  max_thread_idx = it->thread_nr;
289  if(it->is_assert() && !it->cond_value)
290  trace_has_violation = true;
291  }
292 
293  graphml.key_values["sourcecodelang"]="C";
294 
295  const graphmlt::node_indext sink=graphml.add_node();
296  graphml[sink].node_name="sink";
297  graphml[sink].is_violation=false;
298  graphml[sink].has_invariant=false;
299 
300  if(max_thread_idx > 0 && trace_has_violation)
301  {
302  std::vector<graphmlt::node_indext> nodes;
303 
304  for(unsigned int i = 0; i <= max_thread_idx + 1; ++i)
305  {
306  nodes.push_back(graphml.add_node());
307  graphml[nodes.back()].node_name = "N" + std::to_string(i);
308  graphml[nodes.back()].is_violation = i == max_thread_idx + 1;
309  graphml[nodes.back()].has_invariant = false;
310  }
311 
312  for(auto it = nodes.cbegin(); std::next(it) != nodes.cend(); ++it)
313  {
314  xmlt edge("edge");
315  edge.set_attribute("source", graphml[*it].node_name);
316  edge.set_attribute("target", graphml[*std::next(it)].node_name);
317  const auto thread_id = std::distance(nodes.cbegin(), it);
318  xmlt &data = edge.new_element("data");
319  data.set_attribute("key", "createThread");
320  data.data = std::to_string(thread_id);
321  if(thread_id == 0)
322  {
323  xmlt &data = edge.new_element("data");
324  data.set_attribute("key", "enterFunction");
325  data.data = "main";
326  }
327  graphml[*std::next(it)].in[*it].xml_node = edge;
328  graphml[*it].out[*std::next(it)].xml_node = edge;
329  }
330 
331  // we do not provide any further details as CPAchecker does not seem to
332  // handle more detailed concurrency witnesses
333  return;
334  }
335 
336  // step numbers start at 1
337  std::vector<std::size_t> step_to_node(goto_trace.steps.size()+1, 0);
338 
339  goto_tracet::stepst::const_iterator prev_it=goto_trace.steps.end();
340  for(goto_tracet::stepst::const_iterator
341  it=goto_trace.steps.begin();
342  it!=goto_trace.steps.end();
343  it++) // we cannot replace this by a ranged for
344  {
345  if(filter_out(goto_trace, prev_it, it))
346  {
347  step_to_node[it->step_nr]=sink;
348 
349  continue;
350  }
351 
352  // skip declarations followed by an immediate assignment
353  goto_tracet::stepst::const_iterator next=it;
354  ++next;
355  if(
356  next != goto_trace.steps.end() &&
357  next->type == goto_trace_stept::typet::ASSIGNMENT &&
358  it->full_lhs == next->full_lhs &&
359  it->pc->source_location() == next->pc->source_location())
360  {
361  step_to_node[it->step_nr]=sink;
362 
363  continue;
364  }
365 
366  prev_it=it;
367 
368  const source_locationt &source_location = it->pc->source_location();
369 
370  const graphmlt::node_indext node=graphml.add_node();
371  graphml[node].node_name=
372  std::to_string(it->pc->location_number)+"."+std::to_string(it->step_nr);
373  graphml[node].file=source_location.get_file();
374  graphml[node].line=source_location.get_line();
375  graphml[node].is_violation=
376  it->type==goto_trace_stept::typet::ASSERT && !it->cond_value;
377  graphml[node].has_invariant=false;
378 
379  step_to_node[it->step_nr]=node;
380  }
381 
382  unsigned thread_id = 0;
383 
384  // build edges
385  for(goto_tracet::stepst::const_iterator
386  it=goto_trace.steps.begin();
387  it!=goto_trace.steps.end();
388  ) // no ++it
389  {
390  const std::size_t from=step_to_node[it->step_nr];
391 
392  // no outgoing edges from sinks or violation nodes
393  if(from == sink || graphml[from].is_violation)
394  {
395  ++it;
396  continue;
397  }
398 
399  auto next = std::next(it);
400  for(; next != goto_trace.steps.end() &&
401  (step_to_node[next->step_nr] == sink ||
402  pointee_address_equalt{}(it->pc, next->pc)); // NOLINT
403  ++next)
404  {
405  // advance
406  }
407  const std::size_t to=
408  next==goto_trace.steps.end()?
409  sink:step_to_node[next->step_nr];
410 
411  switch(it->type)
412  {
413  case goto_trace_stept::typet::ASSIGNMENT:
414  case goto_trace_stept::typet::ASSERT:
415  case goto_trace_stept::typet::GOTO:
416  case goto_trace_stept::typet::SPAWN:
417  {
418  xmlt edge(
419  "edge",
420  {{"source", graphml[from].node_name},
421  {"target", graphml[to].node_name}},
422  {});
423 
424  {
425  xmlt &data_f = edge.new_element("data");
426  data_f.set_attribute("key", "originfile");
427  data_f.data = id2string(graphml[from].file);
428 
429  xmlt &data_l = edge.new_element("data");
430  data_l.set_attribute("key", "startline");
431  data_l.data = id2string(graphml[from].line);
432 
433  xmlt &data_t = edge.new_element("data");
434  data_t.set_attribute("key", "threadId");
435  data_t.data = std::to_string(it->thread_nr);
436  }
437 
438  const auto lhs_object = it->get_lhs_object();
439  if(
440  it->type == goto_trace_stept::typet::ASSIGNMENT &&
441  lhs_object.has_value())
442  {
443  const std::string &lhs_id = id2string(lhs_object->get_identifier());
444  if(lhs_id.find("pthread_create::thread") != std::string::npos)
445  {
446  xmlt &data_t = edge.new_element("data");
447  data_t.set_attribute("key", "createThread");
448  data_t.data = std::to_string(++thread_id);
449  }
450  else if(
451  !contains_symbol_prefix(
452  it->full_lhs_value, SYMEX_DYNAMIC_PREFIX "::dynamic_object") &&
453  !contains_symbol_prefix(
454  it->full_lhs, SYMEX_DYNAMIC_PREFIX "::dynamic_object") &&
455  lhs_id.find("thread") == std::string::npos &&
456  lhs_id.find("mutex") == std::string::npos &&
457  (!it->full_lhs_value.is_constant() ||
458  !it->full_lhs_value.has_operands() ||
459  !has_prefix(
460  id2string(
461  to_multi_ary_expr(it->full_lhs_value).op0().get(ID_value)),
462  "INVALID-")))
463  {
464  xmlt &val = edge.new_element("data");
465  val.set_attribute("key", "assumption");
466 
467  code_assignt assign{it->full_lhs, it->full_lhs_value};
468  val.data = convert_assign_rec(lhs_id, assign);
469 
470  if(!it->function_id.empty())
471  {
472  xmlt &val_s = edge.new_element("data");
473  val_s.set_attribute("key", "assumption.scope");
474  irep_idt function_id = it->function_id;
475  const symbolt *symbol_ptr = nullptr;
476  if(!ns.lookup(lhs_id, symbol_ptr) && symbol_ptr->is_parameter)
477  {
478  function_id = lhs_id.substr(0, lhs_id.find("::"));
479  }
480  val_s.data = id2string(function_id);
481  }
482 
483  if(has_prefix(val.data, "\\result ="))
484  {
485  xmlt &val_f = edge.new_element("data");
486  val_f.set_attribute("key", "assumption.resultfunction");
487  val_f.data = id2string(it->function_id);
488  }
489  }
490  }
491  else if(it->type == goto_trace_stept::typet::GOTO && it->pc->is_goto())
492  {
493  }
494 
495  graphml[to].in[from].xml_node = edge;
496  graphml[from].out[to].xml_node = edge;
497 
498  break;
499  }
500 
501  case goto_trace_stept::typet::DECL:
502  case goto_trace_stept::typet::FUNCTION_CALL:
503  case goto_trace_stept::typet::FUNCTION_RETURN:
504  case goto_trace_stept::typet::LOCATION:
505  case goto_trace_stept::typet::ASSUME:
506  case goto_trace_stept::typet::INPUT:
507  case goto_trace_stept::typet::OUTPUT:
508  case goto_trace_stept::typet::SHARED_READ:
509  case goto_trace_stept::typet::SHARED_WRITE:
510  case goto_trace_stept::typet::MEMORY_BARRIER:
511  case goto_trace_stept::typet::ATOMIC_BEGIN:
512  case goto_trace_stept::typet::ATOMIC_END:
513  case goto_trace_stept::typet::DEAD:
514  case goto_trace_stept::typet::CONSTRAINT:
515  case goto_trace_stept::typet::NONE:
516  // ignore
517  break;
518  }
519 
520  it=next;
521  }

525 void graphml_witnesst::operator()(const symex_target_equationt &equation)
526 {
527  graphml.key_values["sourcecodelang"]="C";
528 
529  const graphmlt::node_indext sink=graphml.add_node();
530  graphml[sink].node_name="sink";
531  graphml[sink].is_violation=false;
532  graphml[sink].has_invariant=false;
533 
534  // step numbers start at 1
535  std::vector<std::size_t> step_to_node(equation.SSA_steps.size()+1, 0);
536 
537  std::size_t step_nr=1;
538  for(symex_target_equationt::SSA_stepst::const_iterator
539  it=equation.SSA_steps.begin();
540  it!=equation.SSA_steps.end();
541  it++, step_nr++) // we cannot replace this by a ranged for
542  {
543  const source_locationt &source_location = it->source.pc->source_location();
544 
545  if(
546  it->hidden ||
547  (!it->is_assignment() && !it->is_goto() && !it->is_assert()) ||
548  (it->is_goto() && it->source.pc->condition().is_true()) ||
549  source_location.is_nil() || source_location.is_built_in() ||
550  source_location.get_line().empty())
551  {
552  step_to_node[step_nr]=sink;
553 
554  continue;
555  }
556 
557  // skip declarations followed by an immediate assignment
558  symex_target_equationt::SSA_stepst::const_iterator next=it;
559  ++next;
560  if(
561  next != equation.SSA_steps.end() && next->is_assignment() &&
562  it->ssa_full_lhs == next->ssa_full_lhs &&
563  it->source.pc->source_location() == next->source.pc->source_location())
564  {
565  step_to_node[step_nr]=sink;
566 
567  continue;
568  }
569 
570  const graphmlt::node_indext node=graphml.add_node();
571  graphml[node].node_name=
572  std::to_string(it->source.pc->location_number)+"."+
573  std::to_string(step_nr);
574  graphml[node].file=source_location.get_file();
575  graphml[node].line=source_location.get_line();
576  graphml[node].is_violation=false;
577  graphml[node].has_invariant=false;
578 
579  step_to_node[step_nr]=node;
580  }
581 
582  // build edges
583  step_nr=1;
584  for(symex_target_equationt::SSA_stepst::const_iterator
585  it=equation.SSA_steps.begin();
586  it!=equation.SSA_steps.end();
587  ) // no ++it
588  {
589  const std::size_t from=step_to_node[step_nr];
590 
591  if(from==sink)
592  {
593  ++it;
594  ++step_nr;
595  continue;
596  }
597 
598  symex_target_equationt::SSA_stepst::const_iterator next=it;
599  std::size_t next_step_nr=step_nr;
600  for(++next, ++next_step_nr;
601  next!=equation.SSA_steps.end() &&
602  (step_to_node[next_step_nr]==sink || it->source.pc==next->source.pc);
603  ++next, ++next_step_nr)
604  {
605  // advance
606  }
607  const std::size_t to=
608  next==equation.SSA_steps.end()?
609  sink:step_to_node[next_step_nr];
610 
611  switch(it->type)
612  {
613  case goto_trace_stept::typet::ASSIGNMENT:
614  case goto_trace_stept::typet::ASSERT:
615  case goto_trace_stept::typet::GOTO:
616  case goto_trace_stept::typet::SPAWN:
617  {
618  xmlt edge(
619  "edge",
620  {{"source", graphml[from].node_name},
621  {"target", graphml[to].node_name}},
622  {});
623 
624  {
625  xmlt &data_f = edge.new_element("data");
626  data_f.set_attribute("key", "originfile");
627  data_f.data = id2string(graphml[from].file);
628 
629  xmlt &data_l = edge.new_element("data");
630  data_l.set_attribute("key", "startline");
631  data_l.data = id2string(graphml[from].line);
632  }
633 
634  if(
635  (it->is_assignment() || it->is_decl()) && it->ssa_rhs.is_not_nil() &&
636  it->ssa_full_lhs.is_not_nil())
637  {
638  irep_idt identifier = it->ssa_lhs.get_object_name();
639 
640  graphml[to].has_invariant = true;
641  code_assignt assign(it->ssa_lhs, it->ssa_rhs);
642  graphml[to].invariant = convert_assign_rec(identifier, assign);
643  graphml[to].invariant_scope = id2string(it->source.function_id);
644  }
645 
646  graphml[to].in[from].xml_node = edge;
647  graphml[from].out[to].xml_node = edge;
648 
649  break;
650  }
